## 前言

TypeScript 为我们提供了编译时的类型安全，但在运行时，数据验证仍然是一个挑战。来自用户输入、API 响应或数据库的数据可能不符合我们的预期。

Prisma 通过自动生成类型，为我们提供了强大的类型系统基础。结合 Zod 等验证库，我们可以实现真正的端到端类型安全——从数据库到前端，全程类型保护。

- Prisma 类型生成机制
- 类型导出和使用
- Prisma 类型完整参考
- Zod 集成验证

## 类型安全实战

### 类型安全的查询构建器

```typescript
import { Prisma } from "@prisma/client";

function findUser<T extends Prisma.UserFindUniqueArgs>(
  args: Prisma.SelectSubset<T, Prisma.UserFindUniqueArgs>
) {
  return prisma.user.findUnique(args);
}

// 使用
const user = findUser({
  where: { id: userId },
  select: {
    id: true,
    username: true,
  },
});
```

### 类型安全的创建函数

```typescript
type CreateUserInput = {
  email: string;
  username: string;
  password: string;
  role?: "USER" | "ADMIN" | "MODERATOR";
};

async function createUser(data: CreateUserInput) {
  return await prisma.user.create({
    data: {
      email: data.email,
      username: data.username,
      password: await hash(data.password),
      role: data.role || "USER",
    },
  });
}

// 使用
const user = await createUser({
  email: "user@example.com",
  username: "john",
  password: "secret",
  role: "ADMIN", // TypeScript 会提示可选值
});
```

### 类型安全的更新函数

```typescript
type UpdateUserInput = Partial<CreateUserInput>;

async function updateUser(id: string, data: UpdateUserInput) {
  return await prisma.user.update({
    where: { id },
    data: {
      ...data,
      ...(data.password && {
        password: await hash(data.password),
      }),
    },
  });
}
```

## Zod 集成验证

### 安装依赖

```bash
npm install zod
npm install @hono/zod-validator  # 如果使用 Hono
```

### 基础验证

```typescript
import { z } from "zod";

// 定义验证 Schema
const CreateUserSchema = z.object({
  email: z.string().email("无效的邮箱地址"),
  username: z
    .string()
    .min(3, "用户名至少3个字符")
    .max(20, "用户名最多20个字符")
    .regex(/^[a-zA-Z0-9_]+$/, "用户名只能包含字母、数字和下划线"),
  password: z
    .string()
    .min(8, "密码至少8个字符")
    .regex(/^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)/, "密码必须包含大小写字母和数字"),
  role: z.enum(["USER", "ADMIN", "MODERATOR"]).default("USER"),
});

// 类型推断
type CreateUserInput = z.infer<typeof CreateUserSchema>;

// 使用
function validateUserInput(data: unknown): CreateUserInput {
  return CreateUserSchema.parse(data);
}

// 安全解析（不抛出错误）
function validateUserInputSafe(data: unknown) {
  return CreateUserSchema.safeParse(data);
}

const result = validateUserInputSafe({
  email: "user@example.com",
  username: "john",
  password: "Secret123",
  role: "ADMIN",
});

if (result.success) {
  const userData: CreateUserInput = result.data;
  // userData 现在是类型安全的
} else {
  console.log(result.error.errors);
}
```

### 从 Prisma 生成 Zod Schema

```typescript
import { z } from "zod";

// 根据 Prisma Schema 生成 Zod Schema
const CreateUserZodSchema = z.object({
  email: z.string().email(),
  username: z.string().min(3).max(20),
  password: z.string().min(8),
  role: z.enum(["USER", "ADMIN", "MODERATOR"]).optional(),
  avatar: z.string().url().optional(),
  bio: z.string().max(500).optional(),
});

// 与 Prisma 类型结合
type CreateUserInput = z.infer<typeof CreateUserZodSchema>;

async function createUser(data: CreateUserInput) {
  // data 已经是类型安全的
  return await prisma.user.create({
    data: {
      ...data,
      password: await hash(data.password),
    },
  });
}
```

### 更复杂的验证

```typescript
import { z } from "zod";

const RegisterSchema = z
  .object({
    email: z.string().email(),
    username: z.string().min(3).max(20),
    password: z.string().min(8),
    confirmPassword: z.string(),
  })
  .refine((data) => data.password === data.confirmPassword, {
    message: "密码不匹配",
    path: ["confirmPassword"],
  });

// 带自定义验证
const UsernameSchema = z
  .string()
  .min(3)
  .max(20)
  .refine(
    async (username) => {
      const user = await prisma.user.findUnique({
        where: { username },
      });
      return !user; // 用户名不存在
    },
    {
      message: "用户名已被占用",
    }
  );

// 条件验证
const ProfileSchema = z
  .object({
    accountType: z.enum(["individual", "company"]),
    firstName: z.string().optional(),
    lastName: z.string().optional(),
    companyName: z.string().optional(),
    taxId: z.string().optional(),
  })
  .refine(
    (data) => {
      if (data.accountType === "individual") {
        return data.firstName && data.lastName;
      }
      return true;
    },
    {
      message: "个人账户需要填写姓名",
    }
  )
  .refine(
    (data) => {
      if (data.accountType === "company") {
        return data.companyName && data.taxId;
      }
      return true;
    },
    {
      message: "公司账户需要填写公司名称和税号",
    }
  );
```

### 使用 zValidator 中间件

```typescript
import { Hono } from "hono";
import { zValidator } from "@hono/zod-validator";
import { z } from "zod";
import { prisma } from "../lib/prisma";

const app = new Hono();

// 定义验证 Schema
const createUserSchema = z.object({
  email: z.string().email(),
  username: z.string().min(3).max(20),
  password: z.string().min(8),
  role: z.enum(["USER", "ADMIN"]).default("USER"),
});

// POST /users
app.post(
  "/users",
  zValidator("json", createUserSchema, (result, c) => {
    if (!result.success) {
      return c.json(
        {
          success: false,
          errors: result.error.errors.map((e) => ({
            field: e.path.join("."),
            message: e.message,
          })),
        },
        400
      );
    }
  }),
  async (c) => {
    const data = c.req.valid("json"); // 类型已推断

    // data: {
    //   email: string
    //   username: string
    //   password: string
    //   role: 'USER' | 'ADMIN'
    // }

    const user = await prisma.user.create({
      data: {
        ...data,
        password: await hash(data.password),
      },
    });

    return c.json(
      {
        success: true,
        data: {
          id: user.id,
          email: user.email,
          username: user.username,
        },
      },
      201
    );
  }
);
```

### 查询参数验证

```typescript
const getPostsSchema = z.object({
  page: z.string().transform(Number).default("1"),
  limit: z.string().transform(Number).default("10"),
  keyword: z.string().optional(),
  status: z.enum(["DRAFT", "PUBLISHED", "ARCHIVED"]).optional(),
});

app.get("/posts", zValidator("query", getPostsSchema), async (c) => {
  const { page, limit, keyword, status } = c.req.valid("query");

  const where: any = {};
  if (keyword) {
    where.OR = [
      { title: { contains: keyword } },
      { content: { contains: keyword } },
    ];
  }
  if (status) {
    where.status = status;
  }

  const [posts, total] = await Promise.all([
    prisma.post.findMany({
      where,
      skip: (page - 1) * limit,
      take: limit,
      orderBy: { createdAt: "desc" },
    }),
    prisma.post.count({ where }),
  ]);

  return c.json({
    data: posts,
    meta: {
      page,
      limit,
      total,
      totalPages: Math.ceil(total / limit),
    },
  });
});
```

## 自定义验证中间件

### 数据库唯一性验证

```typescript
import { z } from "zod";
import { prisma } from "../lib/prisma";

// 检查邮箱唯一性
const checkEmailUnique = async (email: string) => {
  const user = await prisma.user.findUnique({
    where: { email },
  });
  return !user;
};

// 检查用户名唯一性
const checkUsernameUnique = async (username: string) => {
  const user = await prisma.user.findUnique({
    where: { username },
  });
  return !user;
};

// 组合验证
const RegisterSchema = z
  .object({
    email: z.string().email(),
    username: z.string().min(3).max(20),
    password: z.string().min(8),
  })
  .refine(async (data) => await checkEmailUnique(data.email), {
    message: "邮箱已被注册",
    path: ["email"],
  })
  .refine(async (data) => await checkUsernameUnique(data.username), {
    message: "用户名已被占用",
    path: ["username"],
  });
```

### 业务逻辑验证

```typescript
import { z } from "zod";

// 验证年龄
const ageSchema = z.number().int().min(18, "必须年满18岁").max(120, "年龄无效");

// 验证日期范围
const eventSchema = z
  .object({
    startDate: z.date(),
    endDate: z.date(),
  })
  .refine((data) => data.endDate > data.startDate, {
    message: "结束日期必须晚于开始日期",
    path: ["endDate"],
  });

// 验证库存
const orderItemSchema = z
  .object({
    productId: z.string().uuid(),
    quantity: z.number().positive(),
  })
  .refine(
    async (item) => {
      const product = await prisma.product.findUnique({
        where: { id: item.productId },
      });
      return product ? product.stock >= item.quantity : false;
    },
    {
      message: "库存不足",
      path: ["quantity"],
    }
  );
```

## 类型保护

### 运行时类型检查

```typescript
// 从数据库获取的用户
const user = await prisma.user.findUnique({
  where: { id: userId },
});

// 类型保护
if (user === null) {
  throw new Error("用户不存在");
}

// 现在 TypeScript 知道 user 不为 null
console.log(user.email);

// 或者使用可选链
const email = user?.email ?? "default@example.com";
```

### 自定义类型保护

```typescript
function isPublishedPost(
  post: Prisma.PostCreateInput | null
): post is Prisma.PostGetPayload<{
  include: { author: true };
}> {
  return post !== null && post.status === "PUBLISHED";
}

const post = await prisma.post.findUnique({
  where: { id: postId },
  include: { author: true },
});

if (isPublishedPost(post)) {
  // post 的类型现在包含 author
```

### 类型安全的分页

```typescript
interface PaginatedResult<T> {
  data: T[];
  meta: {
    page: number;
    limit: number;
    total: number;
    totalPages: number;
  };
}

async function paginatePosts<T extends Prisma.PostFindManyArgs>(
  page: number,
  limit: number,
  args?: Prisma.SelectSubset<T, Prisma.UserFindManyArgs>
): Promise<PaginatedResult<Prisma.PostGetPayload<T>>> {
  const [data, total] = await Promise.all([
    prisma.post.findMany({
      ...args,
      skip: (page - 1) * limit,
      take: limit,
    }),
    prisma.post.count(args?.where),
  ]);

  return {
    data,
    meta: {
      page,
      limit,
      total,
      totalPages: Math.ceil(total / limit),
    },
  };
}

// 使用
const result = await paginatePosts(1, 10, {
  where: { status: "PUBLISHED" },
  include: { author: true },
});
```

## 完整示例：验证层

```typescript
// lib/validators/user.ts
import { z } from "zod";
import { prisma } from "../prisma";

export const createUserSchema = z.object({
  email: z
    .string()
    .email("无效的邮箱地址")
    .refine(
      async (email) => {
        const user = await prisma.user.findUnique({ where: { email } });
        return !user;
      },
      { message: "邮箱已被注册" }
    ),
  username: z
    .string()
    .min(3, "用户名至少3个字符")
    .max(20, "用户名最多20个字符")
    .regex(/^[a-zA-Z0-9_]+$/, "只能包含字母、数字和下划线")
    .refine(
      async (username) => {
        const user = await prisma.user.findUnique({ where: { username } });
        return !user;
      },
      { message: "用户名已被占用" }
    ),
  password: z
    .string()
    .min(8, "密码至少8个字符")
    .regex(/^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)/, "必须包含大小写字母和数字"),
  role: z.enum(["USER", "ADMIN", "MODERATOR"]).default("USER"),
});

export const updateUserSchema = createUserSchema.partial();

export type CreateUserInput = z.infer<typeof createUserSchema>;
export type UpdateUserInput = z.infer<typeof updateUserSchema>;

// routes/users.ts
import { Hono } from "hono";
import { zValidator } from "@hono/zod-validator";
import { createUserSchema, updateUserSchema } from "../lib/validators/user";
import { prisma } from "../lib/prisma";

const users = new Hono();

users.post("/", zValidator("json", createUserSchema), async (c) => {
  const data = c.req.valid("json");

  const user = await prisma.user.create({
    data: {
      ...data,
      password: await hash(data.password),
    },
    select: {
      id: true,
      email: true,
      username: true,
      role: true,
      createdAt: true,
    },
  });

  return c.json({ success: true, data: user }, 201);
});

users.put("/:id", zValidator("json", updateUserSchema), async (c) => {
  const id = c.req.param("id");
  const data = c.req.valid("json");

  const user = await prisma.user.update({
    where: { id },
    data: {
      ...data,
      ...(data.password && {
        password: await hash(data.password),
      }),
    },
    select: {
      id: true,
      email: true,
      username: true,
      role: true,
      updatedAt: true,
    },
  });

  return c.json({ success: true, data: user });
});

export default users;
```

## 最佳实践

### 1. 始终使用类型推断

```typescript
// ✅ 好：使用 z.infer
type CreateUserInput = z.infer<typeof CreateUserSchema>;

// ❌ 差：手动定义类型
type CreateUserInput = {
  email: string;
  username: string;
  // ...
};
```

### 2. 分离验证和业务逻辑

```typescript
// ✅ 好：验证层独立
const validatedData = createUserSchema.parse(rawData);
const user = await createUser(validatedData);

// ❌ 差：混在一起
const user = await createUser(rawData); // createUser 内部验证
```

### 3. 提供友好的错误信息

```typescript
// ✅ 好：详细的错误信息
password: z.string()
  .min(8, "密码至少8个字符")
  .regex(/.../, "密码必须包含大小写字母和数字");

// ❌ 差：简单的错误
password: z.string().min(8);
```

### 4. 使用类型保护

```typescript
// ✅ 好：使用类型保护
if (user === null) {
  throw new Error("User not found");
}
user.email; // 类型安全

// ❌ 差：使用非空断言
user!.email; // 不安全
```

## 常见问题

**Q: Zod 和 Prisma 类型可以互相转换吗？**

A: 可以使用 `z.infer` 从 Zod Schema 生成类型，或手动创建对应的 Zod Schema。

**Q: 如何处理嵌套验证？**

A: 使用 Zod 的对象嵌套：

```typescript
const schema = z.object({
  user: z.object({
    email: z.string().email(),
    username: z.string(),
  }),
});
```

**Q: 验证失败如何处理？**

A: 使用 `safeParse` 而不是 `parse`，避免抛出错误：

```typescript
const result = schema.safeParse(data);
if (!result.success) {
  // 处理错误
}
```
