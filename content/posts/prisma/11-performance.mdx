## 查询优化策略

### 只查询需要的字段

```typescript
// ❌ 差：查询所有字段
const users = await prisma.user.findMany();

// ✅ 好：只查询需要的字段
const users = await prisma.user.findMany({
  select: {
    id: true,
    username: true,
    avatar: true,
  },
});
```

### 使用游标分页

```typescript
// ❌ 差：偏移分页（大数据量时性能差）
const posts = await prisma.post.findMany({
  skip: 10000,
  take: 10,
});

// ✅ 好：游标分页
const posts = await prisma.post.findMany({
  take: 10,
  cursor: { id: lastPostId },
  orderBy: { createdAt: "desc" },
});
```

### 避免重复查询

```typescript
// ❌ 差：重复查询相同数据
const user1 = await prisma.user.findUnique({ where: { id } });
const user2 = await prisma.user.findUnique({ where: { id } });

// ✅ 好：缓存结果或一次查询
const user = await prisma.user.findUnique({ where: { id } });
```

### 使用查询延迟

```typescript
// ✅ 好：延迟执行，只在需要时查询
const lazyPosts = prisma.post.findMany({
  where: { status: "PUBLISHED" },
});

// ... 其他操作

// 真正需要时才执行
const posts = await lazyPosts;
```

## 连接池配置

### 连接池大小

```typescript
// 默认配置（通常足够）
const prisma = new PrismaClient();

// 自定义连接池
const prisma = new PrismaClient({
  datasources: {
    db: {
      url: process.env.DATABASE_URL,
    },
  },
});

// 在 DATABASE_URL 中配置
// postgresql://user:password@host:5432/db?connection_limit=10&pool_timeout=20
```

### 连接池参数

```env
# .env
DATABASE_URL="postgresql://user:password@host:5432/db?connection_limit=10&pool_timeout=20&connect_timeout=10"

# connection_limit: 最大连接数（默认：num_cpus * 2 + 1）
# pool_timeout: 连接池超时（秒）
# connect_timeout: 连接超时（秒）
```

### 连接池最佳实践

```typescript
// ✅ 好：使用单例模式
const globalForPrisma = globalThis as unknown as {
  prisma: PrismaClient | undefined;
};

export const prisma =
  globalForPrisma.prisma ??
  new PrismaClient({
    log: process.env.NODE_ENV === "development" ? ["query"] : [],
  });

if (process.env.NODE_ENV !== "production") {
  globalForPrisma.prisma = prisma;
}

// ❌ 差：每次创建新连接
export function getPrisma() {
  return new PrismaClient(); // 每次都创建新实例
}
```

## 批量操作优化

### 批量创建

```typescript
// ❌ 差：循环创建
for (const data of dataArray) {
  await prisma.user.create({ data }); // N 次查询
}

// ✅ 好：批量创建
await prisma.user.createMany({
  data: dataArray, // 1 次查询
});
```

### 批量更新

```typescript
// ❌ 差：循环更新
for (const id of ids) {
  await prisma.user.update({
    where: { id },
    data: { status: "ACTIVE" },
  });
}

// ✅ 好：批量更新
await prisma.user.updateMany({
  where: {
    id: { in: ids },
  },
  data: { status: "ACTIVE" },
});
```

### 事务批量操作

```typescript
// ✅ 好：使用事务批量操作
const result = await prisma.$transaction([
  prisma.user.create({ data: user1 }),
  prisma.user.create({ data: user2 }),
  prisma.user.create({ data: user3 }),
]);

// ✅ 更好：使用 createMany
await prisma.user.createMany({
  data: [user1, user2, user3],
  skipDuplicates: true,
});
```

## 索引优化

### 为常用查询添加索引

```prisma
// schema.prisma
model Post {
  id        String   @id
  authorId  String
  status    String
  createdAt DateTime

  // 为常用查询组合创建索引
  @@index([authorId, status])
  @@index([status, createdAt])
  @@index([authorId, createdAt])
}
```

### 何时使用索引

```typescript
// ✅ 会使用索引的查询
const posts = await prisma.post.findMany({
  where: {
    authorId: userId,
    status: "PUBLISHED",
  },
  // 使用 @@index([authorId, status])
});

// ✅ 会使用索引的查询
const posts = await prisma.post.findMany({
  where: {
    authorId: userId,
  },
  orderBy: {
    createdAt: "desc",
  },
  // 使用 @@index([authorId, createdAt])
});

// ❌ 不会使用索引的查询（函数包装）
const posts = await prisma.post.findMany({
  where: {
    createdAt: {
      gte: new Date(), // 函数调用不使用索引
    },
  },
});
```

### 复合索引顺序

```prisma
// ✅ 好：按查询频率排序
model Post {
  authorId  String
  status    String
  createdAt DateTime

  // 最常用的查询在前
  @@index([authorId, status])
}

// ❌ 差：顺序错误
model Post {
  authorId  String
  status    String
  createdAt DateTime

  // 如果通常按 authorId 查询，status 应该在后面
  @@index([status, authorId])
}
```

### 全文索引

```prisma
model Post {
  id      String @id
  title   String
  content String @db.Text

  @@fulltext([title, content])
}

// 使用全文搜索
const posts = await prisma.post.findMany({
  where: {
    OR: [
      { title: { search: 'Prisma' } },
      { content: { search: 'Prisma' } },
    ],
  },
})
```

### 表结构优化

```prisma
// ✅ 好：使用合适的数据类型
model User {
  id       String @id @default(uuid())
  email    String
  age      Int                    // 整数使用 Int
  price    Decimal @db.Decimal(10, 2)  // 金额使用 Decimal
  isActive Boolean               // 布尔使用 Boolean
  metadata Json?                 // 灵活数据使用 JSON
}

// ❌ 差：所有字段都用 String

  id    String @id
  email String
```

### 数据归档

```typescript
// 定期归档旧数据
async function archiveOldPosts() {
  const sixMonthsAgo = new Date();
  sixMonthsAgo.setMonth(sixMonthsAgo.getMonth() - 6);

  const oldPosts = await prisma.post.findMany({
    where: {
      createdAt: { lte: sixMonthsAgo },
      status: "ARCHIVED",
    },
    select: { id: true },
  });

  // 移动到归档表
  await prisma.postArchive.createMany({
    data: await prisma.post.findMany({
      where: {
        id: { in: oldPosts.map((p) => p.id) },
      },
    }),
  });

  // 删除原表数据
  await prisma.post.deleteMany({
    where: {
      id: { in: oldPosts.map((p) => p.id) },
    },
  });
}
```

## N+1 问题解决

### 问题示例

```typescript
// ❌ N+1 问题
const users = await prisma.user.findMany(); // 1 次查询

for (const user of users) {
  const posts = await prisma.post.findMany({
    // N 次查询
    where: { authorId: user.id },
  });
  user.posts = posts;
}

// 总计：1 + N 次查询
```

### 解决方案 1：预加载

```typescript
// ✅ 使用 Include 一次查询
const users = await prisma.user.findMany({
  include: {
    posts: true,
  },
});

// 总计：1 次查询（使用 JOIN）
```

### 解决方案 2：分离查询

```typescript
// ✅ 先查询用户
const users = await prisma.user.findMany();

// 收集所有用户 ID
const userIds = users.map((u) => u.id);

// 一次性查询所有文章
const posts = await prisma.post.findMany({
  where: {
    authorId: { in: userIds },
  },
});

// 构建 Map
const postsMap = new Map<string, Post[]>();
posts.forEach((post) => {
  if (!postsMap.has(post.authorId)) {
    postsMap.set(post.authorId, []);
  }
  postsMap.get(post.authorId)!.push(post);
});

// 组合数据
const usersWithPosts = users.map((user) => ({
  ...user,
  posts: postsMap.get(user.id) || [],
}));

// 总计：2 次查询
```

### 解决方案 3：使用 DataLoader

```typescript
import DataLoader from "dataloader";

const postLoader = new DataLoader(async (authorIds: string[]) => {
  const posts = await prisma.post.findMany({
    where: {
      authorId: { in: authorIds },
    },
  });

  const postsByAuthor = new Map<string, Post[]>();
  posts.forEach((post) => {
    if (!postsByAuthor.has(post.authorId)) {
      postsByAuthor.set(post.authorId, []);
    }
    postsByAuthor.get(post.authorId)!.push(post);
  });

  return authorIds.map((id) => postsByAuthor.get(id) || []);
});

// 使用
const user = await prisma.user.findUnique({ where: { id } });
const posts = await postLoader.load(user.id);
```

## 性能监控

### 启用查询日志

```typescript
const prisma = new PrismaClient({
  log: [
    { level: "query", emit: "event" },
    { level: "error", emit: "stdout" },
    { level: "warn", emit: "stdout" },
  ],
});

prisma.$on("query" as any, (e: any) => {
  console.log("Query: " + e.query);
  console.log("Duration: " + e.duration + "ms");
});
```

### 自定义性能监控

```typescript
interface QueryStats {
  model: string;
  action: string;
  count: number;
  totalDuration: number;
  avgDuration: number;
  maxDuration: number;
}

const stats = new Map<string, QueryStats>();

prisma.$use(async (params, next) => {
  const start = Date.now();

  const result = await next(params);

  const duration = Date.now() - start;
  const key = `${params.model}.${params.action}`;

  if (!stats.has(key)) {
    stats.set(key, {
      model: params.model,
      action: params.action,
      count: 0,
      totalDuration: 0,
      avgDuration: 0,
      maxDuration: 0,
    });
  }

  const stat = stats.get(key)!;
  stat.count++;
  stat.totalDuration += duration;
  stat.avgDuration = stat.totalDuration / stat.count;
  stat.maxDuration = Math.max(stat.maxDuration, duration);

  return result;
});

// 获取性能统计
function getPerformanceStats() {
  return Array.from(stats.values()).sort(
    (a, b) => b.avgDuration - a.avgDuration
  );
}
```

## 完整优化示例

```typescript
// lib/optimized-queries.ts
import { PrismaClient } from "@prisma/client";

const prisma = new PrismaClient();

// ✅ 优化 1：批量创建
export async function createManyUsers(
  users: Array<{ email: string; username: string }>
) {
  return await prisma.user.createMany({
    data: users,
    skipDuplicates: true,
  });
}

// ✅ 优化 2：分页查询 + 预加载
export async function getPostsWithAuthors(page: number, limit: number) {
  const [posts, total] = await Promise.all([
    prisma.post.findMany({
      skip: (page - 1) * limit,
      take: limit,
      where: { status: "PUBLISHED" },
      include: {
        author: {
          select: {
            id: true,
            username: true,
            avatar: true,
          },
        },
      },
      orderBy: { createdAt: "desc" },
    }),
    prisma.post.count({ where: { status: "PUBLISHED" } }),
  ]);

  return {
    data: posts,
    meta: {
      page,
      limit,
      total,
      totalPages: Math.ceil(total / limit),
    },
  };
}

// ✅ 优化 3：使用游标分页
export async function getPostsCursor(cursor: string | null, limit: number) {
  return await prisma.post.findMany({
    take: limit,
    skip: cursor ? 1 : 0,
    cursor: cursor ? { id: cursor } : undefined,
    where: { status: "PUBLISHED" },
    orderBy: { createdAt: "desc" },
  });
}

// ✅ 优化 4：批量更新
export async function publishPosts(postIds: string[]) {
  return await prisma.post.updateMany({
    where: {
      id: { in: postIds },
    },
    data: {
      status: "PUBLISHED",
      publishedAt: new Date(),
    },
  });
}

// ✅ 优化 5：使用聚合
export async function getUserStats(userId: string) {
  const [postCount, commentCount, totalViews] = await Promise.all([
    prisma.post.count({
      where: { authorId: userId },
    }),
    prisma.comment.count({
      where: { authorId: userId },
    }),
    prisma.post.aggregate({
      where: { authorId: userId },
      _sum: { views: true },
    }),
  ]);

  return {
    postCount,
    commentCount,
    totalViews: totalViews._sum.views || 0,
  };
}
```

```typescript
import { performance } from "perf_hooks";

async function benchmarkQuery() {
  const iterations = 100;

  // 预热
  await prisma.user.findMany({ take: 1 });

  const start = performance.now();

  for (let i = 0; i < iterations; i++) {
    await prisma.user.findMany({

      include: { posts: true },

  const end = performance.now();
  const duration = end - start;

  console.log(`Total: ${duration}ms`);
  console.log(`Average: ${duration / iterations}ms`);
  console.log(`QPS: ${(iterations / duration) * 1000}`);
}
```

### 负载测试

```typescript
import { performance } from "perf_hooks";

async function loadTest(concurrency: number, iterations: number) {
  const start = performance.now();

  const promises = Array.from({ length: concurrency }, async (_, i) => {
    const results = [];
    for (let j = 0; j < iterations; j++) {
      const s = performance.now();
      await prisma.user.findUnique({ where: { id: `user-${i}` } });
      results.push(performance.now() - s);
    }
    return results;
  });

  const allResults = await Promise.all(promises);
  const flatResults = allResults.flat();

  const end = performance.now();

  console.log(`Total time: ${end - start}ms`);
  console.log(`Total queries: ${concurrency * iterations}`);
  console.log(
    `Average latency: ${flatResults.reduce((a, b) => a + b, 0) / flatResults.length}ms`
  );
  console.log(
    `P95 latency: ${flatResults.sort((a, b) => a - b)[Math.floor(flatResults.length * 0.95)]}ms`
  );
  console.log(
    `P99 latency: ${flatResults.sort((a, b) => a - b)[Math.floor(flatResults.length * 0.99)]}ms`
  );
}
```

## 常见问题

**Q: 如何检测慢查询？**

A: 启用查询日志并设置阈值：

```typescript
prisma.$on("query" as any, (e: any) => {
  if (e.duration > 1000) {
    console.warn("Slow query:", e.query, e.duration);
  }
});
```

**Q: Prisma 查询比原生 SQL 慢吗？**

A: Prisma 生成的查询通常与手写 SQL 性能相当。性能差异通常来自：

- 不必要的字段查询
- N+1 问题
- 缺少索引
- 过度的嵌套查询

**Q: 如何优化大数据量查询？**

A：

1. 使用分页（游标分页）
2. 只查询需要的字段
3. 使用流式处理
4. 考虑使用原始 SQL
