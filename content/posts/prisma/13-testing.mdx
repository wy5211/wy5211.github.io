## 测试类型和选择

### 单元测试

测试单个函数或类的行为。

```typescript
// 示例：测试用户验证函数
test("validateEmail should return true for valid email", () => {
  expect(validateEmail("user@example.com")).toBe(true);
});

test("validateEmail should return false for invalid email", () => {
  expect(validateEmail("invalid")).toBe(false);
});
```

**特点**：

- ✅ 快速执行
- ✅ 隔离性好
- ❌ 可能无法发现集成问题

### 集成测试

测试多个组件协同工作。

```typescript
// 示例：测试用户注册流程
test("should register user and send welcome email", async () => {
  const response = await app.request("/auth/register", {
    method: "POST",
    body: JSON.stringify({
      email: "user@example.com",
      password: "Secret123",
    }),
  });

  expect(response.status).toBe(201);

  // 验证数据库中创建了用户
  const user = await prisma.user.findUnique({
    where: { email: "user@example.com" },
  });

  expect(user).toBeTruthy();
});
```

**特点**：

- ✅ 测试真实交互
- ✅ 发现集成问题
- ❌ 执行较慢
- ❌ 可能需要数据库

## 单元测试

### 测试纯函数

```typescript
// lib/utils/user.ts
export function validateEmail(email: string): boolean {
  const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  return emailRegex.test(email);
}

export function hashPassword(password: string): string {
  // 简化版本
  return Buffer.from(password).toString("base64");
}

// tests/unit/user.test.ts
import { describe, it, expect } from "vitest";
import { validateEmail, hashPassword } from "../../lib/utils/user";

describe("validateEmail", () => {
  it("应该接受有效的邮箱地址", () => {
    expect(validateEmail("user@example.com")).toBe(true);
    expect(validateEmail("test.user+tag@domain.co.uk")).toBe(true);
  });

  it("应该拒绝无效的邮箱地址", () => {
    expect(validateEmail("invalid")).toBe(false);
    expect(validateEmail("user@")).toBe(false);
    expect(validateEmail("@example.com")).toBe(false);
    expect(validateEmail("user example.com")).toBe(false);
  });
});

describe("hashPassword", () => {
  it("应该返回哈希后的密码", () => {
    const password = "Secret123";
    const hashed = hashPassword(password);

    expect(hashed).not.toBe(password);
    expect(typeof hashed).toBe("string");
  });

  it("相同密码应该产生不同哈希", () => {
    const password = "Secret123";
    const hash1 = hashPassword(password);
    const hash2 = hashPassword(password);

    expect(hash1).not.toBe(hash2);
  });
});
```

### 测试业务逻辑

```typescript
// lib/services/user.service.ts
import { prisma } from '../db'

export class UserService {
  async createUser(data: { email: string; username: string }) {
    // 检查邮箱是否已存在
    const existing = await prisma.user.findUnique({
      where: { email: data.email },
    })

    if (existing) {
      throw new Error('EMAIL_EXISTS')
    }

    return await prisma.user.create({
      data,
    })
  }

  async getUserById(id: string) {
    const user = await prisma.user.findUnique({
      where: { id },
    })

    if (!user) {
      throw new Error('USER_NOT_FOUND')
    }

    return user
  }
}

// tests/unit/user.service.test.ts
import { describe, it, expect, vi, beforeEach } from 'vitest'
import { UserService } from '../../lib/services/user.service'
import { prisma } from '../../lib/db'

vi.mock('../../lib/db', () => ({
  prisma: {
    user: {
      findUnique: vi.fn(),
      create: vi.fn(),
    },
  },
}))

describe('UserService', () => {
  let userService: UserService

  beforeEach(() => {
    userService = new UserService()
    vi.clearAllMocks()
  })

  describe('createUser', () => {
    it('应该成功创建新用户', async () => {
      const userData = {
        email: 'user@example.com',
        username: 'john',
      }

      vi.mocked(prisma.user.findUnique).mockResolvedValue(null)
      vi.mocked(prisma.user.create).mockResolvedValue({
        id: '1',
        ...userData,
        createdAt: new Date(),
        updatedAt: new Date(),
      } as any)

      const result = await userService.createUser(userData)

      expect(result).toHaveProperty('id')
      expect(prisma.user.create).toHaveBeenCalledWith({
        data: userData,
      })
    })

    it('邮箱已存在时应该抛出错误', async () => {
      const userData = {
        email: 'existing@example.com',
        username: 'john',
      }

      vi.mocked(prisma.user.findUnique).mockResolvedValue({
        id: '1',
        ...userData,
      } as any)

      await expect(userService.createUser(userData)).rejects.toThrow(
        'EMAIL_EXISTS'
      )
    })
  })

  describe('getUserById', () => {
    it('应该返回用户', async () => {
      const user = {
        id: '1',
        email: 'user@example.com',
        username: 'john',
      }

      vi.mocked(prisma.user.findUnique).mockResolvedValue(user as any)

      const result = await userService.getUserById('1')

      expect(result).toEqual(user)
    })

    it('用户不存在时应该抛出错误', async () => {
      vi.mocked(prisma.user.findUnique).mockResolvedValue(null)

      await expect(userService.getUserById('1')).rejects.toThrow(
        'USER_NOT_FOUND'
      )
    })
  }
})
```

### API 测试

```typescript
// tests/integration/auth.test.ts
import { describe, it, expect, beforeEach, afterEach } from "vitest";
import { Hono } from "hono";
import { prisma } from "../lib/db";
import authRoutes from "../routes/auth";

describe("Auth API", () => {
  let app: Hono;

  beforeEach(async () => {
    app = new Hono();
    app.route("/auth", authRoutes);

    // 清理数据库
    await prisma.user.deleteMany();
  });

  afterEach(async () => {
    await prisma.$disconnect();
  });

  describe("POST /auth/register", () => {
    it("应该成功注册新用户", async () => {
      const response = await app.request("/auth/register", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
        },
        body: JSON.stringify({
          email: "user@example.com",
          username: "john",
          password: "Secret123",
        }),
      });

      expect(response.status).toBe(201);

      const json = await response.json();
      expect(json).toHaveProperty("data");
      expect(json.data).toHaveProperty("id");
      expect(json.data).toHaveProperty("email", "user@example.com");

      // 验证数据库
      const user = await prisma.user.findUnique({
        where: { email: "user@example.com" },
      });

      expect(user).toBeTruthy();
    });

    it("邮箱已存在时应该返回错误", async () => {
      // 先创建用户
      await prisma.user.create({
        data: {
          email: "user@example.com",
          username: "john",
          password: "hash",
        },
      });

      const response = await app.request("/auth/register", {
        method: "POST",
        body: JSON.stringify({
          email: "user@example.com",
          username: "jane",
          password: "Secret123",
        }),
      });

      expect(response.status).toBe(409);

      const json = await response.json();
      expect(json).toHaveProperty("error");
    });
  });
});
```

export function createTestPrisma() {

## 测试最佳实践

### 1. 测试命名

```typescript
// ✅ 好：清晰的测试名称
test("should return 404 when user does not exist", async () => {
  // ...
});

// ❌ 差：模糊的测试名称
test("test user", async () => {
  // ...
});
```

### 2. AAA 模式

```typescript
test("should create user", async () => {
  // Arrange（准备）
  const userData = {
    email: "user@example.com",
    username: "john",
  };

  // Act（执行）
  const user = await createUser(userData);

  // Assert（断言）
  expect(user).toHaveProperty("id");
  expect(user.email).toBe(userData.email);
});
```

### 3. 测试隔离

```typescript
// ✅ 好：每个测试独立
test("test 1", async () => {
  const user = await createTestUser();
  // ...
});

test("test 2", async () => {
  const user = await createTestUser();
  // ...
});

// ❌ 差：测试之间有依赖
let userId: string;

test("test 1", async () => {
  const user = await createTestUser();
  userId = user.id;
});

test("test 2", async () => {
  const user = await getUserById(userId); // 依赖 test 1
});
```

### 4. 避免测试实现细节

```typescript
// ✅ 好：测试行为
test("should create user with email", async () => {
  const user = await createUser({
    email: "user@example.com",
  });

  expect(user.email).toBe("user@example.com");
});

// ❌ 差：测试实现细节
test("should call prisma.user.create", async () => {
  await createUser({
    email: "user@example.com",
  });

  expect(prisma.user.create).toHaveBeenCalled();
});
```

## 常见问题

**Q: 应该使用单元测试还是集成测试？**

A: 两者结合：核心业务逻辑用单元测试，API 和数据库交互用集成测试。

**Q: 如何加快测试速度？**

A：

1. 使用 SQLite 作为测试数据库
2. 并行运行测试
3. 使用事务回滚而非清理

**Q: 需要测试所有代码吗？**

A：优先测试核心业务逻辑和边界情况，工具函数和简单逻辑可以不测。
