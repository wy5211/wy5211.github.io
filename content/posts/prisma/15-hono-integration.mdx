## 前言

Hono 是一个轻量级、快速的 Web 框架，特别适合构建 API 和边缘应用。结合 Prisma 的类型安全，可以构建出性能优异且开发体验出色的后端服务。

与 Next.js 不同，Hono 更专注于 API 层，它的体积更小、启动更快，非常适合部署在边缘运行时。

- 项目初始化和配置

### Posts 路由

```typescript
// src/routes/posts.ts
import { Hono } from "hono";
import { zValidator } from "@hono/zod-validator";
import { z } from "zod";

const posts = new Hono();

// 验证 Schema
const createPostSchema = z.object({
  title: z.string().min(1).max(200),
  slug: z.string().min(1).max(200),
  content: z.string().min(1),
  excerpt: z.string().optional(),
  status: z.enum(["DRAFT", "PUBLISHED"]).default("DRAFT"),
  authorId: z.string().uuid(),
});

const updatePostSchema = createPostSchema.partial();

const querySchema = z.object({
  page: z.string().transform(Number).default("1"),
  limit: z.string().transform(Number).default("10"),
  status: z.enum(["DRAFT", "PUBLISHED"]).optional(),
});

// GET /posts - 获取文章列表
posts.get("/", zValidator("query", querySchema), async (c) => {
  const db = c.get("db");
  const { page, limit, status } = c.req.valid("query");

  const where: any = {};
  if (status) {
    where.status = status;
  }

  const [posts, total] = await Promise.all([
    db.post.findMany({
      where,
      skip: (page - 1) * limit,
      take: limit,
      orderBy: { createdAt: "desc" },
      include: {
        author: {
          select: {
            id: true,
            username: true,
            avatar: true,
          },
        },
      },
    }),
    db.post.count({ where }),
  ]);

  return c.json({
    data: posts,
    meta: {
      page,
      limit,
      total,
      totalPages: Math.ceil(total / limit),
    },
  });
});

// GET /posts/:id - 获取单篇文章
posts.get(
  "/:id",
  zValidator(
    "param",
    z.object({
      id: z.string().uuid(),
    })
  ),
  async (c) => {
    const db = c.get("db");
    const { id } = c.req.valid("param");

    const post = await db.post.findUnique({
      where: { id },
      include: {
        author: {
          select: {
            id: true,
            username: true,
            email: true,
          },
        },
      },
    });

    if (!post) {
      return c.json(
        {
          success: false,
          error: "Post not found",
        },
        404
      );
    }

    return c.json({
      success: true,
      data: post,
    });
  }
);

// POST /posts - 创建文章
posts.post("/", zValidator("json", createPostSchema), async (c) => {
  const db = c.get("db");
  const data = c.req.valid("json");

  try {
    const post = await db.post.create({
      data: {
        ...data,
        publishedAt: data.status === "PUBLISHED" ? new Date() : null,
      },
    });

    return c.json(
      {
        success: true,
        data: post,
      },
      201
    );
  } catch (error) {
    return c.json(
      {
        success: false,
        error: "Failed to create post",
      },
      500
    );
  }
});

// PUT /posts/:id - 更新文章
posts.put(
  "/:id",
  zValidator(
    "param",
    z.object({
      id: z.string().uuid(),
    })
  ),
  zValidator("json", updatePostSchema),
  async (c) => {
    const db = c.get("db");
    const { id } = c.req.valid("param");
    const data = c.req.valid("json");

    try {
      const post = await db.post.update({
        where: { id },
        data: {
          ...data,
          updatedAt: new Date(),
        },
      });

      return c.json({
        success: true,
        data: post,
      });
    } catch (error) {
      return c.json(
        {
          success: false,
          error: "Post not found",
        },
        404
      );
    }
  }
);

// DELETE /posts/:id - 删除文章
posts.delete(
  "/:id",
  zValidator(
    "param",
    z.object({
      id: z.string().uuid(),
    })
  ),
  async (c) => {
    const db = c.get("db");
    const { id } = c.req.valid("param");

    try {
      await db.post.delete({
        where: { id },
      });

      return c.json({
        success: true,
        message: "Post deleted",
      });
    } catch (error) {
      return c.json(
        {
          success: false,
          error: "Post not found",
        },
        404
      );
    }
  }
);

export default posts;
```

### Users 路由

```typescript
// src/routes/users.ts
import { Hono } from "hono";
import { zValidator } from "@hono/zod-validator";
import { z } from "zod";

const users = new Hono();

const createUserSchema = z.object({
  email: z.string().email(),
  username: z.string().min(3).max(20),
  password: z.string().min(8),
  bio: z.string().optional(),
});

// GET /users - 获取用户列表
users.get("/", async (c) => {
  const db = c.get("db");

  const page = parseInt(c.req.query("page") || "1");
  const limit = parseInt(c.req.query("limit") || "10");

  const [users, total] = await Promise.all([
    db.user.findMany({
      skip: (page - 1) * limit,
      take: limit,
      select: {
        id: true,
        email: true,
        username: true,
        bio: true,
        avatar: true,
        createdAt: true,
      },
      orderBy: { createdAt: "desc" },
    }),
    db.user.count(),
  ]);

  return c.json({
    data: users,
    meta: {
      page,
      limit,
      total,
      totalPages: Math.ceil(total / limit),
    },
  });
});

// GET /users/:id - 获取用户详情
users.get(
  "/:id",
  zValidator(
    "param",
    z.object({
      id: z.string().uuid(),
    })
  ),
  async (c) => {
    const db = c.get("db");
    const { id } = c.req.valid("param");

    const user = await db.user.findUnique({
      where: { id },
      select: {
        id: true,
        email: true,
        username: true,
        bio: true,
        avatar: true,
        createdAt: true,
        posts: {
          select: {
            id: true,
            title: true,
            slug: true,
            createdAt: true,
          },
          take: 10,
          orderBy: { createdAt: "desc" },
        },
      },
    });

    if (!user) {
      return c.json(
        {
          success: false,
          error: "User not found",
        },
        404
      );
    }

    return c.json({
      success: true,
      data: user,
    });
  }
);

// POST /users - 创建用户
users.post("/", zValidator("json", createUserSchema), async (c) => {
  const db = c.get("db");
  const data = c.req.valid("json");

  try {
    // 检查邮箱是否已存在
    const existingUser = await db.user.findUnique({
      where: { email: data.email },
    });

    if (existingUser) {
      return c.json(
        {
          success: false,
          error: "Email already exists",
        },
        409
      );
    }

    // 哈希密码
    const hashedPassword = await hash(data.password);

    const user = await db.user.create({
      data: {
        email: data.email,
        username: data.username,
        password: hashedPassword,
        bio: data.bio,
      },
      select: {
        id: true,
        email: true,
        username: true,
        bio: true,
        avatar: true,
        createdAt: true,
      },
    });

    return c.json(
      {
        success: true,
        data: user,
      },
      201
    );
  } catch (error) {
    return c.json(
      {
        success: false,
        error: "Failed to create user",
      },
      500
    );
  }
});

export default users;
```

## 认证中间件

### JWT 认证

```typescript
// src/middleware/auth.ts
import { Context, Next } from "hono";
import { verify } from "hono/jwt";

export const authMiddleware = async (c: Context, next: Next) => {
  const authorization = c.req.header("Authorization");

  if (!authorization) {
    return c.json(
      {
        success: false,
        error: "Unauthorized",
      },
      401
    );
  }

  const token = authorization.replace("Bearer ", "");

  try {
    const payload = await verify(token, c.env.JWT_SECRET);

    // 将用户信息注入到 Context
    c.set("user", payload);
    await next();
  } catch (error) {
    return c.json(
      {
        success: false,
        error: "Invalid token",
      },
      401
    );
  }
};

// 保护路由
const protected = new Hono();
protected.use("*", authMiddleware);

protected.get("/me", async (c) => {
  const user = c.get("user");
  return c.json({ user });
});
```

## 错误处理

### 全局错误处理

```typescript
// src/middleware/error.ts
import { Context, Next } from "hono";

export const errorHandler = async (c: Context, next: Next) => {
  try {
    await next();
  } catch (error) {
    console.error("Error:", error);

    // Prisma 错误处理
    if (error.code === "P2002") {
      return c.json(
        {
          success: false,
          error: "Unique constraint violation",
        },
        409
      );
    }

    if (error.code === "P2025") {
      return c.json(
        {
          success: false,
          error: "Record not found",
        },
        404
      );
    }

    // 默认错误响应
    return c.json(
      {
        success: false,
        error: "Internal server error",
      },
      500
    );
  }
};

// 使用
app.use("*", errorHandler);
```

## 事务处理

### 使用事务 API

```typescript
// src/routes/transfer.ts
import { Hono } from "hono";

const transfer = new Hono();

transfer.post("/", async (c) => {
  const db = c.get("db");
  const { fromId, toId, amount } = await c.req.json();

  try {
    const result = await db.$transaction(async (tx) => {
      // 检查发送者余额
      const fromAccount = await tx.account.findUnique({
        where: { id: fromId },
      });

      if (!fromAccount || fromAccount.balance < amount) {
        throw new Error("Insufficient balance");
      }

      // 扣除发送者余额
      await tx.account.update({
        where: { id: fromId },
        data: {
          balance: {
            decrement: amount,
          },
        },
      });

      // 增加接收者余额
      await tx.account.update({
        where: { id: toId },
        data: {
          balance: {
            increment: amount,
          },
        },
      });

      // 创建交易记录
      const transaction = await tx.transaction.create({
        data: {
          fromId,
          toId,
          amount,
        },
      });

      return transaction;
    });

    return c.json({
      success: true,
      data: result,
    });
  } catch (error) {
    return c.json(
      {
        success: false,
        error: error.message,
      },
      400
    );
  }
});

export default transfer;
```

## 边缘运行时支持

### 使用适配器

```typescript
// src/index.ts
import { Hono } from "hono";
import { PrismaClient } from "@prisma/client";
import { PrismaPlanetScale } from "@prisma/adapter-planetscale";
import { Client } from "@planetscale/database";

const app = new Hono();

// 检查是否在边缘环境
if (process.env.EDGE_RUNTIME) {
  const client = new Client({
    url: process.env.DATABASE_URL,
  });

  const adapter = new PrismaPlanetScale(client);

  app.use("*", async (c, next) => {
    c.set("db", new PrismaClient({ adapter }));
    await next();
  });
} else {
  // 传统 Node.js 环境
  const globalForPrisma = globalThis as unknown as {
    prisma: PrismaClient | undefined;
  };

  const prisma = globalForPrisma.prisma ?? new PrismaClient();

  if (process.env.NODE_ENV !== "production") {
    globalForPrisma.prisma = prisma;
  }

  app.use("*", async (c, next) => {
    c.set("db", prisma);
    await next();
  });
}

export default app;
```

## 常见问题

**Q: Hono 和 Express 相比如何？**

A：Hono 更轻量、更快、TypeScript 原生支持。适合新项目和 API 开发。

**Q: 如何处理文件上传？**

A：使用 `@hono/node-server` 的 multipart 支持：

```typescript
import { multipart } from "@hono/node-server";

app.post(
  "/upload",
  multipart(async (c) => {
    const body = await c.req.parseBody();
    // 处理文件...
  })
);
```

**Q: 支持边缘部署吗？**

A：是的！Hono 天生支持边缘运行时，但 Prisma 的边缘支持有限，需要使用特定适配器。
