## 数据库设计

### Schema 定义

```typescript
// src/db/schema.ts
import { sqliteTable, text, integer } from "drizzle-orm/sqlite-core";
import { sql, relations } from "drizzle-orm";

export const users = sqliteTable("users", {
  id: text("id").primaryKey(),
  email: text("email").notNull().unique(),
  username: text("username").notNull().unique(),
  password: text("password").notNull(),
  name: text("name"),
  bio: text("bio"),
  avatar: text("avatar"),
  role: text("role", { enum: ["user", "admin"] })
    .notNull()
    .default("user"),
  createdAt: integer("created_at", { mode: "timestamp" }).notNull(),
});

export const posts = sqliteTable("posts", {
  id: text("id").primaryKey(),
  title: text("title").notNull(),
  slug: text("slug").notNull().unique(),
  content: text("content").notNull(),
  excerpt: text("excerpt"),
  coverImage: text("cover_image"),
  published: integer("published", { mode: "boolean" }).notNull().default(false),
  authorId: text("author_id")
    .notNull()
    .references(() => users.id),
  views: integer("views").notNull().default(0),
  createdAt: integer("created_at", { mode: "timestamp" }).notNull(),
  updatedAt: integer("updated_at", { mode: "timestamp" }).notNull(),
  publishedAt: integer("published_at", { mode: "timestamp" }),
});

export const comments = sqliteTable("comments", {
  id: text("id").primaryKey(),
  content: text("content").notNull(),
  postId: text("post_id")
    .notNull()
    .references(() => posts.id),
  authorId: text("author_id")
    .notNull()
    .references(() => users.id),
  parentId: text("parent_id").references(() => comments.id),
  createdAt: integer("created_at", { mode: "timestamp" }).notNull(),
});

export const tags = sqliteTable("tags", {
  id: text("id").primaryKey(),
  name: text("name").notNull().unique(),
  slug: text("slug").notNull().unique(),
});

export const postTags = sqliteTable("post_tags", {
  postId: text("post_id")
    .notNull()
    .references(() => posts.id),
  tagId: text("tag_id")
    .notNull()
    .references(() => tags.id),
});

// 关系定义
export const usersRelations = relations(users, ({ many }) => ({
  posts: many(posts),
  comments: many(comments),
}));

export const postsRelations = relations(posts, ({ one, many }) => ({
  author: one(users, {
    fields: [posts.authorId],
    references: [users.id],
  }),
  comments: many(comments),
  tags: many(postTags),
}));
```

## 认证系统

### JWT 认证

```typescript
// src/routes/auth.ts
import { Hono } from "hono";
import { sign, verify } from "hono/jwt";
import { hash, compare } from "bcrypt";
import { zValidator } from "@hono/zod-validator";
import { z } from "zod";

const auth = new Hono<{ Bindings: Env }>();

// 注册
auth.post(
  "/register",
  zValidator(
    "json",
    z.object({
      email: z.string().email(),
      username: z.string().min(3),
      password: z.string().min(8),
      name: z.string(),
    })
  ),
  async (c) => {
    const data = c.req.valid("json");

    const hashedPassword = await hash(data.password, 10);

    const [user] = await c.env.DB.prepare(
      "INSERT INTO users (id, email, username, password, name) VALUES (?, ?, ?, ?, ?)"
    )
      .bind(
        crypto.randomUUID(),
        data.email,
        data.username,
        hashedPassword,
        data.name
      )
      .run();

    return c.json({ success: true, data: { id: user.meta.last_row_id } }, 201);
  }
);

// 登录
auth.post(
  "/login",
  zValidator(
    "json",
    z.object({
      email: z.string().email(),
      password: z.string(),
    })
  ),
  async (c) => {
    const { email, password } = c.req.valid("json");

    const result = await c.env.DB.prepare("SELECT * FROM users WHERE email = ?")
      .bind(email)
      .first();

    if (!result || !(await compare(password, result.password as string))) {
      return c.json({ success: false, message: "Invalid credentials" }, 401);
    }

    const token = await sign(
      { sub: result.id, role: result.role },
      c.env.JWT_SECRET
    );

    return c.json({ success: true, data: { token } });
  }
);

export default auth;
```

## 文章 CRUD

### 完整实现

```typescript
// src/routes/posts.ts
import { Hono } from "hono";
import { zValidator } from "@hono/zod-validator";
import { z } from "zod";
import { authMiddleware } from "../middleware/auth";

const posts = new Hono<{ Bindings: Env }>();

// Schema
const postSchema = z.object({
  title: z.string().min(1),
  content: z.string().min(1),
  excerpt: z.string().optional(),
  published: z.boolean().default(false),
  tags: z.array(z.string()).optional(),
});

// GET /posts - 列表
posts.get("/", async (c) => {
  const { page = "1", limit = "10" } = c.req.query();

  const result = await c.env.DB.prepare(
    `
    SELECT * FROM posts
    WHERE published = 1
    ORDER BY created_at DESC
    LIMIT ? OFFSET ?
  `
  )
    .bind(limit, (parseInt(page) - 1) * parseInt(limit))
    .all();

  return c.json({ success: true, data: result.results });
});

// GET /posts/:id - 详情
posts.get("/:id", async (c) => {
  const { id } = c.req.param();

  // 增加浏览量
  await c.env.DB.prepare("UPDATE posts SET views = views + 1 WHERE id = ?")
    .bind(id)
    .run();

  const post = await c.env.DB.prepare("SELECT * FROM posts WHERE id = ?")
    .bind(id)
    .first();

  if (!post) {
    return c.json({ success: false, message: "Not found" }, 404);
  }

  return c.json({ success: true, data: post });
});

// POST /posts - 创建
posts.post("/", authMiddleware, zValidator("json", postSchema), async (c) => {
  const data = c.req.valid("json");
  const user = c.get("user");

  const slug = data.title.toLowerCase().replace(/[^a-z0-9]+/g, "-");

  const [post] = await c.env.DB.prepare(
    `
      INSERT INTO posts (id, title, slug, content, excerpt, published, author_id)
      VALUES (?, ?, ?, ?, ?, ?, ?)
    `
  )
    .bind(
      crypto.randomUUID(),
      data.title,
      slug,
      data.content,
      data.excerpt,
      data.published ? 1 : 0,
      user.id
    )
    .run();

  return c.json({ success: true, data: post }, 201);
});

// PUT /posts/:id - 更新
posts.put(
  "/:id",
  authMiddleware,
  zValidator("json", postSchema.partial()),
  async (c) => {
    const { id } = c.req.param();
    const data = c.req.valid("json");
    const user = c.get("user");

    // 检查权限
    const post = await c.env.DB.prepare("SELECT * FROM posts WHERE id = ?")
      .bind(id)
      .first();

    if (!post || post.authorId !== user.id) {
      return c.json({ success: false, message: "Forbidden" }, 403);
    }

    await c.env.DB.prepare(
      `
      UPDATE posts SET title = ?, content = ?, excerpt = ?, updated_at = ?
      WHERE id = ?
    `
    )
      .bind(
        data.title,
        data.content,
        data.excerpt,
        new Date().toISOString(),
        id
      )
      .run();

    return c.json({ success: true });
  }
);

// DELETE /posts/:id - 删除
posts.delete("/:id", authMiddleware, async (c) => {
  const { id } = c.req.param();
  const user = c.get("user");

  const post = await c.env.DB.prepare("SELECT * FROM posts WHERE id = ?")
    .bind(id)
    .first();

  if (!post || post.authorId !== user.id) {
    return c.json({ success: false, message: "Forbidden" }, 403);
  }

  await c.env.DB.prepare("DELETE FROM posts WHERE id = ?").bind(id).run();

  return c.json({ success: true });
});

export default posts;
```

## 缓存优化

### KV 缓存

```typescript
// src/services/cache.service.ts
export class CacheService {
  constructor(private kv: KVNamespace) {}

  async get<T>(key: string): Promise<T | null> {
    const value = await this.kv.get(key, "json");
    return value as T;
  }

  async set(key: string, value: any, ttl: number): Promise<void> {
    await this.kv.put(key, JSON.stringify(value), { expirationTtl: ttl });
  }

  async delete(key: string): Promise<void> {
    await this.kv.delete(key);
  }

  async invalidatePattern(prefix: string): Promise<void> {
    const keys = await this.kv.list({ prefix });
    await Promise.all(keys.keys.map((key) => this.kv.delete(key.name)));
  }
}

// 使用
const cache = new CacheService(c.env.CACHE);

const cached = await cache.get(`post:${id}`);
if (cached) {
  return c.json(cached);
}

const post = await fetchPost(id);
await cache.set(`post:${id}`, post, 600);
```

## 文件上传

### R2 存储

```typescript
// src/routes/upload.ts
app.post("/upload", authMiddleware, async (c) => {
  const formData = await c.req.formData();
  const file = formData.get("file") as File;

  if (!file) {
    return c.json({ success: false, message: "No file" }, 400);
  }

  const key = `uploads/${Date.now()}-${file.name}`;
  await c.env.STORAGE.put(key, file.stream());

  const url = `https://cdn.example.com/${key}`;

  return c.json({ success: true, data: { url } });
});
```
