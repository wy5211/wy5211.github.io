## 前言

在构建 Web API 时，数据验证是不可或缺的一环。无效或恶意的数据可能导致应用程序崩溃、数据不一致，甚至安全漏洞。传统的验证方式往往冗长、难以维护，而且与 TypeScript 类型系统脱节。

**Zod** 是一个 TypeScript-first 的数据验证库，它让你能够用简洁的语法定义 Schema，同时自动推断 TypeScript 类型。这意味着你只需要写一次验证规则，就能同时获得运行时验证和编译时类型检查。

Hono 与 Zod 的结合更是天衣无缝，通过 `@hono/zod-validator` 中间件，你可以轻松实现请求数据验证，让 API 更加健壮和安全。

本篇文章将带你：

- 🎯 理解数据验证的重要性和 Zod 的优势
- 📝 定义各种类型的 Zod Schema
- ✨ 在 Hono 中集成 Zod 验证中间件
- 🔍 处理复杂的验证场景
- 💡 提供友好的错误提示
- 🚀 掌握验证最佳实践

## 为什么需要数据验证

### 数据验证的重要性

```typescript
// ❌ 没有验证的危险代码
app.post("/users", async (c) => {
  const body = await c.req.json();

  // 直接使用用户数据，没有任何验证
  await db.insert(users).values(body);

  return c.json({ message: "User created" });
});

// 问题：
// 1. 缺少必填字段怎么办？
// 2. email 格式不正确怎么办？
// 3. password 太简单怎么办？
// 4. 恶意用户传入多余字段怎么办？
```

```typescript
// ✅ 带验证的安全代码
app.post("/users", zValidator("json", userSchema), async (c) => {
  const user = c.req.valid("json"); // 已验证的数据
  await db.insert(users).values(user);

  return c.json({ message: "User created" });
});
```

### 传统验证 vs Zod

```typescript
// ❌ 传统验证方式（冗长、重复）
app.post("/users", async (c) => {
  const body = await c.req.json();

  // 手动验证
  if (!body.name || typeof body.name !== "string") {
    return c.json({ error: "Name is required" }, 400);
  }

  if (!body.email || !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(body.email)) {
    return c.json({ error: "Invalid email" }, 400);
  }

  if (!body.password || body.password.length < 8) {
    return c.json({ error: "Password too short" }, 400);
  }

  // 验证通过后，body 的类型还是 any
  await createUser(body);
});

// ✅ Zod 验证（简洁、类型安全）
const userSchema = z.object({
  name: z.string().min(1),
  email: z.string().email(),
  password: z.string().min(8),
});

app.post("/users", zValidator("json", userSchema), async (c) => {
  const user = c.req.valid("json");
  // user 类型自动推断为 { name: string; email: string; password: string }

  await createUser(user);
});
```

### 基本类型验证

```typescript
import { z } from "zod";

// 字符串验证
const nameSchema = z.string();
nameSchema.parse("John"); // ✅ 通过
nameSchema.parse(123); // ❌ 抛出 ZodError

// 带验证规则的字符串
const usernameSchema = z
  .string()
  .min(3, "Username too short") // 最小长度
  .max(20, "Username too long") // 最大长度
  .regex(/^[a-zA-Z0-9_]+$/, "Invalid characters");

// 数字验证
const ageSchema = z
  .number()
  .min(18, "Must be 18 or older")
  .max(120, "Invalid age");

// 布尔值
const isActiveSchema = z.boolean();

// 枚举
const roleSchema = z.enum(["user", "admin", "moderator"]);
roleSchema.parse("user"); // ✅
roleSchema.parse("superadmin"); // ❌

// 可选类型
const phoneSchema = z.string().optional(); // 可以是 undefined
const emailSchema = z.string().nullable(); // 可以是 null

// 带默认值
const limitSchema = z.number().default(10);
const sortSchema = z.enum(["asc", "desc"]).default("desc");
```


## 在 Hono 中使用 Zod

### 基本集成

```typescript
import { Hono } from "hono";
import { zValidator } from "@hono/zod-validator";
import { z } from "zod";

const app = new Hono();

// 定义 Schema
const userSchema = z.object({
  name: z.string().min(2).max(50),
  email: z.string().email(),
  age: z.number().min(18).optional(),
});

// 使用 zValidator 中间件
app.post("/users", zValidator("json", userSchema), async (c) => {
  const user = c.req.valid("json");
  // user 类型是 { name: string; email: string; age?: number }

  // 创建用户...
  return c.json({ message: "User created" }, 201);
});
```

### 验证查询参数

```typescript
// 查询参数 Schema
const querySchema = z.object({
  keyword: z.string().optional(),
  page: z.string().transform(Number).default("1"),
  limit: z.string().transform(Number).default("10"),
  sort: z.enum(["asc", "desc"]).default("desc"),
});

app.get("/users", zValidator("query", querySchema), (c) => {
  const { keyword, page, limit, sort } = c.req.valid("query");
  // keyword?: string
  // page: number (已转换)
  // limit: number (已转换)
  // sort: 'asc' | 'desc'

  return c.json({ keyword, page, limit, sort });
});
```

```typescript

```

### 同时验证多个位置

```typescript
// 定义多个 Schema
const paramsSchema = z.object({
  userId: z.string().uuid(),
});

const querySchema = z.object({
  includePosts: z.enum(["true", "false"]).transform((v) => v === "true"),
});

const bodySchema = z.object({
  name: z.string().min(1),
});

// 组合使用
app.put(
  "/users/:userId",
  zValidator("param", paramsSchema),
  zValidator("query", querySchema),
  zValidator("json", bodySchema),
  async (c) => {
    const { userId } = c.req.valid("param");
    const { includePosts } = c.req.valid("query");
    const { name } = c.req.valid("json");

    return c.json({ userId, name, includePosts });
  }
);
```

```typescript
// 使用 refine 添加自定义验证
const passwordSchema = z
  .string()
  .min(8, "Password must be at least 8 characters")
  .refine(
    (password) => /[A-Z]/.test(password),

  .refine(

    "Password must contain at least one lowercase letter"

  .refine(

    "Password must contain at least one number"

    "Password must contain at least one special character"

// 使用 superRefine 进行复杂验证
const orderSchema = z
  .object({
    items: z.array(
      z.object({
        productId: z.string().uuid(),
        quantity: z.number().min(1),
      })
    ),
    shippingAddress: z.object({
      country: z.string(),
      city: z.string(),
      address: z.string(),
    }),
  })
  .superRefine((data, ctx) => {
    // 检查是否支持该国配送
    if (!isShippingAvailable(data.shippingAddress.country)) {
      ctx.addIssue({
        code: z.ZodIssueCode.custom,
        message: "Shipping not available for this country",
        path: ["shippingAddress", "country"],
      });
    }

    // 检查库存
    for (const item of data.items) {
      if (!checkStock(item.productId, item.quantity)) {
        ctx.addIssue({
          code: z.ZodIssueCode.custom,
          message: `Insufficient stock for product ${item.productId}`,
          path: ["items", data.items.indexOf(item)],
        });
      }
    }
  });
```

### 全局错误处理中间件

```typescript
// 创建统一的验证错误处理器
const validationErrorHandler = (result: z.ZodValidationResult, c: Context) => {
  if (!result.success) {
    return c.json(
      {
        success: false,
        error: "Validation failed",
        details: result.error.errors.map((err) => ({
          field: err.path.join("."),
          message: err.message,
          code: err.code,
        })),
      },
      400
    );
  }
};

// 使用
app.post(
  "/users",
  zValidator("json", userSchema, validationErrorHandler),
  async (c) => {
    const user = c.req.valid("json");
    // ...
  }
);
```

## 实战示例

### 完整的 CRUD API 验证

```typescript
import { Hono } from "hono";
import { zValidator } from "@hono/zod-validator";
import { z } from "zod";
import { HTTPException } from "hono/http-exception";

const app = new Hono();

// ==================== Schema 定义 ====================

// 创建文章的 Schema
const createPostSchema = z.object({
  title: z.string().min(1, "标题不能为空").max(200, "标题最多200个字符"),
  content: z.string().min(1, "内容不能为空").max(10000, "内容最多10000个字符"),
  excerpt: z.string().max(500, "摘要最多500个字符").optional(),
  published: z.boolean().default(false),
  tags: z
    .array(z.string())
    .min(1, "至少需要一个标签")
    .max(5, "最多5个标签")
    .optional(),
  categoryId: z.string().uuid("Invalid category ID"),
  coverImage: z.string().url("Invalid cover image URL").optional(),
});

// 更新文章的 Schema（所有字段可选）
const updatePostSchema = createPostSchema.partial();

// 查询参数 Schema
const postQuerySchema = z.object({
  page: z.string().transform(Number).default("1"),
  limit: z.string().transform(Number).default("10"),
  keyword: z.string().optional(),
  categoryId: z.string().uuid().optional(),
  published: z.enum(["true", "false", "all"]).default("all"),
  sortBy: z.enum(["createdAt", "updatedAt", "title"]).default("createdAt"),
  sortOrder: z.enum(["asc", "desc"]).default("desc"),
});

// 路径参数 Schema
const postParamSchema = z.object({
  id: z.string().uuid("Invalid post ID format"),
});

// ==================== 验证错误处理 ====================

const handleValidationError = (result: z.ZodValidationResult, c: any) => {
  if (!result.success) {
    return c.json(
      {
        success: false,
        message: "Validation failed",
        errors: result.error.errors.map((err) => ({
          field: err.path.join("."),
          message: err.message,
        })),
      },
      400
    );
  }
};

// ==================== 路由定义 ====================

// GET /posts - 获取文章列表
app.get(
  "/posts",
  zValidator("query", postQuerySchema, handleValidationError),
  async (c) => {
    const { page, limit, keyword, categoryId, published, sortBy, sortOrder } =
      c.req.valid("query");

    // 构建查询条件
    const conditions: any[] = [];

    if (keyword) {
      conditions.push(like(posts.title, `%${keyword}%`));
    }

    if (categoryId) {
      conditions.push(eq(posts.categoryId, categoryId));
    }

    if (published !== "all") {
      conditions.push(eq(posts.published, published === "true"));
    }

    // 查询数据
    const [postList, totalCount] = await Promise.all([
      db.query.posts.findMany({
        where: conditions.length > 0 ? and(...conditions) : undefined,
        limit,
        offset: (page - 1) * limit,
        orderBy: sortOrder === "asc" ? asc(posts[sortBy]) : desc(posts[sortBy]),
        with: {
          category: true,
          author: {
            columns: { id: true, name: true, avatar: true },
          },
        },
      }),
      db
        .select({ count: sql<number>`count(*)` })
        .from(posts)
        .where(conditions.length > 0 ? and(...conditions) : undefined),
    ]);

    return c.json({
      success: true,
      data: postList,
      meta: {
        page,
        limit,
        total: totalCount[0].count,
        totalPages: Math.ceil(totalCount[0].count / limit),
      },
    });
  }
);

// GET /posts/:id - 获取文章详情
app.get(
  "/posts/:id",
  zValidator("param", postParamSchema, handleValidationError),
  async (c) => {
    const { id } = c.req.valid("param");

    const post = await db.query.posts.findFirst({
      where: eq(posts.id, id),
      with: {
        category: true,
        author: true,
        tags: true,
        comments: {
          where: eq(comments.approved, true),
          with: {
            author: {
              columns: { id: true, name: true, avatar: true },
            },
          },
        },
      },
    });

    if (!post) {
      throw new HTTPException(404, { message: "Post not found" });
    }

    // 增加浏览量
    await db
      .update(posts)
      .set({ views: sql`${posts.views} + 1` })
      .where(eq(posts.id, id));

    return c.json({
      success: true,
      data: post,
    });
  }
);

// POST /posts - 创建文章
app.post(
  "/posts",
  zValidator("json", createPostSchema, handleValidationError),
  async (c) => {
    const data = c.req.valid("json");
    const currentUser = c.get("user");

    // 检查分类是否存在
    const category = await db.query.categories.findFirst({
      where: eq(categories.id, data.categoryId),
    });

    if (!category) {
      return c.json(
        {
          success: false,
          message: "Category not found",
        },
        404
      );
    }

    // 创建文章
    const post = await db
      .insert(posts)
      .values({
        id: crypto.randomUUID(),
        ...data,
        authorId: currentUser.id,
        views: 0,
        createdAt: new Date(),
        updatedAt: new Date(),
      })
      .returning();

    // 处理标签
    if (data.tags && data.tags.length > 0) {
      const tagRelations = data.tags.map((tagId) => ({
        postId: post[0].id,
        tagId,
      }));
      await db.insert(postTags).values(tagRelations);
    }

    return c.json(
      {
        success: true,
        data: post[0],
        message: "Post created successfully",
      },
      201
    );
  }
);

// PATCH /posts/:id - 更新文章
app.patch(
  "/posts/:id",
  zValidator("param", postParamSchema, handleValidationError),
  zValidator("json", updatePostSchema, handleValidationError),
  async (c) => {
    const { id } = c.req.valid("param");
    const data = c.req.valid("json");
    const currentUser = c.get("user");

    // 检查文章是否存在
    const post = await db.query.posts.findFirst({
      where: eq(posts.id, id),
    });

    if (!post) {
      throw new HTTPException(404, { message: "Post not found" });
    }

    // 检查权限
    if (post.authorId !== currentUser.id && currentUser.role !== "admin") {
      throw new HTTPException(403, { message: "Forbidden" });
    }

    // 更新文章
    const updated = await db
      .update(posts)
      .set({ ...data, updatedAt: new Date() })
      .where(eq(posts.id, id))
      .returning();

    // 更新标签
    if (data.tags) {
      await db.delete(postTags).where(eq(postTags.postId, id));

      if (data.tags.length > 0) {
        const tagRelations = data.tags.map((tagId) => ({
          postId: id,
          tagId,
        }));
        await db.insert(postTags).values(tagRelations);
      }
    }

    return c.json({
      success: true,
      data: updated[0],
      message: "Post updated successfully",
    });
  }
);

// DELETE /posts/:id - 删除文章
app.delete(
  "/posts/:id",
  zValidator("param", postParamSchema, handleValidationError),
  async (c) => {
    const { id } = c.req.valid("param");
    const currentUser = c.get("user");

    const post = await db.query.posts.findFirst({
      where: eq(posts.id, id),
    });

    if (!post) {
      throw new HTTPException(404, { message: "Post not found" });
    }

    if (post.authorId !== currentUser.id && currentUser.role !== "admin") {
      throw new HTTPException(403, { message: "Forbidden" });
    }

    await db.delete(posts).where(eq(posts.id, id));

    return c.json({
      success: true,
      message: "Post deleted successfully",
    });
  }
);

export default app;
```

```typescript
        },

      );

      return c.json(
        {
          success: false,
          message: "Username already taken",
        },
        409
      );
    }
```
