```toml

  { pattern = "api.example.com/*", zone_name = "example.com" }
```


## 应用代码

### 类型定义

```typescript
// src/index.ts
import { Hono } from "hono";

// 定义 Bindings 类型
type Bindings = {
  CACHE: KVNamespace;
  DB: D1Database;
  STORAGE: R2Bucket;
  ENVIRONMENT: string;
  API_URL: string;
  SECRET: string;
};

const app = new Hono<{ Bindings: Bindings }>();

app.get("/", (c) => {
  return c.text(`Running on ${c.env.ENVIRONMENT}`);
});

// 使用 KV
app.get("/cache/:key", async (c) => {
  const { key } = c.req.param();
  const value = await c.env.CACHE.get(key);

  return c.json({ key, value });
});

// 使用 D1
app.get("/users", async (c) => {
  const { results } = await c.env.DB.prepare("SELECT * FROM users").all();

  return c.json({ users: results });
});

// 使用 R2
app.get("/files/:key", async (c) => {
  const { key } = c.req.param();
  const object = await c.env.STORAGE.get(key);

  if (!object) {
    return c.json({ error: "Not found" }, 404);
  }

  return c.body(object.body);
});

export default app;
```

### 环境变量

```typescript
// 使用环境变量
app.get("/config", (c) => {
  return c.json({
    environment: c.env.ENVIRONMENT,
    apiUrl: c.env.API_URL,
  });
});

// 使用密钥
app.post("/webhook", async (c) => {
  const signature = c.req.header("X-Signature");

  if (signature !== c.env.SECRET) {
    return c.json({ error: "Unauthorized" }, 401);
  }

  // 处理 webhook
  return c.json({ success: true });
});
```

## 部署

### 创建资源

```bash
# 创建 KV 命名空间
wrangler kv:namespace create "CACHE"
# 记录返回的 id，添加到 wrangler.toml

# 创建 D1 数据库
wrangler d1 create "my-database"
# 记录返回的 database_id

# 创建 R2 Bucket
wrangler r2 bucket create "my-bucket"
```

### 设置密钥

```bash
# 设置环境密钥
wrangler secret put API_KEY
# 输入密钥值

# 批量设置（从文件）
cat .env | wrangler secret bulk
```

### 部署命令

```bash
# 开发模式
wrangler dev

# 部署到生产环境
wrangler deploy

# 部署到特定环境
wrangler deploy --env development

# 部署并立即查看日志
wrangler deploy --tail
```

## 数据库迁移

### D1 迁移

```bash
# 创建迁移文件
wrangler d1 migrations create my-database init

# 编辑迁移 SQL 文件

# 本地应用迁移
wrangler d1 migrations apply my-database --local

# 生产环境应用迁移
wrangler d1 migrations apply my-database --remote
```

### 迁移示例

```sql
-- migrations/0001_init.sql
CREATE TABLE IF NOT EXISTS users (
  id TEXT PRIMARY KEY,
  email TEXT NOT NULL UNIQUE,
  name TEXT,
  created_at TEXT NOT NULL
);

CREATE TABLE IF NOT EXISTS posts (
  id TEXT PRIMARY KEY,
  title TEXT NOT NULL,
  content TEXT NOT NULL,
  author_id TEXT NOT NULL REFERENCES users(id),
  created_at TEXT NOT NULL
);
```

## CI/CD 集成

### GitHub Actions

```yaml
# .github/workflows/deploy.yml
name: Deploy to Cloudflare Workers

on:
  push:
    branches: [main]

jobs:
  deploy:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3

      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: "20"

      - name: Install dependencies
        run: npm ci

      - name: Run tests
        run: npm test

      - name: Deploy to Cloudflare Workers
        uses: cloudflare/wrangler-action@v2
        with:
          apiToken: ${{ secrets.CLOUDFLARE_API_TOKEN }}
          accountId: ${{ secrets.CLOUDFLARE_ACCOUNT_ID }}
```
