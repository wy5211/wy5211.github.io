## 数据库优化

### 查询优化

```typescript
// ❌ N+1 查询问题
const posts = await db.query.posts.findMany();
for (const post of posts) {
  post.author = await db.query.users.findFirst({
    where: eq(users.id, post.authorId),
  });
}

// ✅ 使用 JOIN 或关系查询
const posts = await db.query.posts.findMany({
  with: {
    author: true,
  },
});

// ✅ 只查询需要的字段
const posts = await db
  .select({
    id: posts.id,
    title: posts.title,
    createdAt: posts.createdAt,
  })
  .from(posts);
```

### 索引优化

```typescript
// 在 Schema 中定义索引
export const posts = sqliteTable(
  "posts",
  {
    id: text("id").primaryKey(),
    title: text("title").notNull(),
    authorId: text("author_id").notNull(),
    published: integer("published", { mode: "boolean" }).notNull(),
    createdAt: integer("created_at", { mode: "timestamp" }).notNull(),
  },
  (table) => ({
    authorIdIdx: index("posts_author_id_idx").on(table.authorId),
    publishedIdx: index("posts_published_idx").on(table.published),
    createdAtIdx: index("posts_created_at_idx").on(table.createdAt),
  })
);
```

## 缓存策略

### 分层缓存

```typescript
class CacheManager {
  private memoryCache = new Map<string, { data: any; expires: number }>();

  async get(key: string, kv: KVNamespace): Promise<any> {
    // L1: 内存缓存（最快，10秒）
    const memData = this.memoryCache.get(key);
    if (memData && memData.expires > Date.now()) {
      return memData.data;
    }

    // L2: KV 缓存（快，1分钟）
    const kvData = await kv.get(key, "json");
    if (kvData) {
      this.memoryCache.set(key, {
        data: kvData,
        expires: Date.now() + 10000,
      });
      return kvData;
    }

    return null;
  }
}
```

### 缓存预热

```typescript
// 定时预热热门内容
export async function warmupCache(env: any) {
  const popularPosts = await db.query.posts.findMany({
    where: eq(posts.featured, true),
    limit: 100,
  });

  await Promise.all(
    popularPosts.map((post) =>
      env.CACHE.put(`post:${post.id}`, JSON.stringify(post), {
        expirationTtl: 3600,
      })
    )
  );
}
```

## 内存优化

### 避免内存泄漏

```typescript
// ❌ 错误：未清理事件监听器
app.get("/stream", (c) => {
  const interval = setInterval(() => {
    // 发送数据
  }, 1000);

  // 忘记清理
});

// ✅ 正确：清理资源
app.get("/stream", (c) => {
  return c.stream(async (stream) => {
    const interval = setInterval(() => {
      stream.write("data\n");
    }, 1000);

    // 清理
    stream.close();
    clearInterval(interval);
  });
});
```

### 限制并发

```typescript
// 限制数据库并发连接
const MAX_CONCURRENT = 10;
let activeQueries = 0;
const queue: Array<() => void> = [];

async function query<T>(fn: () => Promise<T>): Promise<T> {
  while (activeQueries >= MAX_CONCURRENT) {
    await new Promise((resolve) => queue.push(resolve));
  }

  activeQueries++;
  try {
    return await fn();
  } finally {
    activeQueries--;
    const next = queue.shift();
    if (next) next();
  }
}
```

## 监控和调优

### 性能监控

```typescript
// 收集性能指标
const metrics = {
  requestCount: 0,
  totalDuration: 0,
  errors: 0,
};

app.use("*", async (c, next) => {
  const start = Date.now();

  try {
    await next();
    metrics.requestCount++;
  } catch (error) {
    metrics.errors++;
    throw error;
  } finally {
    const duration = Date.now() - start;
    metrics.totalDuration += duration;

    // 定期报告
    if (metrics.requestCount % 100 === 0) {
      console.log({
        avgDuration: metrics.totalDuration / metrics.requestCount,
        errorRate: metrics.errors / metrics.requestCount,
      });
    }
  }
});
```
