    .default(sql`CURRENT_TIMESTAMP`),

    .default(sql`CURRENT_TIMESTAMP`),

  posts: many(posts),
  comments: many(comments),

  author: one(users, {
    fields: [posts.authorId],
    references: [users.id],

  category: one(categories, {
    fields: [posts.categoryId],
    references: [categories.id],

    references: [posts.id],

  author: one(users, {
    fields: [comments.authorId],
    references: [users.id],
  }),
  parent: one(comments, {
    fields: [comments.parentId],
    references: [comments.id],
  }),
  replies: many(comments),
}));

export const tagsRelations = relations(tags, ({ many }) => ({
  posts: many(postTags),
}));

export const categoriesRelations = relations(categories, ({ one, many }) => ({
  parent: one(categories, {
    fields: [categories.parentId],
    references: [categories.id],
  }),
  children: many(categories),
}));

export const postTagsRelations = relations(postTags, ({ one }) => ({
  post: one(posts, {
    fields: [postTags.postId],
    references: [posts.id],

  tag: one(tags, {
    fields: [postTags.tagId],

## 文章 CRUD

### 完整的文章路由

```typescript
// src/routes/posts.ts
import { Hono } from "hono";
import { zValidator } from "@hono/zod-validator";
import { eq, and, like, sql, desc, asc, or } from "drizzle-orm";
import { HTTPException } from "hono/http-exception";
import * as schema from "../db/schema";
import { db } from "../db";
import {
  createPostSchema,
  updatePostSchema,
  postQuerySchema,
} from "../validators/schemas";
import { authMiddleware } from "../middleware/auth";
import { generateSlug } from "../utils/slug";

const posts = new Hono();

// 辅助函数：生成唯一 slug
const ensureUniqueSlug = async (slug: string, excludeId?: string) => {
  let finalSlug = slug;
  let counter = 1;

  while (true) {
    const existing = await db.query.posts.findFirst({
      where: and(
        eq(schema.posts.slug, finalSlug),
        excludeId ? sql`${schema.posts.id} != ${excludeId}` : undefined
      ),
    });

    if (!existing) break;

    finalSlug = `${slug}-${counter}`;
    counter++;
  }

  return finalSlug;
};

// GET /posts - 获取文章列表
posts.get("/", zValidator("query", postQuerySchema), async (c) => {
  const {
    page,
    limit,
    keyword,
    categoryId,
    tagId,
    authorId,
    published,
    featured,
    sortBy,
    sortOrder,
  } = c.req.valid("query");

  const conditions: any[] = [];

  // 关键词搜索
  if (keyword) {
    conditions.push(
      or(
        like(schema.posts.title, `%${keyword}%`),
        like(schema.posts.content, `%${keyword}%`),
        like(schema.posts.excerpt, `%${keyword}%`)
      )
    );
  }

  // 分类过滤
  if (categoryId) {
    conditions.push(eq(schema.posts.categoryId, categoryId));
  }

  // 标签过滤
  if (tagId) {
    const postIds = await db
      .select({
        postId: schema.postTags.postId,
      })
      .from(schema.postTags)
      .where(eq(schema.postTags.tagId, tagId));

    conditions.push(
      sql`${schema.posts.id} IN ${sql.raw(`('${postIds.map((p) => p.postId).join("','")}')`)}`
    );
  }

  // 作者过滤
  if (authorId) {
    conditions.push(eq(schema.posts.authorId, authorId));
  }

  // 发布状态过滤
  if (published !== "all") {
    conditions.push(eq(schema.posts.published, published === "true"));
  }

  // 精选文章过滤
  if (featured !== undefined) {
    conditions.push(eq(schema.posts.featured, featured === "true"));
  }

  const where = conditions.length > 0 ? and(...conditions) : undefined;

  // 查询文章和总数
  const [postsList, totalCount] = await Promise.all([
    db.query.posts.findMany({
      where,
      limit,
      offset: (page - 1) * limit,
      orderBy:
        sortOrder === "asc"
          ? asc(schema.posts[sortBy])
          : desc(schema.posts[sortBy]),
      with: {
        author: {
          columns: {
            id: true,
            name: true,
            username: true,
            avatar: true,
          },
        },
        category: true,
        tags: {
          with: {
            tag: true,
          },
        },
      },
    }),
    db
      .select({ count: sql<number>`count(*)` })
      .from(schema.posts)
      .where(where),
  ]);

  // 格式化标签
  const formattedPosts = postsList.map((post) => ({
    ...post,
    tags: post.tags.map((pt) => pt.tag),
  }));

  return c.json({
    success: true,
    data: formattedPosts,
    meta: {
      page,
      limit,
      total: totalCount[0].count,
      totalPages: Math.ceil(totalCount[0].count / limit),
    },
  });
});

// GET /posts/:id - 获取文章详情
posts.get("/:id", async (c) => {
  const { id } = c.req.param();

  const post = await db.query.posts.findFirst({
    where: eq(schema.posts.id, id),
    with: {
      author: {
        columns: {
          id: true,
          name: true,
          username: true,
          bio: true,
          avatar: true,
        },
      },
      category: true,
      tags: {
        with: {
          tag: true,
        },
      },
    },
  });

  if (!post) {
    throw new HTTPException(404, { message: "Post not found" });
  }

  // 增加浏览量
  await db
    .update(schema.posts)
    .set({ views: sql`${schema.posts.views} + 1` })
    .where(eq(schema.posts.id, id));

  // 格式化标签
  const formattedPost = {
    ...post,
    tags: post.tags.map((pt) => pt.tag),
  };

  return c.json({
    success: true,
    data: formattedPost,
  });
});

// GET /posts/slug/:slug - 通过 slug 获取文章
posts.get("/slug/:slug", async (c) => {
  const { slug } = c.req.param();

  const post = await db.query.posts.findFirst({
    where: eq(schema.posts.slug, slug),
    with: {
      author: {
        columns: {
          id: true,
          name: true,
          username: true,
          avatar: true,
        },
      },
      category: true,
      tags: {
        with: {
          tag: true,
        },
      },
    },
  });

  if (!post) {
    throw new HTTPException(404, { message: "Post not found" });
  }

  return c.json({
    success: true,
    data: post,
  });
});

// POST /posts - 创建文章
posts.post(
  "/",
  authMiddleware,
  zValidator("json", createPostSchema),
  async (c) => {
    const data = c.req.valid("json");
    const user = c.get("user");

    // 生成 slug
    const slug = await ensureUniqueSlug(generateSlug(data.title));

    // 创建文章
    const [post] = await db
      .insert(schema.posts)
      .values({
        id: crypto.randomUUID(),
        title: data.title,
        slug,
        content: data.content,
        excerpt: data.excerpt,
        coverImage: data.coverImage,
        published: data.published,
        featured: data.featured,
        authorId: user.id,
        categoryId: data.categoryId,
        publishedAt: data.published ? new Date() : null,
        createdAt: new Date(),
        updatedAt: new Date(),
      })
      .returning();

    // 关联标签
    if (data.tags && data.tags.length > 0) {
      await db.insert(schema.postTags).values(
        data.tags.map((tagId) => ({
          postId: post.id,
          tagId,
          createdAt: new Date(),
        }))
      );
    }

    return c.json(
      {
        success: true,
        data: post,
        message: "Post created successfully",
      },
      201
    );
  }
);

// PATCH /posts/:id - 更新文章
posts.patch(
  "/:id",
  authMiddleware,
  zValidator("json", updatePostSchema),
  async (c) => {
    const { id } = c.req.param();
    const data = c.req.valid("json");
    const user = c.get("user");

    // 检查文章是否存在
    const post = await db.query.posts.findFirst({
      where: eq(schema.posts.id, id),
    });

    if (!post) {
      throw new HTTPException(404, { message: "Post not found" });
    }

    // 检查权限
    if (post.authorId !== user.id && user.role !== "admin") {
      throw new HTTPException(403, { message: "Forbidden" });
    }

    // 如果标题改变，更新 slug
    let slug = post.slug;
    if (data.title && data.title !== post.title) {
      slug = await ensureUniqueSlug(generateSlug(data.title), id);
    }

    // 更新文章
    const [updated] = await db
      .update(schema.posts)
      .set({
        ...data,
        slug,
        updatedAt: new Date(),
        publishedAt:
          data.published && !post.published ? new Date() : post.publishedAt,
      })
      .where(eq(schema.posts.id, id))
      .returning();

    // 更新标签
    if (data.tags !== undefined) {
      // 删除旧标签
      await db.delete(schema.postTags).where(eq(schema.postTags.postId, id));

      // 添加新标签
      if (data.tags.length > 0) {
        await db.insert(schema.postTags).values(
          data.tags.map((tagId) => ({
            postId: id,
            tagId,
            createdAt: new Date(),
          }))
        );
      }
    }

    return c.json({
      success: true,
      data: updated,
      message: "Post updated successfully",
    });
  }
);

// DELETE /posts/:id - 删除文章
posts.delete("/:id", authMiddleware, async (c) => {
  const { id } = c.req.param();
  const user = c.get("user");

  const post = await db.query.posts.findFirst({
    where: eq(schema.posts.id, id),
  });

  if (!post) {
    throw new HTTPException(404, { message: "Post not found" });
  }

  if (post.authorId !== user.id && user.role !== "admin") {
    throw new HTTPException(403, { message: "Forbidden" });
  }

  await db.delete(schema.posts).where(eq(schema.posts.id, id));

  return c.json({
    success: true,
    message: "Post deleted successfully",
  });
});

// POST /posts/:id/like - 点赞文章
posts.post("/:id/like", async (c) => {
  const { id } = c.req.param();

  await db
    .update(schema.posts)
    .set({ likes: sql`${schema.posts.likes} + 1` })
    .where(eq(schema.posts.id, id));

  return c.json({
    success: true,
    message: "Post liked",
  });
});

export default posts;
```

// src/routes/comments.ts
import { Hono } from "hono";
import { zValidator } from "@hono/zod-validator";
import { eq, and, desc } from "drizzle-orm";
import { HTTPException } from "hono/http-exception";
import * as schema from "../db/schema";
import { db } from "../db";
import { createCommentSchema } from "../validators/schemas";
import { authMiddleware } from "../middleware/auth";

const comments = new Hono();

// GET /comments - 获取评论列表
comments.get("/", async (c) => {
  const postId = c.req.query("postId");
  const page = parseInt(c.req.query("page") || "1");
  const limit = parseInt(c.req.query("limit") || "10");

  if (!postId) {
    return c.json(
      {
        success: false,
        message: "Post ID is required",
      },
      400
    );
  }

  const comments = await db.query.comments.findMany({
    where: and(
      eq(schema.comments.postId, postId),
      eq(schema.comments.approved, true)
    ),
    with: {
      author: {
        columns: {
          id: true,
          name: true,
          username: true,
          avatar: true,
        },
      },
      replies: {
        with: {
          author: {
            columns: {
              id: true,
              name: true,
              username: true,
              avatar: true,
            },
          },
        },
      },
    },
    orderBy: [desc(schema.comments.createdAt)],
    limit,
    offset: (page - 1) * limit,
  });

  // 只返回顶级评论（没有 parentId 的）
  const topLevelComments = comments.filter((c) => !c.parentId);

  return c.json({
    success: true,
    data: topLevelComments,
    meta: { page, limit },
  });
});

// GET /comments/:id - 获取评论详情
comments.get("/:id", async (c) => {
  const { id } = c.req.param();

  const comment = await db.query.comments.findFirst({
    where: eq(schema.comments.id, id),
    with: {
      post: {
        columns: {
          id: true,
          title: true,
          slug: true,
        },
      },
      author: {
        columns: {
          id: true,
          name: true,
          username: true,
          avatar: true,
        },
      },
      parent: {
        with: {
          author: {
            columns: {
              id: true,
              name: true,
              username: true,
            },
          },
        },
      },
      replies: {
        with: {
          author: {
            columns: {
              id: true,
              name: true,
              username: true,
              avatar: true,
            },
          },
        },
      },
    },
  });

  if (!comment) {
    throw new HTTPException(404, { message: "Comment not found" });
  }

  return c.json({
    success: true,
    data: comment,
  });
});

// POST /comments - 创建评论
comments.post(
  "/",
  authMiddleware,
  zValidator("json", createCommentSchema),
  async (c) => {
    const { content, postId, parentId } = await c.req.json();
    const user = c.get("user");

    // 检查文章是否存在
    const post = await db.query.posts.findFirst({
      where: eq(schema.posts.id, postId),
    });

    if (!post) {
      throw new HTTPException(404, { message: "Post not found" });
    }

    // 如果是回复评论，检查评论是否存在
    if (parentId) {
      const parentComment = await db.query.comments.findFirst({
        where: eq(schema.comments.id, parentId),
      });

      if (!parentComment) {
        throw new HTTPException(404, { message: "Parent comment not found" });
      }
    }

    // 创建评论
    const [comment] = await db
      .insert(schema.comments)
      .values({
        id: crypto.randomUUID(),
        content,
        postId,
        authorId: user.id,
        parentId,
        approved: user.role === "admin" || user.role === "moderator",
        createdAt: new Date(),
        updatedAt: new Date(),
      })
      .returning();

    return c.json(
      {
        success: true,
        data: comment,
        message: "Comment created successfully",
      },
      201
    );
  }
);

// PATCH /comments/:id - 更新评论
comments.patch("/:id", authMiddleware, async (c) => {
  const { id } = c.req.param();
  const { content } = await c.req.json();
  const user = c.get("user");

  const comment = await db.query.comments.findFirst({
    where: eq(schema.comments.id, id),
  });

  if (!comment) {
    throw new HTTPException(404, { message: "Comment not found" });
  }

  // 检查权限
  if (comment.authorId !== user.id && user.role !== "admin") {
    throw new HTTPException(403, { message: "Forbidden" });
  }

  const [updated] = await db
    .update(schema.comments)
    .set({
      content,
      updatedAt: new Date(),
    })
    .where(eq(schema.comments.id, id))
    .returning();

  return c.json({
    success: true,
    data: updated,
    message: "Comment updated successfully",
  });
});

// DELETE /comments/:id - 删除评论
comments.delete("/:id", authMiddleware, async (c) => {
  const { id } = c.req.param();
  const user = c.get("user");

  const comment = await db.query.comments.findFirst({
    where: eq(schema.comments.id, id),
  });

  if (!comment) {
    throw new HTTPException(404, { message: "Comment not found" });
  }

  if (comment.authorId !== user.id && user.role !== "admin") {
    throw new HTTPException(403, { message: "Forbidden" });
  }

  await db.delete(schema.comments).where(eq(schema.comments.id, id));

  return c.json({
    success: true,
    message: "Comment deleted successfully",
  });
});

// POST /comments/:id/approve - 审批评论
comments.post("/:id/approve", authMiddleware, async (c) => {

    .update(schema.comments)
    .set({ approved: true })
    .where(eq(schema.comments.id, id))
    .returning();
