## MyBatis 配置文件详解

### properties：属性配置

```xml
<properties>
    <!-- 定义属性 -->
    <property name="driver" value="com.mysql.cj.jdbc.Driver"/>
    <property name="url" value="jdbc:mysql://localhost:3306/mybatis_demo"/>
    <property name="username" value="root"/>
    <property name="password" value="root123"/>
</properties>

<!-- 使用属性 -->
<dataSource type="POOLED">
    <property name="driver" value="${driver}"/>
    <property name="url" value="${url}"/>
    <property name="username" value="${username}"/>
    <property name="password" value="${password}"/>
</dataSource>
```

### 外部配置文件

```properties
# db.properties
driver=com.mysql.cj.jdbc.Driver
url=jdbc:mysql://localhost:3306/mybatis_demo
username=root
password=root123
```

```xml
<!-- 引用外部文件 -->
<properties resource="db.properties"/>
```

### settings：全局设置

```xml
<settings>
    <!-- 开启驼峰命名转换 -->
    <setting name="mapUnderscoreToCamelCase" value="true"/>

    <!-- 开启延迟加载 -->
    <setting name="lazyLoadingEnabled" value="true"/>

    <!-- 允许使用方法参数名 -->
    <setting name="useActualParamName" value="true"/>

    <!-- 开启二级缓存 -->
    <setting name="cacheEnabled" value="true"/>

    <!-- 打印 SQL 日志 -->
    <setting name="logImpl" value="STDOUT_LOGGING"/>
</settings>
```

### typeAliases：类型别名

```xml
<typeAliases>
    <!-- 单个别名 -->
    <typeAlias type="com.example.entity.User" alias="User"/>

    <!-- 包扫描（推荐） -->
    <package name="com.example.entity"/>
</typeAliases>

<!-- 使用别名 -->
<select id="selectById" resultType="User">
    SELECT * FROM users WHERE id = #{id}
</select>
```

**MyBatis 内置别名**：

| 别名   | Java 类型 |
| ------ | --------- |
| \_int  | int       |
| \_long | long      |
| string | String    |
| list   | List      |
| map    | Map       |

### typeHandlers：类型处理器

```java
// 自定义类型处理器
@MappedTypes(LocalDateTime.class)
@MappedJdbcTypes(JdbcType.TIMESTAMP)
public class LocalDateTimeTypeHandler extends BaseTypeHandler<LocalDateTime> {
    @Override
    public void setNonNullParameter(PreparedStatement ps, int i,
        LocalDateTime param, JdbcType jdbcType) throws SQLException {
        ps.setTimestamp(i, Timestamp.valueOf(param));
    }

    @Override
    public LocalDateTime getNullableResult(ResultSet rs, String columnName)
        throws SQLException {
        Timestamp timestamp = rs.getTimestamp(columnName);
        return timestamp != null ? timestamp.toLocalDateTime() : null;
    }
}
```

```xml
<!-- 注册类型处理器 -->
<typeHandlers>
    <typeHandler handler="com.example.handler.LocalDateTimeTypeHandler"/>
</typeHandlers>
```

## Mapper 接口详解

### 接口定义规范

```java
package com.example.mapper;

import com.example.entity.User;
import org.apache.ibatis.annotations.Param;

import java.util.List;

public interface UserMapper {
    // 单个参数
    User selectById(Long id);

    // 多个参数（使用 @Param）
    User selectByUsernameAndEmail(@Param("username") String username,
                                   @Param("email") String email);

    // 对象参数
    int insert(User user);

    // 返回集合
    List<User> selectAll();

    // 返回 Map
    Map<String, Object> selectAsMap(Long id);
}
```

### 方法命名规范

```java
// ✅ 推荐的命名
User selectById(Long id);      // 查询
List<User> selectAll();         // 查询所有
int insert(User user);          // 插入
int update(User user);          // 更新
int deleteById(Long id);        // 删除

// ❌ 避免的命名
User getUser(Long id);          // 不够明确
User findUser(Long id);         // 不推荐
```

### 参数传递方式

**单个参数**：

```java
User selectById(Long id);

<!-- XML -->
<select id="selectById" resultType="User">
    SELECT * FROM users WHERE id = #{id}
</select>
```

**多个参数**：

```java
// 方式 1：使用 @Param（推荐）
User selectByUsernameAndEmail(@Param("username") String username,
                               @Param("email") String email);

<!-- XML -->
<select id="selectByUsernameAndEmail" resultType="User">
    SELECT * FROM users
    WHERE username = #{username} AND email = #{email}
</select>

// 方式 2：使用对象
User selectByUser(User user);

<!-- XML -->
<select id="selectByUser" resultType="User">
    SELECT * FROM users
    WHERE username = #{username} AND email = #{email}
</select>
```

**集合参数**：

```java
List<User> selectByIds(List<Long> ids);

<!-- XML -->
<select id="selectByIds" resultType="User">
    SELECT * FROM users WHERE id IN
    <foreach collection="list" item="id" open="(" close=")" separator=",">
        #{id}
    </foreach>
</select>
```

## SqlSession 使用

### 基本用法

```java
// 获取 SqlSession
SqlSession session = sqlSessionFactory.openSession();

try {
    // 获取 Mapper
    UserMapper mapper = session.getMapper(UserMapper.class);

    // 执行操作
    User user = mapper.selectById(1L);

    // 提交事务（增删改需要）
    session.commit();
} catch (Exception e) {
    // 回滚事务
    session.rollback();
    throw e;
} finally {
    // 关闭 SqlSession
    session.close();
}
```

### SqlSession 方法

```java
// 查询
User user = session.selectOne("com.example.mapper.UserMapper.selectById", 1L);
List<User> users = session.selectList("com.example.mapper.UserMapper.selectAll");

// 插入、更新、删除
int count = session.insert("com.example.mapper.UserMapper.insert", user);
int count = session.update("com.example.mapper.UserMapper.update", user);
int count = session.delete("com.example.mapper.UserMapper.deleteById", 1L);

// 事务控制
session.commit();
session.rollback();
```

## 完整配置示例

### mybatis-config.xml

```xml
<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE configuration
        PUBLIC "-//mybatis.org//DTD Config 3.0//EN"
        "https://mybatis.org/dtd/mybatis-3-config.dtd">
<configuration>
    <!-- 引入外部配置 -->
    <properties resource="db.properties"/>

    <!-- 全局设置 -->
    <settings>
        <setting name="mapUnderscoreToCamelCase" value="true"/>
        <setting name="lazyLoadingEnabled" value="true"/>
        <setting name="logImpl" value="STDOUT_LOGGING"/>
    </settings>

    <!-- 类型别名 -->
    <typeAliases>
        <package name="com.example.entity"/>
    </typeAliases>

    <!-- 类型处理器 -->
    <typeHandlers>
        <typeHandler handler="com.example.handler.LocalDateTimeTypeHandler"/>
    </typeHandlers>

    <!-- 插件 -->
    <plugins>
        <plugin interceptor="com.github.pagehelper.PageInterceptor">
            <property name="helperDialect" value="mysql"/>
        </plugin>
    </plugins>

    <!-- 环境配置 -->
    <environments default="development">
        <environment id="development">
            <transactionManager type="JDBC"/>
            <dataSource type="POOLED">
                <property name="driver" value="${driver}"/>
                <property name="url" value="${url}"/>
                <property name="username" value="${username}"/>
                <property name="password" value="${password}"/>

                <!-- 连接池配置 -->
                <property name="poolMaximumActiveConnections" value="20"/>
                <property name="poolMaximumIdleConnections" value="10"/>
                <property name="poolMaximumCheckoutTime" value="20000"/>
            </dataSource>
        </environment>
    </environments>

    <!-- Mapper 扫描 -->
    <mappers>
        <package name="com.example.mapper"/>
    </mappers>
</configuration>
```

### 2. 使用工具类

```java
public class MyBatisUtil {
    private static final SqlSessionFactory factory;

    static {
        try {
            String resource = "mybatis-config.xml";
            InputStream inputStream = Resources.getResourceAsStream(resource);
            factory = new SqlSessionFactoryBuilder().build(inputStream);
        } catch (IOException e) {
            throw new RuntimeException("MyBatis 初始化失败", e);
        }
    }

    public static SqlSession getSession() {
        return factory.openSession();
    }

    public static SqlSession getSession(boolean autoCommit) {
        return factory.openSession(autoCommit);
    }
}
```

    </select>
