## 前言

一对多关系是最常见的关联关系。本文将讲解如何使用 MyBatis 处理一对多关系，并解决 N+1 问题。

---

## 一对多关系示例

### 场景：用户与订单

```java
@Data
public class User {
    private Long id;
    private String username;
    private List<Order> orders;  // 一对多关系
}

@Data
public class Order {
    private Long id;
    private Long userId;
    private String orderNo;
    private BigDecimal amount;
}
```

---

## collection 标签

### 嵌套查询方式

```xml
<resultMap id="UserWithOrdersMap" type="User">
    <id column="id" property="id"/>
    <result column="username" property="username"/>

    <!-- 集合映射 -->
    <collection property="orders" column="id"
                select="selectOrdersByUserId" fetchType="lazy"/>
</resultMap>

<select id="selectById" resultMap="UserWithOrdersMap">
    SELECT * FROM users WHERE id = #{id}
</select>

<select id="selectOrdersByUserId" resultType="Order">
    SELECT * FROM orders WHERE user_id = #{userId}
</select>
```

### 嵌套结果方式

```xml
<resultMap id="UserWithOrdersMap2" type="User">
    <id column="id" property="id"/>
    <result column="username" property="username"/>

    <!-- 集合映射 -->
    <collection property="orders" ofType="Order">
        <id column="order_id" property="id"/>
        <result column="order_no" property="orderNo"/>
        <result column="amount" property="amount"/>
    </collection>
</resultMap>

<select id="selectByIdWithOrders" resultMap="UserWithOrdersMap2">
    SELECT
        u.id, u.username,
        o.id as order_id, o.order_no, o.amount
    FROM users u
    LEFT JOIN orders o ON u.id = o.user_id
    WHERE u.id = #{id}
</select>
```

## N+1 问题及解决

### 问题示例

```java
// 查询 10 个用户
List<User> users = userMapper.selectAll();

// 访问每个用户的订单（触发 N 次查询）
for (User user : users) {
    List<Order> orders = user.getOrders();  // N 次查询
}
// 总共：1 + N 次查询
```

### 解决方案 1：使用嵌套结果

```xml
<!-- 一次查询获取所有数据 -->
<select id="selectAllWithOrders" resultMap="UserWithOrdersMap2">
    SELECT
        u.id, u.username,
        o.id as order_id, o.order_no, o.amount
    FROM users u
    LEFT JOIN orders o ON u.id = o.user_id
</select>
```

### 解决方案 2：批量加载

```xml
<resultMap id="UserWithOrdersMap" type="User">
    <id column="id" property="id"/>
    <result column="username" property="username"/>

    <collection property="orders"
                select="selectOrdersByUserIds"
                column="id"
                fetchType="lazy"/>
</resultMap>

<!-- 批量查询 -->
<select id="selectOrdersByUserIds" resultType="Order">
    SELECT * FROM orders
    WHERE user_id IN
    <foreach collection="list" item="id" open="(" close=")" separator=",">
        #{id}
    </foreach>
</select>
```
