## 变量声明

### 标准声明方式

Go 使用 `var` 关键字声明变量：

```go
package main

import "fmt"

func main() {
    // 完整声明
    var name string
    name = "张三"
    fmt.Println(name)

    // 声明并初始化
    var age int = 25
    fmt.Println(age)

    // 类型推断（推荐）
    var city = "北京"
    fmt.Println(city)
}
```

### 短变量声明（最常用）

在函数内，可以使用 `:=` 进行短变量声明：

```go
package main

import "fmt"

func main() {
    // 短变量声明（自动类型推断）
    name := "李四"
    age := 30
    isStudent := false

    fmt.Printf("姓名: %s, 年龄: %d, 学生: %t\n", name, age, isStudent)

    // 多变量声明
    x, y := 10, 20
    fmt.Printf("x=%d, y=%d\n", x, y)
}
```

### 批量声明

```go
package main

import "fmt"

func main() {
    // 使用 var 批量声明
    var (
        name   string = "王五"
        age    int    = 28
        salary float64 = 15000.50
    )

    fmt.Printf("姓名: %s, 年龄: %d, 薪资: %.2f\n", name, age, salary)

    // 使用 := 批量声明
    a, b, c := 1, 2, 3
    fmt.Printf("a=%d, b=%d, c=%d\n", a, b, c)
}
```

### 因式分解

```go
package main

import "fmt"

func main() {
    // 相同类型的变量可以因式分解
    var (
        x, y int    // x 和 y 都是 int
        p, q string // p 和 q 都是 string
    )

    x, y = 10, 20
    p, q = "Hello", "World"

    fmt.Printf("x=%d, y=%d\n", x, y)
    fmt.Printf("p=%s, q=%s\n", p, q)
}
```

## 基本数据类型

### 布尔型（bool）

```go
package main

import "fmt"

func main() {
    var isPassed bool = true
    var isFailed bool = false

    fmt.Printf("通过: %t, 失败: %t\n", isPassed, isFailed)

    // 布尔运算
    fmt.Printf("true && false = %t\n", true && false)
    fmt.Printf("true || false = %t\n", true || false)
    fmt.Printf("!true = %t\n", !true)
}
```

### 数值类型

#### 整型

```go
package main

import "fmt"

func main() {
    // 有符号整型
    var i int = 100          // 根据平台自动选择（32位或64位）
    var i8 int8 = 127        // -128 到 127
    var i16 int16 = 32767    // -32768 到 32767
    var i32 int32 = 2147483647
    var i64 int64 = 9223372036854775807

    // 无符号整型
    var u uint = 100         // 根据平台自动选择
    var u8 uint8 = 255       // 0 到 255（byte 是 uint8 的别名）
    var u16 uint16 = 65535
    var u32 uint32 = 4294967295
    var u64 uint64 = 18446744073709551615

    // 字符类型（rune 是 int32 的别名）
    var r rune = '中'

    fmt.Printf("int: %d\n", i)
    fmt.Printf("int8: %d\n", i8)
    fmt.Printf("uint8: %d\n", u8)
    fmt.Printf("rune: %c (Unicode: %U)\n", r, r)
}
```

#### 浮点型

```go
package main

import "fmt"

func main() {
    var f32 float32 = 3.14
    var f64 float64 = 3.141592653589793

    fmt.Printf("float32: %f\n", f32)
    fmt.Printf("float64: %.15f\n", f64)

    // 科学计数法
    var scientific float64 = 1.234e10
    fmt.Printf("科学计数法: %f\n", scientific)
}
```

#### 复数类型

```go
package main

import "fmt"

func main() {
    var c1 complex64 = 3 + 4i
    var c2 complex128 = 1 + 2i

    fmt.Printf("复数1: %v\n", c1)
    fmt.Printf("实部: %f, 虚部: %f\n", real(c1), imag(c1))

    // 复数运算
    sum := c1 + c2
    fmt.Printf("复数相加: %v\n", sum)
}
```

### 字符串类型

```go
package main

import "fmt"

func main() {
    // 字符串声明
    var s1 string = "Hello"
    s2 := "World"

    // 字符串拼接
    message := s1 + " " + s2
    fmt.Println(message)

    // 多行字符串（使用反引号）
    multiline := `
    这是一个
    多行字符串
    示例
    `
    fmt.Println(multiline)

    // 字符串长度（字节数）
    fmt.Printf("字符串长度: %d\n", len(message))

    // 访问字符（字节）
    fmt.Printf("第一个字符: %c\n", message[0])
}
```

## 常量

### 常量声明

```go
package main

import "fmt"

func main() {
    // 常量使用 const 关键字声明
    const pi = 3.14159
    const greeting = "Hello, Go!"

    fmt.Println(pi)
    fmt.Println(greeting)

    // 常量组
    const (
        StatusOK       = 200
        StatusNotFound = 404
        StatusError    = 500
    )

    fmt.Printf("OK: %d, NotFound: %d, Error: %d\n",
        StatusOK, StatusNotFound, StatusError)
}
```

### iota 枚举

`iota` 是 Go 的常量生成器，用于创建枚举：

```go
package main

import "fmt"

func main() {
    // 基础用法
    const (
        Monday = iota    // 0
        Tuesday          // 1
        Wednesday        // 2
        Thursday         // 3
        Friday           // 4
        Saturday         // 5
        Sunday           // 6
    )

    fmt.Printf("Monday: %d, Friday: %d\n", Monday, Friday)

    // 高级用法
    const (
        _  = iota             // 0（忽略）
        KB = 1 << (10 * iota) // 1 << 10 = 1024
        MB                    // 1 << 20 = 1048576
        GB                    // 1 << 30 = 1073741824
        TB                    // 1 << 40
    )

    fmt.Printf("MB: %d, GB: %d\n", MB, GB)

    // 表达式中使用 iota
    const (
        Read = 1 << iota  // 1 (二进制: 0001)
        Write             // 2 (二进制: 0010)
        Execute           // 4 (二进制: 0100)
        ReadWrite = Read | Write  // 3 (二进制: 0011)
    )

    fmt.Printf("Read: %b, Write: %b, ReadWrite: %b\n",
        Read, Write, ReadWrite)
}
```

## 类型转换

### 基本类型转换

Go 要求显式类型转换，不支持隐式转换：

```go
package main

import "fmt"

func main() {
    // 整型转换
    var i int = 42
    var f float64 = float64(i)
    fmt.Printf("int: %d, float64: %f\n", i, f)

    // 浮点型转整型（注意精度丢失）
    var pi float64 = 3.99
    var i2 int = int(pi)
    fmt.Printf("float64: %f, int: %d\n", pi, i2)

    // 字符串和数值转换
    var s string = "100"
    // i3 := int(s)  // 错误！不能直接转换
    fmt.Printf("字符串: %s\n", s)

    // 使用 strconv 包
    import "strconv"
    i4, _ := strconv.Atoi(s)
    fmt.Printf("转换后的整数: %d\n", i4)

    s2 := strconv.Itoa(i4)
    fmt.Printf("转换后的字符串: %s\n", s2)
}
```

### 字符串转换

```go
package main

import (
    "fmt"
    "strconv"
)

func main() {
    // 字符串转整数
    str := "123"
    num, err := strconv.Atoi(str)
    if err != nil {
        fmt.Println("转换失败:", err)
    } else {
        fmt.Printf("转换结果: %d\n", num)
    }

    // 整数转字符串
    num2 := 456
    str2 := strconv.Itoa(num2)
    fmt.Printf("转换结果: %s\n", str2)

    // 字符串转浮点数
    f, _ := strconv.ParseFloat("3.14", 64)
    fmt.Printf("转换结果: %f\n", f)

    // 布尔值转换
    b, _ := strconv.ParseBool("true")
    fmt.Printf("转换结果: %t\n", b)
}
```

## 变量作用域

### 局部变量

```go
package main

import "fmt"

func main() {
    x := 10
    fmt.Println("外部 x:", x)

    {
        // 内部作用域
        x := 20  // 声明新的 x（遮蔽外部 x）
        y := 30
        fmt.Println("内部 x:", x)
        fmt.Println("内部 y:", y)
    }

    fmt.Println("外部 x:", x)
    // fmt.Println(y)  // 错误：y 在外部不可访问
}
```

### 全局变量

```go
package main

import "fmt"

// 全局变量
var globalVar = "我是全局变量"

func function1() {
    fmt.Println("函数1:", globalVar)
}

func function2() {
    fmt.Println("函数2:", globalVar)
}

func main() {
    fmt.Println("主函数:", globalVar)
    function1()
    function2()
}
```

## 变量零值

Go 中变量声明后会有默认的零值：

```go
package main

import "fmt"

func main() {
    var i int
    var f float64
    var b bool
    var s string

    fmt.Printf("int 零值: %d\n", i)       // 0
    fmt.Printf("float64 零值: %f\n", f)   // 0.000000
    fmt.Printf("bool 零值: %t\n", b)      // false
    fmt.Printf("string 零值: '%s'\n", s)  // '' (空字符串)
}
```

**零值对照表**：

| 类型                            | 零值  |
| ------------------------------- | ----- |
| int/int8/int16/int32/int64      | 0     |
| uint/uint8/uint16/uint32/uint64 | 0     |
| float32/float64                 | 0.0   |
| bool                            | false |
| string                          | ""    |
| 指针                            | nil   |
| 切片、map、channel              | nil   |

## 实战示例

### 示例：单位转换器

```go
package main

import "fmt"

func main() {
    const (
        MetersToKilometers = 0.001
        KilometersToMiles = 0.621371
    )

    var meters float64
    fmt.Print("请输入米数: ")
    fmt.Scanln(&meters)

    // 转换为千米
    kilometers := meters * MetersToKilometers
    fmt.Printf("%.2f 米 = %.2f 千米\n", meters, kilometers)

    // 转换为英里
    miles := kilometers * KilometersToMiles
    fmt.Printf("%.2f 千米 = %.2f 英里\n", kilometers, miles)
}
```

### 示例：温度转换

```go
package main

import "fmt"

func celsiusToFahrenheit(c float64) float64 {
    return (c * 9 / 5) + 32
}

func fahrenheitToCelsius(f float64) float64 {
    return (f - 32) * 5 / 9
}

func main() {
    var celsius float64 = 25
    var fahrenheit float64 = 77

    fmt.Printf("%.2f°C = %.2f°F\n",
        celsius, celsiusToFahrenheit(celsius))
    fmt.Printf("%.2f°F = %.2f°C\n",
        fahrenheit, fahrenheitToCelsius(fahrenheit))
}
```

## 最佳实践

### ✅ 推荐做法

1. **使用短变量声明**

```go
// ✅ 推荐
name := "张三"
age := 25

// ❌ 不推荐
var name string = "张三"
var age int = 25
```

2. **使用有意义的变量名**

```go
// ✅ 推荐
userAge := 25
totalPrice := 99.99

// ❌ 不推荐
x := 25
y := 99.99
```

3. **限制变量作用域**

```go
// ✅ 推荐
func process() {
    if condition {
        temp := calculate()
        use(temp)
    }
}

// ❌ 不推荐
func process() {
    var temp int
    if condition {
        temp = calculate()
    }
    use(temp)
}
```

### ❌ 避免做法

1. **不要过度使用类型转换**

```go
// ❌ 不必要的转换
var i int = 42
var i2 int = int(i)

// ✅ 直接使用
var i int = 42
```

2. **不要忽略错误**

```go
// ❌ 忽略错误
num, _ := strconv.Atoi("123")

// ✅ 处理错误
num, err := strconv.Atoi("123")
if err != nil {
    // 处理错误
}
```

## 常见问题

**Q: var 和 := 有什么区别？**

A:

- `var` 可以在任何地方使用
- `:=` 只能在函数内使用
- `:=` 更简洁，推荐在函数内使用

**Q: 字符串是基本类型吗？**

A: 是的，Go 中字符串是基本类型，不是字符数组。

**Q: 如何声明一个未初始化的变量？**

A: 声明变量时会自动初始化为零值：

```go
var i int        // i = 0
var s string     // s = ""
var b bool       // b = false
```
