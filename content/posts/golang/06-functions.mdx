## 为什么需要函数

假设你要编写一个程序来处理用户数据。如果不使用函数，你可能需要在多个地方重复编写相同的逻辑。这不仅让代码变得冗长，而且一旦需要修改逻辑，就要在多处进行更改。

函数的出现解决了这个问题。它将代码封装成可重用的块，让程序更模块化、更易维护。Go 语言的函数设计简洁而强大，支持多返回值、可变参数、闭包等特性，让代码更加灵活。

## 函数基础

### 函数定义与调用

Go 语言的函数定义非常直观。函数名在前，参数列表在后，最后是返回值类型。

```go
package main

import "fmt"

// 无参数无返回值的函数
func sayHello() {
    fmt.Println("Hello, Go!")
}

// 带参数的函数
func greet(name string) {
    fmt.Printf("Hello, %s\n", name)
}

// 带返回值的函数
func add(a int, b int) int {
    return a + b
}

// 参数类型相同时可以简写
func subtract(a, b int) int {
    return a - b
}

func main() {
    sayHello()
    greet("张三")

    result := add(10, 20)
    fmt.Printf("10 + 20 = %d\n", result)

    diff := subtract(10, 3)
    fmt.Printf("10 - 3 = %d\n", diff)
}
```

### 多返回值

Go 语言的一个独特特性是函数可以返回多个值。这在处理错误时特别有用，可以同时返回结果和错误信息。

```go
package main

import (
    "fmt"
    "math"
)

// 返回商和余数
func divide(dividend, divisor int) (int, int) {
    quotient := dividend / divisor
    remainder := dividend % divisor
    return quotient, remainder
}

// 返回结果和错误
func sqrt(value float64) (float64, error) {
    if value < 0 {
        return 0, fmt.Errorf("不能计算负数的平方根")
    }
    return math.Sqrt(value), nil
}

// 命名返回值
func rectangle(width, height float64) (area, perimeter float64) {
    // 直接给命名返回值赋值
    area = width * height
    perimeter = 2 * (width + height)
    // return 语句会自动返回已命名的变量
    return
}

func main() {
    // 接收多个返回值
    q, r := divide(17, 5)
    fmt.Printf("17 ÷ 5 = %d 余 %d\n", q, r)

    // 只接收需要的值，使用 _ 忽略不需要的返回值
    q2, _ := divide(20, 3)
    fmt.Printf("20 ÷ 3 = %d\n", q2)

    // 处理带错误返回的函数
    result, err := sqrt(16)
    if err != nil {
        fmt.Println("计算出错:", err)
    } else {
        fmt.Printf("√16 = %.2f\n", result)
    }

    // 错误情况
    _, err2 := sqrt(-4)
    if err2 != nil {
        fmt.Println("计算出错:", err2)
    }

    // 使用命名返回值
    a, p := rectangle(5, 3)
    fmt.Printf("矩形面积: %.2f, 周长: %.2f\n", a, p)
}
```

## 可变参数函数

有时你无法确定函数会接收多少个参数。Go 语言的切片类型可以很好地处理这种情况。

```go
package main

import "fmt"

// 可变参数函数，numbers 的类型是 []int
func sum(numbers ...int) int {
    total := 0
    // numbers 实际上是一个切片
    for _, num := range numbers {
        total += num
    }
    return total
}

// 混合固定参数和可变参数
func greet(prefix string, names ...string) {
    for _, name := range names {
        fmt.Printf("%s, %s!\n", prefix, name)
    }
}

// 返回可变参数的总数、总和、平均值
func stats(numbers ...int) (count int, sum int, average float64) {
    count = len(numbers)
    for _, num := range numbers {
        sum += num
    }
    if count > 0 {
        average = float64(sum) / float64(count)
    }
    return
}

func main() {
    // 可以传递任意数量的参数
    fmt.Println(sum(1, 2, 3))
    fmt.Println(sum(1, 2, 3, 4, 5))
    fmt.Println(sum())

    // 使用切片展开操作符 ...
    nums := []int{10, 20, 30, 40}
    fmt.Println(sum(nums...))

    // 混合参数
    greet("Hello", "张三", "李四", "王五")

    // 获取统计信息
    c, s, avg := stats(10, 20, 30, 40, 50)
    fmt.Printf("数量: %d, 总和: %d, 平均值: %.2f\n", c, s, avg)
}
```

## 匿名函数与闭包

Go 支持匿名函数，即没有名字的函数。它们可以作为值传递，也可以赋值给变量。闭包是指匿名函数可以访问其外部作用域的变量。

```go
package main

import "fmt"

// 函数作为参数
func applyOperation(x, y int, operation func(int, int) int) int {
    return operation(x, y)
}

// 返回一个函数
func createMultiplier(factor int) func(int) int {
    // 返回的匿名函数可以访问外部的 factor 变量
    return func(x int) int {
        return x * factor
    }
}

// 闭包累加器
func createAccumulator() func(int) int {
    sum := 0
    return func(x int) int {
        sum += x
        return sum
    }
}

func main() {
    // 匿名函数直接调用
    func(msg string) {
        fmt.Println(msg)
    }("直接调用的匿名函数")

    // 匿名函数赋值给变量
    add := func(a, b int) int {
        return a + b
    }
    fmt.Println("5 + 3 =", add(5, 3))

    // 使用高阶函数
    result := applyOperation(10, 5, func(a, b int) int {
        return a * b
    })
    fmt.Println("10 * 5 =", result)

    // 闭包示例：创建乘法函数
    double := createMultiplier(2)
    triple := createMultiplier(3)
    fmt.Println("double(5) =", double(5))
    fmt.Println("triple(5) =", triple(5))

    // 闭包示例：累加器
    acc := createAccumulator()
    fmt.Println(acc(10))
    fmt.Println(acc(20))
    fmt.Println(acc(30))
}
```

## defer 语句

defer 语句用于延迟函数的执行，直到包含它的函数返回。这在资源清理、文件关闭、解锁等场景中非常有用。

```go
package main

import (
    "fmt"
    "os"
)

func createFile(name string) *os.File {
    file, err := os.Create(name)
    if err != nil {
        panic(err)
    }
    // 使用 defer 确保函数返回时关闭文件
    defer file.Close()
    fmt.Printf("文件 %s 创建成功\n", name)
    return file
}

func processItem(index int) {
    // defer 语句在函数返回前执行
    // 多个 defer 按照后进先出(LIFO)的顺序执行
    defer fmt.Printf("清理第 %d 个项目\n", index)
    defer fmt.Printf("处理第 %d 个项目\n", index)

    fmt.Printf("开始处理第 %d 个项目\n", index)
}

// defer 在命名返回值中的作用
func deferExample() (result int) {
    // defer 可以修改命名返回值
    defer func() {
        result *= 2
    }()
    result = 10
    return
}

func main() {
    // defer 的执行顺序
    fmt.Println("开始")
    defer fmt.Println("第 1 个 defer")
    defer fmt.Println("第 2 个 defer")
    defer fmt.Println("第 3 个 defer")
    fmt.Println("结束")

    fmt.Println("\n---\n")

    // defer 在实际应用中
    for i := 1; i <= 3; i++ {
        processItem(i)
    }

    fmt.Println("\n---\n")

    // defer 修改返回值
    value := deferExample()
    fmt.Printf("deferExample 返回值: %d\n", value)
}
```

## 实战示例：任务队列的函数设计

现在我们开始构建任务队列 API 的基础。首先设计一些处理任务的函数。

```go
package main

import (
    "fmt"
    "time"
)

// Task 表示一个任务
type Task struct {
    ID       int
    Name     string
    Priority int
    Status   string
}

// 任务状态常量
const (
    StatusPending   = "待处理"
    StatusProcessing = "处理中"
    StatusCompleted = "已完成"
    StatusFailed    = "失败"
)

// 创建新任务
func createTask(id int, name string, priority int) Task {
    return Task{
        ID:       id,
        Name:     name,
        Priority: priority,
        Status:   StatusPending,
    }
}

// 处理单个任务
func processTask(task Task) Task {
    task.Status = StatusProcessing
    fmt.Printf("处理任务: %s (优先级: %d)\n", task.Name, task.Priority)

    // 模拟任务处理
    time.Sleep(time.Millisecond * 100)

    // 根据优先级决定是否成功
    if task.Priority > 5 {
        task.Status = StatusFailed
    } else {
        task.Status = StatusCompleted
    }

    return task
}

// 批量处理任务，返回成功和失败的任务
func processTasks(tasks []Task) ([]Task, []Task) {
    var completed []Task
    var failed []Task

    for _, task := range tasks {
        processed := processTask(task)
        if processed.Status == StatusCompleted {
            completed = append(completed, processed)
        } else {
            failed = append(failed, processed)
        }
    }

    return completed, failed
}

// 按优先级过滤任务
func filterByPriority(tasks []Task, minPriority int) []Task {
    var filtered []Task
    for _, task := range tasks {
        if task.Priority >= minPriority {
            filtered = append(filtered, task)
        }
    }
    return filtered
}

// 打印任务统计信息
func printStats(completed, failed []Task) {
    fmt.Println("\n--- 任务统计 ---")
    fmt.Printf("已完成: %d 个\n", len(completed))
    fmt.Printf("失败: %d 个\n", len(failed))

    if len(failed) > 0 {
        fmt.Println("\n失败的任务:")
        for _, task := range failed {
            fmt.Printf("  - %s (优先级: %d)\n", task.Name, task.Priority)
        }
    }
}

func main() {
    // 创建一些任务
    tasks := []Task{
        createTask(1, "发送邮件", 2),
        createTask(2, "生成报表", 3),
        createTask(3, "备份数据库", 8),
        createTask(4, "清理缓存", 1),
        createTask(5, "同步数据", 6),
    }

    fmt.Println("=== 任务队列处理系统 ===\n")

    // 批量处理任务
    completed, failed := processTasks(tasks)

    // 打印统计信息
    printStats(completed, failed)

    // 过滤高优先级任务
    highPriority := filterByPriority(tasks, 5)
    fmt.Printf("\n高优先级任务: %d 个\n", len(highPriority))
}
```

## 权衡与思考

### 使用多返回值的优势与代价

**优势**：

- 错误处理更加明确，不会忽略错误
- 返回相关联的多个值，代码更清晰
- 避免了通过参数传递引用的方式

**代价**：

- 调用方需要处理所有返回值，代码可能变多
- 对于简单的场景，可能显得有些繁琐

**何时使用**：

- 需要同时返回结果和错误时
- 返回的多个值在逻辑上相关联时

### 使用闭包的注意事项

**优势**：

- 可以创建有状态的函数
- 实现数据封装和私有变量
- 用于函数工厂和高阶函数

**代价**：

- 可能导致内存泄漏（如果不正确使用）
- 代码可读性可能降低
- 调试时可能较困难

**最佳实践**：

- 只在必要时使用闭包
- 确保闭包的生命周期合理
- 避免在闭包中修改外部变量

### defer 的使用建议

**优势**：

- 确保资源被正确释放
- 代码更整洁，资源清理逻辑集中
- 处理 panic 时也能执行清理

**代价**：

- defer 有轻微的性能开销
- 可能影响代码的可读性（如果 defer 语句过多）

**最佳实践**：

- 在涉及资源清理时使用 defer
- 不要在循环中使用 defer（除非你真的理解其行为）
- 记住 defer 是后进先出的执行顺序

## 总结

函数是 Go 语言的基本构建块。通过函数，你可以：

- 将代码组织成可重用的模块
- 使用多返回值同时返回结果和错误
- 通过可变参数处理不定数量的参数
- 使用闭包创建有状态的函数
- 使用 defer 确保资源被正确释放
