## 函数基础

### 函数定义

```go
package main

import "fmt"

// 无参数无返回值
func sayHello() {
    fmt.Println("Hello, Go!")
}

// 有参数无返回值
func greet(name string) {
    fmt.Println("Hello,", name)
}

// 有参数有返回值
func add(a, b int) int {
    return a + b
}

// 多返回值
func divide(a, b float64) (float64, error) {
    if b == 0 {
        return 0, fmt.Errorf("除数不能为零")
    }
    return a / b, nil
}

func main() {
    sayHello()
    greet("张三")
    result := add(10, 20)
    fmt.Println("10 + 20 =", result)

    quotient, err := divide(10, 3)
    if err != nil {
        fmt.Println("错误:", err)
    } else {
        fmt.Println("商:", quotient)
    }
}
```

### 命名返回值

```go
package main

import "fmt"

func rectangle(width, height float64) (area float64, perimeter float64) {
    area = width * height
    perimeter = 2 * (width + height)
    return  // 自动返回已命名的变量
}

func main() {
    a, p := rectangle(3, 4)
    fmt.Printf("面积: %.2f, 周长: %.2f\n", a, p)
}
```

## 可变参数

```go
package main

import "fmt"

// 可变参数函数
func sum(numbers ...int) int {
    total := 0
    for _, num := range numbers {
        total += num
    }
    return total
}

func main() {
    fmt.Println(sum(1, 2, 3))
    fmt.Println(sum(1, 2, 3, 4, 5))

    // 使用切片
    nums := []int{10, 20, 30}
    fmt.Println(sum(nums...))
}
```

## 匿名函数和闭包

```go
package main

import "fmt"

func main() {
    // 匿名函数
    add := func(a, b int) int {
        return a + b
    }
    fmt.Println(add(3, 4))

    // 闭包
    multiplier := func(factor int) func(int) int {
        return func(x int) int {
            return x * factor
        }
    }

    double := multiplier(2)
    triple := multiplier(3)

    fmt.Println(double(5))  // 10
    fmt.Println(triple(5))  // 15
}
```

## 高阶函数

```go
package main

import "fmt"

// 函数作为参数
func apply(nums []int, op func(int) int) []int {
    result := make([]int, len(nums))
    for i, num := range nums {
        result[i] = op(num)
    }
    return result
}

func square(x int) int {
    return x * x
}

func main() {
    nums := []int{1, 2, 3, 4, 5}
    squared := apply(nums, square)
    fmt.Println(squared)
}
```

## defer 语句

```go
package main

import "fmt"

func main() {
    // defer 延迟执行，常用于资源清理
    defer fmt.Println("最后执行")
    defer fmt.Println("倒数第二执行")

    fmt.Println("首先执行")

    // defer 在函数返回前执行
    defer func() {
        fmt.Println("defer 匿名函数")
    }()
}
```

## 实战示例

### 示例：斐波那契数列

```go
package main

import "fmt"

// 递归实现
func fibonacci(n int) int {
    if n <= 1 {
        return n
    }
    return fibonacci(n-1) + fibonacci(n-2)
}

// 闭包实现
func fibonacciGenerator() func() int {
    a, b := 0, 1
    return func() int {
        a, b = b, a+b
        return a
    }
}

func main() {
    fmt.Println("递归方式:")
    for i := 0; i < 10; i++ {
        fmt.Printf("%d ", fibonacci(i))
    }

    fmt.Println("\n闭包方式:")
    fib := fibonacciGenerator()
    for i := 0; i < 10; i++ {
        fmt.Printf("%d ", fib())
    }
}
```
