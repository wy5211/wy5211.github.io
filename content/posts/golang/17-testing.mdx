## 基础测试

### 编写测试

```go
// calc.go
package calc

func Add(a, b int) int {
    return a + b
}

func Subtract(a, b int) int {
    return a - b
}
```

```go
// calc_test.go
package calc

import "testing"

func TestAdd(t *testing.T) {
    result := Add(2, 3)
    expected := 5
    if result != expected {
        t.Errorf("Add(2, 3) = %d; expected %d", result, expected)
    }
}

func TestSubtract(t *testing.T) {
    result := Subtract(5, 3)
    expected := 2
    if result != expected {
        t.Errorf("Subtract(5, 3) = %d; expected %d", result, expected)
    }
}
```

### 运行测试

```bash
# 运行所有测试
go test

# 运行指定测试
go test -run TestAdd

# 详细输出
go test -v

# 覆盖率
go test -cover
```

## 表格驱动测试

```go
package calc

import "testing"

func TestAddTableDriven(t *testing.T) {
    tests := []struct {
        name     string
        a        int
        b        int
        expected int
    }{
        {"正数相加", 2, 3, 5},
        {"负数相加", -2, -3, -5},
        {"零相加", 0, 5, 5},
    }

    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            result := Add(tt.a, tt.b)
            if result != tt.expected {
                t.Errorf("Add(%d, %d) = %d; expected %d", tt.a, tt.b, result, tt.expected)
            }
        })
    }
}
```

## 基准测试

```go
package calc

import "testing"

func BenchmarkAdd(b *testing.B) {
    for i := 0; i < b.N; i++ {
        Add(100, 200)
    }
}

func BenchmarkSubtract(b *testing.B) {
    for i := 0; i < b.N; i++ {
        Subtract(200, 100)
    }
}
```

```bash
# 运行基准测试
go test -bench=.

# 带内存统计
go test -bench=. -benchmem
```

## 测试模拟

### 使用接口模拟

```go
package main

import (
    "testing"
)

type Database interface {
    GetUser(id int) (string, error)
}

type Service struct {
    db Database
}

func (s *Service) GetUserName(id int) (string, error) {
    return s.db.GetUser(id)
}

// Mock Database
type MockDB struct {
    GetUserFunc func(int) (string, error)
}

func (m *MockDB) GetUser(id int) (string, error) {
    return m.GetUserFunc(id)
}

func TestService_GetUserName(t *testing.T) {
    mockDB := &MockDB{
        GetUserFunc: func(id int) (string, error) {
            return "张三", nil
        },
    }

    service := &Service{db: mockDB}
    name, err := service.GetUserName(1)

    if err != nil {
        t.Errorf("unexpected error: %v", err)
    }

    if name != "张三" {
        t.Errorf("expected 张三, got %s", name)
    }
}
```

## 测试辅助函数

### testify/assert

```bash
go get github.com/stretchr/testify
```

```go
package calc

import (
    "testing"
    "github.com/stretchr/testify/assert"
)

func TestAddWithAssert(t *testing.T) {
    result := Add(2, 3)
    assert.Equal(t, 5, result)
    assert.NotEqual(t, 6, result)
}
```
