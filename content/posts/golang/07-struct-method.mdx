## 结构体进阶

### 结构体嵌套

```go
package main

import "fmt"

type Address struct {
    City    string
    Street  string
}

type Person struct {
    Name    string
    Age     int
    Address // 匿名嵌套
}

func main() {
    p := Person{
        Name: "张三",
        Age:  25,
        Address: Address{
            City:   "北京",
            Street: "长安街",
        },
    }

    // 直接访问嵌套字段
    fmt.Println(p.Name)
    fmt.Println(p.City)
    fmt.Println(p.Street)
}
```

### 结构体比较

```go
package main

import "fmt"

type Point struct {
    X, Y int
}

func main() {
    p1 := Point{1, 2}
    p2 := Point{1, 2}
    p3 := Point{2, 3}

    fmt.Println("p1 == p2:", p1 == p2)  // true
    fmt.Println("p1 == p3:", p1 == p3)  // false
}
```

## 方法

### 值接收者 vs 指针接收者

```go
package main

import "fmt"

type Counter struct {
    count int
}

// 值接收者（不修改原对象）
func (c Counter) Value() int {
    return c.count
}

// 指针接收者（可以修改原对象）
func (c *Counter) Increment() {
    c.count++
}

func main() {
    counter := Counter{count: 0}

    fmt.Println("初始值:", counter.Value())

    counter.Increment()
    fmt.Println("增量后:", counter.Value())
}
```

### 方法表达式

```go
package main

import "fmt"

type Math struct{}

func (m Math) Add(a, b int) int {
    return a + b
}

func main() {
    m := Math{}

    // 方法调用
    result := m.Add(10, 20)
    fmt.Println(result)

    // 方法表达式
    addFunc := Math.Add
    result2 := addFunc(m, 30, 40)
    fmt.Println(result2)
}
```

## 接口

### 接口定义

```go
package main

import "fmt"

// 定义接口
type Shape interface {
    Area() float64
    Perimeter() float64
}

// 实现接口
type Rectangle struct {
    Width, Height float64
}

func (r Rectangle) Area() float64 {
    return r.Width * r.Height
}

func (r Rectangle) Perimeter() float64 {
    return 2 * (r.Width + r.Height)
}

type Circle struct {
    Radius float64
}

func (c Circle) Area() float64 {
    return 3.14159 * c.Radius * c.Radius
}

func (c Circle) Perimeter() float64 {
    return 2 * 3.14159 * c.Radius
}

func main() {
    var s Shape

    s = Rectangle{Width: 3, Height: 4}
    fmt.Printf("矩形面积: %.2f\n", s.Area())

    s = Circle{Radius: 5}
    fmt.Printf("圆形面积: %.2f\n", s.Area())
}
```

### 空接口

```go
package main

import "fmt"

func printAnything(v interface{}) {
    fmt.Printf("值: %v, 类型: %T\n", v, v)
}

func main() {
    printAnything(42)
    printAnything("Hello")
    printAnything(3.14)
}
```

## 类型断言

```go
package main

import "fmt"

func main() {
    var i interface{} = "Hello"

    // 类型断言
    s, ok := i.(string)
    if ok {
        fmt.Println("字符串:", s)
    }

    // 类型选择
    switch v := i.(type) {
    case int:
        fmt.Println("整数:", v)
    case string:
        fmt.Println("字符串:", v)
    default:
        fmt.Println("未知类型")
    }
}
```

## 实战示例

### 示例：银行账户

```go
package main

import "fmt"

type BankAccount struct {
    owner   string
    balance float64
}

func NewBankAccount(owner string, initialBalance float64) *BankAccount {
    return &BankAccount{
        owner:   owner,
        balance: initialBalance,
    }
}

func (ba *BankAccount) Deposit(amount float64) {
    if amount > 0 {
        ba.balance += amount
        fmt.Printf("存入 %.2f，余额: %.2f\n", amount, ba.balance)
    }
}

func (ba *BankAccount) Withdraw(amount float64) bool {
    if amount > 0 && ba.balance >= amount {
        ba.balance -= amount
        fmt.Printf("取出 %.2f，余额: %.2f\n", amount, ba.balance)
        return true
    }
    fmt.Println("余额不足")
    return false
}

func (ba *BankAccount) Balance() float64 {
    return ba.balance
}

func main() {
    account := NewBankAccount("张三", 1000)

    account.Deposit(500)
    account.Withdraw(300)
    account.Withdraw(2000)

    fmt.Printf("最终余额: %.2f\n", account.Balance())
}
```
