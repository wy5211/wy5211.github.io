## 为什么控制流很重要

程序不是按顺序执行指令就够了，它需要根据不同情况做出决策，重复执行某些操作，跳过不必要的步骤。在任务队列 API 中，你需要根据任务状态决定是否重试，遍历任务列表找到符合条件的任务，处理不同类型的任务请求。

Go 的控制流设计很简洁。它只有一种循环语句（for），没有 while 或 do-while；switch 语句更强大，不需要显式 break；if 语句可以带初始化子句。这些设计减少了语法噪音，让代码更清晰。

这篇文章会用任务处理的实际场景来讲解 Go 的各种控制流语句。

## if-else 语句

if 语句用于条件判断，Go 的 if 有个独特之处：可以包含一个初始化语句，这在错误处理和资源清理时特别有用。

```go
package main

import "fmt"

func main() {
    // 基本 if-else
    taskStatus := "pending"
    if taskStatus == "pending" {
        fmt.Println("任务等待执行")
    } else if taskStatus == "running" {
        fmt.Println("任务正在执行")
    } else {
        fmt.Println("任务已完成或失败")
    }

    // if 带初始化语句（Go 的特色）
    // 变量作用域仅限于 if 块
    if retryCount := 3; retryCount > 0 {
        fmt.Printf("还有 %d 次重试机会\n", retryCount)
    }
    // retryCount 在这里不可访问

    // 实际应用：验证任务存在性
    taskMap := map[string]int{
        "task_1001": 1001,
        "task_1002": 1002,
    }

    // if 初始化语句常用于检查 map 键是否存在
    if taskID, exists := taskMap["task_1001"]; exists {
        fmt.Printf("找到任务 ID: %d\n", taskID)
    } else {
        fmt.Println("任务不存在")
    }

    // 复杂条件判断
    priority := 8
    isUrgent := true
    hasPermission := true

    // 可以用括号让复杂条件更清晰
    if (priority >= 7 && isUrgent) || hasPermission {
        fmt.Println("任务满足执行条件")
    }
}
```

if 语句的初始化子句是个强大的特性，它让你能在条件判断前准备好变量，同时限制变量的作用域。这在处理错误、检查资源可用性时特别有用。

## switch 语句

Go 的 switch 比其他语言的更灵活。它不需要显式 break（默认不会继续执行下一个 case），支持多值匹配，甚至可以不带表达式实现类似 if-else 链的功能。

```go
package main

import "fmt"

func main() {
    // 基本 switch
    taskType := "backup"
    switch taskType {
    case "backup":
        fmt.Println("执行备份任务")
    case "cleanup":
        fmt.Println("执行清理任务")
    case "report":
        fmt.Println("生成报表")
    default:
        fmt.Println("未知任务类型")
    }

    // 多值匹配（一个 case 匹配多个值）
    dayOfWeek := 6
    switch dayOfWeek {
    case 1, 2, 3, 4, 5:
        fmt.Println("工作日")
    case 6, 7:
        fmt.Println("周末")
    default:
        fmt.Println("无效日期")
    }

    // 无表达式的 switch（类似 if-else 链）
    score := 85
    switch {
    case score >= 90:
        fmt.Println("优秀")
    case score >= 80:
        fmt.Println("良好")
    case score >= 60:
        fmt.Println("及格")
    default:
        fmt.Println("不及格")
    }

    // switch 带初始化语句
    switch priority := getTaskPriority(1001); priority {
    case 1, 2, 3:
        fmt.Println("低优先级任务")
    case 4, 5, 6:
        fmt.Println("中优先级任务")
    case 7, 8, 9:
        fmt.Println("高优先级任务")
    case 10:
        fmt.Println("紧急任务")
    default:
        fmt.Println("优先级超出范围")
    }

    // fallthrough：强制执行下一个 case（很少用）
    status := "running"
    switch status {
    case "pending":
        fmt.Println("任务等待中")
        fallthrough  // 继续执行下一个 case
    case "running":
        fmt.Println("任务正在运行")
    case "completed":
        fmt.Println("任务已完成")
    }
}

func getTaskPriority(taskID int) int {
    // 模拟根据任务 ID 获取优先级
    return 7
}
```

switch 语句的灵活性让代码更简洁。多值匹配避免了重复的 case 语句，无表达式的 switch 让复杂的条件判断更清晰，fallthrough 虽然存在但很少使用。

## for 循环：Go 的唯一循环

Go 只有一种循环语句：for。但它足够灵活，可以替代其他语言中的 while、do-while 等循环形式。

```go
package main

import "fmt"

func main() {
    // 标准的 for 循环（类似 C 的 for）
    fmt.Println("=== 标准循环 ===")
    for i := 0; i < 5; i++ {
        fmt.Printf("计数: %d\n", i)
    }

    // while 循环形式（只有条件）
    fmt.Println("\n=== While 形式 ===")
    count := 0
    for count < 3 {
        fmt.Printf("计数: %d\n", count)
        count++
    }

    // 无限循环（用作服务器主循环很常见）
    fmt.Println("\n=== 带退出条件的无限循环 ===")
    attempts := 0
    maxAttempts := 3
    success := false

    for {
        attempts++
        fmt.Printf("尝试 %d/%d\n", attempts, maxAttempts)

        // 模拟任务执行
        if attempts == 2 {
            success = true
            fmt.Println("任务成功！")
            break
        }

        if attempts >= maxAttempts {
            fmt.Println("达到最大尝试次数")
            break
        }
    }

    if !success {
        fmt.Println("任务最终失败")
    }

    // range 遍历（最常用）
    fmt.Println("\n=== 遍历切片 ===")
    tasks := []string{"备份", "清理", "报表"}
    for index, task := range tasks {
        fmt.Printf("位置 %d: %s\n", index, task)
    }

    // 只要值，忽略索引
    for _, task := range tasks {
        fmt.Printf("任务: %s\n", task)
    }

    // 只要索引，忽略值
    for index := range tasks {
        fmt.Printf("索引: %d\n", index)
    }

    // 遍历 map
    fmt.Println("\n=== 遍历 Map ===")
    taskPriority := map[string]int{
        "备份": 8,
        "清理": 5,
        "报表": 6,
    }
    for task, priority := range taskPriority {
        fmt.Printf("%s: 优先级 %d\n", task, priority)
    }

    // 遍历字符串（按 Unicode 字符遍历）
    fmt.Println("\n=== 遍历字符串 ===")
    text := "Hello世界"
    for index, char := range text {
        fmt.Printf("位置 %d: %c (Unicode: %U)\n", index, char, char)
    }
}
```

for 循环的 range 形式是 Go 中最常用的遍历方式。它适用于切片、数组、map、字符串、channel 等多种类型。遍历 map 时要注意，顺序是随机的，每次遍历可能不同。

## break 和 continue

break 用于立即退出循环，continue 用于跳过当前迭代直接进入下一次。这两个语句配合条件判断，能实现复杂的循环逻辑。

```go
package main

import "fmt"

func main() {
    // break：找到第一个符合条件的元素就停止
    fmt.Println("=== 使用 break ===")
    tasks := []string{"备份", "清理", "报表", "邮件"}
    target := "报表"

    for i, task := range tasks {
        if task == target {
            fmt.Printf("找到目标任务 '%s' 在位置 %d\n", target, i)
            break  // 找到后就退出循环
        }
        fmt.Printf("检查任务: %s\n", task)
    }

    // continue：跳过不符合条件的元素
    fmt.Println("\n=== 使用 continue ===")
    priorities := []int{3, 8, 2, 9, 5}
    highPriorityCount := 0

    for _, priority := range priorities {
        if priority < 7 {
            continue  // 跳过低优先级
        }
        fmt.Printf("高优先级任务: %d\n", priority)
        highPriorityCount++
    }

    fmt.Printf("高优先级任务总数: %d\n", highPriorityCount)

    // 嵌套循环中的标签（跳出多层循环）
    fmt.Println("\n=== 标签跳出多层循环 ===")
outer:
    for i := 0; i < 3; i++ {
        for j := 0; j < 3; j++ {
            if i == 1 && j == 1 {
                fmt.Printf("在 (%d, %d) 跳出所有循环\n", i, j)
                break outer  // 跳出外层循环
            }
            fmt.Printf("位置: (%d, %d)\n", i, j)
        }
    }
}
```

标签跳转是 Go 的一个特色，它让你能从多层嵌套循环中直接退出。这在搜索算法、矩阵处理等场景中很有用。但不要过度使用标签，它会让代码变得难以理解。

## defer 语句

defer 语句用于延迟函数的执行，直到包含它的函数返回。常用于资源清理、解锁互斥锁、关闭文件等操作。

```go
package main

import (
    "fmt"
    "time"
)

func main() {
    // 基本用法：defer 在函数返回前执行
    fmt.Println("=== defer 基本用法 ===")
    defer fmt.Println("这条语句最后执行")
    fmt.Println("这条语句先执行")

    // 多个 defer 按 LIFO 顺序执行（后进先出）
    fmt.Println("\n=== 多个 defer ===")
    defer fmt.Println("第一个 defer（最后执行）")
    defer fmt.Println("第二个 defer")
    defer fmt.Println("第三个 defer（最先执行）")

    // 实际应用：资源清理
    fmt.Println("\n=== 资源清理 ===")
    processTask("重要任务")

    // defer 在返回前执行，即使发生 panic
    fmt.Println("\n=== defer 和 panic ===")
    safeOperation()
}

func processTask(taskName string) {
    fmt.Printf("开始处理任务: %s\n", taskName)

    // defer 确保资源总是被释放
    defer fmt.Printf("任务 '%s' 的资源已清理\n", taskName)

    // 模拟任务处理
    fmt.Printf("处理任务 '%s' 中...\n", taskName)
    time.Sleep(100 * time.Millisecond)

    fmt.Printf("任务 '%s' 处理完成\n", taskName)
}

func safeOperation() {
    defer func() {
        if r := recover(); r != nil {
            fmt.Printf("捕获到 panic: %v\n", r)
        }
    }()

    fmt.Println("执行危险操作...")
    panic("出错了！")
    fmt.Println("这行不会执行")  // panic 后的代码不会执行
}
```

defer 是 Go 处理资源和错误的优雅方式。它保证了清理代码总会执行，无论函数是正常返回还是发生 panic。多个 defer 语句按后进先出顺序执行，这很重要，因为后获取的资源通常需要先释放。

## 实战例子：任务队列处理器

把控制流语句整合起来，实现一个简单的任务队列处理器。

```go
package main

import (
    "fmt"
    "time"
)

// Task 表示一个任务
type Task struct {
    ID          int
    Name        string
    Priority    int
    Status      string
    MaxRetries  int
    RetryCount  int
}

// Processor 任务处理器
type Processor struct {
    tasks       []Task
    running     bool
}

func NewProcessor() *Processor {
    return &Processor{
        tasks:   make([]Task, 0),
        running: false,
    }
}

func (p *Processor) AddTask(name string, priority int, maxRetries int) {
    task := Task{
        ID:         len(p.tasks) + 1001,
        Name:       name,
        Priority:   priority,
        Status:     "pending",
        MaxRetries: maxRetries,
        RetryCount: 0,
    }
    p.tasks = append(p.tasks, task)
    fmt.Printf("添加任务: %s (ID: %d, 优先级: %d)\n", name, task.ID, priority)
}

func (p *Processor) Start() {
    if p.running {
        fmt.Println("处理器已经在运行")
        return
    }

    p.running = true
    fmt.Println("\n=== 启动任务处理器 ===")

    // 按优先级排序任务（简单的冒泡排序）
    for i := 0; i < len(p.tasks); i++ {
        for j := i + 1; j < len(p.tasks); j++ {
            if p.tasks[i].Priority < p.tasks[j].Priority {
                p.tasks[i], p.tasks[j] = p.tasks[j], p.tasks[i]
            }
        }
    }

    // 处理所有任务
    for i := 0; i < len(p.tasks); i++ {
        task := &p.tasks[i]

        // 跳过已完成的任务
        if task.Status == "completed" {
            continue
        }

        fmt.Printf("\n处理任务 %d: %s (优先级: %d)\n",
            task.ID, task.Name, task.Priority)

        // 带重试的任务执行循环
        success := false
        for task.RetryCount < task.MaxRetries {
            task.Status = "running"
            task.RetryCount++

            fmt.Printf("  尝试 %d/%d...", task.RetryCount, task.MaxRetries)

            // 模拟任务执行
            time.Sleep(200 * time.Millisecond)

            // 根据任务 ID 决定是否成功（仅用于演示）
            if task.ID%3 != 0 {  // ID 不被 3 整除的任务会成功
                fmt.Println(" 成功！")
                task.Status = "completed"
                success = true
                break
            } else {
                fmt.Println(" 失败")
                if task.RetryCount < task.MaxRetries {
                    fmt.Printf("  等待重试...\n")
                }
            }
        }

        // 根据执行结果设置最终状态
        if !success {
            task.Status = "failed"
            fmt.Printf("任务 %d 失败：超过最大重试次数\n", task.ID)
        }
    }

    p.running = false
    fmt.Println("\n=== 所有任务处理完成 ===")
}

func (p *Processor) PrintReport() {
    fmt.Println("\n=== 任务报告 ===")
    for _, task := range p.tasks {
        statusSymbol := "✗"
        if task.Status == "completed" {
            statusSymbol = "✓"
        }
        fmt.Printf("%s 任务 %d: %s\n", statusSymbol, task.ID, task.Name)
        fmt.Printf("   状态: %s, 优先级: %d, 重试次数: %d/%d\n",
            task.Status, task.Priority, task.RetryCount, task.MaxRetries)
    }
}

func main() {
    processor := NewProcessor()

    // 添加不同优先级的任务
    processor.AddTask("数据备份", 8, 3)
    processor.AddTask("日志清理", 5, 2)
    processor.AddTask("报表生成", 6, 1)
    processor.AddTask("邮件发送", 4, 2)
    processor.AddTask("系统更新", 9, 3)

    // 启动处理器
    processor.Start()

    // 打印报告
    processor.PrintReport()

    // 演示 defer 在实际应用中的作用
    fmt.Println("\n=== defer 演示 ===")
    demonstrateDefer()
}

func demonstrateDefer() {
    fmt.Println("进入函数")

    // defer 确保函数退出时清理资源
    defer fmt.Println("清理资源")

    // 模拟工作
    fmt.Println("执行工作...")

    // 即使发生错误，defer 也会执行
    // defer fmt.Println("这条永远不执行")
    // panic("出错了")
}
```

## 权衡与考虑

Go 的控制流设计体现了简洁性原则。只有一种循环语句减少了认知负担，但需要适应不同的 for 形式。switch 默认不贯穿避免了常见错误，但也意味着需要显式使用 fallthrough。

defer 是处理资源的优雅方案，但要注意多个 defer 的执行顺序是后进先出。defer 中的变量在 defer 声明时求值，这可能导致预期外的结果。

控制流语句的选择应该让代码意图更清晰。简单的条件用 if，多分支用 switch，固定次数或遍历集合用 for，资源清理用 defer。避免过度嵌套，复杂逻辑考虑提取成函数。
