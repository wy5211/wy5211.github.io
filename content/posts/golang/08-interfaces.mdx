## 前言

接口是 Go 语言实现多态和抽象的核心机制。Go 的接口是隐式实现的，非常灵活。

---

## 接口基础

### 接口定义和实现

```go
package main

import "fmt"

// 定义接口
type Writer interface {
    Write(data string) error
}

// 实现接口（隐式）
type File struct {
    name string
}

func (f *File) Write(data string) error {
    fmt.Printf("写入文件 %s: %s\n", f.name, data)
    return nil
}

type Console struct{}

func (c *Console) Write(data string) error {
    fmt.Println("控制台输出:", data)
    return nil
}

func main() {
    var w Writer

    w = &File{name: "test.txt"}
    w.Write("Hello")

    w = &Console{}
    w.Write("World")
}
```

## 接口组合

```go
package main

import "fmt"

type Reader interface {
    Read() string
}

type Writer interface {
    Write(data string) error
}

// 接口组合
type ReadWriter interface {
    Reader
    Writer
}

type File struct {
    content string
}

func (f *File) Read() string {
    return f.content
}

func (f *File) Write(data string) error {
    f.content = data
    return nil
}

func main() {
    file := &File{}

    var rw ReadWriter = file
    rw.Write("Hello, Go!")
    fmt.Println(rw.Read())
}
```

## 空接口

### 空接口的使用

```go
package main

import "fmt"

func printAny(v interface{}) {
    fmt.Printf("值: %v, 类型: %T\n", v, v)
}

func main() {
    printAny(42)
    printAny("Hello")
    printAny([]int{1, 2, 3})

    // 空接口作为 map 值
    m := map[string]interface{}{
        "name":   "张三",
        "age":    25,
        "scores": []int{90, 85, 95},
    }
    fmt.Println(m)
}
```

## 类型断言和类型选择

```go
package main

import "fmt"

func processValue(v interface{}) {
    // 类型断言
    s, ok := v.(string)
    if ok {
        fmt.Println("字符串:", s)
        return
    }

    // 类型选择
    switch val := v.(type) {
    case int:
        fmt.Println("整数:", val)
    case string:
        fmt.Println("字符串:", val)
    case []int:
        fmt.Println("整数切片:", val)
    default:
        fmt.Println("未知类型")
    }
}

func main() {
    processValue(42)
    processValue("Hello")
    processValue([]int{1, 2, 3})
}
```

## 常用标准接口

### Stringer 接口

```go
package main

import "fmt"

type Person struct {
    Name string
    Age  int
}

// 实现 Stringer 接口
func (p Person) String() string {
    return fmt.Sprintf("%s(%d岁)", p.Name, p.Age)
}

func main() {
    p := Person{"张三", 25}
    fmt.Println(p)  // 自动调用 String()
}
```

### error 接口

```go
package main

import "fmt"

// 自定义错误类型
type ValidationError struct {
    Field   string
    Message string
}

func (e ValidationError) Error() string {
    return fmt.Sprintf("%s: %s", e.Field, e.Message)
}

func validateAge(age int) error {
    if age < 0 {
        return ValidationError{"age", "年龄不能为负数"}
    }
    if age > 150 {
        return ValidationError{"age", "年龄不能超过150"}
    }
    return nil
}

func main() {
    err := validateAge(-5)
    if err != nil {
        fmt.Println("验证失败:", err)
    }
}
```

## 接口最佳实践

### 小接口原则

```go
// ✅ 推荐：小接口
type Reader interface {
    Read(p []byte) (n int, err error)
}

type Writer interface {
    Write(p []byte) (n int, err error)
}

// ❌ 避免：大接口
type FileOperation interface {
    Read(p []byte) (n int, err error)
    Write(p []byte) (n int, err error)
    Close() error
    Seek(offset int64, whence int) (int64, error)
    Sync() error
    // ... 更多方法
}
```

### 接口接受，结构体返回

```go
// ✅ 推荐
func Process(w io.Writer) {
    w.Write([]byte("Hello"))
}

func NewWriter() *io.PipeWriter {
    return &io.PipeWriter{}
}

// ❌ 避免
func Process(w *os.File) {
    w.Write([]byte("Hello"))
}
```

## 实战示例

### 示例：排序接口

```go
package main

import (
    "fmt"
    "sort"
)

type Person struct {
    Name string
    Age  int
}

type ByAge []Person

func (a ByAge) Len() int           { return len(a) }
func (a ByAge) Swap(i, j int)      { a[i], a[j] = a[j], a[i] }
func (a ByAge) Less(i, j int) bool { return a[i].Age < a[j].Age }

func main() {
    people := []Person{
        {"张三", 30},
        {"李四", 25},
        {"王五", 35},
    }

    sort.Sort(ByAge(people))
    fmt.Println(people)
}
```
