## 读取文件

### 读取整个文件

```go
package main

import (
    "fmt"
    "os"
)

func main() {
    // 读取整个文件
    data, err := os.ReadFile("test.txt")
    if err != nil {
        fmt.Println("错误:", err)
        return
    }
    fmt.Println(string(data))
}
```

### 按行读取

```go
package main

import (
    "bufio"
    "fmt"
    "os"
)

func main() {
    file, err := os.Open("test.txt")
    if err != nil {
        fmt.Println("错误:", err)
        return
    }
    defer file.Close()

    scanner := bufio.NewScanner(file)
    for scanner.Scan() {
        fmt.Println(scanner.Text())
    }

    if err := scanner.Err(); err != nil {
        fmt.Println("错误:", err)
    }
}
```

## 写入文件

### 写入文件

```go
package main

import (
    "fmt"
    "os"
)

func main() {
    // 写入文件（覆盖）
    err := os.WriteFile("output.txt", []byte("Hello, Go!"), 0644)
    if err != nil {
        fmt.Println("错误:", err)
        return
    }

    // 追加写入
    f, err := os.OpenFile("output.txt", os.O_APPEND|os.O_WRONLY, 0644)
    if err != nil {
        fmt.Println("错误:", err)
        return
    }
    defer f.Close()

    _, err = f.WriteString("\n追加内容")
    if err != nil {
        fmt.Println("错误:", err)
    }
}
```

### 使用 bufio 写入

```go
package main

import (
    "bufio"
    "fmt"
    "os"
)

func main() {
    file, err := os.Create("output.txt")
    if err != nil {
        fmt.Println("错误:", err)
        return
    }
    defer file.Close()

    writer := bufio.NewWriter(file)
    defer writer.Flush()

    for i := 1; i <= 5; i++ {
        fmt.Fprintf(writer, "行 %d\n", i)
    }
}
```

## 目录操作

### 创建目录

```go
package main

import (
    "fmt"
    "os"
)

func main() {
    // 创建单个目录
    err := os.Mkdir("dir", 0755)
    if err != nil {
        fmt.Println("错误:", err)
    }

    // 创建多级目录
    err = os.MkdirAll("path/to/dir", 0755)
    if err != nil {
        fmt.Println("错误:", err)
    }
}
```

### 读取目录

```go
package main

import (
    "fmt"
    "os"
)

func main() {
    entries, err := os.ReadDir(".")
    if err != nil {
        fmt.Println("错误:", err)
        return
    }

    for _, entry := range entries {
        if entry.IsDir() {
            fmt.Printf("[目录] %s\n", entry.Name())
        } else {
            fmt.Printf("[文件] %s\n", entry.Name())
        }
    }
}
```

## 文件路径

### path/filepath 包

```go
package main

import (
    "fmt"
    "path/filepath"
)

func main() {
    // Join
    path := filepath.Join("dir", "subdir", "file.txt")
    fmt.Println(path)

    // Base
    fmt.Println(filepath.Base(path))

    // Dir
    fmt.Println(filepath.Dir(path))

    // Ext
    fmt.Println(filepath.Ext(path))

    // Abs
    abs, _ := filepath.Abs(".")
    fmt.Println(abs)
}
```
