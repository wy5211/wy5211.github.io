## 前言

channel 是 Go 语言并发模型的核心，提供了一种安全的方式在 goroutine 之间传递数据。

---

## channel 基础

### 创建和使用 channel

```go
package main

import "fmt"

func main() {
    // 创建 channel
    ch := make(chan int)

    // 发送数据
    go func() {
        ch <- 42
    }()

    // 接收数据
    val := <-ch
    fmt.Println("接收到的值:", val)
}
```

### 缓冲 channel

```go
package main

import "fmt"

func main() {
    // 缓冲 channel（容量为 3）
    ch := make(chan int, 3)

    ch <- 1  // 不会阻塞
    ch <- 2
    ch <- 3

    fmt.Println(<-ch)
    fmt.Println(<-ch)
    fmt.Println(<-ch)
}
```

## channel 操作

### 关闭 channel

```go
package main

import "fmt"

func main() {
    ch := make(chan int, 2)

    ch <- 1
    ch <- 2
    close(ch)

    // 可以继续接收
    for val := range ch {
        fmt.Println(val)
    }

    // 检查是否关闭
    val, ok := <-ch
    fmt.Printf("值: %d, 关闭: %t\n", val, ok)
}
```

### 单向 channel

```go
package main

import "fmt"

// 只发送 channel
func sender(ch chan<- int) {
    ch <- 42
}

// 只接收 channel
func receiver(ch <-chan int) {
    val := <-ch
    fmt.Println("接收:", val)
}

func main() {
    ch := make(chan int)

    go sender(ch)
    receiver(ch)
}
```

## select 语句

### 多 channel 选择

```go
package main

import (
    "fmt"
    "time"
)

func main() {
    ch1 := make(chan string)
    ch2 := make(chan string)

    go func() {
        time.Sleep(100 * time.Millisecond)
        ch1 <- "one"
    }()

    go func() {
        time.Sleep(200 * time.Millisecond)
        ch2 <- "two"
    }()

    for i := 0; i < 2; i++ {
        select {
        case msg1 := <-ch1:
            fmt.Println("从 ch1 接收:", msg1)
        case msg2 := <-ch2:
            fmt.Println("从 ch2 接收:", msg2)
        }
    }
}
```

### 超时处理

```go
package main

import (
    "fmt"
    "time"
)

func main() {
    ch := make(chan string)

    go func() {
        time.Sleep(2 * time.Second)
        ch <- "结果"
    }()

    select {
    case result := <-ch:
        fmt.Println("接收到:", result)
    case <-time.After(1 * time.Second):
        fmt.Println("超时")
    }
}
```

## channel 模式

### 生产者-消费者模式

```go
package main

import "fmt"

func producer(ch chan<- int) {
    for i := 1; i <= 5; i++ {
        ch <- i
    }
    close(ch)
}

func consumer(ch <-chan int) {
    for val := range ch {
        fmt.Println("消费:", val)
    }
}

func main() {
    ch := make(chan int)

    go producer(ch)
    consumer(ch)
}
```

### Fan-Out 模式

```go
package main

import "fmt"

func worker(id int, jobs <-chan int, results chan<- int) {
    for job := range jobs {
        fmt.Printf("Worker %d 处理任务 %d\n", id, job)
        results <- job * 2
    }
}

func main() {
    jobs := make(chan int, 100)
    results := make(chan int, 100)

    // 启动多个 worker
    for w := 1; w <= 3; w++ {
        go worker(w, jobs, results)
    }

    // 发送任务
    for j := 1; j <= 5; j++ {
        jobs <- j
    }
    close(jobs)

    // 收集结果
    for r := 1; r <= 5; r++ {
        <-results
    }
}
```
