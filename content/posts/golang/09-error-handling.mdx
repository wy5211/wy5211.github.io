## 前言

Go 通过显式的错误返回值处理错误，而不是异常机制。这种方式让错误处理更清晰、更可控。

---

## 错误基础

### 基本错误处理

```go
package main

import (
    "errors"
    "fmt"
)

func divide(a, b float64) (float64, error) {
    if b == 0 {
        return 0, errors.New("除数不能为零")
    }
    return a / b, nil
}

func main() {
    result, err := divide(10, 0)
    if err != nil {
        fmt.Println("错误:", err)
        return
    }
    fmt.Println("结果:", result)
}
```

### fmt.Errorf

```go
package main

import "fmt"

func validateAge(age int) error {
    if age < 0 {
        return fmt.Errorf("年龄 %d 无效：不能为负数", age)
    }
    if age > 150 {
        return fmt.Errorf("年龄 %d 无效：不能超过150", age)
    }
    return nil
}

func main() {
    err := validateAge(-5)
    if err != nil {
        fmt.Println("验证失败:", err)
    }
}
```

## 自定义错误类型

### 实现错误接口

```go
package main

import "fmt"

// 自定义错误类型
type ValidationError struct {
    Field   string
    Message string
}

func (e *ValidationError) Error() string {
    return fmt.Sprintf("%s: %s", e.Field, e.Message)
}

func validateEmail(email string) error {
    if email == "" {
        return &ValidationError{
            Field:   "email",
            Message: "邮箱不能为空",
        }
    }
    return nil
}

func main() {
    err := validateEmail("")
    if err != nil {
        fmt.Println("错误:", err)

        // 类型断言
        if ve, ok := err.(*ValidationError); ok {
            fmt.Printf("字段: %s, 消息: %s\n", ve.Field, ve.Message)
        }
    }
}
```

## 错误包装

### Go 1.13+ 错误包装

```go
package main

import (
    "errors"
    "fmt"
)

func readFile() error {
    return errors.New("文件不存在")
}

func processData() error {
    err := readFile()
    if err != nil {
        return fmt.Errorf("处理数据失败: %w", err)
    }
    return nil
}

func main() {
    err := processData()
    if err != nil {
        fmt.Println("错误:", err)

        // 检查特定错误
        if errors.Is(err, errors.New("文件不存在")) {
            fmt.Println("这是文件不存在错误")
        }

        // 解包错误
        wrappedErr := errors.Unwrap(err)
        fmt.Println("原始错误:", wrappedErr)
    }
}
```

### errors.As

```go
package main

import (
    "errors"
    "fmt"
)

type MyError struct {
    Code int
    Msg  string
}

func (e *MyError) Error() string {
    return e.Msg
}

func doSomething() error {
    return &MyError{Code: 404, Msg: "资源未找到"}
}

func main() {
    err := doSomething()
    if err != nil {
        var myErr *MyError
        if errors.As(err, &myErr) {
            fmt.Printf("错误代码: %d, 消息: %s\n", myErr.Code, myErr.Msg)
        }
    }
}
```

## panic 和 recover

### 使用场景

```go
package main

import "fmt"

func divide(a, b int) int {
    if b == 0 {
        panic("除数不能为零")
    }
    return a / b
}

func safeDivide(a, b int) (result int) {
    defer func() {
        if r := recover(); r != nil {
            fmt.Println("捕获 panic:", r)
            result = 0
        }
    }()

    return divide(a, b)
}

func main() {
    result := safeDivide(10, 0)
    fmt.Println("结果:", result)
}
```

## 错误处理最佳实践

### 尽早返回

```go
// ✅ 推荐
func process(data string) (string, error) {
    if data == "" {
        return "", errors.New("数据不能为空")
    }

    result := processData(data)
    return result, nil
}

// ❌ 避免
func process(data string) (string, error) {
    var result string
    var err error

    if data == "" {
        err = errors.New("数据不能为空")
    } else {
        result = processData(data)
    }

    return result, err
}
```

### 包装错误时添加上下文

```go
// ✅ 推荐
func saveUser(user User) error {
    if err := db.Save(user); err != nil {
        return fmt.Errorf("保存用户 %s 失败: %w", user.ID, err)
    }
    return nil
}

// ❌ 避免
func saveUser(user User) error {
    if err := db.Save(user); err != nil {
        return err
    }
    return nil
}
```

## 实战示例

### 示例：HTTP 服务器错误处理

```go
package main

import (
    "encoding/json"
    "errors"
    "fmt"
    "net/http"
)

// 错误响应
type ErrorResponse struct {
    Error string `json:"error"`
}

func handleError(w http.ResponseWriter, err error, statusCode int) {
    w.Header().Set("Content-Type", "application/json")
    w.WriteHeader(statusCode)
    json.NewEncoder(w).Encode(ErrorResponse{Error: err.Error()})
}

func handler(w http.ResponseWriter, r *http.Request) {
    // 处理请求
    data, err := fetchData()
    if err != nil {
        handleError(w, err, http.StatusInternalServerError)
        return
    }

    // 返回成功响应
    json.NewEncoder(w).Encode(data)
}

func fetchData() (interface{}, error) {
    return nil, errors.New("数据库连接失败")
}

func main() {
    http.HandleFunc("/", handler)
    fmt.Println("服务器启动在 :8080")
    http.ListenAndServe(":8080", nil)
}
```
