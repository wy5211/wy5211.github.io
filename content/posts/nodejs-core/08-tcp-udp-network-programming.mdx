## 前言

在 Node.js 后端开发中，**网络编程**是核心技能。虽然 HTTP 是应用层协议，但理解底层的 TCP/UDP 对于构建高性能应用、实现自定义协议至关重要。

**核心内容**：

- TCP 三次握手与四次挥手
- net.Socket 事件与状态管理
- UDP 数据报的无连接特性
- 实现自定义协议
- 实战：TCP 聊天服务器、UDP 广播

## 一、TCP 编程

### 1.1 TCP 服务器

```javascript
const net = require("net");

const server = net.createServer((socket) => {
  console.log("Client connected");

  socket.on("data", (data) => {
    console.log("Received:", data.toString());
    socket.write("Echo: " + data);
  });

  socket.on("end", () => {
    console.log("Client disconnected");
  });

  socket.on("error", (err) => {
    console.error("Error:", err);
  });
});

server.listen(8080, () => {
  console.log("Server listening on port 8080");
});
```

### 1.2 TCP 客户端

```javascript
const net = require("net");

const client = net.createConnection({ port: 8080 }, () => {
  console.log("Connected to server");
  client.write("Hello Server");
});

client.on("data", (data) => {
  console.log("Received:", data.toString());
  client.end();
});

client.on("end", () => {
  console.log("Disconnected from server");
});
```

### 1.3 聊天服务器

```javascript
const net = require("net");

const clients = [];

const server = net.createServer((socket) => {
  const clientId = clients.length + 1;
  console.log(`Client ${clientId} connected`);

  clients.push(socket);

  socket.write(`Welcome! You are client ${clientId}\n`);

  socket.on("data", (data) => {
    const message = `Client ${clientId}: ${data}`;

    clients.forEach((client) => {
      if (client !== socket) {
        client.write(message);
      }
    });
  });

  socket.on("end", () => {
    console.log(`Client ${clientId} disconnected`);
    const index = clients.indexOf(socket);
    if (index !== -1) {
      clients.splice(index, 1);
    }
  });
});

server.listen(8080);
```

## 二、UDP 编程

### 2.1 UDP 服务器

```javascript
const dgram = require("dgram");

const server = dgram.createSocket("udp4");

server.on("message", (msg, rinfo) => {
  console.log(`Received: ${msg} from ${rinfo.address}:${rinfo.port}`);

  const response = Buffer.from("Echo: " + msg);
  server.send(response, rinfo.port, rinfo.address);
});

server.on("listening", () => {
  const address = server.address();
  console.log(`Server listening ${address.address}:${address.port}`);
});

server.bind(8080);
```

### 2.2 UDP 客户端

```javascript
const dgram = require("dgram");

const client = dgram.createSocket("udp4");

const message = Buffer.from("Hello UDP Server");

client.send(message, 8080, "localhost", (err) => {
  if (err) {
    console.error("Error:", err);
    client.close();
  }
});

client.on("message", (msg, rinfo) => {
  console.log("Received:", msg.toString());
  client.close();
});
```

### 2.3 UDP 广播

```javascript
// 广播服务器
const dgram = require("dgram");

const server = dgram.createSocket("udp4");

server.bind(8080, () => {
  server.setBroadcast(true);

  setInterval(() => {
    const message = Buffer.from("Broadcast message");
    server.send(message, 8081, "255.255.255.255");
  }, 1000);
});
```

```javascript
// 广播客户端
const dgram = require("dgram");

const client = dgram.createSocket("udp4");

client.bind(8081);

client.on("message", (msg, rinfo) => {
  console.log("Received broadcast:", msg.toString());
});
```

## 三、自定义协议

### 3.1 协议设计

```
[4 bytes: length][2 bytes: type][n bytes: data]
```

### 3.2 协议实现

```javascript
const net = require("net");

class Protocol {
  static encode(type, data) {
    const dataBuffer = Buffer.from(JSON.stringify(data));
    const typeBuffer = Buffer.alloc(2);
    typeBuffer.writeUInt16BE(type);

    const lengthBuffer = Buffer.alloc(4);
    lengthBuffer.writeUInt32BE(dataBuffer.length);

    return Buffer.concat([lengthBuffer, typeBuffer, dataBuffer]);
  }

  static decode(buffer) {
    if (buffer.length < 6) return null;

    const length = buffer.readUInt32BE(0);
    const type = buffer.readUInt16BE(4);

    if (buffer.length < 6 + length) return null;

    const data = JSON.parse(buffer.slice(6, 6 + length).toString());

    return { type, data };
  }
}

const server = net.createServer((socket) => {
  let buffer = Buffer.alloc(0);

  socket.on("data", (data) => {
    buffer = Buffer.concat([buffer, data]);

    while (true) {
      const message = Protocol.decode(buffer);

      if (!message) break;

      console.log("Received:", message);

      buffer = buffer.slice(6 + message.data.length);
    }
  });
});

server.listen(8080);
```
