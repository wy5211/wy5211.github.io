## 前言

在 Node.js 后端开发中，**加密与安全**是必备技能。crypto 模块提供了完整的密码学功能，包括哈希、对称加密、非对称加密、签名验证等。

**核心内容**：

- 哈希函数：md5、sha1、sha256
- 对称加密：aes-256-cbc
- 非对称加密：rsa、ecc
- 签名与验证：HMAC、sign/verify
- 实战：密码存储、JWT 签名、数据加密

## 一、哈希函数

### 1.1 基础哈希

```javascript
const crypto = require("crypto");

// MD5（不推荐，不安全）
const md5 = crypto.createHash("md5").update("hello").digest("hex");
console.log("MD5:", md5);

// SHA-1（不推荐）
const sha1 = crypto.createHash("sha1").update("hello").digest("hex");
console.log("SHA-1:", sha1);

// SHA-256（推荐）
const sha256 = crypto.createHash("sha256").update("hello").digest("hex");
console.log("SHA-256:", sha256);

// SHA-512
const sha512 = crypto.createHash("sha512").update("hello").digest("hex");
console.log("SHA-512:", sha512);
```

### 1.2 文件哈希

```javascript
const crypto = require("crypto");
const fs = require("fs");

function getFileHash(filePath, algorithm = "sha256") {
  return new Promise((resolve, reject) => {
    const hash = crypto.createHash(algorithm);
    const stream = fs.createReadStream(filePath);

    stream.on("data", (data) => hash.update(data));
    stream.on("end", () => resolve(hash.digest("hex")));
    stream.on("error", reject);
  });
}

getFileHash("file.txt").then((hash) => {
  console.log("File hash:", hash);
});
```

## 二、对称加密

### 2.1 AES 加密

```javascript
const crypto = require("crypto");

const algorithm = "aes-256-cbc";
const key = crypto.randomBytes(32); // 256 位密钥
const iv = crypto.randomBytes(16); // 初始化向量

function encrypt(text) {
  const cipher = crypto.createCipheriv(algorithm, key, iv);
  let encrypted = cipher.update(text, "utf8", "hex");
  encrypted += cipher.final("hex");
  return encrypted;
}

function decrypt(encrypted) {
  const decipher = crypto.createDecipheriv(algorithm, key, iv);
  let decrypted = decipher.update(encrypted, "hex", "utf8");
  decrypted += decipher.final("utf8");
  return decrypted;
}

const plaintext = "Hello World";
const encrypted = encrypt(plaintext);
const decrypted = decrypt(encrypted);

console.log("Plaintext:", plaintext);
console.log("Encrypted:", encrypted);
console.log("Decrypted:", decrypted);
```

### 2.2 HMAC 签名

```javascript
const crypto = require("crypto");

const secret = "my-secret-key";
const data = "important data";

// 生成 HMAC
const hmac = crypto.createHmac("sha256", secret).update(data).digest("hex");

console.log("HMAC:", hmac);

// 验证 HMAC
const hmac2 = crypto.createHmac("sha256", secret).update(data).digest("hex");

console.log("Verified:", hmac === hmac2);
```

## 三、非对称加密

### 3.1 生成密钥对

```javascript
const crypto = require("crypto");

// 生成 RSA 密钥对
const { publicKey, privateKey } = crypto.generateKeyPairSync("rsa", {
  modulusLength: 2048,
});

console.log("Public Key:", publicKey.export({ type: "spki", format: "pem" }));
console.log(
  "Private Key:",
  privateKey.export({ type: "pkcs8", format: "pem" })
);
```

### 3.2 加密与解密

```javascript
const crypto = require("crypto");

const data = "sensitive data";

// 公钥加密
const encrypted = crypto.publicEncrypt(publicKey, Buffer.from(data));

console.log("Encrypted:", encrypted.toString("base64"));

// 私钥解密
const decrypted = crypto.privateDecrypt(privateKey, encrypted);

console.log("Decrypted:", decrypted.toString());
```

### 3.3 签名与验证

```javascript
const crypto = require("crypto");

const data = "important message";

// 私钥签名
const sign = crypto.createSign("sha256");
sign.update(data);
const signature = sign.sign(privateKey, "hex");

console.log("Signature:", signature);

// 公钥验证
const verify = crypto.createVerify("sha256");
verify.update(data);
const verified = verify.verify(publicKey, signature, "hex");

console.log("Verified:", verified);
```

## 四、实战案例

### 4.1 密码存储

```javascript
const crypto = require("crypto");

// 使用 PBKDF2
function hashPassword(password, salt) {
  return new Promise((resolve, reject) => {
    crypto.pbkdf2(password, salt, 100000, 64, "sha512", (err, derivedKey) => {
      if (err) reject(err);
      else resolve(derivedKey.toString("hex"));
    });
  });
}

async function register(username, password) {
  const salt = crypto.randomBytes(16).toString("hex");
  const hash = await hashPassword(password, salt);

  // 存储 { username, hash, salt }
  return { username, hash, salt };
}

async function login(username, password, storedHash, storedSalt) {
  const hash = await hashPassword(password, storedSalt);
  return hash === storedHash;
}
```

### 4.2 JWT 签名

```javascript
const crypto = require("crypto");

function signJWT(payload, secret) {
  const header = { alg: "HS256", typ: "JWT" };
  const encodedHeader = Buffer.from(JSON.stringify(header)).toString(
    "base64url"
  );
  const encodedPayload = Buffer.from(JSON.stringify(payload)).toString(
    "base64url"
  );

  const signature = crypto
    .createHmac("sha256", secret)
    .update(`${encodedHeader}.${encodedPayload}`)
    .digest("base64url");

  return `${encodedHeader}.${encodedPayload}.${signature}`;
}

function verifyJWT(token, secret) {
  const [encodedHeader, encodedPayload, signature] = token.split(".");

  const expectedSignature = crypto
    .createHmac("sha256", secret)
    .update(`${encodedHeader}.${encodedPayload}`)
    .digest("base64url");

  if (signature !== expectedSignature) {
    throw new Error("Invalid signature");
  }

  const payload = JSON.parse(
    Buffer.from(encodedPayload, "base64url").toString()
  );
  return payload;
}

const payload = { userId: 123, exp: Date.now() + 3600000 };
const secret = "my-secret";

const token = signJWT(payload, secret);
console.log("Token:", token);

const decoded = verifyJWT(token, secret);
console.log("Decoded:", decoded);
```
