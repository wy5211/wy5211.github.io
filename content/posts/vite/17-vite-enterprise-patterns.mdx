## 前言

企业级项目需要更高的可维护性、可扩展性和稳定性。本文将探讨 Vite 在企业级项目中的实践模式。

## CI/CD 集成

### GitHub Actions

```yaml
# .github/workflows/ci.yml
name: CI

on:
  push:
    branches: [main, develop]
  pull_request:

jobs:
  test:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v3

      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: "20"
          cache: "pnpm"

      - name: Install dependencies
        run: pnpm install

      - name: Lint
        run: pnpm run lint

      - name: Type check
        run: pnpm run type-check

      - name: Test
        run: pnpm run test

      - name: Build
        run: pnpm run build

      - name: Upload artifacts
        uses: actions/upload-artifact@v3
        with:
          name: dist
          path: apps/*/dist
```

### 部署流程

```yaml
# .github/workflows/deploy.yml
name: Deploy

on:
  push:
    tags:
      - "v*"

jobs:
  deploy:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v3

      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: "20"
          cache: "pnpm"

      - name: Install dependencies
        run: pnpm install

      - name: Build
        run: pnpm run build

      - name: Deploy to Vercel
        uses: amondnet/vercel-action@v25
        with:
          vercel-token: ${{ secrets.VERCEL_TOKEN }}
          vercel-org-id: ${{ secrets.ORG_ID }}
          vercel-project-id: ${{ secrets.PROJECT_ID }}
          vercel-args: "--prod"
```

### 性能监控

```typescript
// src/monitoring/performance.ts
import { onCLS, onFID, onFCP, onLCP, onTTFB } from "web-vitals";

export function reportWebVitals(metric) {
  const { name, value, id } = metric;

  // 上报到监控平台
  fetch("/api/analytics", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ name, value, id }),
  });
}

// 使用
onCLS(reportWebVitals);
onFID(reportWebVitals);
onFCP(reportWebVitals);
onLCP(reportWebVitals);
onTTFB(reportWebVitals);
```

### 自定义监控

```typescript
// src/monitoring/custom.ts
export class Monitoring {
  static logError(error: Error, context?: any) {
    console.error("[Monitoring]", error, context);

    // 上报
    fetch("/api/error", {
      method: "POST",
      body: JSON.stringify({
        message: error.message,
        stack: error.stack,
        context,
      }),
    });
  }

  static logEvent(name: string, data?: any) {
    console.log("[Event]", name, data);

    fetch("/api/event", {
      method: "POST",
      body: JSON.stringify({ name, data }),
    });
  }
}
```

## 配置管理

### 环境配置中心

```typescript
// config/env.config.ts
interface EnvConfig {
  apiUrl: string;
  sentryDsn: string;
  features: {
    newUI: boolean;
    beta: boolean;
  };
}

export const loadConfig = (): EnvConfig => {
  const env = import.meta.env;

  return {
    apiUrl: env.VITE_API_URL,
    sentryDsn: env.VITE_SENTRY_DSN,
    features: {
      newUI: env.VITE_ENABLE_NEW_UI === "true",
      beta: env.VITE_ENABLE_BETA === "true",
    },
  };
};

export const config = loadConfig();
```

### 远程配置

```typescript
// config/remote-config.ts
export async function loadRemoteConfig() {
  try {
    const response = await fetch("/api/config");
    const remoteConfig = await response.json();

    return {
      ...config,
      ...remoteConfig,
    };
  } catch (error) {
    console.error("Failed to load remote config:", error);
    return config;
  }
}
```

### 安全检查

```typescript
// scripts/security-check.ts
export function checkSecurity() {
  const issues: string[] = [];

  // 检查敏感信息
  if (process.env.API_KEY) {
    issues.push("API key exposed in process.env");
  }

  // 检查依赖漏洞
  const deps = require("./package.json");
  Object.keys(devDependencies).forEach((dep) => {
    if (dep.includes("insecure")) {
      issues.push(`Insecure dependency: ${dep}`);
    }
  });

  return issues;
}
```

## 常见问题

### Q1: Monorepo 构建慢？

A: 使用 Turborepo 或 Nx：

```bash
# 安装
npm install -D turbo

# 配置
turbo run build --parallel
```

### Q2: 团队配置不一致？

A: 使用统一的配置包：

```bash
# 创建配置包
mkdir packages/config

# 使用
npm install -D @my-org/config
```

### Q3: 如何规范 Git 提交？

A: 使用 Commitlint：

```bash
npm install -D @commitlint/cli @commitlint/config-conventional

echo "export default { extends: ['@commitlint/config-conventional'] };" > commitlint.config.js
```

5. **配置管理**：环境变量 + 远程配置
6. **团队协作**：规范文档 + 审查清单
