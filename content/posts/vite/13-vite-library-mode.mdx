## 前言

Vite 不仅是应用构建工具，也是库开发的利器。本文将深入探讨如何使用 Vite 库模式开发高质量的开源库。

## 库模式配置

### 基础配置

```typescript
// vite.config.ts
import { defineConfig } from "vite";
import dts from "vite-plugin-dts";

export default defineConfig({
  build: {
    lib: {
      // 入口文件
      entry: "src/index.ts",

      // 输出格式
      name: "MyLibrary",
      fileName: (format) => `my-library.${format}.js`,

      // 输出格式
      formats: ["es", "cjs", "umd"],
    },

    rollupOptions: {
      // 外部化依赖
      external: ["react", "react-dom"],

      output: {
        // 全局变量
        globals: {
          react: "React",
          "react-dom": "ReactDOM",
        },
      },
    },
  },

  plugins: [
    // 生成类型声明
    dts({
      rollupTypes: true,
    }),
  ],
});
```

### 多入口配置

```typescript
export default defineConfig({
  build: {
    lib: {
      entry: {
        index: "src/index.ts",
        hooks: "src/hooks/index.ts",
        utils: "src/utils/index.ts",
      },
      name: "MyLibrary",
    },
  },
});
```

## 类型声明

### 生成类型

```bash
# 安装插件
npm install -D vite-plugin-dts

# 配置
import dts from 'vite-plugin-dts';

export default defineConfig({
  plugins: [
    dts({
      // 包含文件
      include: ['src'],

      // 排除文件
      exclude: ['**/*.test.ts', '**/*.spec.ts'],

      // Rollup 类型
      rollupTypes: true,

      // 输出目录
      outDir: 'dist'
    })
  ]
});
```

### 类型声明文件

```typescript
// src/index.ts
export function useFeature(): boolean;
export const version: string;

export interface Config {
  debug?: boolean;
  timeout?: number;
}

export class MyClass {
  constructor(config: Config);
  method(): void;
}
```

### 手动类型声明

```typescript
// dist/index.d.ts
export declare function useFeature(): boolean;
export declare const version: string;

export interface Config {
  debug?: boolean;
  timeout?: number;
}

export declare class MyClass {
  constructor(config: Config);
  method(): void;
}
```

## 依赖管理

### peerDependencies

```json
{
  "peerDependencies": {
    "react": "^18.0.0",
    "react-dom": "^18.0.0"
  },
  "peerDependenciesMeta": {
    "react": {
      "optional": false
    }
  }
}
```

### external 配置

```typescript
export default defineConfig({
  build: {
    rollupOptions: {
      external: [
        "react",
        "react-dom",
        /^node:.*/, // Node.js 内置模块
      ],
    },
  },
});
```

## 常见问题

### Q1: 如何处理 React 依赖？

A: 使用 peerDependencies：

```json
{
  "peerDependencies": {
    "react": "^18.0.0"
  }
}
```

### Q2: 如何支持 Tree Shaking？

A: 使用 ESM 格式和 sideEffects: false：

```json
{
  "sideEffects": false,
  "module": "./dist/index.mjs"
}
```

### Q3: 类型声明不正确？

A: 使用 vite-plugin-dts 生成类型：

```typescript
import dts from "vite-plugin-dts";
```
