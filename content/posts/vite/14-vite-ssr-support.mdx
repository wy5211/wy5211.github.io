## 前言

服务端渲染（SSR）对 SEO 和首屏加载至关重要。Vite 对 SSR 提供了优秀的支持。本文将深入探讨 Vite 的 SSR 实现方案。

## SSR 基础

### 什么是 SSR

```
CSR（客户端渲染）
浏览器 → 下载 HTML → 加载 JS → 渲染

SSR（服务端渲染）
服务器 → 渲染 HTML → 返回 HTML → 浏览器直接显示
```

### SSR 优势

- **SEO 优化**：搜索引擎可以抓取完整内容
- **首屏速度**：减少首屏加载时间
- **社交分享**：分享卡片显示完整内容

## Vite SSR 配置

```typescript
// vite.config.ts
import { defineConfig } from "vite";
import react from "@vitejs/plugin-react";

export default defineConfig({
```

## 开发服务器

### SSR 开发服务器

```typescript
// server-dev.ts
import fs from "fs";
import path from "path";
import { createServer } from "vite";

async function createDevServer() {
  const vite = await createServer({
    server: { middlewareMode: true },
    appType: "custom",
  });

  const template = fs.readFileSync(
    path.resolve(__dirname, "index.html"),
    "utf-8"
  );

  return { vite, template };
}

export { createDevServer };
```

### 开发服务器实现

```typescript
// server.ts
import express from "express";
import { createDevServer } from "./server-dev";

const app = express();

app.use("*", async (req, res) => {
  const { vite, template } = await createDevServer();

  const { render } = await vite.ssrLoadModule("/src/entry-server.tsx");

  const appHtml = await render(req.url);

  const html = template.replace(`<!--app-html-->`, appHtml);

  res.status(200).set({ "Content-Type": "text/html" }).end(html);
});

app.listen(3000, () => {
  console.log("Server running at http://localhost:3000");
});
```

### 构建脚本

```typescript

import fs from "fs";
import path from "path";
import { build } from "vite";

async function buildSSR() {
  // 客户端构建
  await build({
    build: {
      rollupOptions: {
        input: {
          client: "src/entry-client.tsx",
        },
        output: {
          entryFileNames: "[name].js",
          chunkFileNames: "[name].js",
        },
      },
    },
  });

  // 服务端构建
  await build({
    ssr: {
      noExternal: true,
    },
    build: {
      rollupOptions: {
        input: "src/entry-server.tsx",
        output: {
          entryFileNames: "server.js",
          format: "cjs",
        },
      },
    },
  });
}

buildSSR();
```

## 数据预取

### 数据预取模式

```typescript
// src/entry-server.tsx
import { renderToString } from 'react-dom/server';
import { StaticRouter } from 'react-router-dom/server';
import App, { loadData } from './App';

export async function render(url: string) {
  // 预取数据
  const data = await loadData(url);

  // 渲染应用
  const html = renderToString(
    <StaticRouter location={url}>
      <App data={data} />
    </StaticRouter>
  );

  return { html, data };
}
```

### 路由级数据加载

```typescript
// src/routes.tsx
export const routes = [
  {
    path: "/",
    component: Home,
    loader: async () => {
      return fetch("/api/home").then((r) => r.json());
    },
  },
  {
    path: "/about",
    component: About,
    loader: async () => {
      return fetch("/api/about").then((r) => r.json());
    },
  },
];

// 数据预取
export async function loadData(url: string) {
  const route = routes.find((r) => r.path === url);

  if (route?.loader) {
    return await route.loader();
  }

  return {};
}
```

### 客户端数据同步

```typescript
// src/entry-client.tsx
import { hydrateRoot } from 'react-dom/client';
import { BrowserRouter } from 'react-router-dom';
import App from './App';

// 从 window 获取预取的数据

hydrateRoot(
  document.getElementById('root')!,
  <BrowserRouter>
    <App preloadedData={preloadedData} />
  </BrowserRouter>

```

## Streaming SSR

### 基础配置

```typescript
// src/entry-server.tsx
import { renderToPipeableStream } from 'react-dom/server';
import { StaticRouter } from 'react-router-dom/server';
import App from './App';

export function renderStream(url: string, res: any) {
  const stream = renderToPipeableStream(
    <StaticRouter location={url}>
      <App />
    </StaticRouter>,
    {
      // 所有内容准备好后注入 shell
      onShellReady() {
        res.statusCode = 200;
        res.setHeader('Content-type', 'text/html');
        stream.pipe(res);
      },

      // 错误处理
      onShellError(error) {
        res.statusCode = 500;
        res.send('<!doctype html><p>Error loading app...</p>');
      }
    }
  );
}
```

## 常见问题

### Q1: SSR 和客户端状态不一致？

A: 确保数据预取和同步：

```typescript
// 服务端渲染时传递数据
const html = renderToString(<App data={data} />);

// 客户端 hydration 时使用相同数据
hydrateRoot(document, <App data={preloadedData} />);
```

### Q2: 如何处理路由？

A: 使用 StaticRouter（SSR）和 BrowserRouter（客户端）：

```typescript
// SSR: StaticRouter
<StaticRouter location={url}>
  <App />
</StaticRouter>

// 客户端: BrowserRouter
<BrowserRouter>
  <App />
</BrowserRouter>
```

### Q3: 优化 SSR 性能？

A: 使用 Streaming 和缓存：

```typescript
// Streaming SSR
renderToPipeableStream(app, {
  onShellReady() {
    stream.pipe(res);
  },
});
```
