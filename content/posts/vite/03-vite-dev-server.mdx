## 前言

Vite 开发服务器是其最强大的特性之一。本文将深入探讨开发服务器的内部机制、高级配置和调试技巧，帮助你充分利用 Vite 的开发体验。

## 开发服务器架构

### 核心组件

```
Vite Dev Server
    ↓
┌─────────────────────────────────┐
│  Connect 中间件系统              │
│  ├─ HMR 中间件                  │
│  ├─ 转换中间件                  │
│  ├─ 静态资源中间件              │
│  ├─ 代理中间件                  │
│  └─ 索引 HTML 中间件            │
└─────────────────────────────────┘
    ↓
┌─────────────────────────────────┐
│  插件系统                        │
│  ├─ transform 钩子              │
│  ├─ configureServer 钩子        │
│  └─ handleHotUpdate 钩子        │
└─────────────────────────────────┘
```

### 请求处理流程

```
浏览器请求 /src/App.tsx
    ↓
Vite 接收请求
    ↓
判断请求类型
    ├─ /@fs/...      → 文件系统请求
    ├─ /@vite/...    → Vite 内部模块
    ├─ /node_modules/ → 预构建依赖
    └─ 源码文件       → 按需编译
    ↓
esbuild 转换（如果是 TS/JSX）
    ↓
返回 ESM 格式代码
    ↓
浏览器执行并建立 HMR 连接
```

## 服务器配置详解

### 基础配置

```typescript
// vite.config.ts
export default defineConfig({
  server: {
    // 监听地址
    host: "0.0.0.0", // 监听所有地址
    port: 3000, // 端口号
    strictPort: false, // 端口被占用时自动尝试下一个

    // 自动打开浏览器
    open: true,
    open: "/docs", // 打开指定路径

    // CORS
    cors: true,

    // HTTPS
    https: false,

    // 监听文件变化
    watch: {
      usePolling: false, // 轮询（WSL/docker 环境）
      interval: 100, // 轮询间隔
      ignored: ["**/node_modules/**", "**/.git/**"],
    },
  },
});
```

### 代理中间件

```typescript
import { defineConfig } from "vite";
import proxy from "http-proxy-middleware";

export default defineConfig({
  server: {
    proxy: {
      "/api": {
        target: "http://localhost:8080",
        configure: (proxy, options) => {
          // 自定义代理处理
          proxy.on("proxyReq", (proxyReq, req, res) => {
            console.log("[Proxy]", req.method, req.url);

            // 添加请求头
            proxyReq.setHeader("X-Custom-Header", "value");
          });

          proxy.on("proxyRes", (proxyRes, req, res) => {
            console.log("[Proxy Response]", proxyRes.statusCode);

            // 修改响应头
            proxyRes.headers["X-Added-Header"] = "value";
          });

          proxy.on("error", (err, req, res) => {
            console.error("[Proxy Error]", err);

            res.statusCode = 500;
            res.end("Proxy Error");
          });
        },
      },
    },
  },
});
```

### Mock 数据集成

```typescript
// vite.config.ts
export default defineConfig({
  server: {
    proxy: {
      "/api": {
        target: "http://localhost:8080",
        changeOrigin: true,

        // 开发环境使用 Mock
        bypass: (req, res, options) => {
          if (process.env.MOCK === "true") {
            const mock = getMockData(req.url);

            if (mock) {
              res.setHeader("Content-Type", "application/json");
              res.end(JSON.stringify(mock));
              return false; // 不转发到真实服务器
            }
          }
        },
      },
    },
  },
});

// mock 数据
function getMockData(url: string) {
  const mocks: Record<string, any> = {
    "/api/users": [{ id: 1, name: "Alice" }],
    "/api/posts": [{ id: 1, title: "Hello" }],
  };

  return mocks[url];
}
```

## HMR 深入

### HMR API

```typescript
// src/main.tsx
if (import.meta.hot) {
  // 接受当前模块的更新
  import.meta.hot.accept((newModule) => {
    console.log("Module updated:", newModule);
  });

  // 接受依赖模块的更新
  import.meta.hot.accept("./helpers", (newHelpers) => {
    console.log("Helpers updated:", newHelpers);
  });

  // 销毁回调
  import.meta.hot.dispose((data) => {
    // 清理副作用
    console.log("Disposing module");
  });

  // 发生错误时的回调
  import.meta.hot.accept((err) => {
    console.error("HMR error:", err);

    // 可以选择刷新页面
    window.location.reload();
  });
}
```

### React Fast Refresh

```typescript
// vite.config.ts
export default defineConfig({
  plugins: [
    react({
      // 启用 Fast Refresh
      babel: {
        plugins: ["react-refresh/babel"],
      },
    }),
  ],
});
```

### 组件级 HMR

```typescript
// src/components/MyComponent.tsx
import { useEffect, useState } from "react";

if (import.meta.hot) {
  // 保存组件状态
  const savedState = new WeakMap();

  import.meta.hot.accept((newModule) => {
    // 恢复状态
    console.log("Component updated with state preserved");
  });

  import.meta.hot.dispose((data) => {
    // 保存状态到 data
    // 可以在 accept 中恢复
  });
}
```

### 自定义 HMR 处理

```typescript
// vite.config.ts
export default defineConfig({
  plugins: [
    {
      name: "custom-hmr",
      handleHotUpdate({ file, server, modules }) {
        // 自定义 HMR 处理逻辑

        if (file.endsWith(".custom")) {
          server.ws.send({
            type: "custom",
            event: "custom-update",
            data: { file },
          });

          // 返回空数组阻止默认 HMR
          return [];
        }

        // 返回模块列表让 Vite 处理
        return modules;
      },
    },
  ],
});
```

### HMR 错误处理

```typescript
// vite.config.ts
export default defineConfig({
  server: {
    hmr: {
      // HMR 覆盖层
      overlay: true,
    },
  },
});

// 自定义错误处理
if (import.meta.hot) {
  import.meta.hot.on("vite:error", (err) => {
    console.error("HMR Error:", err);

    // 上报错误
    reportError(err);
  });
}
```

## 调试技巧

### 服务器日志

```typescript
// vite.config.ts
export default defineConfig({
  plugins: [
    {
      name: "debug-plugin",
      configureServer(server) {
        // 服务器启动时的日志
        server.httpServer?.once("listening", () => {
          console.log("Server started!");

          // 获取实际监听的端口
          const address = server.httpServer?.address();
          console.log("Address:", address);
        });

        // 请求日志
        server.middlewares.use((req, res, next) => {
          console.log(`[${req.method}] ${req.url}`);
          next();
        });
      },
    },
  ],
});
```

### Vite 调试模式

```bash
# 启用 Vite 调试日志
DEBUG=vite:* npm run dev

# 只查看特定模块的日志
DEBUG=vite:deps npm run dev
DEBUG=vite:hmr npm run dev
```

### 插件调试

```typescript
// vite.config.ts
export default defineConfig({
  plugins: [
    {
      name: "debug-plugin",
      configureServer(server) {
        // 打印配置
        console.log("Vite config:", server.config);

        // 打印插件列表
        console.log("Plugins:", server.config.plugins);
      },

      transform(code, id) {
        // 打印转换信息
        if (id.includes("debug-me")) {
          console.log("Transforming:", id);
          console.log("Code:", code);
        }

        return null; // 不修改代码
      },
    },
  ],
});
```

## 常见问题

### Q1: HMR 不工作？

A: 检查以下几点：

1. 确保浏览器支持 WebSocket
2. 检查防火墙是否阻止了 WebSocket
3. 查看 HMR 日志：`DEBUG=vite:hmr npm run dev`

### Q2: 代理 404？

A: 检查代理配置：

```typescript
proxy: {
  '/api': {
    target: 'http://localhost:8080',
    changeOrigin: true,  // 必须设置
    rewrite: (path) => path.replace(/^\/api/, '')
  }
}
```

### Q3: 如何在 Docker 中使用？

A: 使用轮询模式：

```typescript
server: {
  watch: {
    usePolling: true,
    interval: 100
  },
  host: '0.0.0.0'
}
```
