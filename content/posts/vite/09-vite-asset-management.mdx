## 前言

静态资源管理是前端工程化的重要组成部分。Vite 提供了灵活的静态资源处理机制。本文将深入探讨 Vite 的资源管理策略。

## 资源导入方式

### 基础导入

```typescript
// 方式 1：导入 URL
import imgUrl from "./image.png";
document.getElementById("hero").src = imgUrl;

// 方式 2：显式 URL 导入
import imgUrl2 from "./image.png?url";

// 方式 3：导入原始内容
import imgContent from "./image.png?raw";

// 方式 4：导入为 Worker
import Worker from "./worker.js?worker";

// 方式 5：URL 方式（推荐）
const url = new URL("./image.png", import.meta.url).href;
img.src = url;
```

### CSS 中使用

```css
.logo {
  background: url("./logo.png");
}

/* 或使用别名 */
.logo {
  background: url("@/assets/logo.png");
}
```

### 基础配置

```typescript
export default defineConfig({
```

### 图片压缩插件

```typescript
import viteImagemin from "vite-plugin-imagemin";

export default defineConfig({
  plugins: [
    viteImagemin({
```

### 响应式图片

```typescript
// src/utils/image.ts
export function getResponsiveImage(
  name: string,
  size: 'small' | 'medium' | 'large'
) {
  const sizes = {
    small: 400,
    medium: 800,
    large: 1200
  };

  return new URL(
    `./images/${name}-${sizes[size]}.jpg`,
    import.meta.url
  ).href;
}

// 使用
const imageUrl = getResponsiveImage('hero', 'large');
<img src={imageUrl} alt="Hero" />
```

### 图片懒加载

```typescript
// src/hooks/useLazyImage.ts
export function useLazyImage(src: string) {
  const [imageSrc, setImageSrc] = useState<string>('');

  useEffect(() => {
    const img = new Image();

    img.onload = () => {
      setImageSrc(src);
    };

    img.src = src;
  }, [src]);

  return imageSrc;
}

// 使用
function LazyImage({ src, alt }: { src: string; alt: string }) {
  const imageSrc = useLazyImage(src);

  return (
    <img
      src={imageSrc}
      alt={alt}
      loading="lazy"
      style={{ opacity: imageSrc ? 1 : 0, transition: 'opacity 0.3s' }}
    />
  );
}
```

## 字体管理

### 字体配置

```typescript
export default defineConfig({
  build: {
    rollupOptions: {
      output: {
        assetFileNames: (assetInfo) => {
          // 字体单独处理
          if (/\.(woff2?|eot|ttf|otf)$/.test(assetInfo.name || "")) {
            return "assets/fonts/[name]-[hash][extname]";
          }
          return "assets/[ext]/[name]-[hash][extname]";
        },
      },
    },
  },
});
```

### 字体加载

```css
/* global.css */
@font-face {
  font-family: "Custom Font";
  src:
    url("@/assets/fonts/custom.woff2") format("woff2"),
    url("@/assets/fonts/custom.woff") format("woff");
  font-weight: normal;
  font-style: normal;
  font-display: swap; /* 优化加载 */
}

body {
  font-family: "Custom Font", sans-serif;
}
```

## SVG 处理

### SVG 导入

```typescript
// 方式 1：导入为 URL
import logoUrl from './logo.svg?url';
<img src={logoUrl} alt="Logo" />

// 方式 2：导入为字符串
import logoString from './logo.svg?raw';
<div dangerouslySetInnerHTML={{ __html: logoString }} />

// 方式 3：使用插件（推荐）
import vitePluginSvgr from 'vite-plugin-svgr';

export default defineConfig({
  plugins: [
    vitePluginSvgr({
      svgrOptions: {
        icon: true,
        replaceAttrValues: { '#000': 'currentColor' }
      }
    })
  ]
});

// 使用
import { ReactComponent as Logo } from './logo.svg';
<Logo />
```

### SVG 优化

```bash
# 安装 SVGO
npm install -D svgo

# 优化 SVG
svgo input.svg -o output.svg
```

```typescript
// vite.config.ts
import { optimize } from "svgo";

export default defineConfig({
  plugins: [
    {
      name: "svgo-plugin",
      transform(code, id) {
        if (!id.endsWith(".svg")) return null;

        const result = optimize(code, {
          plugins: [
            { name: "removeViewBox", active: false },
            { name: "removeEmptyAttrs", active: false },
          ],
        });

        return {
          code: result.data,
          map: null,
        };
      },
    },
  ],
});
```

### 动态 CDN

```typescript
// src/utils/cdn.ts
export function getCDNUrl(path: string) {
  const cdnBase = import.meta.env.VITE_CDN_URL || "";

  return cdnBase ? `${cdnBase}/${path}` : `/${path}`;
}

// 使用
const imageUrl = getCDNUrl("assets/logo.png");
```

### 图片 CDN

```typescript
// 使用图片处理服务
export function getOptimizedImageUrl(
  originalUrl: string,
  width: number,
  quality: number = 80
) {
  const cdn = "https://cdn.example.com";

  return `${cdn}/${originalUrl}?w=${width}&q=${quality}`;
```

## 资源预加载

### Link 预加载

```html
<!-- index.html -->
<link rel="preload" href="/assets/logo.png" as="image" />
<link rel="preload" href="/assets/font.woff2" as="font" crossorigin />
```

### 动态预加载

```typescript
// src/hooks/usePreload.ts
export function usePreload(href: string, as: string) {
  useEffect(() => {
    const link = document.createElement('link');
    link.rel = 'preload';
    link.href = href;
    link.as = as;

    if (as === 'font') {
      link.crossOrigin = 'anonymous';
    }

    document.head.appendChild(link);

    return () => {
      document.head.removeChild(link);
    };
  }, [href, as]);
}

// 使用
function App() {
  usePreload('/assets/logo.png', 'image');
  usePreload('/assets/font.woff2', 'font');

  return <div>App</div>;
}
```

## 常见问题

### Q1: 图片路径错误？

A: 使用正确的导入方式：

```typescript
// ✅ 正确
import img from "./image.png";

// ❌ 错误（相对路径问题）
const img = "./image.png";
```

### Q2: 字体不显示？

A: 检查路径和 MIME 类型：

```css
@font-face {
  src: url("@/assets/fonts/font.woff2") format("woff2");
}
```

### Q3: SVG 样式丢失？

A: 使用 inline SVG 或正确设置：

```typescript
// 使用 ?url
import logo from './logo.svg?url';
<img src={logo} />
```
