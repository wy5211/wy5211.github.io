## 前言

合理的项目初始化和配置是高效开发的基础。本文将深入讲解 Vite 项目的初始化流程和配置体系，帮助你搭建企业级的前端项目。

## 项目初始化

### 官方脚手架

```bash
# 创建新项目
npm create vite@latest my-app

# 交互式选择框架和语言
✔ Select a framework: › React
✔ Select a variant: › TypeScript + SWC

# 快速创建（跳过交互）
npm create vite@latest my-app -- --template react-ts
```

### 支持的模板

```bash
# Vanilla
--template vanilla
--template vanilla-ts

# React
--template react
--template react-ts

# Vue
--template vue
--template vue-ts

# 其他
--template preact
--template preact-ts
--template lit
--template lit-ts
--template svelte
--template svelte-ts
```

### 手动初始化

```bash
# 1. 创建项目目录
mkdir my-vite-app && cd my-vite-app

# 2. 初始化 package.json
npm init -y

# 3. 安装依赖
npm install vite @vitejs/plugin-react react react-dom

# 4. 安装开发依赖
npm install -D typescript @types/react @types/react-dom

# 5. 创建配置文件
touch vite.config.ts
touch tsconfig.json

# 6. 创建入口文件
mkdir src
touch src/main.tsx
touch index.html
```

## 核心配置详解

### 完整配置示例

```typescript
// vite.config.ts
import { defineConfig, loadEnv } from "vite";
import react from "@vitejs/plugin-react";
import path from "path";

// https://vitejs.dev/config/
export default defineConfig(({ mode }) => {
  // 加载环境变量
  const env = loadEnv(mode, process.cwd(), "");

  return {
    // 插件
    plugins: [
      react({
        // 使用 SWC（更快）
        jsxImportSource: "react",
        babel: {
          plugins:
            mode === "development" ? ["babel-plugin-styled-components"] : [],
        },
      }),
    ],

    // 开发服务器配置
    server: {
      host: "0.0.0.0",
      port: 3000,
      open: true,
      cors: true,

      // 代理配置
      proxy: {
        "/api": {
          target: "http://localhost:8080",
          changeOrigin: true,
          rewrite: (path) => path.replace(/^\/api/, ""),
        },
      },
    },

    // 预览服务器配置
    preview: {
      port: 4000,
      open: true,
    },

    // 依赖预构建
    optimizeDeps: {
      include: ["react", "react-dom", "react-router-dom"],
      exclude: ["your-local-package"],
    },

    // 解析配置
    resolve: {
      alias: {
        "@": path.resolve(__dirname, "./src"),
        "@components": path.resolve(__dirname, "./src/components"),
        "@utils": path.resolve(__dirname, "./src/utils"),
      },
      extensions: [".js", ".jsx", ".ts", ".tsx", ".json"],
    },

    // 构建配置
    build: {
      outDir: "dist",
      assetsDir: "assets",
      sourcemap: mode === "development",
      minify: "terser",
      target: "es2015",

      // 分包策略
      rollupOptions: {
        output: {
          manualChunks: {
            "react-vendor": ["react", "react-dom"],
            router: ["react-router-dom"],
          },
          chunkFileNames: "assets/js/[name]-[hash].js",
          entryFileNames: "assets/js/[name]-[hash].js",
          assetFileNames: "assets/[ext]/[name]-[hash].[ext]",
        },
      },

      // 压缩配置
      terserOptions: {
        compress: {
          drop_console: mode === "production",
          drop_debugger: mode === "production",
        },
      },

      // Chunk 大小警告
      chunkSizeWarningLimit: 1000,
    },

    // CSS 配置
    css: {
      modules: {
        localsConvention: "camelCaseOnly",
        generateScopedName: "[name]__[local]__[hash:base64:5]",
      },
      preprocessorOptions: {
        scss: {
          additionalData: `@import "@/styles/variables.scss";`,
        },
      },
    },

    // 全局常量
    define: {
      __APP_VERSION__: JSON.stringify(env.npm_package_version),
    },
  };
});
```

### 环境变量文件

```bash

# .env.production

# .env.staging

```

### 使用环境变量

```typescript
// src/config/env.ts
export const config = {
  apiUrl: import.meta.env.VITE_API_URL,
  appTitle: import.meta.env.VITE_APP_TITLE,
  isDev: import.meta.env.DEV,
  isProd: import.meta.env.PROD,
  mode: import.meta.env.MODE,
  enableMock: import.meta.env.VITE_ENABLE_MOCK === "true",
};

// src/api/base.ts
import axios from "axios";
import { config } from "@/config/env";

const api = axios.create({
  baseURL: config.apiUrl,
  timeout: 10000,
});

export default api;
```

### 环境变量类型安全

```typescript
// src/types/env.d.ts
interface ImportMetaEnv {
  // 必须以 VITE_ 开头
  readonly VITE_API_URL: string;
  readonly VITE_APP_TITLE: string;
  readonly VITE_ENABLE_MOCK: "true" | "false";
}

// 验证环境变量
const validateEnv = () => {
  const required: (keyof ImportMetaEnv)[] = ["VITE_API_URL", "VITE_APP_TITLE"];

  const missing = required.filter((key) => !import.meta.env[key]);

  if (missing.length > 0) {
    throw new Error(
      `Missing required environment variables: ${missing.join(", ")}`
    );
  }
};

// 在入口文件验证
// src/main.tsx
validateEnv();
```

## 配置最佳实践

### 1. 配置文件拆分

```typescript
// vite.config.ts
import { defineConfig } from "vite";
import react from "@vitejs/plugin-react";
import path from "path";

export default defineConfig({
  plugins: [react()],
  resolve: {
    alias: getAlias(),
  },
  build: {
    rollupOptions: {
      output: {
        manualChunks: getChunkSplit(),
      },
    },
  },
});

// vite.config.utils.ts
export function getAlias() {
  return {
    "@": path.resolve(__dirname, "./src"),
    "@components": path.resolve(__dirname, "./src/components"),
  };
}

export function getChunkSplit() {
  return {
    "react-vendor": ["react", "react-dom"],
    router: ["react-router-dom"],
  };
}
```

### 2. 环境特定配置

```typescript
// vite.config.ts
export default defineConfig(({ mode }) => {
  const isProd = mode === "production";

  return {
    build: {
      sourcemap: !isProd,
      minify: isProd ? "terser" : false,
      terserOptions: isProd
        ? {
            compress: {
              drop_console: true,
            },
          }
        : undefined,
    },
  };
});
```

### 3. 类型安全的环境变量

```typescript
// src/config/index.ts
export const config = {
```

## 常见问题

### Q1: 路径别名不生效？

A: 确保 Vite 和 TypeScript 都配置了路径别名：

```typescript
// vite.config.ts
resolve: {
  alias: {
    '@': '/src'
  }
}

// tsconfig.json
{
  "compilerOptions": {
    "baseUrl": ".",
    "paths": {
      "@/*": ["./src/*"]
    }
  }
}
```

### Q2: 环境变量在代码中为 undefined？

A: 确保变量名以 `VITE_` 开头：

```bash
# ✅ 正确
VITE_API_URL=http://localhost:8080

# ❌ 错误（不会暴露给客户端）
API_URL=http://localhost:8080
```

### Q3: TypeScript 报错找不到模块？

A: 检查 `tsconfig.json` 的 `include` 和 `compilerOptions.moduleResolution`：

```json
{
  "include": ["src"],
  "compilerOptions": {
    "moduleResolution": "bundler"
  }
}
```
