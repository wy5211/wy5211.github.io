## 缓存穿透

### 问题

查询不存在的数据，缓存和数据库都没有，导致每次请求都打到数据库。

### 解决方案 1：缓存空值

```typescript
async function getWithNullCache(key: string) {
  let data = await redis.get(key);

  if (data) {
    if (data === "NULL") {
      return null;
    }
    return JSON.parse(data);
  }

  data = await db.query(key);

  if (!data) {
    // 缓存空值，短过期时间
    await redis.setex(key, 60, "NULL");
    return null;
  }

  await redis.setex(key, 3600, JSON.stringify(data));
  return data;
}
```

### 解决方案 2：布隆过滤器

```typescript
import { BloomFilter } from "bloom-filters";

// 初始化布隆过滤器
const bloomFilter = new BloomFilter(10000, 0.01);

// 预加载所有存在的 key
async function initBloomFilter() {
  const allKeys = await db.getAllKeys();
  for (const key of allKeys) {
    bloomFilter.add(key);
  }
}

// 查询时先检查布隆过滤器
async function getWithBloomFilter(key: string) {
  // 布隆过滤器判断 key 不存在
  if (!bloomFilter.has(key)) {
    return null;
  }

  // 可能存在，继续查询
  let data = await redis.get(key);
  if (data) {
    return JSON.parse(data);
  }

  data = await db.query(key);
  if (data) {
    await redis.setex(key, 3600, JSON.stringify(data));
  }

  return data;
}
```

### 解决方案 3：Redis 布隆过滤器

```bash

redis-cli BF.ADD bloomfilter key1
redis-cli BF.EXISTS bloomfilter key1
```

## 缓存击穿

### 问题

热点数据过期时，大量请求同时打到数据库。

### 解决方案 1：互斥锁

```typescript
async function getWithLock(key: string) {
  let data = await redis.get(key);
  if (data) {
    return JSON.parse(data);
  }

  const lockKey = `lock:${key}`;
  const lockValue = Date.now().toString();

  // 尝试获取锁
  const lock = await redis.set(lockKey, lockValue, "NX", "EX", 10);

  if (lock === "OK") {
    try {
      // 获取锁成功，查询数据库
      data = await db.query(key);
      if (data) {
        await redis.setex(key, 3600, JSON.stringify(data));
      }
      return data;
    } finally {
      // 释放锁
      const script = `
        if redis.call("get", KEYS[1]) == ARGV[1] then
          return redis.call("del", KEYS[1])
        else
          return 0
        end
      `;
      await redis.eval(script, 1, lockKey, lockValue);
    }
  }

  // 获取锁失败，等待后重试
  await new Promise((resolve) => setTimeout(resolve, 50));
  return await getWithLock(key);
}
```

### 解决方案 2：热点数据永不过期

```typescript
async function getWithNeverExpire(key: string) {
  // 逻辑过期时间存储在值中
  const cached = await redis.get(key);
  if (cached) {
    const data = JSON.parse(cached);

    // 检查逻辑过期时间
    if (data.expireTime > Date.now()) {
      return data.value;
    }

    // 过期了，异步更新
    refreshCache(key);
    return data.value; // 返回旧数据
  }

  // 缓存不存在，直接查询
  const data = await db.query(key);

  // 设置逻辑过期时间（物理不过期）
  await redis.set(
    key,
    JSON.stringify({
      value: data,
      expireTime: Date.now() + 3600 * 1000,
    })
  );

  return data;
}

async function refreshCache(key: string) {
  const lockKey = `refresh:${key}`;
  const lock = await redis.set(lockKey, "1", "NX", "EX", 10);

  if (lock === "OK") {
    try {
      const data = await db.query(key);
      await redis.set(
        key,
        JSON.stringify({
          value: data,
          expireTime: Date.now() + 3600 * 1000,
        })
      );
    } finally {
      await redis.del(lockKey);
    }
  }
}
```

## 缓存雪崩

### 问题

大量缓存同时失效，导致请求全部打到数据库。

### 解决方案 1：过期时间加随机值

```typescript
async function setWithRandomTTL(
  key: string,
  value: any,
  baseTTL: number = 3600
) {
  // 添加随机值（±10%）
  const randomTTL = baseTTL + Math.random() * baseTTL * 0.2 - baseTTL * 0.1;
  await redis.setex(key, Math.floor(randomTTL), JSON.stringify(value));
}
```

### 解决方案 2：缓存预热

```typescript
async function warmupCache() {
  const hotKeys = await db.getHotKeys();

  for (const key of hotKeys) {
    const data = await db.query(key);
    if (data) {
      // 错开过期时间
      const ttl = 3600 + Math.random() * 600;
      await redis.setex(key, Math.floor(ttl), JSON.stringify(data));
    }
  }
}

// 应用启动时预热
warmupCache();
```

### 解决方案 4：熔断降级

```typescript
class CircuitBreaker {
  private failures = 0;
  private lastFailTime = 0;
  private state = "CLOSED"; // CLOSED, OPEN, HALF_OPEN

  async execute(fn: Function) {
    if (this.state === "OPEN") {
      if (Date.now() - this.lastFailTime > 60000) {
        this.state = "HALF_OPEN";
      } else {
        throw new Error("服务熔断");
      }
    }

    try {
      const result = await fn();
      this.onSuccess();
      return result;
    } catch (error) {
      this.onFailure();
      throw error;
    }
  }

  private onSuccess() {
    this.failures = 0;
    this.state = "CLOSED";
  }

  private onFailure() {
    this.failures++;
    this.lastFailTime = Date.now();

    if (this.failures > 5) {
      this.state = "OPEN";
    }
  }
}

// 使用
const breaker = new CircuitBreaker();

async function getDataWithBreaker(key: string) {
  return await breaker.execute(async () => {
    let data = await redis.get(key);
    if (data) return JSON.parse(data);

    data = await db.query(key);
    if (data) {
      await redis.setex(key, 3600, JSON.stringify(data));
    }

    return data;
  });
}
```

## 完整解决方案

```typescript
class RobustCache {
  private bloomFilter: BloomFilter;
  private circuitBreaker: CircuitBreaker;

  constructor(
    private redis: Redis,
    private db: any
  ) {
    this.bloomFilter = new BloomFilter(10000, 0.01);
    this.circuitBreaker = new CircuitBreaker();
    this.initBloomFilter();
  }

  private async initBloomFilter() {
    const allKeys = await this.db.getAllKeys();
    for (const key of allKeys) {
      this.bloomFilter.add(key);
    }
  }

  async get(key: string) {
    // 1. 布隆过滤器检查（防穿透）
    if (!this.bloomFilter.has(key)) {
      return null;
    }

    // 2. 查询缓存
    let data = await this.redis.get(key);
    if (data) {
      if (data === "NULL") return null;
      return JSON.parse(data);
    }

    // 3. 使用互斥锁查询数据库（防击穿）
    return await this.getWithLock(key);
  }

  private async getWithLock(key: string) {
    const lockKey = `lock:${key}`;
    const lockValue = Date.now().toString();

    const lock = await this.redis.set(lockKey, lockValue, "NX", "EX", 10);

    if (lock === "OK") {
      try {
        // 使用熔断器（防雪崩）
        return await this.circuitBreaker.execute(async () => {
          const data = await this.db.query(key);

          if (!data) {
            // 缓存空值
            await this.redis.setex(key, 60, "NULL");
            return null;
          }

          // 设置随机过期时间
          const ttl = 3600 + Math.random() * 600;
          await this.redis.setex(key, Math.floor(ttl), JSON.stringify(data));

          return data;
        });
      } finally {
        const script = `
          if redis.call("get", KEYS[1]) == ARGV[1] then
            return redis.call("del", KEYS[1])
          else
            return 0
          end
        `;
        await this.redis.eval(script, 1, lockKey, lockValue);
      }
    }

    // 获取锁失败，等待重试
    await new Promise((resolve) => setTimeout(resolve, 50));
    return await this.get(key);
  }
}
```

## 总结

本文我们学习了三大缓存问题的解决方案：

✅ **缓存穿透**：布隆过滤器、缓存空值
✅ **缓存击穿**：互斥锁、热点数据永不过期
✅ **缓存雪崩**：随机过期时间、缓存预热、熔断降级
