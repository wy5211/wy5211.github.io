### 3.1 完整的 RAG 实现

```typescript
/**
 * 完整的 RAG 系统实现
 */

import { OpenAI } from "openai";
import { Pinecone } from "@pinecone-database/pinecone";
import { RecursiveCharacterTextSplitter } from "langchain/text_splitter";

// 初始化
const openai = new OpenAI();
const pinecone = new Pinecone({ apiKey: process.env.PINECONE_API_KEY! });

/**
 * 第一步：文档处理
 */
class DocumentProcessor {
  private splitter = new RecursiveCharacterTextSplitter({
    chunkSize: 1000, // 每块大小
    chunkOverlap: 200, // 重叠部分
  });

  /**
   * 处理文档：分割成小块
   */
  async processDocument(content: string, metadata: Record<string, any> = {}) {
    // 分割文档
    const chunks = await this.splitter.splitText(content);

    // 为每个块添加元数据
    return chunks.map((chunk, index) => ({
      id: `${metadata.id || "doc"}-${index}`,
      content: chunk,
      metadata: {
        ...metadata,
        chunkIndex: index,
      },
    }));
  }

  /**
   * 批量处理多个文档
   */
  async processDocuments(
    documents: Array<{ content: string; metadata?: any }>
  ) {
    const allChunks = [];

    for (const doc of documents) {
      const chunks = await this.processDocument(doc.content, doc.metadata);
      allChunks.push(...chunks);
    }

    return allChunks;
  }
}

/**
 * 第二步：向量化与存储
 */
class VectorStore {
  private index;
  private openai = new OpenAI();

  constructor(indexName: string) {
    this.index = pinecone.index(indexName);
  }

  /**
   * 生成向量
   */
  private async generateEmbedding(text: string): Promise<number[]> {
    const response = await this.openai.embeddings.create({
      model: "text-embedding-3-small",
      input: text,
    });
    return response.data[0].embedding;
  }

  /**
   * 批量生成向量
   */
  private async generateEmbeddings(texts: string[]): Promise<number[][]> {
    const response = await this.openai.embeddings.create({
      model: "text-embedding-3-small",
      input: texts,
    });
    return response.data.map((d) => d.embedding);
  }

  /**
   * 存储文档
   */
  async addDocuments(
    documents: Array<{ id: string; content: string; metadata: any }>
  ) {
    // 批量生成向量
    const embeddings = await this.generateEmbeddings(
      documents.map((d) => d.content)
    );

    // 准备数据
    const vectors = documents.map((doc, i) => ({
      id: doc.id,
      values: embeddings[i],
      metadata: {
        ...doc.metadata,
        text: doc.content,
      },
    }));

    // 批量上传（每批 100 条）
    const batchSize = 100;
    for (let i = 0; i < vectors.length; i += batchSize) {
      const batch = vectors.slice(i, i + batchSize);
      await this.index.upsert(batch);
    }

    console.log(`已索引 ${documents.length} 个文档块`);
  }

  /**
   * 相似度搜索
   */
  async similaritySearch(query: string, topK: number = 5) {
    // 查询向量化
    const queryEmbedding = await this.generateEmbedding(query);

    // 搜索
    const results = await this.index.query({
      vector: queryEmbedding,
      topK,
      includeMetadata: true,
    });

    return results.matches.map((match) => ({
      content: match.metadata?.text as string,
      score: match.score || 0,
      metadata: match.metadata,
    }));
  }
}

/**
 * 第三步：RAG 查询
 */
class RAGQuery {
  private vectorStore: VectorStore;
  private openai = new OpenAI();

  constructor(vectorStore: VectorStore) {
    this.vectorStore = vectorStore;
  }

  /**
   * 构建提示词
   */
  private buildPrompt(query: string, contexts: string[]): string {
    return `你是一个智能助手。请根据以下参考内容回答用户问题。

如果参考内容中没有相关信息，请诚实地说"根据现有资料，我无法回答这个问题"。

参考内容：
${contexts.map((c, i) => `[文档${i + 1}]\n${c}`).join("\n\n")}

用户问题：${query}

请提供准确、有帮助的回答：`;
  }

  /**
   * 执行查询
   */
  async query(question: string) {
    // 1. 检索相关文档
    const relevantDocs = await this.vectorStore.similaritySearch(question, 5);

    if (relevantDocs.length === 0) {
      return {
        answer: "抱歉，我没有找到相关的信息来回答你的问题。",
        sources: [],
      };
    }

    // 2. 过滤低相关度的结果
    const filteredDocs = relevantDocs.filter((doc) => doc.score > 0.7);

    // 3. 构建上下文
    const contexts = filteredDocs.map((doc) => doc.content);

    // 4. 生成回答
    const prompt = this.buildPrompt(question, contexts);

    const response = await this.openai.chat.completions.create({
      model: "gpt-4o-mini",
      messages: [
        { role: "system", content: "你是一个专业、准确的助手。" },
        { role: "user", content: prompt },
      ],
      temperature: 0.3, // 较低的温度，保证准确性
    });

    return {
      answer: response.choices[0].message.content,
      sources: filteredDocs.map((doc) => ({
        content: doc.content.slice(0, 200) + "...",
        score: doc.score,
        metadata: doc.metadata,
      })),
    };
  }
}

/**
 * 使用示例
 */
async function main() {
  // 初始化组件
  const processor = new DocumentProcessor();
  const vectorStore = new VectorStore("my-knowledge-base");
  const ragQuery = new RAGQuery(vectorStore);

  // 示例：索引文档
  const documents = [
    {
      content: `# 公司请假制度

员工享有以下假期类型：
1. 年假：工作满一年后，享有5天带薪年假
2. 病假：每年享有10天带薪病假
3. 事假：需提前申请，经主管批准

请假流程：
1. 在OA系统提交申请
2. 选择请假类型和时间
3. 等待主管审批
4. 审批通过后生效`,
      metadata: { id: "policy-leave", type: "policy", title: "请假制度" },
    },
    {
      content: `# 报销流程

报销类型：
1. 差旅费：交通、住宿、餐饮
2. 办公用品：经批准的采购
3. 业务招待：需提前申请

报销步骤：
1. 收集原始发票
2. 在财务系统录入报销单
3. 上传发票扫描件
4. 提交审批
5. 财务审核通过后打款`,
      metadata: { id: "policy-expense", type: "policy", title: "报销流程" },
    },
  ];

  // 处理并索引文档
  const chunks = await processor.processDocuments(documents);
  await vectorStore.addDocuments(chunks);

  // 查询
  const result1 = await ragQuery.query("请告诉我年假有几天？");
  console.log("回答:", result1.answer);
  console.log("来源:", result1.sources);

  const result2 = await ragQuery.query("报销需要什么材料？");
  console.log("回答:", result2.answer);
}
```

```typescript
// 2. 混合检索
async function hybridSearch(query: string) {
  // 向量检索
  const vectorResults = await vectorStore.similaritySearch(query, 5);

  // 关键词检索（简化示例）
  const keywordResults = await keywordSearch(query, 5);

  // 合并去重
  const merged = [...vectorResults, ...keywordResults];
  const unique = new Map();
  merged.forEach((item) => {
    const key = item.content.slice(0, 100);
    if (!unique.has(key) || unique.get(key).score < item.score) {
      unique.set(key, item);
    }
  });

  return Array.from(unique.values())
    .sort((a, b) => b.score - a.score)
    .slice(0, 5);
}

// 3. 重排序
async function rerankResults(query: string, documents: any[]) {
  const prompt = `问题：${query}

以下是检索到的文档片段，请对每个文档与问题的相关性打分（1-10）：

${documents
  .map(
    (doc, i) => `
[文档${i + 1}]
${doc.content.slice(0, 300)}
`
  )
  .join("\n")}

请以JSON格式输出评分：
{"scores": [分数1, 分数2, ...]}`;

  const response = await openai.chat.completions.create({
    model: "gpt-4o-mini",
    messages: [{ role: "user", content: prompt }],
    response_format: { type: "json_object" },
  });

  const scores = JSON.parse(
    response.choices[0].message.content || '{"scores":[]}'
  ).scores;

  return documents
    .map((doc, i) => ({ ...doc, rerankScore: scores[i] || 0 }))
    .sort((a, b) => b.rerankScore - a.rerankScore);
}
```

```typescript
/**

import { ChatOpenAI } from "@langchain/openai";
import {
  HumanMessage,
  AIMessage,
  SystemMessage,
} from "@langchain/core/messages";

// Agent 基类
abstract class BaseAgent {
  protected llm: ChatOpenAI;
  protected name: string;
  protected role: string;

  constructor(name: string, role: string) {
    this.llm = new ChatOpenAI({ modelName: "gpt-4o-mini", temperature: 0.7 });
    this.name = name;
    this.role = role;
  }

  abstract execute(task: string, context?: string): Promise<string>;

  protected async generateResponse(prompt: string): Promise<string> {
    const response = await this.llm.invoke([
      new SystemMessage(`你是${this.name}，${this.role}`),
      new HumanMessage(prompt),
    ]);
    return response.content as string;
  }
}

// 研究员 Agent
class ResearcherAgent extends BaseAgent {
  constructor() {
    super("研究员", "负责收集和分析信息，提供事实和数据支持");
  }

  async execute(task: string): Promise<string> {
    const prompt = `请针对以下任务进行研究分析：

任务：${task}

请提供：
1. 关键信息点
2. 相关数据或事实
3. 潜在问题和风险`;

    return this.generateResponse(prompt);
  }
}

// 写作者 Agent
class WriterAgent extends BaseAgent {
  constructor() {
    super("写作者", "负责撰写和编辑内容，保证文字质量");
  }

  async execute(task: string, context?: string): Promise<string> {
    const prompt = `请根据以下信息和要求撰写内容：

任务：${task}

${context ? `参考资料：\n${context}` : ""}

请撰写清晰、专业的内容。`;

    return this.generateResponse(prompt);
  }
}

// 审核者 Agent
class ReviewerAgent extends BaseAgent {
  constructor() {
    super("审核者", "负责审核内容质量，提出改进建议");
  }

  async execute(task: string, context: string): Promise<string> {
    const prompt = `请审核以下内容：

原始任务：${task}

待审核内容：
${context}

请从以下方面进行审核：
1. 内容准确性
2. 逻辑清晰度
3. 语言表达
4. 改进建议`;

    return this.generateResponse(prompt);
  }
}

// 协调者 - 管理多 Agent 协作
class AgentOrchestrator {
  private agents: Map<string, BaseAgent>;
  private maxIterations: number = 3;

  constructor() {
    this.agents = new Map([
      ["researcher", new ResearcherAgent()],
      ["writer", new WriterAgent()],
      ["reviewer", new ReviewerAgent()],
    ]);
  }

  /**
   * 执行协作任务
   */
  async execute(task: string): Promise<string> {
    // 1. 研究阶段
    console.log("📊 研究员正在分析...");
    const research = await this.agents.get("researcher")!.execute(task);
    console.log("研究完成");

    // 2. 撰写阶段（可能多轮迭代）
    let draft = "";
    let review = "";
    let iterations = 0;

    while (iterations < this.maxIterations) {
      // 撰写/修改
      console.log(`✍️ 写作者正在${iterations === 0 ? "撰写" : "修改"}...`);
      draft = await this.agents
        .get("writer")!
        .execute(
          task,
          `${research}\n\n${review ? `审核意见：\n${review}` : ""}`
        );

      // 审核
      console.log("🔍 审核者正在审核...");
      review = await this.agents.get("reviewer")!.execute(task, draft);

      // 检查是否需要修改
      if (review.includes("通过") || review.includes("无需修改")) {
        console.log("✅ 审核通过");
        break;
      }

      iterations++;
      console.log(`🔄 需要修改 (第 ${iterations} 轮)`);
    }

    return draft;
  }
}

  console.log(result);
}
```

### 4.2 Agent 通信协议

```typescript
/**
 * Agent 之间的通信协议
 */

// 消息类型
interface AgentMessage {
  from: string; // 发送者
  to: string; // 接收者
  type: "request" | "response" | "broadcast";
  content: string; // 消息内容
  metadata?: Record<string, any>;
  timestamp: number;
}

// 消息总线
class MessageBus {
  private subscribers: Map<string, (message: AgentMessage) => void> = new Map();

  // 订阅
  subscribe(agentName: string, handler: (message: AgentMessage) => void) {
    this.subscribers.set(agentName, handler);
  }

  // 发送消息
  send(message: AgentMessage) {
    const handler = this.subscribers.get(message.to);
    if (handler) {
      handler(message);
    }
  }

  // 广播
  broadcast(from: string, content: string) {
    this.subscribers.forEach((handler, agentName) => {
      if (agentName !== from) {
        handler({
          from,
          to: agentName,
          type: "broadcast",
          content,
          timestamp: Date.now(),
        });
      }
    });
  }
}

// 带通信能力的 Agent
class CommunicatingAgent extends BaseAgent {
  private bus: MessageBus;

  constructor(name: string, role: string, bus: MessageBus) {
    super(name, role);
    this.bus = bus;

    // 订阅消息
    this.bus.subscribe(name, this.handleMessage.bind(this));
  }

  private async handleMessage(message: AgentMessage) {
    console.log(
      `[${this.name}] 收到来自 ${message.from} 的消息: ${message.content.slice(0, 50)}...`
    );

    // 处理消息并回复
    const response = await this.execute(message.content);

    this.bus.send({
      from: this.name,
      to: message.from,
      type: "response",
      content: response,
      timestamp: Date.now(),
    });
  }

  // 请求其他 Agent
  async request(targetAgent: string, task: string): Promise<string> {
    return new Promise((resolve) => {
      // 设置一次性监听
      const tempHandler = (message: AgentMessage) => {
        if (message.from === targetAgent) {
          resolve(message.content);
        }
      };

      this.bus.subscribe(`${this.name}-temp`, tempHandler);

      // 发送请求
      this.bus.send({
        from: this.name,
        to: targetAgent,
        type: "request",
        content: task,
        timestamp: Date.now(),
      });
    });
  }

  async execute(task: string): Promise<string> {
    return this.generateResponse(task);
  }
}
```

```typescript
// 1. 缓存层
class AgentCache {
  private redis: Redis;

  constructor() {
    this.redis = new Redis(process.env.REDIS_URL!);
  }

  async get(key: string): Promise<string | null> {
    return this.redis.get(`agent:${key}`);
  }

  async set(key: string, value: string, ttl: number = 3600): Promise<void> {
    await this.redis.setex(`agent:${key}`, ttl, value);
  }

  // 生成缓存 key
  static generateKey(prompt: string, model: string): string {
    const hash = require("crypto")
      .createHash("md5")
      .update(`${model}:${prompt}`)
      .digest("hex");
    return hash;
  }
}

// 2. LLM 调用层（带缓存和重试）
class LLMService {
  private cache: AgentCache;
  private openai: OpenAI;

  constructor() {
    this.cache = new AgentCache();
    this.openai = new OpenAI();
  }

  async chat(
    messages: any[],
    options: { model?: string; useCache?: boolean } = {}
  ) {
    const model = options.model || "gpt-4o-mini";

    // 检查缓存
    if (options.useCache !== false) {
      const cacheKey = AgentCache.generateKey(JSON.stringify(messages), model);
      const cached = await this.cache.get(cacheKey);
      if (cached) {
        console.log("Cache hit");
        return JSON.parse(cached);
      }
    }

    // 调用 LLM（带重试）
    let lastError;
    for (let attempt = 0; attempt < 3; attempt++) {
      try {
        const response = await this.openai.chat.completions.create({
          model,
          messages,
        });

        const result = response.choices[0].message.content;

        // 缓存结果
        if (options.useCache !== false) {
          const cacheKey = AgentCache.generateKey(
            JSON.stringify(messages),
            model
          );
          await this.cache.set(cacheKey, JSON.stringify(result));
        }

        return result;
      } catch (error) {
        lastError = error;
        console.log(`Attempt ${attempt + 1} failed, retrying...`);
        await new Promise((r) => setTimeout(r, 1000 * (attempt + 1)));
      }
    }

    throw lastError;
  }
}
```
