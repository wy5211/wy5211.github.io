```typescript
/**
 * Token 是 LLM 计费的基本单位
 * 理解 Token 有助于成本优化
 */

class TokenCalculator {
  // 中文：1 个汉字约 1-2 个 Token
  // 英文：1 个单词约 1 个 Token
  // 代码：根据格式，通常比自然语言更密集

  estimateTokens(text: string): number {
    // 粗略估算
    const chineseChars = (text.match(/[\u4e00-\u9fa5]/g) || []).length;
    const englishWords = text.split(/\s+/).filter((w) => w).length;
    const codeChars = text.length - chineseChars - englishWords * 5;

    return Math.ceil(
      chineseChars * 1.5 + // 中文
        englishWords * 1 + // 英文
        codeChars * 0.3 // 代码和其他
    );
  }

  estimateCost(tokens: number, model: string): number {
    const pricing = {
      "gpt-4": { input: 0.03 / 1000, output: 0.06 / 1000 },
      "gpt-3.5": { input: 0.0015 / 1000, output: 0.002 / 1000 },
      "claude-3": { input: 0.015 / 1000, output: 0.075 / 1000 },
    };

    // 假设输入输出各占一半
    const p = pricing[model];
    return tokens * 0.5 * p.input + tokens * 0.5 * p.output;
  }
}
```

#### 模式一：ReAct 架构

```typescript
/**
 * ReAct: Reasoning + Acting
 * 最常用的 Agent 架构模式
 */

interface ReActStep {
  thought: string; // 思考：分析当前情况
  action: string; // 行动：决定采取什么行动
  observation: string; // 观察：行动的结果
}

class ReActAgent {
  async run(task: string): Promise<string> {
    const steps: ReActStep[] = [];
    let isComplete = false;

    while (!isComplete) {
      // 思考阶段
      const thought = await this.think(task, steps);
      console.log(`💭 Thought: ${thought}`);

      // 如果任务完成
      if (thought.includes("Task complete")) {
        isComplete = true;
        break;
      }

      // 行动阶段
      const action = await this.decideAction(thought);
      console.log(`🎯 Action: ${action.tool}(${JSON.stringify(action.args)})`);

      // 观察阶段
      const observation = await this.executeAction(action);
      console.log(`👁️ Observation: ${observation}`);

      steps.push({ thought, action: action.tool, observation });

      // 安全限制
      if (steps.length > 20) {
        throw new Error("Max steps exceeded");
      }
    }

    return this.generateFinalAnswer(steps);
  }

  private async think(task: string, history: ReActStep[]): Promise<string> {
    const prompt = `
任务: ${task}

历史步骤:
${history
  .map(
    (s, i) => `
步骤 ${i + 1}:
- 思考: ${s.thought}
- 行动: ${s.action}
- 观察: ${s.observation}
`
  )
  .join("\n")}

请思考下一步该做什么。
如果任务已完成，请说 "Task complete: [答案]"
    `;

    return await this.llm.generate(prompt);
  }
}
```

```typescript
  name: string;
  responsibility: string;
  tools: string[];
}

// 研究员 Agent
const researcher: AgentRole = {
  name: "Researcher",
  responsibility: "收集和分析信息",
  tools: ["searchWeb", "readDocument", "extractData"],
};

// 写作者 Agent
const writer: AgentRole = {
  name: "Writer",
  responsibility: "撰写和编辑内容",
  tools: ["generateText", "formatContent"],
};

// 审核者 Agent
const reviewer: AgentRole = {
  name: "Reviewer",
  responsibility: "审核和改进质量",
  tools: ["checkGrammar", "verifyFacts", "suggestImprovements"],
};

// 协调系统
class MultiAgentSystem {
  private agents: Map<string, Agent>;

  constructor() {
    this.agents = new Map([
      ["researcher", new Agent(researcher)],
      ["writer", new Agent(writer)],
      ["reviewer", new Agent(reviewer)],
    ]);
  }

  async writeArticle(topic: string) {
    // 1. 研究员收集信息
    const research = await this.agents
      .get("researcher")!
      .run(`研究主题: ${topic}，收集关键信息`);

    // 2. 写作者撰写初稿
    const draft = await this.agents
      .get("writer")!
      .run(`根据以下研究内容撰写文章:\n${research}`);

    // 3. 审核者审核
    const review = await this.agents
      .get("reviewer")!
      .run(`审核以下文章并提出改进建议:\n${draft}`);

    // 4. 写作者根据审核修改
    const finalArticle = await this.agents
      .get("writer")!
      .run(`根据审核意见修改文章:\n原稿: ${draft}\n审核意见: ${review}`);

    return finalArticle;
  }
}
```

```typescript
/**
 * 工具开发最佳实践
 */

// 1. 清晰的工具定义
interface ToolDefinition {
  name: string; // 工具名称（唯一标识）
  description: string; // 工具描述（Agent 需要理解）
  parameters: JSONSchema; // 参数 Schema
  returns: JSONSchema; // 返回值 Schema
}

// 2. 示例：天气查询工具
const weatherTool: ToolDefinition = {
  name: "get_weather",
  description: "获取指定城市的当前天气信息",
  parameters: {
    type: "object",
    properties: {
      city: {
        type: "string",
        description: '城市名称，如"北京"、"上海"',
      },
      unit: {
        type: "string",
        enum: ["celsius", "fahrenheit"],
        description: "温度单位，默认 celsius",
      },
    },
    required: ["city"],
  },
  returns: {
    type: "object",
    properties: {
      temperature: { type: "number", description: "温度" },
      condition: { type: "string", description: "天气状况" },
      humidity: { type: "number", description: "湿度百分比" },
    },
  },
};

// 3. 工具实现
class WeatherTool {
  definition = weatherTool;

  async execute(args: { city: string; unit?: string }) {
    // 参数验证
    if (!args.city) {
      throw new ToolError("City is required", "INVALID_ARGS");
    }

    try {
      // 调用天气 API
      const response = await fetch(
        `https://api.weather.com/current?city=${encodeURIComponent(args.city)}`
      );

      if (!response.ok) {
        throw new ToolError(
          `Weather API error: ${response.status}`,
          "API_ERROR"
        );
      }

      const data = await response.json();

      // 返回标准化格式
      return {
        success: true,
        data: {
          temperature: this.convertUnit(data.temp, args.unit),
          condition: data.condition,
          humidity: data.humidity,
        },
      };
    } catch (error) {
      // 统一错误处理
      return {
        success: false,
        error: error.message,
        code: error.code || "UNKNOWN_ERROR",
      };
    }
  }

  private convertUnit(temp: number, unit?: string): number {
    if (unit === "fahrenheit") {
      return (temp * 9) / 5 + 32;
    }
    return temp;
  }
}

// 4. 工具注册
class ToolRegistry {
  private tools: Map<string, Tool> = new Map();

  register(tool: Tool) {
    this.tools.set(tool.definition.name, tool);
  }

  get(name: string): Tool | undefined {
    return this.tools.get(name);
  }

  // 获取所有工具定义（供 LLM 使用）
  getDefinitions(): ToolDefinition[] {
    return Array.from(this.tools.values()).map((t) => t.definition);
  }
}
```

### 6.2 Agent 测试框架

```typescript
/**
 * Agent 测试框架
 */

// 1. 评估数据集
interface TestCase {
  input: string;
  expectedOutput?: string;
  expectedTools?: string[];
  evaluationCriteria: EvaluationCriteria;
}

interface EvaluationCriteria {
  accuracy?: number; // 准确率阈值
  toolUsage?: string[]; // 期望使用的工具
  latency?: number; // 响应时间要求
  costLimit?: number; // 成本限制
}

// 2. 评估器
class AgentEvaluator {
  /**
   * 运行评估
   */
  async evaluate(
    agent: Agent,
    testCases: TestCase[]
  ): Promise<EvaluationReport> {
    const results: TestResult[] = [];

    for (const testCase of testCases) {
      const startTime = Date.now();
      const result = await agent.run(testCase.input);
      const latency = Date.now() - startTime;

      const evaluation = {
        passed: this.evaluateResult(result, testCase),
        latency,
        toolsUsed: result.toolsUsed,
        cost: result.cost,
      };

      results.push({
        testCase,
        result,
        evaluation,
      });
    }

    return this.generateReport(results);
  }

  /**
   * 评估单个结果
   */
  private evaluateResult(result: AgentResult, testCase: TestCase): boolean {
    const criteria = testCase.evaluationCriteria;

    // 检查输出
    if (testCase.expectedOutput) {
      const similarity = this.calculateSimilarity(
        result.output,
        testCase.expectedOutput
      );
      if (similarity < (criteria.accuracy || 0.8)) {
        return false;
      }
    }

    // 检查工具使用
    if (criteria.toolUsage) {
      const usedTools = result.toolsUsed.map((t) => t.name);
      for (const expectedTool of criteria.toolUsage) {
        if (!usedTools.includes(expectedTool)) {
          return false;
        }
      }
    }

    // 检查延迟
    if (criteria.latency && result.latency > criteria.latency) {
      return false;
    }

    // 检查成本
    if (criteria.costLimit && result.cost > criteria.costLimit) {
      return false;
    }

    return true;
  }
}
```
