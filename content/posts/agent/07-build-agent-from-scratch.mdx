### 2.2 核心流程

```typescript
/**
 * Agent 核心处理流程
 */
async function processUserMessage(
  message: string,
  context: ConversationContext
) {
  // 1. 意图识别
  const intent = await intentAgent.recognize(message);
  // 例如：{ type: 'order_query', confidence: 0.95, entities: { orderId: '12345' } }

  // 2. 槽位填充（提取必要信息）
  const slots = await slotFilling(message, intent);
  // 例如：{ orderId: '12345' }

  // 3. 检查信息完整性
  if (!isComplete(slots, intent)) {
    // 询问缺失信息
    return askForMissingInfo(slots, intent);
  }

  // 4. 执行任务
  const result = await taskAgent.execute(intent, slots);

  // 5. 生成回复
  const response = await responseAgent.generate(intent, result);

  return response;
}
```

### 3.2 核心代码实现

#### 协调器（Orchestrator）

```typescript
// lib/agent/orchestrator.ts

import { ChatOpenAI } from "@langchain/openai";
import { ConversationChain } from "langchain/chains";
import { BufferMemory } from "langchain/memory";

import { IntentAgent } from "./intent-agent";
import { TaskAgent } from "./task-agent";
import { ResponseAgent } from "./response-agent";
import { ConversationMemory } from "../memory/conversation";

/**
 * Agent 协调器
 * 管理多个 Agent 的协作
 */
export class AgentOrchestrator {
  private llm: ChatOpenAI;
  private intentAgent: IntentAgent;
  private taskAgent: TaskAgent;
  private responseAgent: ResponseAgent;
  private memory: ConversationMemory;

  constructor() {
    // 初始化 LLM
    this.llm = new ChatOpenAI({
      modelName: "gpt-4o-mini",
      temperature: 0.3,
    });

    // 初始化各 Agent
    this.intentAgent = new IntentAgent(this.llm);
    this.taskAgent = new TaskAgent(this.llm);
    this.responseAgent = new ResponseAgent(this.llm);
    this.memory = new ConversationMemory();
  }

  /**
   * 处理用户消息
   */
  async handleMessage(
    message: string,
    sessionId: string
  ): Promise<AgentResponse> {
    // 1. 获取对话历史
    const history = await this.memory.getHistory(sessionId);

    // 2. 意图识别
    const intent = await this.intentAgent.recognize(message, history);
    console.log("[Intent]", intent);

    // 3. 检查是否需要人工
    if (intent.type === "human_handoff") {
      return {
        type: "handoff",
        message: "正在为您转接人工客服，请稍候...",
        metadata: { reason: intent.reason },
      };
    }

    // 4. 执行任务
    let taskResult;
    if (intent.requiresAction) {
      // 检查信息是否完整
      if (!this.isInfoComplete(intent)) {
        return {
          type: "question",
          message: this.generateQuestion(intent),
          missingInfo: intent.missingSlots,
        };
      }

      // 执行任务
      taskResult = await this.taskAgent.execute(intent);
      console.log("[Task Result]", taskResult);
    }

    // 5. 生成回复
    const response = await this.responseAgent.generate(
      message,
      intent,
      taskResult,
      history
    );

    // 6. 保存对话历史
    await this.memory.addMessage(sessionId, {
      role: "user",
      content: message,
    });
    await this.memory.addMessage(sessionId, {
      role: "assistant",
      content: response.message,
    });

    return {
      type: "response",
      message: response.message,
      intent: intent.type,
      data: taskResult?.data,
    };
  }

  /**
   * 检查信息是否完整
   */
  private isInfoComplete(intent: Intent): boolean {
    const requiredSlots = this.getRequiredSlots(intent.type);
    return requiredSlots.every((slot) => intent.entities[slot]);
  }

  /**
   * 获取必需的槽位
   */
  private getRequiredSlots(intentType: string): string[] {
    const slotMap: Record<string, string[]> = {
      order_query: ["orderId"],
      order_cancel: ["orderId"],
      order_modify: ["orderId"],
      product_query: ["productId"],
      return_request: ["orderId", "reason"],
      complaint: ["orderId", "issue"],
    };
    return slotMap[intentType] || [];
  }

  /**
   * 生成询问缺失信息的提示
   */
  private generateQuestion(intent: Intent): string {
    const questions: Record<string, string> = {
      orderId: "请提供您的订单号，通常是以 ORD 开头的编号。",
      productId: "请告诉我您想了解的产品名称或编号。",
      reason: "请说明您要退货的原因。",
      issue: "请详细描述您遇到的问题。",
    };

    const missingQuestions = intent.missingSlots
      .map((slot) => questions[slot])
      .filter(Boolean);

    return missingQuestions.join("\n");
  }
}

// 类型定义
interface Intent {
  type: string;
  confidence: number;
  entities: Record<string, any>;
  requiresAction: boolean;
  missingSlots: string[];
  reason?: string;
}

interface AgentResponse {
  type: "response" | "question" | "handoff";
  message: string;
  intent?: string;
  data?: any;
  missingInfo?: string[];
  metadata?: Record<string, any>;
}
```

#### 意图识别 Agent

```typescript
// lib/agent/intent-agent.ts

import { ChatOpenAI } from "@langchain/openai";
import { PromptTemplate } from "@langchain/core/prompts";
import { StructuredOutputParser } from "langchain/output_parsers";
import { z } from "zod";

/**
 * 意图识别 Agent
 * 分析用户消息，识别意图和实体
 */
export class IntentAgent {
  private llm: ChatOpenAI;
  private parser: StructuredOutputParser<z.infer<typeof intentSchema>>;

  // 意图 Schema
  private static intentSchema = z.object({
    intent: z.enum([
      "order_query", // 订单查询
      "order_cancel", // 取消订单
      "order_modify", // 修改订单
      "product_query", // 产品咨询
      "product_recommend", // 产品推荐
      "return_request", // 退货申请
      "complaint", // 投诉
      "general_question", // 一般问题
      "human_handoff", // 转人工
    ]),
    confidence: z.number().min(0).max(1),
    entities: z.object({
      orderId: z.string().optional(),
      productId: z.string().optional(),
      productName: z.string().optional(),
      reason: z.string().optional(),
      issue: z.string().optional(),
    }),
    requiresAction: z.boolean(),
    missingSlots: z.array(z.string()),
    handoffReason: z.string().optional(),
  });

  constructor(llm: ChatOpenAI) {
    this.llm = llm;
    this.parser = StructuredOutputParser.fromZodSchema(
      IntentAgent.intentSchema
    );
  }

  /**
   * 识别意图
   */
  async recognize(message: string, history: Message[]): Promise<Intent> {
    const prompt = PromptTemplate.fromTemplate(`
你是一个客服意图识别专家。分析用户消息，识别用户意图和相关实体。

对话历史：
{history}

用户消息：{message}

{format_instructions}

注意事项：
1. 订单号通常是 ORD 开头的字符串
2. 如果用户情绪激动或问题复杂，标记为 human_handoff
3. 实体可能是用户直接提供的，也可能需要从上下文推断
4. missingSlots 应该包含执行该意图还需要但缺失的信息
    `);

    const formattedPrompt = await prompt.format({
      message,
      history: history.map((h) => `${h.role}: ${h.content}`).join("\n"),
      format_instructions: this.parser.getFormatInstructions(),
    });

    const response = await this.llm.invoke(formattedPrompt);
    const parsed = await this.parser.parse(response.content as string);

    return {
      type: parsed.intent,
      confidence: parsed.confidence,
      entities: parsed.entities,
      requiresAction: parsed.requiresAction,
      missingSlots: parsed.missingSlots,
      reason: parsed.handoffReason,
    };
  }
}
```

#### 任务执行 Agent

```typescript
// lib/agent/task-agent.ts

import { ChatOpenAI } from "@langchain/openai";
import { orderTool } from "../tools/order-tool";
import { productTool } from "../tools/product-tool";
import { ticketTool } from "../tools/ticket-tool";

/**
 * 任务执行 Agent
 * 根据意图执行相应的操作
 */
export class TaskAgent {
  private llm: ChatOpenAI;
  private tools: Map<string, Tool>;

  constructor(llm: ChatOpenAI) {
    this.llm = llm;
    this.tools = new Map([
      ["order", orderTool],
      ["product", productTool],
      ["ticket", ticketTool],
    ]);
  }

  /**
   * 执行任务
   */
  async execute(intent: Intent): Promise<TaskResult> {
    switch (intent.type) {
      case "order_query":
        return this.queryOrder(intent.entities.orderId);

      case "order_cancel":
        return this.cancelOrder(intent.entities.orderId);

      case "order_modify":
        return this.modifyOrder(intent.entities);

      case "product_query":
        return this.queryProduct(intent.entities);

      case "product_recommend":
        return this.recommendProducts(intent.entities);

      case "return_request":
        return this.createReturnRequest(intent.entities);

      case "complaint":
        return this.createComplaint(intent.entities);

      default:
        return { success: false, error: "未知意图类型" };
    }
  }

  /**
   * 查询订单
   */
  private async queryOrder(orderId?: string): Promise<TaskResult> {
    if (!orderId) {
      return { success: false, error: "缺少订单号" };
    }

    try {
      const order = await orderTool.getOrder(orderId);
      const logistics = await orderTool.getLogistics(orderId);

      return {
        success: true,
        data: {
          order,
          logistics,
        },
      };
    } catch (error) {
      return { success: false, error: "查询订单失败" };
    }
  }

  /**
   * 取消订单
   */
  private async cancelOrder(orderId?: string): Promise<TaskResult> {
    if (!orderId) {
      return { success: false, error: "缺少订单号" };
    }

    try {
      // 检查订单状态
      const order = await orderTool.getOrder(orderId);

      if (!["pending", "confirmed"].includes(order.status)) {
        return {
          success: false,
          error: "该订单已发货，无法取消。如需退货，请申请退货。",
        };
      }

      // 执行取消
      await orderTool.cancelOrder(orderId);

      return {
        success: true,
        data: { message: "订单已成功取消，退款将在 3-5 个工作日内到账。" },
      };
    } catch (error) {
      return { success: false, error: "取消订单失败" };
    }
  }

  /**
   * 创建退货申请
   */
  private async createReturnRequest(entities: any): Promise<TaskResult> {
    try {
      const ticket = await ticketTool.createTicket({
        type: "return",
        orderId: entities.orderId,
        reason: entities.reason,
        status: "pending",
      });

      return {
        success: true,
        data: {
          ticketId: ticket.id,
          message: "退货申请已提交，我们将在 24 小时内处理。",
        },
      };
    } catch (error) {
      return { success: false, error: "创建退货申请失败" };
    }
  }
}

// 类型定义
interface TaskResult {
  success: boolean;
  data?: any;
  error?: string;
}

interface Tool {
  [key: string]: (...args: any[]) => Promise<any>;
}
```

#### 回复生成 Agent

```typescript
// lib/agent/response-agent.ts

import { ChatOpenAI } from "@langchain/openai";
import { PromptTemplate } from "@langchain/core/prompts";

/**
 * 回复生成 Agent
 * 根据意图和执行结果生成友好的回复
 */
export class ResponseAgent {
  private llm: ChatOpenAI;

  constructor(llm: ChatOpenAI) {
    this.llm = llm;
  }

  /**
   * 生成回复
   */
  async generate(
    userMessage: string,
    intent: Intent,
    taskResult: TaskResult | undefined,
    history: Message[]
  ): Promise<{ message: string }> {
    // 构建回复模板
    const prompt = PromptTemplate.fromTemplate(`
你是一个友好、专业的客服助手。请根据以下信息生成回复。

用户消息：{userMessage}
识别意图：{intent}
执行结果：{taskResult}
对话历史：{history}

回复要求：
1. 语气友好、专业
2. 直接回答用户问题，不要重复
3. 如果有数据，以清晰的方式呈现
4. 如果需要后续操作，给出明确指引
5. 回复要简洁，不超过 200 字
    `);

    const formattedPrompt = await prompt.format({
      userMessage,
      intent: JSON.stringify(intent, null, 2),
      taskResult: JSON.stringify(taskResult, null, 2),
      history: history
        .slice(-3)
        .map((h) => `${h.role}: ${h.content}`)
        .join("\n"),
    });

    const response = await this.llm.invoke(formattedPrompt);

    return {
      message: response.content as string,
    };
  }

  /**
   * 根据意图生成快速回复建议
   */
  generateQuickReplies(intent: string): string[] {
    const quickReplies: Record<string, string[]> = {
      order_query: ["查看物流详情", "申请售后", "联系商家"],
      product_query: ["查看库存", "查看评价", "加入购物车"],
      return_request: ["查看退货政策", "联系客服", "取消申请"],
    };

    return quickReplies[intent] || ["转人工客服", "返回首页"];
  }
}
```

```typescript
// lib/tools/order-tool.ts

import { prisma } from "@/lib/db";

/**
 * 订单工具
 * 处理订单相关的操作
 */
export const orderTool = {
  /**
   * 获取订单信息
   */
  async getOrder(orderId: string) {
    const order = await prisma.order.findUnique({
      where: { id: orderId },
      include: {
        items: {
          include: {
            product: true,
          },
        },
      },
    });

    if (!order) {
      throw new Error("订单不存在");
    }

    return {
      id: order.id,
      status: order.status,
      totalAmount: order.totalAmount,
      createdAt: order.createdAt,
      items: order.items.map((item) => ({
        name: item.product.name,
        quantity: item.quantity,
        price: item.price,
      })),
    };
  },

  /**
   * 获取物流信息
   */
  async getLogistics(orderId: string) {
    // 模拟物流 API 调用
    const logistics = await prisma.logistics.findFirst({
      where: { orderId },
    });

    return logistics
      ? {
          status: logistics.status,
          trackingNumber: logistics.trackingNumber,
          carrier: logistics.carrier,
          estimatedDelivery: logistics.estimatedDelivery,
          timeline: logistics.timeline,
        }
      : null;
  },

  /**
   * 取消订单
   */
  async cancelOrder(orderId: string) {
    const order = await prisma.order.update({
      where: { id: orderId },
      data: {
        status: "cancelled",
        cancelledAt: new Date(),
      },
    });

    // 触发退款流程
    // await refundService.process(order);

    return order;
  },

  /**
   * 修改订单地址
   */
  async modifyAddress(orderId: string, newAddress: Address) {
    // 检查订单状态
    const order = await prisma.order.findUnique({
      where: { id: orderId },
    });

    if (order?.status !== "pending") {
      throw new Error("订单已发货，无法修改地址");
    }

    return prisma.order.update({
      where: { id: orderId },
      data: {
        shippingAddress: newAddress,
      },
    });
  },
};
```

```typescript
import { NextRequest, NextResponse } from "next/server";
import { AgentOrchestrator } from "@/lib/agent/orchestrator";

const orchestrator = new AgentOrchestrator();

export async function POST(request: NextRequest) {
  try {
    const { message, sessionId } = await request.json();

    if (!message || !sessionId) {
      return NextResponse.json({ error: "缺少必要参数" }, { status: 400 });

}
```

```typescript
/**

// 1. 响应缓存
const responseCache = new Map<string, CachedResponse>();

async function getCachedOrFetch(message: string, sessionId: string) {
  const cacheKey = `${sessionId}:${message}`;

  if (responseCache.has(cacheKey)) {
    const cached = responseCache.get(cacheKey)!;
    if (Date.now() - cached.timestamp < 300000) {
      // 5分钟缓存
      return cached.response;
    }
  }

  const response = await orchestrator.handleMessage(message, sessionId);
  responseCache.set(cacheKey, { response, timestamp: Date.now() });

  return response;
}

// 2. 并行处理
async function parallelProcessing(intent: Intent) {
  // 并行执行多个任务
  const [orderInfo, productInfo, knowledgeBaseResult] = await Promise.all([
    intent.entities.orderId
      ? orderTool.getOrder(intent.entities.orderId)
      : null,
    intent.entities.productId
      ? productTool.getProduct(intent.entities.productId)
      : null,
    ragRetriever.search(intent.originalMessage),
  ]);

  return { orderInfo, productInfo, knowledgeBaseResult };
}

// 3. 流式响应
async function* streamResponse(message: string, sessionId: string) {
  // 先返回意图识别结果
  const intent = await intentAgent.recognize(message);
  yield { type: "intent", data: intent };

  // 执行任务
  const result = await taskAgent.execute(intent);
  yield { type: "result", data: result };

  // 生成回复（流式）
  for await (const chunk of responseAgent.streamGenerate(intent, result)) {
    yield { type: "text", data: chunk };
  }
}
```
