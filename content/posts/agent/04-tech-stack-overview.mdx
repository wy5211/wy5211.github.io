### 2.2 LLM API 调用基础

```typescript
/**
 * OpenAI API 调用示例
 */

import OpenAI from "openai";

// 初始化客户端
const client = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY,
});

// 基础对话
async function basicChat(message: string) {
  const response = await client.chat.completions.create({
    model: "gpt-4o",
    messages: [
      { role: "system", content: "你是一个有帮助的助手。" },
      { role: "user", content: message },
    ],
    temperature: 0.7, // 创造性程度 (0-2)
    max_tokens: 1000, // 最大输出长度
  });

  return response.choices[0].message.content;
}

// Function Calling（工具调用）
async function chatWithTools(message: string) {
  // 定义可用工具
  const tools = [
    {
      type: "function" as const,
      function: {
        name: "get_weather",
        description: "获取指定城市的天气信息",
        parameters: {
          type: "object",
          properties: {
            city: {
              type: "string",
              description: "城市名称",
            },
          },
          required: ["city"],
        },
      },
    },
  ];

  const response = await client.chat.completions.create({
    model: "gpt-4o",
    messages: [{ role: "user", content: message }],
    tools,
    tool_choice: "auto", // 让模型决定是否使用工具
  });

  const message_response = response.choices[0].message;

  // 检查是否需要调用工具
  if (message_response.tool_calls) {
    const toolCall = message_response.tool_calls[0];
    const args = JSON.parse(toolCall.function.arguments);

    // 执行工具
    const result = await getWeather(args.city);

    // 将工具结果返回给模型
    const finalResponse = await client.chat.completions.create({
      model: "gpt-4o",
      messages: [
        { role: "user", content: message },
        message_response,
        {
          role: "tool",
          tool_call_id: toolCall.id,
          content: JSON.stringify(result),
        },
      ],
    });

    return finalResponse.choices[0].message.content;
  }

  return message_response.content;
}

// 辅助函数：获取天气
async function getWeather(city: string) {
  // 实际应用中调用天气 API
  return {
    city,
    temperature: 25,
    condition: "晴天",
  };
}
```

```typescript
async function claudeWithTools(message: string) {

    messages: [{ role: "user", content: message }],
    tools: [
      {
        name: "get_weather",
        description: "获取指定城市的天气信息",
        input_schema: {
          type: "object",
          properties: {
            city: {
              type: "string",
              description: "城市名称",
            },
          },
          required: ["city"],
        },
      },
    ],
  });

  // 处理工具调用
  for (const block of response.content) {
    if (block.type === "tool_use") {
      const result = await getWeather(block.input.city);

      // 继续对话，返回工具结果
      const finalResponse = await anthropic.messages.create({
        model: "claude-sonnet-4-6-20250514",
        max_tokens: 1024,
        messages: [
          { role: "user", content: message },
          { role: "assistant", content: response.content },
          {
            role: "user",
            content: [
              {
                type: "tool_result",
                tool_use_id: block.id,
                content: JSON.stringify(result),
              },
            ],
          },
        ],
      });

      return finalResponse;
    }
  }

  return response;
}
```

```typescript
import { ChatOpenAI } from "@langchain/openai";

import { StringOutputParser } from "@langchain/core/output_parsers";

  modelName: "gpt-4o",
  temperature: 0.7,
});

// 执行链
```

```typescript
  new DynamicTool({
    name: "search",
    description: "搜索互联网获取信息。输入应该是搜索关键词。",
    func: async (input: string) => {
      // 实际应用中调用搜索 API
      return `关于"${input}"的搜索结果...`;
    },

];

  modelName: "gpt-4o",
  temperature: 0,
```

## 四、RAG 技术：给 Agent 知识

### 4.1 RAG 是什么

**RAG（Retrieval-Augmented Generation，检索增强生成）** 是让 Agent 能够利用外部知识的技术。

```
传统 LLM：
用户问题 → LLM → 回答
（只能基于训练数据回答）

RAG 增强：
用户问题 → 检索相关文档 → 组合上下文 → LLM → 回答
（可以基于最新、私有数据回答）
```

### 4.3 RAG 代码实现

```typescript
/**
 * RAG 系统实现
 */

import { OpenAI } from "openai";
import { Pinecone } from "@pinecone-database/pinecone";

// 初始化
const openai = new OpenAI();
const pinecone = new Pinecone({
  apiKey: process.env.PINECONE_API_KEY!,
});

// 1. 文档向量化与存储
async function indexDocuments(documents: string[]) {
  const index = pinecone.index("my-knowledge-base");

  for (const doc of documents) {
    // 生成向量
    const embedding = await createEmbedding(doc);

    // 存储到向量数据库
    await index.upsert([
      {
        id: generateId(),
        values: embedding,
        metadata: { text: doc },
      },
    ]);
  }
}

// 2. 生成向量
async function createEmbedding(text: string): Promise<number[]> {
  const response = await openai.embeddings.create({
    model: "text-embedding-3-small",
    input: text,
  });

  return response.data[0].embedding;
}

// 3. 相似度检索
async function retrieveRelevantDocs(query: string, topK: number = 3) {
  const index = pinecone.index("my-knowledge-base");

  // 查询向量化
  const queryEmbedding = await createEmbedding(query);

  // 向量检索
  const results = await index.query({
    vector: queryEmbedding,
    topK,
    includeMetadata: true,
  });

  return results.matches.map((match) => ({
    text: match.metadata?.text as string,
    score: match.score,
  }));
}

// 4. RAG 查询
async function ragQuery(question: string) {
  // 检索相关文档
  const relevantDocs = await retrieveRelevantDocs(question);

  // 构建上下文
  const context = relevantDocs
    .map((doc, i) => `[文档${i + 1}] ${doc.text}`)
    .join("\n\n");

  // 生成回答
  const response = await openai.chat.completions.create({
    model: "gpt-4o",
    messages: [
      {
        role: "system",
        content: `你是一个助手，请根据以下知识库内容回答问题。
如果知识库中没有相关信息，请诚实地说不知道。

知识库内容：
${context}`,
      },
      { role: "user", content: question },
    ],
  });

  return {
    answer: response.choices[0].message.content,
    sources: relevantDocs,
  };
}

// 使用示例
const result = await ragQuery("公司的请假政策是什么？");
console.log("回答：", result.answer);
console.log("参考来源：", result.sources);
```

```typescript
        cloud: 'aws',
        region: 'us-east-1'
      }
    }
  });
}

// 使用索引
async function useIndex() {
  const index = pc.index('my-index');

  // 插入向量
  await index.upsert([
    {
      id: 'doc-1',
      values: [0.1, 0.2, ...],  // 1536 维向量
      metadata: {
        title: '文档标题',
        content: '文档内容...',
        category: '技术'
      }
    }
  ]);

  // 查询
  const results = await index.query({
    vector: [0.1, 0.2, ...],  // 查询向量
    topK: 10,
    includeMetadata: true,
    filter: {
      category: { $eq: '技术' }  // 元数据过滤
    }
  });

  return results;
}
```

### 6.2 CrewAI

```python
# CrewAI 多 Agent 协作

from crewai import Agent, Task, Crew

# 定义 Agent
researcher = Agent(
    role='研究员',
    goal='收集和分析信息',
    backstory='你是一位经验丰富的研究员',
    allow_delegation=False
)

writer = Agent(
    role='作家',
    goal='撰写高质量文章',
    backstory='你是一位专业的内容创作者',
    allow_delegation=True
)

# 定义任务
research_task = Task(
    description='研究 AI Agent 的最新发展',
    agent=researcher
)

write_task = Task(
    description='根据研究结果撰写一篇文章',
    agent=writer
)

# 组建团队
crew = Crew(
    agents=[researcher, writer],
    tasks=[research_task, write_task]
)

# 执行
result = crew.kickoff()
```
