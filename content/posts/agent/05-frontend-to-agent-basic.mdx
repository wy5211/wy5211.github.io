### 2.3 多轮对话

```typescript
/**
 * 多轮对话：保持对话上下文
 */

class ChatSession {
  private messages: Array<{ role: string; content: string }> = [];

  constructor(private systemPrompt: string) {
    this.messages.push({
      role: "system",
      content: systemPrompt,
    });
  }

  async chat(userMessage: string): Promise<string> {
    // 添加用户消息
    this.messages.push({
      role: "user",
      content: userMessage,
    });

    // 调用 API
    const response = await openai.chat.completions.create({
      model: "gpt-4o-mini",
      messages: this.messages,
    });

    const assistantMessage = response.choices[0].message.content;

    // 添加助手消息到历史
    this.messages.push({
      role: "assistant",
      content: assistantMessage,
    });

    return assistantMessage;
  }

  // 查看对话历史
  getHistory() {
    return this.messages;
  }
}

// 使用示例
const session = new ChatSession("你是一个前端技术面试官");

const q1 = await session.chat("你好，我来面试前端岗位");
console.log(q1); // 欢迎语

const q2 = await session.chat("我有 3 年 React 开发经验");
console.log(q2); // 根据上下文回应

const q3 = await session.chat("请问我刚才说我有什么经验？");
console.log(q3); // 能记住上下文
```

### 3.4 添加工具调用

```typescript
/**
 * 工具调用：让 AI 能执行操作
 */

import { openai } from "@ai-sdk/openai";
import { generateText, tool } from "ai";
import { z } from "zod"; // 用于参数验证

async function agentWithTools(userMessage: string) {
  const { text, toolCalls, toolResults } = await generateText({
    model: openai("gpt-4o-mini"),
    tools: {
      // 工具1：获取天气
      getWeather: tool({
        description: "获取指定城市的当前天气",
        parameters: z.object({
          city: z.string().describe("城市名称"),
        }),
        execute: async ({ city }) => {
          // 模拟天气 API 调用
          const weatherData: Record<string, any> = {
            北京: { temp: 22, condition: "晴天" },
            上海: { temp: 25, condition: "多云" },
            广州: { temp: 30, condition: "小雨" },
          };

          return weatherData[city] || { temp: 20, condition: "未知" };
        },
      }),

      // 工具2：计算器
      calculate: tool({
        description: "计算数学表达式",
        parameters: z.object({
          expression: z.string().describe('数学表达式，如 "2 + 2"'),
        }),
        execute: async ({ expression }) => {
          // 注意：生产环境应使用安全的表达式解析器
          try {
            const result = eval(expression);
            return { result };
          } catch {
            return { error: "无法计算" };
          }
        },
      }),
    },
    maxSteps: 5, // 最多调用 5 次工具
    prompt: userMessage,
  });

  return {
    text,
    toolCalls,
    toolResults,
  };
}

// 使用示例
const result1 = await agentWithTools("北京今天天气怎么样？");
console.log(result1.text);
// 输出：北京今天天气晴朗，气温 22 度。

const result2 = await agentWithTools("帮我算一下 (25 + 17) * 2");
console.log(result2.text);
// 输出：(25 + 17) * 2 = 84
```

```tsx
// components/TodoAgent.tsx
"use client";

import { useState, useEffect } from "react";

interface Task {
  id: string;
  title: string;
  description?: string;
  dueDate?: string;
  priority: "high" | "medium" | "low";
  category: string;
  completed: boolean;
  createdAt: string;
}

export default function TodoAgent() {
  const [tasks, setTasks] = useState<Task[]>([]);
  const [input, setInput] = useState("");
  const [isLoading, setIsLoading] = useState(false);

  // 从 localStorage 加载任务
  useEffect(() => {
    const saved = localStorage.getItem("tasks");
    if (saved) {
      setTasks(JSON.parse(saved));
    }
  }, []);

  // 保存到 localStorage
  useEffect(() => {
    localStorage.setItem("tasks", JSON.stringify(tasks));
  }, [tasks]);

  // 添加任务
  const addTask = async () => {
    if (!input.trim()) return;

    setIsLoading(true);

    try {
      // 调用 AI 解析任务
      const response = await fetch("/api/parse-task", {
        method: "POST",
        body: JSON.stringify({ input }),
      });

      const taskData = await response.json();

      // 创建新任务
      const newTask: Task = {
        id: Date.now().toString(),
        ...taskData,
        completed: false,
        createdAt: new Date().toISOString(),
      };

      setTasks((prev) => [newTask, ...prev]);
      setInput("");
    } catch (error) {
      console.error("添加任务失败:", error);
    } finally {
      setIsLoading(false);
    }
  };

  // 切换完成状态
  const toggleComplete = (id: string) => {
    setTasks((prev) =>
      prev.map((task) =>
        task.id === id ? { ...task, completed: !task.completed } : task
      )
    );
  };

  // 删除任务
  const deleteTask = (id: string) => {
    setTasks((prev) => prev.filter((task) => task.id !== id));
  };

  const priorityColors = {
    high: "bg-red-100 text-red-800",
    medium: "bg-yellow-100 text-yellow-800",
    low: "bg-green-100 text-green-800",
  };

  return (
    <div className="max-w-2xl mx-auto p-4">
      <h1 className="text-2xl font-bold mb-6">智能 TODO</h1>

      {/* 输入区域 */}
      <div className="flex gap-2 mb-6">
        <input
          value={input}
          onChange={(e) => setInput(e.target.value)}
          onKeyDown={(e) => e.key === "Enter" && addTask()}
          placeholder="用自然语言描述任务，如：明天下午3点开会讨论项目进度"
          className="flex-1 p-3 border rounded-lg"
          disabled={isLoading}
        />
        <button
          onClick={addTask}
          disabled={isLoading || !input.trim()}
          className="px-4 py-2 bg-blue-500 text-white rounded-lg disabled:opacity-50"
        >
          {isLoading ? "添加中..." : "添加"}
        </button>
      </div>

      {/* 示例提示 */}
      <div className="mb-6 p-3 bg-gray-50 rounded-lg text-sm text-gray-600">
        <p className="font-medium mb-1">试试这样说：</p>
        <ul className="space-y-1">
          <li>"下周一前完成项目报告，高优先级"</li>
          <li>"每天早上9点站会"</li>
          <li>"这周末之前看完《深入理解计算机系统》第3章"</li>
        </ul>
      </div>

      {/* 任务列表 */}
      <div className="space-y-3">
        {tasks.map((task) => (
          <div
            key={task.id}
            className={`p-4 border rounded-lg ${
              task.completed ? "bg-gray-50 opacity-60" : "bg-white"
            }`}
          >
            <div className="flex items-start justify-between">
              <div className="flex items-start gap-3">
                <input
                  type="checkbox"
                  checked={task.completed}
                  onChange={() => toggleComplete(task.id)}
                  className="mt-1"
                />
                <div>
                  <h3
                    className={`font-medium ${task.completed ? "line-through" : ""}`}
                  >
                    {task.title}
                  </h3>
                  {task.description && (
                    <p className="text-sm text-gray-600 mt-1">
                      {task.description}
                    </p>
                  )}
                  <div className="flex gap-2 mt-2">
                    <span
                      className={`text-xs px-2 py-1 rounded ${priorityColors[task.priority]}`}
                    >
                      {task.priority === "high"
                        ? "高"
                        : task.priority === "medium"
                          ? "中"
                          : "低"}
                    </span>
                    <span className="text-xs px-2 py-1 rounded bg-gray-100">
                      {task.category}
                    </span>
                    {task.dueDate && (
                      <span className="text-xs text-gray-500">
                        截止：{new Date(task.dueDate).toLocaleDateString()}
                      </span>
                    )}
                  </div>
                </div>
              </div>
              <button
                onClick={() => deleteTask(task.id)}
                className="text-gray-400 hover:text-red-500"
              >
                删除
              </button>
            </div>
          </div>
        ))}

        {tasks.length === 0 && (
          <div className="text-center text-gray-400 py-8">
            暂无任务，添加一个试试！
          </div>
        )}
      </div>
    </div>
  );
}
```
