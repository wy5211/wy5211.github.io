## 单元测试

### 测试工具函数

```typescript
// lib/utils.ts
export function formatDate(date: Date): string {
  return date.toLocaleDateString("zh-CN");
}

export function slugify(text: string): string {
  return text
    .toLowerCase()
    .trim()
    .replace(/\s+/g, "-")
    .replace(/[^\w\-]+/g, "");
}
```

```typescript
// lib/utils.test.ts
import { describe, it, expect } from "vitest";
import { formatDate, slugify } from "./utils";

describe("formatDate", () => {
  it("应该正确格式化日期", () => {
    const date = new Date("2026-03-23");
    expect(formatDate(date)).toBe("2026/3/23");
  });

  it("应该处理不同时区的日期", () => {
    const date = new Date("2026-03-23T00:00:00Z");
    const result = formatDate(date);
    expect(result).toBeTruthy();
  });
});

describe("slugify", () => {
  it("应该将文本转换为 slug", () => {
    expect(slugify("Hello World")).toBe("hello-world");
    expect(slugify("Next.js 是一个框架")).toBe("nextjs-是一个框架");
  });

  it("应该移除特殊字符", () => {
    expect(slugify("Hello @#$% World")).toBe("hello-world");
  });

  it("应该处理多个空格", () => {
    expect(slugify("Hello    World")).toBe("hello-world");
  });
});
```

### 测试 React Hooks

```typescript
// hooks/useCounter.ts
import { useState } from "react";

export function useCounter(initialValue = 0) {
  const [count, setCount] = useState(initialValue);

  const increment = () => setCount((prev) => prev + 1);
  const decrement = () => setCount((prev) => prev - 1);
  const reset = () => setCount(initialValue);

  return { count, increment, decrement, reset };
}
```

```typescript
// hooks/useCounter.test.ts
import { describe, it, expect } from "vitest";
import { renderHook, act } from "@testing-library/react";
import { useCounter } from "./useCounter";

describe("useCounter", () => {
  it("应该使用初始值", () => {
    const { result } = renderHook(() => useCounter(5));
    expect(result.current.count).toBe(5);
  });

  it("应该增加计数", () => {
    const { result } = renderHook(() => useCounter());

    act(() => {
      result.current.increment();
    });

    expect(result.current.count).toBe(1);
  });

  it("应该减少计数", () => {
    const { result } = renderHook(() => useCounter());

    act(() => {
      result.current.decrement();
    });

    expect(result.current.count).toBe(-1);
  });

  it("应该重置计数", () => {
    const { result } = renderHook(() => useCounter(10));

    act(() => {
      result.current.increment();
      result.current.reset();
    });

    expect(result.current.count).toBe(10);
  });
});
```

## Server Actions 测试

### 测试 Server Actions

```typescript
// app/actions.test.ts
import { describe, it, expect, vi, beforeEach } from "vitest";
import { createPost } from "./actions";
import { db } from "@/lib/db";

vi.mock("@/lib/db", () => ({
  db: {
    posts: {
      create: vi.fn(),
    },
  },
}));

describe("createPost", () => {
  beforeEach(() => {
    vi.clearAllMocks();
  });

  it("应该创建文章", async () => {
    const mockPost = {
      id: "1",
      title: "测试文章",
      content: "文章内容",
    };

    vi.mocked(db.posts.create).mockResolvedValue(mockPost);

    const formData = new FormData();
    formData.set("title", "测试文章");
    formData.set("content", "文章内容");

    const result = await createPost(formData);

    expect(result.success).toBe(true);
    expect(result.post).toEqual(mockPost);
    expect(db.posts.create).toHaveBeenCalledWith({
      data: {
        title: "测试文章",
        content: "文章内容",
      },
    });
  });

  it("应该处理验证错误", async () => {
    const formData = new FormData();
    formData.set("title", "");
    formData.set("content", "");

    const result = await createPost(formData);

    expect(result.error).toBeTruthy();
    expect(db.posts.create).not.toHaveBeenCalled();
  });
});
```

## API Routes 测试

### 测试 API 端点

```typescript
// app/api/posts/route.test.ts
import { describe, it, expect, vi, beforeEach } from "vitest";
import { POST } from "./route";
import { db } from "@/lib/db";

vi.mock("@/lib/db", () => ({
  db: {
    posts: {
      create: vi.fn(),
    },
  },
}));

describe("POST /api/posts", () => {
  beforeEach(() => {
    vi.clearAllMocks();
  });

  it("应该创建文章", async () => {
    const mockPost = {
      id: "1",
      title: "测试文章",
      content: "文章内容",
    };

    vi.mocked(db.posts.create).mockResolvedValue(mockPost);

    const request = new Request("http://localhost:3000/api/posts", {
      method: "POST",
      body: JSON.stringify({
        title: "测试文章",
        content: "文章内容",
      }),
    });

    const response = await POST(request);
    const data = await response.json();

    expect(response.status).toBe(201);
    expect(data).toEqual(mockPost);
  });

  it("应该处理验证错误", async () => {
    const request = new Request("http://localhost:3000/api/posts", {
      method: "POST",
      body: JSON.stringify({
        title: "",
        content: "",
      }),
    });

    const response = await POST(request);
    const data = await response.json();

    expect(response.status).toBe(400);
    expect(data.error).toBeTruthy();
  });
});
```

## E2E 测试

### Playwright 测试

```typescript
// e2e/blog.spec.ts
import { test, expect } from "@playwright/test";

test.describe("博客功能", () => {
  test("应该显示文章列表", async ({ page }) => {
    await page.goto("/blog");

    // 等待页面加载
    await expect(page.locator("h1")).toContainText("博客");

    // 验证文章列表
    const articles = page.locator("article");
    await expect(articles).toHaveCount(3);
  });

  test("应该能够阅读文章", async ({ page }) => {
    await page.goto("/blog");

    // 点击第一篇文章
    await page.locator("article").first().click();

    // 验证导航到文章页
    await expect(page).toHaveURL(/\/blog\/.+/);
    await expect(page.locator("h1")).toBeVisible();
  });

  test("应该能够搜索文章", async ({ page }) => {
    await page.goto("/blog");

    // 输入搜索关键词
    await page.fill('input[name="search"]', "Next.js");
    await page.press('input[name="search"]', "Enter");

    // 验证搜索结果
    await expect(page.locator("article")).toHaveCount(1);
  });
});
```

test.describe("用户认证流程", () => {

    // 退出登录
    await page.click('button:has-text("退出")');

    // 验证重定向到首页
    await expect(page).toHaveURL("/");
  });
});

## 测试最佳实践

### ✅ 推荐做法

1. **测试行为而非实现**

```typescript
// ✅ 好的测试
it('应该显示用户名', () => {
  render(<UserProfile user={{ name: 'John' }} />);
  expect(screen.getByText('John')).toBeInTheDocument();
});

// ❌ 不好的测试
it('应该使用 h1 标签显示用户名', () => {
  render(<UserProfile user={{ name: 'John' }} />);
  expect(screen.getByRole('heading', { level: 1 })).toHaveTextContent('John');
});
```

2. **使用描述性的测试名称**

```typescript
// ✅ 好的测试名称
it("当用户点击购买按钮时，应该显示确认对话框");

// ❌ 不好的测试名称
it("测试购买功能");
```

3. **保持测试独立**

```typescript
// ✅ 每个测试独立运行
it("应该创建文章", async () => {
  const formData = new FormData();
  // ...
});

it("应该验证文章标题", async () => {
  const formData = new FormData();
  // ...
});
```

4. **使用测试数据工厂**

```typescript
// lib/test-factories.ts
export function createPost(overrides = {}) {
  return {
    id: "1",
    title: "测试文章",
    content: "文章内容",
    ...overrides,
  };
}
```

### ❌ 避免做法

1. **不要测试第三方库**

```typescript
// ❌ 不需要测试
it("useState 应该工作", () => {
  const [count, setCount] = useState(0);
  setCount(1);
  expect(count).toBe(1);
});
```

2. **不要过度测试**

```typescript
// ❌ 过度测试
it('按钮应该有 type="submit" 属性', () => {
  render(<button type="submit">提交</button>);
  expect(screen.getByRole('button')).toHaveAttribute('type', 'submit');
});
```

3. **不要忽略测试清理**

```typescript
// ✅ 正确的清理
afterEach(() => {
  cleanup();
  vi.clearAllMocks();
});
```

## 常见问题

**Q: 单元测试、集成测试、E2E 测试有什么区别？**

A:

- **单元测试**：测试独立的函数或组件
- **集成测试**：测试多个组件的交互
- **E2E 测试**：测试完整的用户流程

**Q: 如何测试需要认证的页面？**

A: 在 E2E 测试中模拟用户登录：

```typescript
test.beforeEach(async ({ page }) => {
  await page.goto("/login");
  await page.fill('input[name="email"]', "test@example.com");
  await page.fill('input[name="password"]', "password");
  await page.click('button[type="submit"]');
  await page.waitForURL("/dashboard");
});
```

**Q: 如何测试异步操作？**

A: 使用 async/await 和 waitFor：

```typescript
it('应该加载数据', async () => {
  render(<UserProfile userId="123" />);

  await waitFor(() => {
    expect(screen.getByText('用户名')).toBeInTheDocument();
  });
});
```
