在全栈应用中，数据验证和类型安全是确保应用稳定性和安全性的关键。Next.js 结合 TypeScript 和 Zod，提供了强大的端到端类型安全保障。

**学习目标**：

- ✅ 掌握 Zod 验证库的使用
- ✅ 实现 Server Actions 数据验证
- ✅ 实现 API Routes 数据验证
- ✅ 达到端到端类型安全
- ✅ 学习错误处理模式

```typescript
// 可选字段

// 可空字段

// 转换
```

## Server Actions 验证

### 基础验证

```typescript
// app/actions.ts
"use server";

import { z } from "zod";
import { db } from "@/lib/db";
import { revalidatePath } from "next/cache";

// 定义 Schema
const createPostSchema = z.object({
  title: z.string().min(5).max(100),
  content: z.string().min(50),
  category: z.enum(["tech", "life", "tutorial"]),
});

export async function createPost(formData: FormData) {
  // 提取数据
  const rawData = {
    title: formData.get("title"),
    content: formData.get("content"),
    category: formData.get("category"),
  };

  // 验证
  const result = createPostSchema.safeParse(rawData);

  if (!result.success) {
    return {
      error: "验证失败",
      fieldErrors: result.error.flatten().fieldErrors,
    };
  }

  // 保存到数据库
  const post = await db.posts.create({
    data: result.data,
  });

  revalidatePath("/blog");

  return { success: true, post };
}
```

### 类型推断

```typescript
// app/actions.ts

const createPostSchema = z.object({

//   content: string;

export async function createPost(data: CreatePostInput) {
  // data 是完全类型安全的
  const post = await db.posts.create({ data });
  return post;
```

```typescript
const initialState = {

          name="content"

        {state.fieldErrors?.content && (
          <span className="error">
            {state.fieldErrors.content[0]}
          </span>
        )}
      </div>

      <button type="submit">发布</button>

      {state.error && !state.fieldErrors && (
        <div className="error">
          {state.error}
        </div>
      )}
    </form>
  );
}
```

## API Routes 验证

### 请求体验证

```typescript
// app/api/posts/route.ts
import { NextResponse } from "next/server";
import { z } from "zod";
import { db } from "@/lib/db";

const createPostSchema = z.object({
  title: z.string().min(5).max(100),
  content: z.string().min(50),
  category: z.enum(["tech", "life", "tutorial"]),
  tags: z.array(z.string()).min(1).max(5),
});

export async function POST(request: Request) {
  try {
    const body = await request.json();

    // 验证
    const result = createPostSchema.safeParse(body);

    if (!result.success) {
      return NextResponse.json(
        {
          error: "验证失败",
          details: result.error.flatten(),
        },
        { status: 400 }
      );
    }

    // 创建文章
    const post = await db.posts.create({
      data: result.data,
    });

    return NextResponse.json(post, { status: 201 });
  } catch (error) {
    return NextResponse.json({ error: "服务器错误" }, { status: 500 });
  }
}
```

### 查询参数验证

```typescript
// app/api/posts/route.ts
const querySchema = z.object({
  page: z.coerce.number().min(1).default(1),
  limit: z.coerce.number().min(1).max(100).default(10),
  category: z.enum(["tech", "life", "tutorial"]).optional(),
  search: z.string().optional(),
});

export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);

  // 验证查询参数
  const query = querySchema.safeParse(Object.fromEntries(searchParams));

  if (!query.success) {
    return NextResponse.json(
      { error: "无效的查询参数", details: query.error.flatten() },
      { status: 400 }
    );
  }

  const { page, limit, category, search } = query.data;

  // 构建查询
  const where = {
    ...(category && { category }),
    ...(search && {
      OR: [{ title: { contains: search } }, { content: { contains: search } }],
    }),
  };

  const [posts, total] = await Promise.all([
    db.posts.findMany({
      where,
      take: limit,
      skip: (page - 1) * limit,
      orderBy: { createdAt: "desc" },
    }),
    db.posts.count({ where }),
  ]);

  return NextResponse.json({
    posts,
    pagination: {
      page,
      limit,
      total,
      totalPages: Math.ceil(total / limit),
    },
  });
}
```

## 端到端类型安全

### 共享类型

```typescript
// types/post.ts
import { z } from "zod";

export const postSchema = z.object({
  id: z.string().uuid(),
  title: z.string(),
  content: z.string(),
  excerpt: z.string().nullable(),
  category: z.enum(["tech", "life", "tutorial"]),
  tags: z.array(z.string()),
  published: z.boolean(),
  createdAt: z.date(),
  updatedAt: z.date(),
});

// 推断类型
export type Post = z.infer<typeof postSchema>;

// 输入类型
export const createPostInputSchema = postSchema.omit({
  id: true,
  createdAt: true,
  updatedAt: true,
});

export type CreatePostInput = z.infer<typeof createPostInputSchema>;

// 更新类型
export const updatePostInputSchema = postSchema.partial().required({
  id: true,
});

export type UpdatePostInput = z.infer<typeof updatePostInputSchema>;
```

### API 路由类型

```typescript
// app/api/posts/route.ts
```

## 错误处理模式

### 统一错误响应

```typescript
// lib/errors.ts
export class ValidationError extends Error {
  constructor(public details: Record<string, string[]>) {
    super("验证失败");
    this.name = "ValidationError";
  }
}

export class NotFoundError extends Error {
  constructor(resource: string) {
    super(`${resource} 未找到`);
    this.name = "NotFoundError";
  }
}

export class UnauthorizedError extends Error {
  constructor(message = "未授权") {
    super(message);
    this.name = "UnauthorizedError";
  }
}
```

### 错误处理中间件

```typescript
// lib/api-handler.ts
import { NextResponse } from "next/server";
import { ZodError } from "zod";
import { ValidationError } from "./errors";

export function withErrorHandler(
  handler: (request: Request) => Promise<NextResponse>
) {
  return async (request: Request) => {
    try {
      return await handler(request);
    } catch (error) {
      // Zod 验证错误
      if (error instanceof ZodError) {
        return NextResponse.json(
          {
            error: "验证失败",
            details: error.flatten(),
          },
          { status: 400 }
        );
      }

      // 自定义验证错误
      if (error instanceof ValidationError) {
        return NextResponse.json(
          {
            error: error.message,
            details: error.details,
          },
          { status: 400 }
        );
      }

      // 未找到错误
      if (error instanceof NotFoundError) {
        return NextResponse.json({ error: error.message }, { status: 404 });
      }

      // 未授权错误
      if (error instanceof UnauthorizedError) {
        return NextResponse.json({ error: error.message }, { status: 401 });
      }

      // 未知错误
      console.error("Unhandled error:", error);
      return NextResponse.json({ error: "服务器错误" }, { status: 500 });
    }
  };
}
```

### 使用错误处理

```typescript
// app/api/posts/[id]/route.ts
import { NextResponse } from "next/server";
import { z } from "zod";
import { db } from "@/lib/db";
import { withErrorHandler } from "@/lib/api-handler";
import { NotFoundError } from "@/lib/errors";

const updatePostSchema = z.object({
  title: z.string().min(5).optional(),
  content: z.string().min(50).optional(),
  category: z.enum(["tech", "life", "tutorial"]).optional(),
});

export const PUT = withErrorHandler(
  async (request: Request, { params }: { params: { id: string } }) => {
    const body = await request.json();
    const result = updatePostSchema.safeParse(body);

    if (!result.success) {
      throw new ZodError(result.error.issues);
    }

    // 检查文章是否存在
    const existingPost = await db.posts.findUnique({
      where: { id: params.id },
    });

    if (!existingPost) {
      throw new NotFoundError("文章");
    }

    // 更新文章
    const post = await db.posts.update({
      where: { id: params.id },
      data: result.data,
    });

    return NextResponse.json(post);
  }
);
```

## 实战：用户注册

```typescript
// lib/schemas.ts
import { z } from "zod";

export const registerSchema = z
  .object({
    email: z
      .string()
      .email("无效的邮箱地址")
      .refine(
        async (email) => {
          const user = await db.user.findUnique({ where: { email } });
          return !user;
        },
        { message: "邮箱已被注册" }
      ),

    username: z
      .string()
      .min(3, "用户名至少需要 3 个字符")
      .max(20, "用户名不能超过 20 个字符")
      .regex(/^[a-zA-Z0-9_]+$/, "用户名只能包含字母、数字和下划线")
      .refine(
        async (username) => {
          const user = await db.user.findUnique({ where: { username } });
          return !user;
        },
        { message: "用户名已被占用" }
      ),

    password: z
      .string()
      .min(8, "密码至少需要 8 个字符")
      .regex(/[A-Z]/, "密码必须包含大写字母")
      .regex(/[a-z]/, "密码必须包含小写字母")
      .regex(/[0-9]/, "密码必须包含数字"),

    confirmPassword: z.string(),
  })
  .refine((data) => data.password === data.confirmPassword, {
    message: "两次密码不一致",
    path: ["confirmPassword"],
  });

export type RegisterInput = z.infer<typeof registerSchema>;
```

```typescript
// app/actions.ts
"use server";

import { registerSchema } from "@/lib/schemas";
import { db } from "@/lib/db";
import { hash } from "bcryptjs";

export async function register(formData: FormData) {
  const rawData = {
    email: formData.get("email"),
    username: formData.get("username"),
    password: formData.get("password"),
    confirmPassword: formData.get("confirmPassword"),
  };

  const result = registerSchema.safeParse(rawData);

  if (!result.success) {
    return {
      error: "验证失败",
      fieldErrors: result.error.flatten().fieldErrors,
    };
  }

  // 创建用户
  const hashedPassword = await hash(result.data.password, 10);

  const user = await db.user.create({
    data: {
      email: result.data.email,
      username: result.data.username,
      password: hashedPassword,
    },
    select: {
      id: true,
      email: true,
      username: true,
    },
  });

  return { success: true, user };
}
```

## 最佳实践

### ✅ 推荐做法

1. **始终验证输入**

```typescript
const result = schema.safeParse(input);
if (!result.success) {
  return { error: result.error.flatten() };
}
```

2. **使用类型推断**

```typescript
type Input = z.infer<typeof schema>;
```

3. **集中管理 Schema**

```typescript
// lib/schemas.ts
export const postSchemas = {
  create: z.object({
    /* ... */
  }),
  update: z.object({
    /* ... */
  }),
  delete: z.object({
    /* ... */
  }),
};
```

4. **提供友好的错误消息**

```typescript
min(5, "标题至少需要 5 个字符");
```

### ❌ 避免做法

1. **不要跳过验证**

```typescript
// ❌ 不安全
export async function createPost(data: any) {
  await db.posts.create({ data });
}

// ✅ 安全
export async function createPost(data: unknown) {
  const result = schema.safeParse(data);
  if (!result.success) {
    return { error: result.error };
  }
  await db.posts.create({ data: result.data });
}
```

2. **不要使用 `any` 类型**

```typescript
// ❌ 错误
function process(data: any) {}

// ✅ 正确
function process(data: z.infer<typeof schema>) {}
```

3. **不要忽略验证错误**

```typescript
// ❌ 忽略错误
schema.parse(input); // 可能抛出异常

// ✅ 处理错误
const result = schema.safeParse(input);
if (!result.success) {
  // 处理错误
}
```

## 常见问题

**Q: Zod 和 TypeScript 类型有什么区别？**

A: TypeScript 类型只在编译时检查，Zod 在运行时验证。使用 Zod 可以确保数据在运行时也是有效的。

**Q: 如何在客户端和服务端共享类型？**

A: 创建共享的类型文件，使用 `z.infer` 推断类型：

```typescript
// types/post.ts
export const postSchema = z.object({
  /* ... */
});
export type Post = z.infer<typeof postSchema>;
```

**Q: 如何处理复杂的验证逻辑？**

A: 使用 `.refine()` 添加自定义验证：

```typescript
.refine(async (val) => {
  // 自定义验证逻辑
  return isValid;
}, { message: '验证失败' });
```
