## 代码分割和懒加载

### 动态导入组件

```typescript
// components/HeavyChart.tsx
"use client";

import { useState } from 'react';

export function HeavyChart() {
  // ... 复杂的图表组件
  return <div>图表内容</div>;
}
```

```typescript
// app/dashboard/page.tsx
import dynamic from 'next/dynamic';

// 懒加载重型组件
const HeavyChart = dynamic(
  () => import('@/components/HeavyChart'),
  {
    loading: () => <p>加载图表中...</p>,
    ssr: false, // 仅客户端渲染
  }
);

export default function DashboardPage() {
  return (
    <div>
      <h1>仪表板</h1>
      <HeavyChart />
    </div>
  );
}
```

### 路由级代码分割

Next.js 默认为每个路由自动进行代码分割：

```
app/
├── page.tsx              → page.js
├── about/
│   └── page.tsx          → about/page.js
└── blog/
    └── page.tsx          → blog/page.js
```

### 条件加载

```typescript
// components/AdminPanel.tsx
"use client";

import dynamic from 'next/dynamic';
import { useSession } from 'next-auth/react';

export function AdminPanel() {
  const { data: session } = useSession();

  // 只有管理员才加载管理面板
  const AdminDashboard = dynamic(
    () => import('./AdminDashboard'),
    {
      loading: () => <p>加载中...</p>,
    }
  );

  if (session?.user?.role !== 'ADMIN') {
    return null;
  }

  return <AdminDashboard />;
}
```

## Bundle 分析

### 使用 @next/bundle-analyzer

```bash
npm install @next/bundle-analyzer
```

```typescript
// next.config.ts
import bundleAnalyzer from "@next/bundle-analyzer";

const nextConfig = {
  experimental: {
    instrumentationHook: true,
  },

```

```bash

```

### 识别大依赖

运行分析后，检查：

1. **页面级别的 Bundle**：每个路由的 JS 大小
2. **共享依赖**：多个页面共享的代码
3. **大依赖库**：考虑替换或优化

### 优化策略

```typescript
// ❌ 不好的做法：导入整个库
import _ from "lodash";

// ✅ 好的做法：只导入需要的函数
import debounce from "lodash/debounce";

// ❌ 不好的做法：导入所有图标
import * as Icons from "some-icon-library";

// ✅ 好的做法：按需导入
import { IconName } from "some-icon-library";
```

## 并行数据获取

### 串行 vs 并行

```typescript
// ❌ 串行：慢
async function Page() {
  const user = await getUser();
  const posts = await getPosts(); // 等待 getUser 完成
  const comments = await getComments(); // 等待 getPosts 完成

  return <div>{/* ... */}</div>;
}

// ✅ 并行：快
async function Page() {
  const [user, posts, comments] = await Promise.all([
    getUser(),
    getPosts(),
    getComments(),
  ]);

  return <div>{/* ... */}</div>;
}
```

### React 并行渲染

```typescript
// app/dashboard/page.tsx
import { Suspense } from 'react';

async function UserStats() {
  const stats = await getUserStats();
  return <div>{/* 统计数据 */}</div>;
}

async function RecentActivity() {
  const activities = await getRecentActivities();
  return <div>{/* 活动列表 */}</div>;
}

export default function DashboardPage() {
  return (
    <div>
      <Suspense fallback={<div>加载统计中...</div>}>
        <UserStats />
      </Suspense>

      <Suspense fallback={<div>加载活动中...</div>}>
        <RecentActivity />
      </Suspense>
    </div>
  );
}
```

## Core Web Vitals 优化

### LCP（Largest Contentful Paint）

```typescript
// 优化 LCP
export default function HomePage() {
  return (
    <>
      {/* 1. 优先加载关键图片 */}
      <Image
        src="/hero.jpg"
        alt="Hero"
        width={1200}
        height={600}
        priority
      />

      {/* 2. 预连接到关键域名 */}
      <link rel="preconnect" href="https://api.example.com" />

      {/* 3. 内联关键 CSS */}
      <style jsx>{`
        .hero {
          /* 关键样式 */
        }
      `}</style>
    </>
  );
}
```

### FID（First Input Delay）

```typescript

const Analytics = dynamic(() => import("@/components/Analytics"), {
  ssr: false,
});

// 2. 使用 Web Workers 处理复杂计算
// lib/worker.ts
export function processLargeData(data: any[]) {
  // 在 Web Worker 中处理
}
```

### CLS（Cumulative Layout Shift）

```typescript
// 防止布局偏移
export function ArticleCard({ post }: { post: Post }) {
  return (
    <div className="article-card">
      {/* 1. 为图片预留空间 */}
      <div className="image-container" style={{ aspectRatio: '16/9' }}>
        <Image
          src={post.image}
          alt={post.title}
          fill
          sizes="(max-width: 768px) 100vw, 50vw"
        />
      </div>

      {/* 2. 骨架屏 */}
      {!post.title && <Skeleton />}

      <h2>{post.title}</h2>
      <p>{post.excerpt}</p>
    </div>
  );
}
```

### 自定义性能追踪

```typescript
// lib/analytics.ts
export function reportWebVitals(metric: any) {
  const { name, value, id } = metric;

  // 发送到分析服务
  fetch("/api/analytics", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      name,
      value: Math.round(name === "CLS" ? value * 1000 : value),
      id,
    }),
  });
}
```

## 实战：优化博客页面

### 优化前

```typescript
// ❌ 未优化的博客页面
import { getAllPosts } from '@/lib/posts';
import { CommentSection } from '@/components/CommentSection';
import { RelatedPosts } from '@/components/RelatedPosts';
import { ShareButtons } from '@/components/ShareButtons';

export default async function BlogPage({ params }: { params: { slug: string } }) {
  // 串行数据获取
  const post = await getPost(params.slug);
  const comments = await getComments(params.slug);
  const related = await getRelatedPosts(post.category);

  return (
    <article>
      <h1>{post.title}</h1>
      <img src={post.image} alt={post.title} />
      <div>{post.content}</div>

      {/* 所有组件都会立即加载 */}
      <CommentSection postId={post.id} />
      <RelatedPosts posts={related} />
      <ShareButtons post={post} />
    </article>
  );
}
```

### 优化后

```typescript
// ✅ 优化的博客页面
import { Suspense } from 'react';
import Image from 'next/image';
import dynamic from 'next/dynamic';

// 懒加载非关键组件
const CommentSection = dynamic(
  () => import('@/components/CommentSection'),
  { loading: () => <div>加载评论中...</div> }
);

const RelatedPosts = dynamic(
  () => import('@/components/RelatedPosts'),
  { loading: () => <div>加载相关文章中...</div> }
);

const ShareButtons = dynamic(
  () => import('@/components/ShareButtons'),
  { ssr: false }
);

async function BlogPost({ slug }: { slug: string }) {
  const post = await getPost(slug);
  return (
    <article>
      <h1>{post.title}</h1>
      <Image
        src={post.image}
        alt={post.title}
        width={1200}
        height={600}
        priority
      />
      <div>{post.content}</div>
    </article>
  );
}

export default async function BlogPage({ params }: { params: { slug: string } }) {
  return (
    <>
      <BlogPost slug={params.slug} />

      {/* Suspense 实现流式渲染 */}
      <Suspense fallback={<div>加载评论中...</div>}>
        <CommentSection postId={params.slug} />
      </Suspense>

      <Suspense fallback={<div>加载相关文章中...</div>}>
        <RelatedPosts slug={params.slug} />
      </Suspense>

      <ShareButtons slug={params.slug} />
    </>
  );
}
```

## 最佳实践

### ✅ 推荐做法

1. **使用动态导入**

```typescript
const HeavyComponent = dynamic(() => import("./HeavyComponent"));
```

2. **优化图片**

```typescript
<Image
  src={src}
  alt={alt}
  width={width}
  height={height}
  loading="lazy"
/>
```

3. **并行数据获取**

```typescript
const [data1, data2] = await Promise.all([fetchData1(), fetchData2()]);
```

4. **使用 Suspense**

```typescript
<Suspense fallback={<Loading />}>
  <AsyncComponent />
</Suspense>
```

### ❌ 避免做法

1. **不要导入整个库**

```typescript
// ❌ 导入整个 lodash
import _ from "lodash";

// ✅ 只导入需要的函数
import debounce from "lodash/debounce";
```

2. **不要忽略图片优化**

```typescript
// ❌ 使用原生 img 标签
<img src="/image.jpg" alt="Image" />

// ✅ 使用 next/image
<Image src="/image.jpg" alt="Image" width={500} height={300} />
```

3. **不要串行数据获取**

```typescript
// ❌ 串行
const data1 = await fetch1();
const data2 = await fetch2();

// ✅ 并行
const [data1, data2] = await Promise.all([fetch1(), fetch2()]);
```

## 常见问题

**Q: 如何选择要懒加载的组件？**

A: 懒加载以下类型的组件：

- 大型组件（>100KB）
- 不在首屏显示的组件
- 低频使用的组件（如模态框、下拉菜单）

**Q: next/image 和原生 img 有什么区别？**

A: next/image 提供：

- 自动响应式
- 懒加载
- 格式优化
- 尺寸优化
- 模糊占位符

**Q: 如何监控 Core Web Vitals？**

A: 使用工具：

- PageSpeed Insights
- Lighthouse
- Vercel Analytics
- 自定义 Web Vitals 追踪
