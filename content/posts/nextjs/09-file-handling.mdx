## 文件上传基础

### 客户端上传表单

```typescript
// components/FileUpload.tsx
"use client";

import { useState } from 'react';
import { useFormStatus } from 'react-dom';

export function FileUpload() {
  const [preview, setPreview] = useState<string>();

  function handleFileChange(e: React.ChangeEvent<HTMLInputElement>) {
    const file = e.target.files?.[0];
    if (!file) return;

    // 创建预览
    const reader = new FileReader();
    reader.onloadend = () => {
      setPreview(reader.result as string);
    };
    reader.readAsDataURL(file);
  }

  return (
    <form action={uploadFile}>
      <input
        type="file"
        name="file"
        onChange={handleFileChange}
        accept="image/*"
      />

      {preview && (
        <img
          src={preview}
          alt="预览"
          style={{ maxWidth: '200px', maxHeight: '200px' }}
        />
      )}

      <SubmitButton />
    </form>
  );
}

function SubmitButton() {
  const { pending } = useFormStatus();

  return (
    <button type="submit" disabled={pending}>
      {pending ? '上传中...' : '上传'}
    </button>
  );
}
```

### Server Action 处理上传

```typescript
// app/actions.ts
"use server";

import { writeFile } from "fs/promises";
import { join } from "path";

export async function uploadFile(formData: FormData) {
  const file = formData.get("file") as File;

  if (!file) {
    return { error: "没有文件" };
  }

  // 验证文件类型
  if (!file.type.startsWith("image/")) {
    return { error: "只能上传图片" };
  }

  // 验证文件大小（5MB）
  const MAX_SIZE = 5 * 1024 * 1024;
  if (file.size > MAX_SIZE) {
    return { error: "文件大小不能超过 5MB" };
  }

  // 读取文件
  const bytes = await file.arrayBuffer();
  const buffer = Buffer.from(bytes);

  // 生成唯一文件名
  const uniqueName = `${Date.now()}-${file.name}`;

  // 保存到 public/uploads
  const path = join(process.cwd(), "public", "uploads", uniqueName);
  await writeFile(path, buffer);

  return {
    success: true,
    url: `/uploads/${uniqueName}`,
  };
}
```

### 上传到 Blob

```typescript
// app/actions.ts
"use server";

import { put } from "@vercel/blob";

export async function uploadToBlob(formData: FormData) {
  const file = formData.get("file") as File;

  if (!file) {
    return { error: "没有文件" };
  }

  try {
    const blob = await put(file.name, file, {
      access: "public",
    });

    return {
      success: true,
      url: blob.url,
      downloadUrl: blob.downloadUrl,
    };
  } catch (error) {
    return { error: "上传失败" };
  }
}
```

## Cloudflare R2 存储

### 配置 R2

```typescript
// lib/r2.ts
import { S3Client, PutObjectCommand } from "@aws-sdk/client-s3";

const r2Client = new S3Client({
  region: "auto",
  endpoint: process.env.CLOUDFLARE_R2_ENDPOINT,
  credentials: {
    accessKeyId: process.env.CLOUDFLARE_R2_ACCESS_KEY_ID!,
    secretAccessKey: process.env.CLOUDFLARE_R2_SECRET_ACCESS_KEY!,
  },
});

export async function uploadToR2({ file, key }: { file: File; key: string }) {
  const arrayBuffer = await file.arrayBuffer();
  const buffer = Buffer.from(arrayBuffer);

  const command = new PutObjectCommand({
    Bucket: process.env.CLOUDFLARE_R2_BUCKET!,
    Key: key,
    Body: buffer,
    ContentType: file.type,
  });

  await r2Client.send(command);

  return {
    url: `${process.env.CLOUDFLARE_R2_PUBLIC_URL}/${key}`,
  };
}
```

### 使用 R2 上传

```typescript
// app/actions.ts
"use server";

import { uploadToR2 } from "@/lib/r2";

export async function uploadFile(formData: FormData) {
  const file = formData.get("file") as File;

  if (!file) {
    return { error: "没有文件" };
  }

  // 生成唯一键
  const key = `${Date.now()}-${file.name}`;

  try {
    const { url } = await uploadToR2({ file, key });

    return {
      success: true,
      url,
    };
  } catch (error) {
    return { error: "上传失败" };
  }
}
```

## 大文件上传

### 分块上传

```typescript
// lib/chunked-upload.ts
const CHUNK_SIZE = 5 * 1024 * 1024; // 5MB

export async function uploadLargeFile(file: File) {
  const chunks = Math.ceil(file.size / CHUNK_SIZE);
  const uploadId = `${Date.now()}-${file.name}`;

  for (let i = 0; i < chunks; i++) {
    const start = i * CHUNK_SIZE;
    const end = Math.min(start + CHUNK_SIZE, file.size);
    const chunk = file.slice(start, end);

    const formData = new FormData();
    formData.append("chunk", chunk);
    formData.append("chunkIndex", i.toString());
    formData.append("totalChunks", chunks.toString());
    formData.append("uploadId", uploadId);

    await fetch("/api/upload-chunk", {
      method: "POST",
      body: formData,
    });
  }

  // 合并分块
  await fetch("/api/merge-chunks", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ uploadId, fileName: file.name }),
  });
}
```

"use client";

export function DownloadButton({ filename }: { filename: string }) {
  const handleDownload = async () => {
    const response = await fetch(`/api/download/${filename}`);
    const blob = await response.blob();

    const url = window.URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;

    a.click();
    window.URL.revokeObjectURL(url);

3. **不要忘记清理临时文件**

```typescript
// ✅ 定期清理临时文件
async function cleanupTempFiles() {
  const tempDir = "./temp";
  const files = await fs.readdir(tempDir);

  for (const file of files) {
    const filePath = join(tempDir, file);
    const stats = await fs.stat(filePath);

    // 删除超过 1 小时的文件
    if (Date.now() - stats.mtimeMs > 3600000) {
      await fs.unlink(filePath);
    }
  }
}
```

**Q: 如何处理大文件上传？**
