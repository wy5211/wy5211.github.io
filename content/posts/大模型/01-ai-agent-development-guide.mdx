## 什么是 AI Agent

### 从 Chatbot 到 Agent

传统 LLM 应用的局限性：

```typescript
// 传统单轮问答
const response = await openai.chat.completions.create({
  model: "gpt-4",
  messages: [{ role: "user", content: "今天北京天气如何？" }],
});

// 问题：LLM 无法获取实时数据，只能回答"我不知道"
```

**AI Agent 的核心能力：**

| 能力         | 描述                   | 示例                                            |
| ------------ | ---------------------- | ----------------------------------------------- |
| **规划**     | 将复杂任务分解为子任务 | "帮我订明天去上海的机票" → 查航班 → 比价 → 下单 |
| **工具使用** | 调用外部 API 和服务    | 搜索、查询数据库、发送邮件                      |
| **推理**     | 基于上下文做出决策     | "用户想省钱，优先推荐经济舱"                    |
| **记忆**     | 记住对话历史和用户偏好 | "记住我喜欢靠窗座位"                            |

## 核心技术原理

### ReAct 模式

ReAct (Reasoning + Acting) 是 Agent 的核心推理模式：

```
Thought: 用户想知道北京天气，我需要调用天气 API
Action: get_weather("北京")
Observation: 北京今天晴，温度 15-25°C
Thought: 获得了天气信息，现在可以回答用户
Answer: 北京今天天气晴朗，温度 15-25°C，适合外出。
```

**TypeScript 实现：**

```typescript
interface ReActStep {
  thought: string;    // 思考
  action?: string;    // 行动
  observation?: string; // 观察
}

async function reactLoop(
  query: string,
  tools: Map&lt;string, Function&gt;,
  maxIterations: number = 5
): Promise&lt;string&gt; {
  const steps: ReActStep[] = [];
  let iteration = 0;

  while (iteration &lt; maxIterations) {
    // 1. 思考阶段
    const thought = await llmThink(query, steps);

    // 2. 决定是否需要行动
    if (thought.action) {
      // 3. 执行工具
      const observation = await executeTool(tools, thought.action);
      steps.push({ ...thought, observation });
    } else {
      // 4. 得出最终答案
      return thought.answer;
    }

    iteration++;
  }

  return "达到最大迭代次数，任务未完成";
}
```

**2. 实现工具执行器**

```typescript
// executor.ts
async function getWeather(city: string): Promise&lt;string&gt; {
  // 实际项目中调用真实天气 API
  const mockData: Record&lt;string, string&gt; = {
    "北京": "晴天，15-25°C，空气质量良好",
    "上海": "多云，18-26°C，有轻微雾霾",
    "深圳": "阴天，22-28°C，可能有小雨"
  };

  return mockData[city] || `未找到 ${city} 的天气信息`;
}

async function calculate(expression: string): Promise&lt;string&gt; {
  try {
    // 注意：生产环境应使用安全的数学表达式解析库
    const result = Function(`"use strict"; return (${expression})`)();
    return `计算结果：${result}`;
  } catch (error) {
    return `计算错误：${error}`;
  }
}

export async function executeFunctionCall(
  functionName: string,
  args: Record&lt;string, any&gt;
): Promise&lt;string&gt; {
  switch (functionName) {
    case "get_weather":
      return await getWeather(args.city);
    case "calculate":
      return await calculate(args.expression);
    default:
      return `未知函数：${functionName}`;
  }
}
```

**3. 完整的 Function Calling 流程**

```typescript
// agent-basic.ts
import OpenAI from "openai";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

async function runAgent(userMessage: string) {
  const messages: OpenAI.ChatCompletionMessageParam[] = [
    {
      role: "system",
      content: "你是一个有用的助手，可以使用工具来帮助用户。",
    },
    { role: "user", content: userMessage },
  ];

  const tools = [weatherTool, calculatorTool];

  // 第一次调用：LLM 决定是否使用工具
  let response = await openai.chat.completions.create({
    model: "gpt-4-turbo-preview",
    messages,
    tools,
    tool_choice: "auto",
  });

  // 循环处理工具调用
  while (response.choices[0].finish_reason === "tool_calls") {
    const assistantMessage = response.choices[0].message;
    messages.push(assistantMessage);

    // 执行所有工具调用
    for (const toolCall of assistantMessage.tool_calls || []) {
      const args = JSON.parse(toolCall.function.arguments);
      const result = await executeFunctionCall(toolCall.function.name, args);

      // 将工具结果添加到消息历史
      messages.push({
        role: "tool",
        tool_call_id: toolCall.id,
        content: result,
      });
    }

    // 第二次调用：LLM 基于工具结果生成最终回复
    response = await openai.chat.completions.create({
      model: "gpt-4-turbo-preview",
      messages,
      tools,
    });
  }

  return response.choices[0].message.content;
}

// 测试
const result = await runAgent("北京今天天气怎么样？");
console.log(result);
// 输出：根据查询，北京今天是晴天，温度在 15-25°C 之间，空气质量良好...
```

## 构建基础 Agent

### 封装 Agent 类

```typescript
// agent.ts
interface Tool {
  name: string;
  description: string;
  parameters: any;
  execute: (args: any) => Promise&lt;string&gt;;
}

class SimpleAgent {
  private messages: OpenAI.ChatCompletionMessageParam[] = [];
  private tools: Tool[] = [];
  private maxIterations: number;

  constructor(
    private model: string = "gpt-4-turbo-preview",
    options: { maxIterations?: number } = {}
  ) {
    this.maxIterations = options.maxIterations || 10;
  }

  // 注册工具
  registerTool(tool: Tool) {
    this.tools.push(tool);
  }

  // 转换为 OpenAI 工具格式
  private getOpenAITools() {
    return this.tools.map(tool => ({
      type: "function" as const,
      function: {
        name: tool.name,
        description: tool.description,
        parameters: tool.parameters
      }
    }));
  }

  // 执行工具
  private async executeTool(name: string, args: any): Promise&lt;string&gt; {
    const tool = this.tools.find(t => t.name === name);
    if (!tool) {
      return `错误：未找到工具 ${name}`;
    }
    return await tool.execute(args);
  }

  // 运行 Agent
  async run(userInput: string): Promise&lt;string&gt; {
    this.messages = [
      {
        role: "system",
        content: this.buildSystemPrompt()
      },
      { role: "user", content: userInput }
    ];

    for (let i = 0; i &lt; this.maxIterations; i++) {
      const response = await openai.chat.completions.create({
        model: this.model,
        messages: this.messages,
        tools: this.getOpenAITools(),
        tool_choice: "auto"
      });

      const assistantMessage = response.choices[0].message;
      this.messages.push(assistantMessage);

      // 检查是否需要调用工具
      if (response.choices[0].finish_reason === "tool_calls") {
        // 执行所有工具调用
        for (const toolCall of assistantMessage.tool_calls || []) {
          const args = JSON.parse(toolCall.function.arguments);
          const result = await this.executeTool(
            toolCall.function.name,
            args
          );

          this.messages.push({
            role: "tool",
            tool_call_id: toolCall.id,
            content: result
          });

          console.log(`[Tool] ${toolCall.function.name}(${JSON.stringify(args)}) => ${result}`);
        }
      } else {
        // 任务完成
        return assistantMessage.content || "";
      }
    }

    return "达到最大迭代次数";
  }

  private buildSystemPrompt(): string {
    const toolDescriptions = this.tools
      .map(t => `- ${t.name}: ${t.description}`)
      .join("\n");

    return `你是一个智能助手，可以使用以下工具：

${toolDescriptions}

请根据用户需求，合理使用工具来完成任务。`;
  }
}

// 使用示例
const agent = new SimpleAgent();

agent.registerTool({
  name: "search",
  description: "搜索互联网信息",
  parameters: {
    type: "object",
    properties: {
      query: { type: "string", description: "搜索关键词" }
    },
    required: ["query"]
  },
  execute: async (args) => {
    // 实际项目中调用搜索 API
    return `搜索结果：${args.query} 的相关信息...`;
  }
});

agent.registerTool({
  name: "get_current_time",
  description: "获取当前时间",
  parameters: { type: "object", properties: {} },
  execute: async () => {
    return new Date().toLocaleString("zh-CN");
  }
});

const answer = await agent.run("现在几点了？帮我搜索一下最新的 AI 新闻");
console.log(answer);
```

const agentExecutor = new AgentExecutor({
  agent,
  tools,
  memory: new BufferMemory({

    memoryKey: "chat_history",
    inputKey: "input",
    outputKey: "output"
  }),
  verbose: true // 打印执行过程
});

// 5. 运行
const result = await agentExecutor.invoke({

import { RetryPolicy } from "langchain/experimental/chat_models";

const llmWithRetry = new ChatOpenAI({

// 工具执行也应有重试机制
async function executeWithRetry(
  fn: () =&gt; Promise&lt;string&gt;,
  maxRetries: number = 3
): Promise&lt;string&gt; {
  let lastError: Error | null = null;

  for (let i = 0; i &lt; maxRetries; i++) {
    try {
      return await fn();
    } catch (error) {
      lastError = error as Error;
      await new Promise(resolve =&gt; setTimeout(resolve, 1000 * (i + 1)));
    }
  }

  return `工具执行失败：${lastError?.message}`;
}
```

  "12345": {
    status: "已发货",
    trackingNumber: "SF123456789",
    items: ["iPhone 15 Pro", "AirPods Pro"]
  },

    const order = orderDatabase[orderId];
    if (!order) {
      return `未找到订单 ${orderId}`;
    }
    return JSON.stringify(order);
  }
});

const checkUserOrdersTool = new DynamicTool({
  name: "check_user_orders",
  description: "查询用户的所有订单。输入用户ID。",
  func: async (userId: string) =&gt; {
    const user = userDatabase[userId];
    if (!user) {
      return `未找到用户 ${userId}`;
    }
    return `用户 ${user.name} 的订单：${user.orders.join(", ")}`;
  }
});

const createTicketTool = new DynamicTool({
  name: "create_ticket",
  description: "创建客服工单。输入问题描述。",
  func: async (description: string) =&gt; {
    const ticketId = `TK${Date.now()}`;
    console.log(`创建工单 ${ticketId}: ${description}`);
    return `已创建工单 ${ticketId}，我们的客服会在 24 小时内联系您`;
  }
});

// 创建 Agent
const llm = new ChatOpenAI({
  modelName: "gpt-4-turbo-preview",
  temperature: 0
});

const tools = [queryOrderTool, checkUserOrdersTool, createTicketTool];

const prompt = ChatPromptTemplate.fromMessages([
  ["system", `你是一个友好的客服助手，可以使用工具帮助用户：
- 查询订单状态
- 查看用户订单
- 创建客服工单

请礼貌、专业地回答用户问题。如果无法解决，主动创建工单。`],
  ["placeholder", "{chat_history}"],
  ["human", "{input}"],
  ["placeholder", "{agent_scratchpad}"]
]);

const agent = await createOpenAIFunctionsAgent({
  llm,
  tools,
  prompt
});

const agentExecutor = new AgentExecutor({
  agent,
  tools,
  memory: new BufferMemory({
    returnMessages: true,
    memoryKey: "chat_history",
    inputKey: "input",
    outputKey: "output"
  }),
  verbose: true,
  maxIterations: 5
});

// 测试对话
async function chat(userInput: string) {
  const result = await agentExecutor.invoke({ input: userInput });
  console.log("客服:", result.output);
  return result.output;
}

await chat("你好，我是张三，用户ID是 user123");
await chat("帮我查一下我的订单");
await chat("订单 12345 现在什么状态？");
await chat("好的，我还有一个问题想反馈");
```
