## 前言

Next.js 14 引入的 App Router 和 React Server Components（RSC）改变了我们构建全栈应用的方式。将 tRPC 与 Next.js 结合，可以获得：

✅ **端到端类型安全**：从数据库到前端的完整类型系统
✅ **服务端组件优势**：直接在服务端调用 tRPC，无需额外的 API 层
✅ **最佳性能**：利用 RSC 减少客户端 JavaScript
✅ **卓越的开发体验**：完整的类型推断和自动补全

今天，我们将学习：

- Next.js App Router 与 tRPC 集成
- 服务端组件和客户端组件中的 tRPC 使用
- Server Actions vs tRPC
- 认证和中间件集成
- 完整的全栈应用实战

## 服务端配置

### 创建 tRPC Router

```typescript
// src/lib/trpc/router.ts

import { z } from "zod";
import { TRPCError } from "@trpc/server";
import { prisma } from "@/lib/prisma";
import { createTRPCRouter, protectedProcedure } from "./server";

export const appRouter = createTRPCRouter({
  // ========== 用户相关 ==========
  users: createTRPCRouter({
    // 获取当前用户信息
    me: protectedProcedure.query(async ({ ctx }) => {
      return ctx.user;
    }),

    // 获取用户列表
    list: protectedProcedure
      .input(
        z.object({
          limit: z.number().min(1).max(100).default(20),
          cursor: z.string().optional(),
        })
      )
      .query(async ({ input }) => {
        const { limit, cursor } = input;

        const users = await prisma.user.findMany({
          take: limit + 1,
          ...(cursor && {
            cursor: { id: cursor },
            skip: 1,
          }),
          orderBy: { createdAt: "desc" },
          select: {
            id: true,
            name: true,
            email: true,
            avatar: true,
            createdAt: true,
          },
        });

        let nextCursor: string | undefined = undefined;
        if (users.length > limit) {
          const nextItem = users.pop();
          nextCursor = nextItem!.id;
        }

        return {
          items: users,
          nextCursor,
        };
      }),

    // 根据 ID 获取用户
    byId: protectedProcedure
      .input(z.object({ id: z.string() }))
      .query(async ({ input }) => {
        const user = await prisma.user.findUnique({
          where: { id: input.id },
          select: {
            id: true,
            name: true,
            email: true,
            avatar: true,
            bio: true,
            createdAt: true,
          },
        });

        if (!user) {
          throw new TRPCError({
            code: "NOT_FOUND",
            message: "用户不存在",
          });
        }

        return user;
      }),
  }),

  // ========== 文章相关 ==========
  posts: createTRPCRouter({
    // 获取文章列表
    list: protectedProcedure
      .input(
        z.object({
          page: z.number().default(1),
          limit: z.number().min(1).max(100).default(20),
          published: z.boolean().optional(),
        })
      )
      .query(async ({ input }) => {
        const { page, limit, published } = input;
        const skip = (page - 1) * limit;

        const [posts, total] = await Promise.all([
          prisma.post.findMany({
            where: published !== undefined ? { published } : undefined,
            skip,
            take: limit,
            orderBy: { createdAt: "desc" },
            include: {
              author: {
                select: {
                  id: true,
                  name: true,
                  avatar: true,
                },
              },
              _count: {
                select: {
                  comments: true,
                  likes: true,
                },
              },
            },
          }),
          prisma.post.count({
            where: published !== undefined ? { published } : undefined,
          }),
        ]);

        return {
          posts,
          pagination: {
            page,
            limit,
            total,
            totalPages: Math.ceil(total / limit),
          },
        };
      }),

    // 根据 ID 获取文章
    byId: protectedProcedure
      .input(
        z.object({
          id: z.string(),
          includeComments: z.boolean().default(true),
        })
      )
      .query(async ({ input }) => {
        const post = await prisma.post.findUnique({
          where: { id: input.id },
          include: {
            author: {
              select: {
                id: true,
                name: true,
                email: true,
                avatar: true,
                bio: true,
              },
            },
            ...(input.includeComments && {
              comments: {
                include: {
                  author: {
                    select: {
                      id: true,
                      name: true,
                      avatar: true,
                    },
                  },
                },
                orderBy: { createdAt: "desc" },
              },
            }),
            likes: {
              select: {
                user: {
                  select: {
                    id: true,
                    name: true,
                  },
                },
              },
            },
          },
        });

        if (!post) {
          throw new TRPCError({
            code: "NOT_FOUND",
            message: "文章不存在",
          });
        }

        return post;
      }),

    // 创建文章
    create: protectedProcedure
      .input(
        z.object({
          title: z.string().min(1).max(200),
          content: z.string().min(1),
          published: z.boolean().default(false),
        })
      )
      .mutation(async ({ input, ctx }) => {
        const post = await prisma.post.create({
          data: {
            ...input,
            authorId: ctx.user.id,
          },
          include: {
            author: {
              select: {
                id: true,
                name: true,
                avatar: true,
              },
            },
          },
        });

        return post;
      }),

    // 更新文章
    update: protectedProcedure
      .input(
        z.object({
          id: z.string(),
          title: z.string().min(1).max(200).optional(),
          content: z.string().min(1).optional(),
          published: z.boolean().optional(),
        })
      )
      .mutation(async ({ input, ctx }) => {
        const { id, ...data } = input;

        // 检查权限
        const post = await prisma.post.findUnique({
          where: { id },
        });

        if (!post) {
          throw new TRPCError({
            code: "NOT_FOUND",
            message: "文章不存在",
          });
        }

        if (post.authorId !== ctx.user.id) {
          throw new TRPCError({
            code: "FORBIDDEN",
            message: "只能编辑自己的文章",
          });
        }

        const updated = await prisma.post.update({
          where: { id },
          data,
          include: {
            author: {
              select: {
                id: true,
                name: true,
                avatar: true,
              },
            },
          },
        });

        return updated;
      }),

    // 删除文章
    delete: protectedProcedure
      .input(z.object({ id: z.string() }))
      .mutation(async ({ input, ctx }) => {
        const post = await prisma.post.findUnique({
          where: { id: input.id },
        });

        if (!post) {
          throw new TRPCError({
            code: "NOT_FOUND",
            message: "文章不存在",
          });
        }

        if (post.authorId !== ctx.user.id) {
          throw new TRPCError({
            code: "FORBIDDEN",
            message: "只能删除自己的文章",
          });
        }

        await prisma.post.delete({
          where: { id: input.id },
        });

        return { success: true };
      }),
  }),

  // ========== 评论相关 ==========
  comments: createTRPCRouter({
    // 创建评论
    create: protectedProcedure
      .input(
        z.object({
          postId: z.string(),
          content: z.string().min(1).max(1000),
        })
      )
      .mutation(async ({ input, ctx }) => {
        const comment = await prisma.comment.create({
          data: {
            ...input,
            authorId: ctx.user.id,
          },
          include: {
            author: {
              select: {
                id: true,
                name: true,
                avatar: true,
              },
            },
          },
        });

        return comment;
      }),
  }),

  // ========== 统计相关 ==========
  stats: createTRPCRouter({
    overview: protectedProcedure.query(async () => {
      const [userCount, postCount, commentCount] = await Promise.all([
        prisma.user.count(),
        prisma.post.count(),
        prisma.comment.count(),
      ]);

      return {
        users: userCount,
        posts: postCount,
        comments: commentCount,
      };
    }),
  }),
});

export type AppRouter = typeof appRouter;
```

### 创建 tRPC 服务器配置

```typescript
// src/lib/trpc/server.ts

import { initTRPC, TRPCError } from "@trpc/server";
import { CreateNextContextOptions } from "@trpc/server/adapters/next";
import { getSession } from "next-auth/react";
import superjson from "superjson";

// ========== Context 类型定义 ==========

interface User {
  id: string;
  name: string;
  email: string;
  avatar: string | null;
  role: "USER" | "MODERATOR" | "ADMIN";
}

interface CreateContextOptions {
  session: Session | null;
}

// ========== 创建 Context ==========

export const createContext = async ({ req, res }: CreateNextContextOptions) => {
  // 获取会话
  const session = await getSession({ req });

  return {
    session,
    user: session?.user || null,
  };
};

export type Context = Awaited<ReturnType<typeof createContext>>;

// ========== 初始化 tRPC ==========

const t = initTRPC.context<Context>().create({
  transformer: superjson,
  errorFormatter({ shape, error }) {
    return {
      ...shape,
      data: {
        ...shape.data,
        zodError:
          error.code === "BAD_REQUEST" && error.cause instanceof z.ZodError
            ? error.cause.flatten()
            : null,
      },
    };
  },
});

// ========== 中间件 ==========

// 是否已登录
const isAuthed = t.middleware(({ ctx, next }) => {
  if (!ctx.user) {
    throw new TRPCError({
      code: "UNAUTHORIZED",
      message: "未登录",
    });
  }
  return next({
    ctx: {
      ...ctx,
      user: ctx.user, // 非 null 断言
    },
  });
});

// ========== 导出 ==========

export const createTRPCRouter = t.router;
export const publicProcedure = t.procedure;
export const protectedProcedure = t.procedure.use(isAuthed);
export const middleware = t.middleware;
```

```typescript
// src/components/providers/trpc-provider.tsx

'use client';

import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
import { httpBatchLink, loggerLink } from '@trpc/client';
import { useState } from 'react';
import superjson from 'superjson';
import { trpc } from '@/lib/trpc/client';

export function TRPCProvider({ children }: { children: React.ReactNode }) {
  const [queryClient] = useState(() => new QueryClient({
    defaultOptions: {
      queries: {
        refetchOnWindowFocus: false,
        staleTime: 1000 * 60 * 5  // 5分钟
      }
    }
  }));

  const [trpcClient] = useState(() =>
    trpc.createClient({
      links: [
        loggerLink({
          enabled: (opts) =>
            process.env.NODE_ENV === 'development' ||
            (opts.direction === 'down' && opts.result instanceof Error)
        }),
        httpBatchLink({
          transformer: superjson,
          url: getBaseUrl() + '/api/trpc',
          headers() {
            return {
              // 可添加认证头
            };
          }
        })
      ]
    })
  );

  return (
    <trpc.Provider client={trpcClient} queryClient={queryClient}>
      <QueryClientProvider client={queryClient}>
        {children}
      </QueryClientProvider>
    </trpc.Provider>
  );
}

function getBaseUrl() {
  if (typeof window !== 'undefined') {
    return window.location.origin;
  }
  if (process.env.VERCEL_URL) {
    return `https://${process.env.VERCEL_URL}`;
  }
  return `http://localhost:${process.env.PORT ?? 3000}`;
}
```

## 客户端组件中使用 tRPC

### 使用 React Hooks

```typescript
// src/components/comments.tsx

'use client';

import { trpc } from '@/lib/trpc/client';
import { useState } from 'react';

interface CommentsProps {
  postId: string;
  initialComments?: Comment[];
}

export function Comments({ postId }: CommentsProps) {
  const [newComment, setNewComment] = useState('');

  // ✅ 使用 useQuery 获取数据
  const { data, isLoading, refetch } = trpc.comments.list.useQuery({
    postId
  });

  // ✅ 使用 useMutation 创建评论
  const createComment = trpc.comments.create.useMutation({
    onSuccess: () => {
      setNewComment('');
      refetch();
    }
  });

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    if (!newComment.trim()) return;

    createComment.mutate({
      postId,
      content: newComment
    });
  };

  if (isLoading) {
    return <div>加载评论中...</div>;
  }

  return (
    <div className="mt-12">
      <h2 className="text-2xl font-bold mb-6">
        评论 ({data?.length || 0})
      </h2>

      {/* 创建评论表单 */}
      <form onSubmit={handleSubmit} className="mb-8">
        <textarea
          value={newComment}
          onChange={(e) => setNewComment(e.target.value)}
          placeholder="写下你的评论..."
          className="w-full p-4 border rounded-lg mb-4"
          rows={4}
        />
        <button
          type="submit"
          disabled={createComment.isLoading}
          className="px-6 py-2 bg-blue-600 text-white rounded-lg disabled:opacity-50"
        >
          {createComment.isLoading ? '发表中...' : '发表评论'}
        </button>
      </form>

      {/* 评论列表 */}
      <div className="space-y-6">
        {data?.map((comment) => (
          <div key={comment.id} className="flex gap-4">
            <img
              src={comment.author.avatar || '/default-avatar.png'}
              alt={comment.author.name}
              className="w-10 h-10 rounded-full"
            />
            <div className="flex-1">
              <div className="font-semibold mb-1">
                {comment.author.name}
              </div>
              <div className="text-gray-700">{comment.content}</div>
              <div className="text-sm text-gray-500 mt-2">
                {new Date(comment.createdAt).toLocaleString()}
              </div>
            </div>
          </div>
        ))}
      </div>
    </div>
  );
}
```

### 无限滚动

```typescript
// src/components/posts-infinite.tsx

'use client';

import { trpc } from '@/lib/trpc/client';
import { useEffect, useRef } from 'react';

export function PostsInfinite() {
  const { data, fetchNextPage, hasNextPage, isFetchingNextPage } =
    trpc.posts.infinite.useInfiniteQuery(
      { limit: 20 },
      {
        getNextPageParam: (lastPage) => lastPage.nextCursor
      }
    );

  const observerTarget = useRef<HTMLDivElement>(null);

  useEffect(() => {
    const observer = new IntersectionObserver(
      (entries) => {
        if (entries[0].isIntersecting && hasNextPage) {
          fetchNextPage();
        }
      },
      { threshold: 1 }
    );

    if (observerTarget.current) {
      observer.observe(observerTarget.current);
    }

    return () => observer.disconnect();
  }, [hasNextPage, fetchNextPage]);

  return (
    <div className="space-y-6">
      {data?.pages.map((page) =>
        page.items.map((post) => (
          <article key={post.id} className="border rounded-lg p-6">
            <h2 className="text-2xl font-semibold mb-2">
              {post.title}
            </h2>
            <p className="text-gray-600">{post.content}</p>
          </article>
        ))
      )}

      {/* 观察目标 */}
      <div ref={observerTarget} />

      {isFetchingNextPage && <div>加载更多...</div>}
    </div>
  );
}
```

### 创建文章模态框

```typescript
// src/components/create-post-modal.tsx

'use client';

import { useState } from 'react';
import { trpc } from '@/lib/trpc/client';
import { useRouter } from 'next/navigation';

export function CreatePostButton() {
  const [isOpen, setIsOpen] = useState(false);
  const router = useRouter();
  const utils = trpc.useContext();

  const createPost = trpc.posts.create.useMutation({
    onSuccess: (data) => {
      setIsOpen(false);
      utils.posts.list.invalidate();
      router.push(`/posts/${data.id}`);
    }
  });

  return (
    <>
      <button
        onClick={() => setIsOpen(true)}
        className="px-6 py-2 bg-blue-600 text-white rounded-lg hover:bg-blue-700"
      >
        写文章
      </button>

      {isOpen && (
        <div className="fixed inset-0 bg-black/50 flex items-center justify-center">
          <div className="bg-white rounded-lg p-6 max-w-2xl w-full mx-4">
            <h2 className="text-2xl font-bold mb-4">创建文章</h2>

            <form
              onSubmit={(e) => {
                e.preventDefault();
                const formData = new FormData(e.currentTarget);
                createPost.mutate({
                  title: formData.get('title') as string,
                  content: formData.get('content') as string,
                  published: false
                });
              }}
              className="space-y-4"
            >
              <div>
                <label className="block mb-2 font-semibold">标题</label>
                <input
                  type="text"
                  name="title"
                  required
                  className="w-full px-4 py-2 border rounded-lg"
                  placeholder="输入文章标题..."
                />
              </div>

              <div>
                <label className="block mb-2 font-semibold">内容</label>
                <textarea
                  name="content"

                />
```

### 3. 错误处理

```typescript
'use client';

import { trpc } from '@/lib/trpc/client';
import { TRPCClientError } from '@trpc/client';

export function UserProfile() {
  const { data, error, isLoading } = trpc.users.me.useQuery();

  if (isLoading) return <div>加载中...</div>;

  if (error) {
    if (error instanceof TRPCClientError) {
      if (error.code === 'UNAUTHORIZED') {
        return <div>请先登录</div>;
      }
    }
    return <div>发生错误：{error.message}</div>;
  }

  return <div>欢迎，{data?.name}</div>;
}
```

   - 缓存策略
   - 服务端渲染优化

```
用户请求

返回 HTML + 数据
```
