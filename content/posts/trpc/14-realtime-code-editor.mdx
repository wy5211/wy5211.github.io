## 前言

协作编辑是现代应用的典型场景。使用 tRPC + WebSocket + CRDT（Conflict-free Replicated Data Types），我们可以构建一个类型安全的实时协作应用。

**项目特性**：

- ✅ 实时协作编辑（CRDT）
- ✅ 多光标同步
- ✅ 用户在线状态
- ✅ 语法高亮
- ✅ 版本历史
- ✅ 权限控制

今天，我们将构建一个完整的协作代码编辑器。

### 创建 tRPC Router

```typescript
// packages/trpc/src/routers/documents.router.ts

import { z } from 'zod';
import { observable } from '@trpc/server/observable';
import { EventEmitter } from 'events';
import { Context } from '../context';

const ee = new EventEmitter();

export const documentsRouter = t.router<Context>({
  // ========== 获取文档 ==========
  byId: t.procedure
  .input(z.object({ id: z.string() }))
  .query(async ({ input, ctx }) => {
    const document = await ctx.prisma.document.findUnique({
      where: { id: input.id },
      include: {
        users: {
          include: {
            user: {
              select: {
                id: true,
                name: true,
                email: true,
                avatar: true
              }
            }
          }
        }
      }
    });

    if (!document) {
      throw new t.TRPCError({
        code: 'NOT_FOUND',
        message: '文档不存在'
      });
    }

    // 检查权限
    const hasAccess = document.users.some(
      du => du.userId === ctx.user?.id
    );

    if (!hasAccess) {
      throw new t.TRPCError({
        code: 'FORBIDDEN',
        message: '无权访问此文档'
      });
    }

    return document;
  }),

  // ========== 创建文档 ==========
  create: t.procedure
  .input(z.object({
    title: z.string().min(1).max(200),
    content: z.string(),
    language: z.enum(['javascript', 'typescript', 'python', 'go']),
    initialUsers: z.array(z.string()).default([])
  }))
  .mutation(async ({ input, ctx }) => {
    const document = await ctx.prisma.document.create({
      data: {
        title: input.title,
        content: input.content,
        language: input.language,
        ownerId: ctx.user.id
      }
    });

    // 添加文档用户
    await ctx.prisma.documentUser.createMany({
      data: [
        {
          userId: ctx.user.id,
          documentId: document.id,
          role: 'owner'
        },
        ...input.initialUsers.map(userId => ({
          userId,
          documentId: document.id,
          role: 'editor'
        }))
      ]
    });

    return document;
  }),

  // ========== 更新文档内容 ==========
  updateContent: t.procedure
  .input(z.object({
    documentId: z.string(),
    updates: z.array(z.object({
      index: z.number(),
      insert: z.string().optional(),
      delete: z.number().optional(),
      retain: z.number().optional()
    }))
  }))
  .mutation(async ({ input, ctx }) => {
    // 验证权限
    const docUser = await ctx.prisma.documentUser.findFirst({
      where: {
        documentId: input.documentId,
        userId: ctx.user.id
      }
    });

    if (!docUser || docUser.role === 'viewer') {
      throw new t.TRPCError({
        code: 'FORBIDDEN',
        message: '无权编辑此文档'
      });
    }

    // 获取当前文档内容
    const document = await ctx.prisma.document.findUnique({
      where: { id: input.documentId },
      select: { content: true }
    });

    if (!document) {
      throw new t.TRPCError({
        code: 'NOT_FOUND',
        message: '文档不存在'
      });
    }

    // 应用 CRDT 更新
    const ydoc = new Y.Doc();
    ydoc.getText().insert(0, document.content);

    // 应用更新
    ydoc.getText().delete(index, delete);
    if (insert) {
      ydoc.getText().insert(index, insert);
    }

    const newContent = ydoc.toString();

    // 更新数据库
    const updated = await ctx.prisma.document.update({
      where: { id: input.documentId },
      data: { content: newContent }
    });

    // 触发更新事件
    ee.emit('document:updated', {
      documentId: input.documentId,
      content: newContent,
      updates: input.updates,
      userId: ctx.user.id
    });

    return updated;
  }),

  // ========== 订阅文档更新 ==========
  onDocumentChange: t.procedure
  .input(z.object({ documentId: z.string() }))
  .subscription(({ input }) => {
    return observable<DocumentUpdate>((emit) => {
      const handler = (update: DocumentUpdate) => {
        if (update.documentId === input.documentId) {
          emit.next(update);
        }
      };

      ee.on('document:updated', handler);

      return () => {
        ee.off('document:updated', handler);
      };
    });
  }),

  // ========== 订阅用户光标 ==========
  onCursors: t.procedure
  .input(z.object({ documentId: z.string() }))
  .subscription(({ input }) => {
    return observable<Cursor>((emit) => {
      const handler = (cursor: Cursor) => {
        if (cursor.documentId === input.documentId) {
          emit.next(cursor);
        }
      };

      ee.on('cursor:move', handler);

      return () => {
        ee.off('cursor:move', handler);
      };
    });
  }),

  // ========== 更新光标位置 ==========
  updateCursor: t.procedure
  .input(z.object({
    documentId: z.string(),
    position: z.number(),
    selection: z.object({
      from: z.number(),
      to: z.number()
    }).optional()
  }))
  .mutation(async ({ input, ctx }) => {
    const cursor: Cursor = {
      userId: ctx.user.id,
      userName: ctx.user.name,
      color: getUserColor(ctx.user.id),
      ...input
    };

    ee.emit('cursor:move', cursor);

    return cursor;
  })
});

function getUserColor(userId: string): string {
  const colors = [
    '#FF6B6B', '#4ECDC4', '#45B7D1', '#FFA07A',
    '#98D8C8', '#F7DC6F', '#FF85A2', '#A8DADC'
  ];

  let hash = 0;
  for (let i = 0; i < userId.length; i++) {
    hash = userId.charCodeAt(i) + ((hash << 5) - hash);
  }

  return colors[Math.abs(hash) % colors.length];
}
```

## 前端实现

### 编辑器组件

```typescript
// apps/web/src/components/editor/MonacoEditor.tsx

'use client';

import { useEffect, useRef, useState } from 'react';
import Editor from '@monaco-editor/react';
import { trpc } from '@/lib/trpc/client';
import { useDebounce } from '@/lib/hooks/useDebounce';

interface MonacoEditorProps {
  documentId: string;
  initialContent: string;
  language: string;
  readOnly?: boolean;
}

export function MonacoEditor({
  documentId,
  initialContent,
  language,
  readOnly = false
}: MonacoEditorProps) {
  const [content, setContent] = useState(initialContent);

  // 订阅文档更新
  trpc.documents.onDocumentChange.useSubscription(
    { documentId },
    {
      onData: (update) => {
        setContent(update.content);
      }
    }
  );

  // 更新内容
  const updateMutation = trpc.documents.updateContent.useMutation({
    onSuccess: () => {
      // 成功后无需操作，订阅会自动更新
    }
  });

  const handleEditorChange = (value: string | undefined) => {
    if (!value || value === content) return;

    setContent(value);

    // 防抖更新
    updateMutation.mutate({
      documentId,
      updates: [{
        index: 0,
        retain: value.length
      }]
    });
  };

  return (
    <Editor
      height="600px"
      language={language}
      value={content}
      onChange={handleEditorChange}
      theme="vs-dark"
      options={{
        readOnly,
        minimap: { enabled: true },
        fontSize: 14,
        lineNumbers: 'on',
        scrollBeyondLastLine: false,
        automaticLayout: true
      }}
    />
  );
}
```

### 光标组件

```typescript
// apps/web/src/components/editor/Cursors.tsx

'use client';

import { useEffect } from 'react';
import { trpc } from '@/lib/trpc/client';
import { Cursor } from '@my-monorepo/trpc';

interface CursorsProps {
  documentId: string;
}

export function Cursors({ documentId }: CursorsProps) {
  const [cursors, setCursors] = useState<Cursor[]>([]);

  // 订阅光标更新
  trpc.documents.onCursors.useSubscription(
    { documentId },
    {
      onData: (cursor) => {
        setCursors(prev => {
          const filtered = prev.filter(c => c.userId !== cursor.userId);
          return [...filtered, cursor];
        });
      }
    }
  );

  return (
    <>
      {cursors.map((cursor) => (
        <CursorOverlay key={cursor.userId} cursor={cursor} />
      ))}
    </>
  );
}

function CursorOverlay({ cursor }: { cursor: Cursor }) {
  const calculatePosition = () => {
    // 计算光标在编辑器中的位置
    const editor = document.querySelector('.monaco-editor');
    const line = editor?.querySelector(`[line="${cursor.position}"]`);

    if (line) {
      const rect = line.getBoundingClientRect();
      return {
        top: rect.top,
        left: rect.left
      };
    }

    return { top: 0, left: 0 };
  };

  const position = calculatePosition();

  return (
    <div
      className="absolute pointer-events-none"
      style={{
        top: position.top,
        left: position.left,
        borderLeft: `2px solid ${cursor.color}`,
        backgroundColor: `${cursor.color}20`,
        padding: '2px 0'
      }}
    >
      <span
        className="text-xs px-1 py-0.5 text-white"
        style={{ backgroundColor: cursor.color }}
      >
        {cursor.userName}
      </span>
    </div>
  );
}
```

```typescript
'use client';

import { useEffect, useState } from 'react';
import { trpc } from '@/lib/trpc/client';
import { Avatar } from '@/components/ui/avatar';

interface UsersListProps {
  documentId: string;

export function UsersList({ documentId }: UsersListProps) {
```

## 性能优化

### 1. 批量更新

```typescript
// 使用 Yjs 的批量更新
const applyBatchUpdates = (ydoc: Y.Doc, updates: DocumentUpdate[]) => {
  ydoc.transact(() => {
    updates.forEach((update) => {
      const { index, insert, delete: del, retain } = update;

      if (del !== undefined) {
        ydoc.getText().delete(index, del);
      }

      if (insert !== undefined) {
        ydoc.getText().insert(index, insert);
      }

      if (retain !== undefined) {
        // 保留指定长度
        const length = ydoc.getText().length;
        if (length > retain) {
          ydoc.getText().delete(retain, length - retain);
        }
      }
    });
  });
};
```

### 3. 差异同步

```typescript
// 只同步差异
interface TextDelta {
  type: "insert" | "delete" | "retain";
  position: number;
  content?: string;
  length?: number;
}

const applyDelta = (content: string, delta: TextDelta): string => {
  switch (delta.type) {
    case "insert":
      return (
        content.slice(0, delta.position) +
        delta.content +
        content.slice(delta.position)
      );
    case "delete":
      return (
        content.slice(0, delta.position) +
        content.slice(delta.position + delta.length)
      );
    case "retain":
      return content.slice(0, delta.position);
    default:
      return content;
  }
};
```
