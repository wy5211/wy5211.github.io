## 前言

电商系统是全栈开发的典型应用场景。使用 tRPC + NestJS + Next.js + Stripe，我们可以构建一个类型安全、功能完整的电商平台：

**项目特性**：

- ✅ 商品目录和搜索
- ✅ 购物车管理
- ✅ 订单处理和状态跟踪
- ✅ Stripe 支付集成
- ✅ 库存管理和预警
- ✅ 实时库存更新
- ✅ 用户评价和推荐
- ✅ 后台管理系统

今天，我们将构建一个完整的全栈电商系统。

### 商品 Router

```typescript
// apps/api/src/modules/products/products.router.ts

import { z } from "zod";
import { t } from "@my-monorepo/trpc";
import { PrismaService } from "@my-monorepo/database";
import { TRPCError } from "@trpc/server";

export const productsRouter = t.router({
  // ========== 商品列表 ==========
  list: t.procedure
    .input(
      z.object({
        page: z.number().default(1),
        limit: z.number().default(20),
        category: z.string().optional(),
        search: z.string().optional(),
        sortBy: z
          .enum(["name", "price", "createdAt", "popular"])
          .default("createdAt"),
        sortOrder: z.enum(["asc", "desc"]).default("desc"),
        featured: z.boolean().optional(),
        minPrice: z.number().optional(),
        maxPrice: z.number().optional(),
        tags: z.array(z.string()).optional(),
      })
    )
    .query(async ({ input, ctx }) => {
      const {
        page,
        limit,
        category,
        search,
        sortBy,
        sortOrder,
        featured,
        minPrice,
        maxPrice,
        tags,
      } = input;

      const where: any = {
        active: true,
      };

      // 分类过滤
      if (category) {
        where.categories = {
          some: {
            slug: category,
          },
        };
      }

      // 搜索
      if (search) {
        where.OR = [
          { name: { contains: search, mode: "insensitive" } },
          { description: { contains: search, mode: "insensitive" } },
          { tags: { hasSome: [search] } },
        ];
      }

      // 特色商品
      if (featured !== undefined) {
        where.featured = featured;
      }

      // 价格范围
      if (minPrice !== undefined || maxPrice !== undefined) {
        where.price = {};
        if (minPrice !== undefined) where.price.gte = minPrice;
        if (maxPrice !== undefined) where.price.lte = maxPrice;
      }

      // 标签过滤
      if (tags && tags.length > 0) {
        where.tags = {
          hasSome: tags,
        };
      }

      const [products, total] = await Promise.all([
        ctx.prisma.product.findMany({
          where,
          take: limit,
          skip: (page - 1) * limit,
          orderBy: {
            [sortBy]: sortOrder,
          },
          include: {
            categories: true,
          },
        }),
        ctx.prisma.product.count({ where }),
      ]);

      return {
        products,
        pagination: {
          page,
          limit,
          total,
          totalPages: Math.ceil(total / limit),
        },
      };
    }),

  // ========== 商品详情 ==========
  byId: t.procedure
    .input(
      z.object({
        id: z.string(),
      })
    )
    .query(async ({ input, ctx }) => {
      const product = await ctx.prisma.product.findUnique({
        where: { id: input.id },
        include: {
          categories: true,
          reviews: {
            include: {
              user: {
                select: {
                  id: true,
                  name: true,
                },
              },
            },
            take: 10,
          },
        },
      });

      if (!product) {
        throw new TRPCError({
          code: "NOT_FOUND",
          message: "商品不存在",
        });
      }

      return product;
    }),

  // ========== 商品搜索建议 ==========
  searchSuggestions: t.procedure
    .input(
      z.object({
        query: z.string().min(2),
      })
    )
    .query(async ({ input, ctx }) => {
      const products = await ctx.prisma.product.findMany({
        where: {
          active: true,
          OR: [
            { name: { contains: input.query, mode: "insensitive" } },
            { tags: { hasSome: [input.query] } },
          ],
        },
        take: 5,
        select: {
          id: true,
          name: true,
          images: true,
          price: true,
        },
      });

      return products;
    }),

  // ========== 特色商品 ==========
  featured: t.procedure
    .input(
      z.object({
        limit: z.number().default(10),
      })
    )
    .query(async ({ input, ctx }) => {
      return await ctx.prisma.product.findMany({
        where: {
          featured: true,
          active: true,
          stock: { gt: 0 },
        },
        take: input.limit,
        include: {
          categories: true,
        },
        orderBy: {
          createdAt: "desc",
        },
      });
    }),

  // ========== 相关商品 ==========
  related: t.procedure
    .input(
      z.object({
        productId: z.string(),
        limit: z.number().default(4),
      })
    )
    .query(async ({ input, ctx }) => {
      const product = await ctx.prisma.product.findUnique({
        where: { id: input.productId },
        include: {
          categories: true,
          tags: true,
        },
      });

      if (!product) {
        throw new TRPCError({
          code: "NOT_FOUND",
          message: "商品不存在",
        });
      }

      // 基于分类和标签找相关商品
      return await ctx.prisma.product.findMany({
        where: {
          id: { not: input.productId },
          active: true,
          OR: [
            {
              categories: {
                some: {
                  id: { in: product.categories.map((c) => c.id) },
                },
              },
            },
            {
              tags: {
                hasSome: product.tags,
              },
            },
          ],
        },
        take: input.limit,
        include: {
          categories: true,
        },
      });
    }),

  // ========== 库存检查 ==========
  checkStock: t.procedure
    .input(
      z.object({
        items: z.array(
          z.object({
            productId: z.string(),
            quantity: z.number(),
          })
        ),
      })
    )
    .query(async ({ input, ctx }) => {
      const productIds = input.items.map((item) => item.productId);

      const products = await ctx.prisma.product.findMany({
        where: {
          id: { in: productIds },
        },
        select: {
          id: true,
          stock: true,
          active: true,
        },
      });

      const stockMap = new Map(products.map((p) => [p.id, p]));

      const results = input.items.map((item) => {
        const product = stockMap.get(item.productId);
        return {
          productId: item.productId,
          inStock: product ? product.stock >= item.quantity : false,
          availableStock: product?.stock || 0,
          active: product?.active || false,
        };
      });

      return results;
    }),
});
```

### 购物车 Router

```typescript
// apps/api/src/modules/cart/cart.router.ts

import { z } from "zod";
import { t } from "@my-monorepo/trpc";
import { TRPCError } from "@trpc/server";

export const cartRouter = t.router({
  // ========== 获取购物车 ==========
  get: t.procedure.query(async ({ ctx }) => {
    if (!ctx.user) {
      throw new TRPCError({
        code: "UNAUTHORIZED",
        message: "请先登录",
      });
    }

    const cart = await ctx.prisma.cart.findUnique({
      where: { userId: ctx.user.id },
      include: {
        items: {
          include: {
            product: {
              include: {
                categories: true,
              },
            },
          },
        },
      },
    });

    // 如果购物车不存在，创建一个
    if (!cart) {
      return await ctx.prisma.cart.create({
        data: {
          userId: ctx.user.id,
        },
        include: {
          items: {
            include: {
              product: {
                include: {
                  categories: true,
                },
              },
            },
          },
        },
      });
    }

    return cart;
  }),

  // ========== 添加商品到购物车 ==========
  addItem: t.procedure
    .input(
      z.object({
        productId: z.string(),
        quantity: z.number().min(1).max(99),
      })
    )
    .mutation(async ({ input, ctx }) => {
      if (!ctx.user) {
        throw new TRPCError({
          code: "UNAUTHORIZED",
          message: "请先登录",
        });
      }

      // 检查商品是否存在且有库存
      const product = await ctx.prisma.product.findUnique({
        where: { id: input.productId },
      });

      if (!product || !product.active) {
        throw new TRPCError({
          code: "NOT_FOUND",
          message: "商品不存在或已下架",
        });
      }

      if (product.stock < input.quantity) {
        throw new TRPCError({
          code: "BAD_REQUEST",
          message: "库存不足",
        });
      }

      // 获取或创建购物车
      let cart = await ctx.prisma.cart.findUnique({
        where: { userId: ctx.user.id },
      });

      if (!cart) {
        cart = await ctx.prisma.cart.create({
          data: {
            userId: ctx.user.id,
          },
        });
      }

      // 检查商品是否已在购物车中
      const existingItem = await ctx.prisma.cartItem.findUnique({
        where: {
          cartId_productId: {
            cartId: cart.id,
            productId: input.productId,
          },
        },
      });

      if (existingItem) {
        // 更新数量
        const newQuantity = existingItem.quantity + input.quantity;

        if (newQuantity > product.stock) {
          throw new TRPCError({
            code: "BAD_REQUEST",
            message: "库存不足",
          });
        }

        await ctx.prisma.cartItem.update({
          where: { id: existingItem.id },
          data: { quantity: newQuantity },
        });
      } else {
        // 添加新商品
        await ctx.prisma.cartItem.create({
          data: {
            cartId: cart.id,
            productId: input.productId,
            quantity: input.quantity,
          },
        });
      }

      // 返回更新后的购物车
      return await ctx.prisma.cart.findUnique({
        where: { userId: ctx.user.id },
        include: {
          items: {
            include: {
              product: {
                include: {
                  categories: true,
                },
              },
            },
          },
        },
      });
    }),

  // ========== 更新购物车商品数量 ==========
  updateQuantity: t.procedure
    .input(
      z.object({
        itemId: z.string(),
        quantity: z.number().min(0).max(99),
      })
    )
    .mutation(async ({ input, ctx }) => {
      if (!ctx.user) {
        throw new TRPCError({
          code: "UNAUTHORIZED",
          message: "请先登录",
        });
      }

      const cartItem = await ctx.prisma.cartItem.findUnique({
        where: { id: input.itemId },
        include: {
          product: true,
          cart: true,
        },
      });

      if (!cartItem) {
        throw new TRPCError({
          code: "NOT_FOUND",
          message: "购物车商品不存在",
        });
      }

      if (cartItem.cart.userId !== ctx.user.id) {
        throw new TRPCError({
          code: "FORBIDDEN",
          message: "无权操作此购物车",
        });
      }

      if (input.quantity === 0) {
        // 删除商品
        await ctx.prisma.cartItem.delete({
          where: { id: input.itemId },
        });
      } else {
        // 检查库存
        if (input.quantity > cartItem.product.stock) {
          throw new TRPCError({
            code: "BAD_REQUEST",
            message: "库存不足",
          });
        }

        // 更新数量
        await ctx.prisma.cartItem.update({
          where: { id: input.itemId },
          data: { quantity: input.quantity },
        });
      }

      // 返回更新后的购物车
      return await ctx.prisma.cart.findUnique({
        where: { userId: ctx.user.id },
        include: {
          items: {
            include: {
              product: {
                include: {
                  categories: true,
                },
              },
            },
          },
        },
      });
    }),

  // ========== 删除购物车商品 ==========
  removeItem: t.procedure
    .input(
      z.object({
        itemId: z.string(),
      })
    )
    .mutation(async ({ input, ctx }) => {
      if (!ctx.user) {
        throw new TRPCError({
          code: "UNAUTHORIZED",
          message: "请先登录",
        });
      }

      const cartItem = await ctx.prisma.cartItem.findUnique({
        where: { id: input.itemId },
        include: { cart: true },
      });

      if (!cartItem) {
        throw new TRPCError({
          code: "NOT_FOUND",
          message: "购物车商品不存在",
        });
      }

      if (cartItem.cart.userId !== ctx.user.id) {
        throw new TRPCError({
          code: "FORBIDDEN",
          message: "无权操作此购物车",
        });
      }

      await ctx.prisma.cartItem.delete({
        where: { id: input.itemId },
      });

      return { success: true };
    }),

  // ========== 清空购物车 ==========
  clear: t.procedure.mutation(async ({ ctx }) => {
    if (!ctx.user) {
      throw new TRPCError({
        code: "UNAUTHORIZED",
        message: "请先登录",
      });
    }

    await ctx.prisma.cartItem.deleteMany({
      where: {
        cart: {
          userId: ctx.user.id,
        },
      },
    });

    return { success: true };
  }),

  // ========== 获取购物车数量 ==========
  getItemCount: t.procedure.query(async ({ ctx }) => {
    if (!ctx.user) {
      return 0;
    }

    const cart = await ctx.prisma.cart.findUnique({
      where: { userId: ctx.user.id },
      include: {
        items: true,
      },
    });

    if (!cart) {
      return 0;
    }

    return cart.items.reduce((sum, item) => sum + item.quantity, 0);
  }),
});
```

```typescript
// apps/api/src/modules/orders/orders.router.ts

import { z } from "zod";
import { t } from "@my-monorepo/trpc";
import { TRPCError } from "@trpc/server";

export const ordersRouter = t.router({
  // ========== 创建订单 ==========
  create: t.procedure
    .input(
      z.object({
        items: z.array(
          z.object({
            productId: z.string(),
            quantity: z.number(),
          })
        ),
        shippingAddress: z.object({
          name: z.string(),
          email: z.string().email(),
          phone: z.string(),
          address: z.string(),
          city: z.string(),
          state: z.string(),
          zip: z.string(),
          country: z.string(),
        }),
        paymentMethod: z.enum(["CREDIT_CARD", "DEBIT_CARD", "PAYPAL"]),
        notes: z.string().optional(),
      })
    )
    .mutation(async ({ input, ctx }) => {
      if (!ctx.user) {
        throw new TRPCError({
          code: "UNAUTHORIZED",
          message: "请先登录",
        });
      }

      // 获取商品信息
      const productIds = input.items.map((item) => item.productId);
      const products = await ctx.prisma.product.findMany({
        where: {
          id: { in: productIds },
        },
      });

      const productMap = new Map(products.map((p) => [p.id, p]));

      // 验证库存
      for (const item of input.items) {
        const product = productMap.get(item.productId);
        if (!product || product.stock < item.quantity) {
          throw new TRPCError({
            code: "BAD_REQUEST",
            message: `商品 ${product?.name || item.productId} 库存不足`,
          });
        }
      }

      // 计算订单金额
      let subtotal = 0;
      const orderItems = input.items.map((item) => {
        const product = productMap.get(item.productId)!;
        const itemTotal = Number(product.price) * item.quantity;
        subtotal += itemTotal;

        return {
          productId: item.productId,
          quantity: item.quantity,
          price: product.price,
        };
      });

      const tax = subtotal * 0.08; // 8% 税
      const shipping = subtotal > 100 ? 0 : 9.99; // 满100免运费
      const total = subtotal + tax + shipping;

      // 生成订单号
      const orderNumber = `ORD-${Date.now()}-${Math.random().toString(36).substr(2, 9).toUpperCase()}`;

      // 创建订单
      const order = await ctx.prisma.$transaction(async (tx) => {
        const order = await tx.order.create({
          data: {
            orderNumber,
            userId: ctx.user.id,
            items: {
              create: orderItems,
            },
            status: "PENDING",
            subtotal,
            tax,
            shipping,
            total,
            shippingName: input.shippingAddress.name,
            shippingEmail: input.shippingAddress.email,
            shippingPhone: input.shippingAddress.phone,
            shippingAddress: input.shippingAddress.address,
            shippingCity: input.shippingAddress.city,
            shippingState: input.shippingAddress.state,
            shippingZip: input.shippingAddress.zip,
            shippingCountry: input.shippingAddress.country,
            paymentMethod: input.paymentMethod,
            notes: input.notes,
          },
        });

        // 扣减库存
        for (const item of input.items) {
          await tx.product.update({
            where: { id: item.productId },
            data: {
              stock: {
                decrement: item.quantity,
              },
            },
          });
        }

        // 清空购物车
        await tx.cartItem.deleteMany({
          where: {
            cart: {
              userId: ctx.user.id,
            },
          },
        });

        return order;
      });

      return order;
    }),

  // ========== 获取订单列表 ==========
  list: t.procedure
    .input(
      z.object({
        page: z.number().default(1),
        limit: z.number().default(10),
        status: z
          .enum([
            "PENDING",
            "PROCESSING",
            "SHIPPED",
            "DELIVERED",
            "CANCELLED",
            "REFUNDED",
          ])
          .optional(),
      })
    )
    .query(async ({ input, ctx }) => {
      if (!ctx.user) {
        throw new TRPCError({
          code: "UNAUTHORIZED",
          message: "请先登录",
        });
      }

      const where: any = {
        userId: ctx.user.id,
      };

      if (input.status) {
        where.status = input.status;
      }

      const [orders, total] = await Promise.all([
        ctx.prisma.order.findMany({
          where,
          take: input.limit,
          skip: (input.page - 1) * input.limit,
          orderBy: {
            createdAt: "desc",
          },
          include: {
            items: {
              include: {
                product: true,
              },
            },
          },
        }),
        ctx.prisma.order.count({ where }),
      ]);

      return {
        orders,
        pagination: {
          page: input.page,
          limit: input.limit,
          total,
          totalPages: Math.ceil(total / input.limit),
        },
      };
    }),

  // ========== 获取订单详情 ==========
  byId: t.procedure
    .input(
      z.object({
        id: z.string(),
      })
    )
    .query(async ({ input, ctx }) => {
      if (!ctx.user) {
        throw new TRPCError({
          code: "UNAUTHORIZED",
          message: "请先登录",
        });
      }

      const order = await ctx.prisma.order.findUnique({
        where: { id: input.id },
        include: {
          items: {
            include: {
              product: true,
            },
          },
        },
      });

      if (!order) {
        throw new TRPCError({
          code: "NOT_FOUND",
          message: "订单不存在",
        });
      }

      if (order.userId !== ctx.user.id && ctx.user.role !== "ADMIN") {
        throw new TRPCError({
          code: "FORBIDDEN",
          message: "无权访问此订单",
        });
      }

      return order;
    }),

  // ========== 取消订单 ==========
  cancel: t.procedure
    .input(
      z.object({
        id: z.string(),
      })
    )
    .mutation(async ({ input, ctx }) => {
      if (!ctx.user) {
        throw new TRPCError({
          code: "UNAUTHORIZED",
          message: "请先登录",
        });
      }

      const order = await ctx.prisma.order.findUnique({
        where: { id: input.id },
        include: {
          items: true,
        },
      });

      if (!order) {
        throw new TRPCError({
          code: "NOT_FOUND",
          message: "订单不存在",
        });
      }

      if (order.userId !== ctx.user.id) {
        throw new TRPCError({
          code: "FORBIDDEN",
          message: "无权操作此订单",
        });
      }

      if (order.status !== "PENDING") {
        throw new TRPCError({
          code: "BAD_REQUEST",
          message: "订单已处理，无法取消",
        });
      }

      // 恢复库存
      await ctx.prisma.$transaction(async (tx) => {
        for (const item of order.items) {
          await tx.product.update({
            where: { id: item.productId },
            data: {
              stock: {
                increment: item.quantity,
              },
            },
          });
        }

        await tx.order.update({
          where: { id: input.id },
          data: {
            status: "CANCELLED",
          },
        });
      });

      return { success: true };
    }),
});
```

```typescript
import { z } from "zod";
import { t } from "@my-monorepo/trpc";
import { TRPCError } from "@trpc/server";
import Stripe from "stripe";

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!, {
  apiVersion: "2024-11-20.acacia",
});

export const paymentsRouter = t.router({
  // ========== 创建支付意图 ==========
  createPaymentIntent: t.procedure
    .input(
      z.object({
        orderId: z.string(),
      })
    )
    .mutation(async ({ input, ctx }) => {
      if (!ctx.user) {
        throw new TRPCError({
          code: "UNAUTHORIZED",
          message: "请先登录",
        });
      }

      const order = await ctx.prisma.order.findUnique({
        where: { id: input.orderId },
      });

      if (!order) {
        throw new TRPCError({
          code: "NOT_FOUND",
          message: "订单不存在",
        });
      }

      if (order.userId !== ctx.user.id) {
        throw new TRPCError({
          code: "FORBIDDEN",
          message: "无权操作此订单",
        });
      }

      if (order.paymentStatus === "COMPLETED") {
        throw new TRPCError({
          code: "BAD_REQUEST",
          message: "订单已支付",
        });
      }

      // 创建 Stripe Payment Intent
      const paymentIntent = await stripe.paymentIntents.create({
        amount: Math.round(Number(order.total) * 100), // 转换为分
        currency: "usd",
        metadata: {
          orderId: order.id,
          orderNumber: order.orderNumber,
        },
      });

      // 更新订单
      await ctx.prisma.order.update({
        where: { id: input.orderId },
        data: {
          stripePaymentIntentId: paymentIntent.id,
        },
      });

      return {
        clientSecret: paymentIntent.client_secret,
        paymentIntentId: paymentIntent.id,
      };
    }),

  // ========== Webhook 处理 ==========
  handleWebhook: t.procedure.mutation(async ({ ctx }) => {
    const sig = ctx.req.headers["stripe-signature"];

    if (!sig) {
      throw new TRPCError({
        code: "BAD_REQUEST",
        message: "无效的签名",
      });
    }

    let event: Stripe.Event;

    try {
      event = stripe.webhooks.constructEvent(
        ctx.req.body,
        sig,
        process.env.STRIPE_WEBHOOK_SECRET!
      );
    } catch (err) {
      throw new TRPCError({
        code: "BAD_REQUEST",
        message: "Webhook 签名验证失败",
      });
    }

    // 处理事件
    switch (event.type) {
      case "payment_intent.succeeded":
        const paymentIntent = event.data.object as Stripe.PaymentIntent;
        const orderId = paymentIntent.metadata.orderId;

        await ctx.prisma.order.update({
          where: { id: orderId },
          data: {
            paymentStatus: "COMPLETED",
            status: "PROCESSING",
          },
        });

        // 发送确认邮件
        await sendOrderConfirmationEmail(orderId);
        break;

      case "payment_intent.payment_failed":
        const failedPayment = event.data.object as Stripe.PaymentIntent;
        const failedOrderId = failedPayment.metadata.orderId;

        await ctx.prisma.order.update({
          where: { id: failedOrderId },
          data: {
            paymentStatus: "FAILED",
          },
        });
        break;
    }

    return { received: true };
  }),
```

```typescript
// apps/web/src/hooks/useCart.ts

"use client";

import { useMutation, useQuery, useQueryClient } from "@tanstack/react-query";
import { trpc } from "@/lib/trpc/client";
import { toast } from "react-hot-toast";

export function useCart() {
  const queryClient = useQueryClient();

  // 获取购物车
  const { data: cart, isLoading } = trpc.cart.get.useQuery();

  // 添加到购物车
  const addToCartMutation = trpc.cart.addItem.useMutation({
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: [["cart", "get"]] });
      queryClient.invalidateQueries({ queryKey: [["cart", "getItemCount"]] });
      toast.success("已添加到购物车");
    },
    onError: (error) => {
      toast.error(error.message);
    },
  });

  // 更新数量
  const updateQuantityMutation = trpc.cart.updateQuantity.useMutation({
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: [["cart", "get"]] });
      queryClient.invalidateQueries({ queryKey: [["cart", "getItemCount"]] });
    },
  });

  // 删除商品
  const removeItemMutation = trpc.cart.removeItem.useMutation({
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: [["cart", "get"]] });
      queryClient.invalidateQueries({ queryKey: [["cart", "getItemCount"]] });
      toast.success("已删除");
    },
  });

  // 获取购物车数量
  const { data: itemCount } = trpc.cart.getItemCount.useQuery();

  const addToCart = (productId: string, quantity: number) => {
    addToCartMutation.mutate({ productId, quantity });
  };

  const updateQuantity = (itemId: string, quantity: number) => {
    updateQuantityMutation.mutate({ itemId, quantity });
  };

  const removeItem = (itemId: string) => {
    removeItemMutation.mutate({ itemId });
  };

  // 计算总价
  const subtotal =
    cart?.items.reduce((sum, item) => {
      return sum + Number(item.product.price) * item.quantity;
    }, 0) || 0;

  return {
    cart,
    isLoading,
    itemCount: itemCount || 0,
    addToCart,
    updateQuantity,
    removeItem,
    subtotal,
  };
}
```

```typescript
                inStock: update.stock > 0,
```

```typescript
    private readonly eventEmitter: EventEmitter2
  ) {}

  async onModuleInit() {
    // 检查低库存商品
    await this.checkLowStock();

    // 定时检查
    setInterval(() => this.checkLowStock(), 5 * 60 * 1000); // 每5分钟
  }

  async checkLowStock() {
    const lowStockProducts = await this.prisma.product.findMany({
      where: {
        stock: {
          lt: 10,
        },
        active: true,
      },
      include: {
        categories: true,
      },
    });

    if (lowStockProducts.length > 0) {
      this.eventEmitter.emit("inventory.low", {
        products: lowStockProducts,
        timestamp: new Date(),
      });

      // 发送邮件通知管理员
      await this.sendLowStockAlert(lowStockProducts);
    }
  }

  async onStockUpdated(productId: string, newStock: number) {
    this.eventEmitter.emit("inventory.updated", {

    if (newStock < 10) {
```
