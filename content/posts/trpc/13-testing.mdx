✅ **可预测的 API**：统一的数据格式

✅ **集成测试友好**：标准的 HTTP 接口

## 单元测试

### 测试 Procedure

```typescript
// src/trpc/routers/__tests__/users.router.test.ts

import { describe, it, expect, beforeEach } from "vitest";
import { t } from "../index";
import { PrismaService } from "@/database/prisma.service";

describe("Users Router", () => {
  let prisma: PrismaService;

  beforeEach(async () => {
    prisma = new PrismaService();
    await prisma.cleanDb();
  });

  describe("me", () => {
    it("应该返回当前用户信息", async () => {
      const caller = createCaller({
        user: {
          id: "1",
          name: "Alice",
          email: "alice@example.com",
          role: "USER",
        },
      });

      const result = await caller.users.me.query();

      expect(result).toEqual({
        id: "1",
        name: "Alice",
        email: "alice@example.com",
        role: "USER",
      });
    });

    it("未登录应该抛出错误", async () => {
      const caller = createCaller({
        user: null,
      });

      await expect(caller.users.me.query()).rejects.toThrowError(
        "UNAUTHORIZED"
      );
    });
  });

  describe("list", () => {
    it("应该返回用户列表", async () => {
      // 准备测试数据
      await prisma.user.createMany({
        data: [
          { id: "1", name: "Alice", email: "alice@example.com" },
          { id: "2", name: "Bob", email: "bob@example.com" },
        ],
      });

      const caller = createCaller({
        user: {
          id: "1",
          name: "Alice",
          email: "alice@example.com",
          role: "USER",
        },
      });

      const result = await caller.users.list.query({
        page: 1,
        limit: 20,
      });

      expect(result.users).toHaveLength(2);
      expect(result.pagination.total).toBe(2);
    });

    it("应该支持分页", async () => {
      // 创建 25 个用户
      const users = Array.from({ length: 25 }, (_, i) => ({
        id: `user-${i}`,
        name: `User ${i}`,
        email: `user${i}@example.com`,
      }));

      await prisma.user.createMany({
        data: users,
      });

      const caller = createCaller({
        user: {
          id: "1",
          name: "Admin",
          email: "admin@example.com",
          role: "ADMIN",
        },
      });

      const page1 = await caller.users.list.query({
        page: 1,
        limit: 10,
      });

      const page2 = await caller.users.list.query({
        page: 2,
        limit: 10,
      });

      expect(page1.users).toHaveLength(10);
      expect(page2.users).toHaveLength(10);
      expect(page1.pagination.totalPages).toBe(3);
    });
  });
});
```

### 测试工具函数

```typescript
// src/test/utils.ts

import { inferRouterInputs, inferRouterOutputs } from "@trpc/server";
import type { AppRouter } from "@/trpc";

// 创建测试用的 Caller
export function createCaller({
  user,
  prisma,
}: {
  user: User | null;
  prisma: PrismaService;
}) {
  const { createContext } = await import("@/trpc/context");

  const testContext = await createContext({
    req: {
      headers: new Headers({
        authorization: user ? `Bearer ${mockToken(user)}` : "",
      }),
    },
    res: {} as any,
  });

  // 如果需要认证用户
  if (user) {
    testContext.user = user;
  }

  const { t } = await import("@/trpc");

  return t.createCallerFactory<AppRouter>()(testContext);
}

// Mock 数据生成器
export function mockUser(overrides: Partial<User> = {}): User {
  return {
    id: "1",
    name: "Test User",
    email: "test@example.com",
    role: "USER",
    ...overrides,
  };
}

export function mockPost(overrides: Partial<Post> = {}): Post {
  return {
    id: "1",
    title: "Test Post",
    content: "Test Content",
    published: false,
    authorId: "1",
    createdAt: new Date(),
    updatedAt: new Date(),
    ...overrides,
  };
}
```

## 集成测试

### 测试 API 端点

```typescript
// src/test/api/users.test.ts

import { describe, it, expect, beforeAll } from "vitest";
import { set } from "zod";
import { createTRPCProxyClient, httpBatchLink } from "@trpc/client";
import { AppRouter } from "@/trpc/routers";
import { prisma } from "@/database/prisma.service";

describe("Users API", () => {
  let client: ReturnType<typeof createTRPCProxyClient<AppRouter>>;

  beforeAll(async () => {
    // 清空数据库
    await prisma.cleanDb();

    // 创建测试用户
    await prisma.user.create({
      data: {
        id: "1",
        name: "Test User",
        email: "test@example.com",
        role: "USER",
      },
    });

    // 创建 HTTP 服务器
    const { server } = await import("@/server");

    const httpServer = server.listen(0);

    const port = (httpServer.address() as any).port;

    // 创建 tRPC 客户端
    client = createTRPCProxyClient<AppRouter>({
      links: [
        httpBatchLink({
          url: `http://localhost:${port}/trpc`,
        }),
      ],
    });
  });

  it("应该获取用户列表", async () => {
    const result = await client.users.list.query({
      page: 1,
      limit: 20,
    });

    expect(result.users).toHaveLength(1);
    expect(result.users[0]).toHaveProperty("id");
    expect(result.users[0]).toHaveProperty("name");
  });

  it("应该验证输入参数", async () => {
    // 测试无效的页码
    await expect(
      client.users.list.query({
        page: -1, // 无效
        limit: 20,
      })
    ).rejects.toThrow();
  });

  it("应该处理认证错误", async () => {
    // 未登录调用需要认证的 API
    const unauthClient = createTRPCProxyClient<AppRouter>({
      links: [
        httpBatchLink({
          url: `http://localhost:${port}/trpc`,
        }),
      ],
    });

    await expect(unauthClient.users.me.query()).rejects.toThrow("UNAUTHORIZED");
  });
});
```

```typescript
// src/test/integration/posts.test.ts

import { describe, it, expect, beforeEach } from "vitest";
import { PrismaClient } from "@prisma/client";

describe("Posts Integration", () => {
  let prisma: PrismaClient;

  beforeEach(async () => {
    prisma = new PrismaClient();
    await prisma.post.deleteMany();
  });

  it("应该创建文章并关联用户", async () => {
    const user = await prisma.user.create({
      data: {
        id: "1",
        name: "Test User",
        email: "test@example.com",
        role: "USER",
      },
    });

    const post = await prisma.post.create({
      data: {
        title: "Test Post",
        content: "Test Content",
        authorId: user.id,
      },
    });

    expect(post).toHaveProperty("id");
    expect(post.title).toBe("Test Post");
    expect(post.authorId).toBe(user.id);
  });

  it("应该支持事务", async () => {
    await prisma.$transaction(async (tx) => {
      const user = await tx.user.create({
        data: {
          id: "1",
          name: "Test User",
          email: "test@example.com",
          role: "USER",
        },
      });

      await tx.post.create({
        data: {
          title: "Test Post",
          content: "Test Content",
          authorId: user.id,
        },
      });
    });

    const count = await prisma.post.count();
    expect(count).toBe(1);
  });
});
```

```typescript
// src/test/performance/posts.test.ts

import { describe, it, expect } from "vitest";
import { createCaller } from "../utils";

describe("Posts Performance", () => {
  it("应该在 100ms 内返回文章列表", async () => {
    const caller = createCaller({
      user: mockUser(),
    });

    const start = Date.now();

      page: 1,
      limit: 20,
    });

    const duration = Date.now() - start;

    expect(duration).toBeLessThan(100);

      caller.posts.list.query({ page: 3 }),
    ]);

    // 并发请求应该比顺序请求更快
    expect(duration).toBeLessThan(300);
  });
});
```

### 2. Factory 模式

```typescript
// src/test/factories/user.factory.ts

export class UserFactory {
  static async create(overrides: Partial<User> = {}) {
    return await prisma.user.create({
      data: {
        id: crypto.randomUUID(),
        name: "Test User",
        email: `test-${Date.now()}@example.com`,
        role: "USER",
        ...overrides,
      },
    });
  }

  static async createMany(count: number) {
    const users = Array.from({ length: count }, (_, i) => ({
      id: crypto.randomUUID(),
      name: `User ${i}`,
      email: `user${i}@example.com`,
      role: "USER" as const,
    }));

    return await prisma.user.createMany({
      data: users,
    });
  }
}

// 使用
const user = await UserFactory.create({ name: "Custom Name" });
const users = await UserFactory.createMany(10);
```
