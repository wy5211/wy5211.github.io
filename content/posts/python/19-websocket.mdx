# WebSocket 实时通信

## 前言

HTTP 协议是"请求-响应"模式，客户端主动请求，服务端被动响应。但在很多场景下（聊天室、实时通知、协作编辑），我们需要服务端主动向客户端推送数据。WebSocket 正是为了解决这一问题而生的全双工通信协议。

FastAPI 内置了对 WebSocket 的完整支持，让我们可以轻松构建实时功能。

**本文将带你：**

1. 理解 WebSocket 与 HTTP 的区别
2. 使用 FastAPI WebSocket API
3. 设计连接管理器（ConnectionManager）
4. 实现房间机制和广播/定向推送
5. 添加心跳检测和认证

---

## 核心概念

### WebSocket vs HTTP

```
HTTP 通信模式（半双工）：
客户端 ──── 请求 ────► 服务端
客户端 ◄─── 响应 ──── 服务端
（每次通信需要重新建立连接，Header 冗余大）

WebSocket 通信模式（全双工）：
客户端 ◄──────────────────► 服务端
（一次握手后持久连接，双向实时通信，开销极小）
```

**WebSocket 握手过程**：

```
1. 客户端发送 HTTP 请求，带有 Upgrade: websocket 头
2. 服务端返回 101 Switching Protocols
3. 之后双方通过同一个 TCP 连接双向通信
4. 客户端或服务端都可以随时发送消息
```

### 示例 1：连接管理器（ConnectionManager）

```python
"""
博客系统 - WebSocket 连接管理器
"""
import asyncio
import json
from datetime import datetime
from typing import Dict, Set, Optional
from fastapi import WebSocket, WebSocketDisconnect
import logging

logger = logging.getLogger(__name__)


class ConnectionManager:
    """
    WebSocket 连接管理器
    管理所有活跃连接，支持广播、房间、定向推送
    """

    def __init__(self):
        # 活跃连接: {websocket: user_id}
        self.active_connections: Dict[WebSocket, int] = {}
        # 房间: {room_id: set(websocket)}
        self.rooms: Dict[str, Set[WebSocket]] = {}
        # 用户到连接的映射（一个用户可能有多个连接）
        self.user_connections: Dict[int, Set[WebSocket]] = {}

    async def connect(self, websocket: WebSocket, user_id: int):
        """接受新的 WebSocket 连接"""
        await websocket.accept()
        self.active_connections[websocket] = user_id

        # 记录用户连接
        if user_id not in self.user_connections:
            self.user_connections[user_id] = set()
        self.user_connections[user_id].add(websocket)

        logger.info(f"用户 {user_id} 已连接，当前在线: {len(self.active_connections)}")

    def disconnect(self, websocket: WebSocket):
        """断开连接"""
        user_id = self.active_connections.pop(websocket, None)

        # 从所有房间中移除
        for room_id, members in self.rooms.items():
            members.discard(websocket)
            if not members:
                del self.rooms[room_id]

        # 从用户连接映射中移除
        if user_id and user_id in self.user_connections:
            self.user_connections[user_id].discard(websocket)
            if not self.user_connections[user_id]:
                del self.user_connections[user_id]

        logger.info(f"用户 {user_id} 已断开，当前在线: {len(self.active_connections)}")

    async def send_personal_message(self, message: dict, websocket: WebSocket):
        """向指定连接发送消息"""
        try:
            await websocket.send_json(message)
        except Exception as e:
            logger.error(f"发送消息失败: {e}")
            self.disconnect(websocket)

    async def send_to_user(self, user_id: int, message: dict):
        """向指定用户的所有连接发送消息"""
        if user_id in self.user_connections:
            for ws in self.user_connections[user_id].copy():
                await self.send_personal_message(message, ws)

    async def broadcast(self, message: dict):
        """向所有连接广播消息"""
        for connection in list(self.active_connections.keys()):
            await self.send_personal_message(message, connection)

    async def join_room(self, websocket: WebSocket, room_id: str):
        """加入房间"""
        if room_id not in self.rooms:
            self.rooms[room_id] = set()
        self.rooms[room_id].add(websocket)

        # 通知房间内其他成员
        user_id = self.active_connections.get(websocket)
        await self.broadcast_to_room(
            room_id,
            {
                "type": "user_joined",
                "user_id": user_id,
                "room_id": room_id,
                "online_count": len(self.rooms[room_id]),
            },
            exclude=websocket,
        )

    async def leave_room(self, websocket: WebSocket, room_id: str):
        """离开房间"""
        if room_id in self.rooms:
            self.rooms[room_id].discard(websocket)

            user_id = self.active_connections.get(websocket)
            await self.broadcast_to_room(
                room_id,
                {
                    "type": "user_left",
                    "user_id": user_id,
                    "room_id": room_id,
                    "online_count": len(self.rooms.get(room_id, set())),
                },
            )

    async def broadcast_to_room(
        self,
        room_id: str,
        message: dict,
        exclude: Optional[WebSocket] = None,
    ):
        """向房间内所有成员广播"""
        if room_id not in self.rooms:
            return
        for ws in self.rooms[room_id].copy():
            if ws != exclude:
                await self.send_personal_message(message, ws)

    def get_online_users(self, room_id: Optional[str] = None) -> list[int]:
        """获取在线用户列表"""
        if room_id and room_id in self.rooms:
            return [
                self.active_connections[ws]
                for ws in self.rooms[room_id]
                if ws in self.active_connections
            ]
        return list(self.user_connections.keys())


# 全局连接管理器
manager = ConnectionManager()
```

### 示例 2：实时聊天室

```python
"""
博客系统 - WebSocket 实时聊天室
"""
from datetime import datetime
from fastapi import FastAPI, WebSocket, WebSocketDisconnect, Query
from typing import Optional
import json
import uuid

app = FastAPI(title="博客系统 API - WebSocket 聊天")


@app.websocket("/ws/chat/{room_id}")
async def chat_websocket(
    websocket: WebSocket,
    room_id: str,
    token: Optional[str] = Query(None),
):
    """
    聊天室 WebSocket 端点

    消息格式:
    - 客户端发送: {"type": "message", "content": "你好"}
    - 服务端广播: {"type": "message", "user_id": 1, "username": "admin",
                   "content": "你好", "timestamp": "...", "message_id": "..."}
    """
    # ========== 认证 ==========
    user_id = authenticate_ws_token(token)
    if user_id is None:
        await websocket.close(code=4001, reason="认证失败")
        return

    # ========== 连接 ==========
    await manager.connect(websocket, user_id)
    await manager.join_room(websocket, room_id)

    # 获取用户信息
    username = f"User_{user_id}"

    # 发送加入欢迎消息
    await manager.send_personal_message(
        {
            "type": "system",
            "content": f"欢迎 {username} 加入聊天室",
            "room_id": room_id,
            "timestamp": datetime.utcnow().isoformat(),
        },
        websocket,
    )

    # ========== 消息循环 ==========
    try:
        while True:
            # 接收客户端消息
            data = await websocket.receive_text()

            try:
                message = json.loads(data)
            except json.JSONDecodeError:
                await manager.send_personal_message(
                    {"type": "error", "content": "无效的 JSON 格式"},
                    websocket,
                )
                continue

            msg_type = message.get("type")

            if msg_type == "message":
                # 聊天消息：广播给房间内所有人
                chat_msg = {
                    "type": "message",
                    "message_id": str(uuid.uuid4()),
                    "user_id": user_id,
                    "username": username,
                    "content": message.get("content", "")[:500],  # 限制长度
                    "room_id": room_id,
                    "timestamp": datetime.utcnow().isoformat(),
                }

                await manager.broadcast_to_room(room_id, chat_msg)

            elif msg_type == "typing":
                # 正在输入提示
                await manager.broadcast_to_room(
                    room_id,
                    {
                        "type": "typing",
                        "user_id": user_id,
                        "username": username,
                    },
                    exclude=websocket,
                )

            elif msg_type == "ping":
                # 心跳响应
                await manager.send_personal_message(
                    {"type": "pong"},
                    websocket,
                )

    except WebSocketDisconnect:
        manager.disconnect(websocket)
    except Exception as e:
        logger.error(f"WebSocket 异常: {e}")
        manager.disconnect(websocket)


def authenticate_ws_token(token: Optional[str]) -> Optional[int]:
    """简单的 Token 认证（实际项目中验证 JWT）"""
    if token and token.startswith("valid_"):
        return int(token.split("_")[1])
    return None
```

### 示例 4：心跳检测机制

```python
import asyncio
from fastapi import WebSocket

class HeartbeatManager:
    """心跳检测管理器"""

    def __init__(self, interval: float = 30.0, timeout: float = 60.0):
        self.interval = interval   # 心跳间隔（秒）
        self.timeout = timeout     # 超时时间（秒）
        # 记录每个连接的最后活跃时间
        self.last_active: dict[WebSocket, float] = {}

    def update_activity(self, websocket: WebSocket):
        """更新连接的活跃时间"""
        self.last_active[websocket] = asyncio.get_event_loop().time()

    def is_alive(self, websocket: WebSocket) -> bool:
        """检查连接是否存活"""
        if websocket not in self.last_active:
            return False
        elapsed = asyncio.get_event_loop().time() - self.last_active[websocket]
        return elapsed < self.timeout

    async def start_heartbeat(self, websocket: WebSocket):
        """启动心跳检测任务"""
        self.update_activity(websocket)
        try:
            while True:
                await asyncio.sleep(self.interval)
                # 发送 ping
                try:
                    await websocket.send_json({"type": "ping"})
                except Exception:
                    break

                # 等待 pong（设置超时）
                if not self.is_alive(websocket):
                    logger.warning(f"心跳超时，关闭连接")
                    await websocket.close(code=4002, reason="心跳超时")
                    break

        except asyncio.CancelledError:
            pass

heartbeat_mgr = HeartbeatManager(interval=30, timeout=90)


@app.websocket("/ws/heartbeat")
async def heartbeat_websocket(websocket: WebSocket):
    """带心跳检测的 WebSocket 连接"""
    await websocket.accept()

    # 启动心跳任务
    heartbeat_task = asyncio.create_task(heartbeat_mgr.start_heartbeat(websocket))

    try:
        while True:
            data = await websocket.receive_text()
            message = json.loads(data)

            if message.get("type") == "pong":
                # 收到心跳响应，更新活跃时间
                heartbeat_mgr.update_activity(websocket)

            # 处理其他消息...

    except WebSocketDisconnect:
        heartbeat_task.cancel()
    except Exception:
        heartbeat_task.cancel()
```

// 前端断线重连示例
let ws;
let reconnectAttempts = 0;
const MAX_RECONNECT = 5;

function connect() {
  ws = new WebSocket(`ws://localhost:8000/ws/chat/${roomId}?token=${token}`);

  ws.onopen = () => {
    reconnectAttempts = 0;
  };
  ws.onclose = () => {
    if (reconnectAttempts < MAX_RECONNECT) {
      reconnectAttempts++;
      setTimeout(connect, 1000 * reconnectAttempts); // 指数退避
    }
  };
}

## 常见问题

**Q: WebSocket 连接突然断开怎么办？**

A: 网络不稳定、代理超时、服务器重启都可能导致断开。实现心跳检测（ping/pong）来发现断连，前端使用指数退避策略重连。

**Q: FastAPI 的 WebSocket 支持多少并发连接？**

A: 取决于 Uvicorn 的配置和服务器资源。默认单进程大约能支持几千个并发连接。生产环境建议使用多 Worker + Nginx 反向代理。

**Q: WebSocket 如何做认证？**

A: 通过查询参数传递 Token（`ws://host/ws?token=xxx`），在连接建立时验证。WebSocket 协议不支持自定义 Header（浏览器限制）。

**Q: 需要水平扩展时怎么办？**

A: 单机 WebSocket 连接无法直接跨进程/跨服务器共享。水平扩展需要使用 Redis Pub/Sub 作为消息中间件，每个服务器实例订阅同一个频道，收到消息后转发给本地连接的客户端。
