## pathlib：路径操作

`pathlib` 是 Python 3.4 引入的面向对象路径操作库，全面替代 `os.path`：

```python
from pathlib import Path, PurePosixPath

# ==================== 基础操作 ====================

# 获取当前工作目录
cwd = Path.cwd()
print(cwd)  # /Users/you/project

# 构造路径（自动处理分隔符）
config_path = Path("src") / "config" / "settings.toml"
print(config_path)  # src/config/settings.toml

# 路径拼接
logs_dir = Path("logs")
log_file = logs_dir / "app.log"

# 获取路径属性
print(config_path.name)       # settings.toml（文件名）
print(config_path.stem)       # settings（不含扩展名）
print(config_path.suffix)     # .toml（扩展名）
print(config_path.parent)     # src/config（父目录）
print(config_path.parents)    # PosixPath.parents 对象

# ==================== 文件读写 ====================

def read_config() -> str:
    """读取配置文件"""
    config = Path("config") / "settings.toml"
    if not config.exists():
        return "default config"
    return config.read_text(encoding="utf-8")

def write_log(message: str) -> None:
    """写入日志文件"""
    log_dir = Path("logs")
    log_dir.mkdir(parents=True, exist_ok=True)  # 父目录不存在时自动创建

    log_file = log_dir / "app.log"
    log_file.write_text(message + "\n", encoding="utf-8")

# ==================== 目录遍历 ====================

def list_python_files(directory: str) -> list[Path]:
    """递归列出所有 Python 文件"""
    base = Path(directory)
    return list(base.rglob("*.py"))  # rglob 递归搜索，glob 只搜索当前目录

def organize_downloads(download_dir: str) -> dict[str, list[Path]]:
    """
    按扩展名整理下载目录
    返回分类后的文件字典
    """
    base = Path(download_dir)
    categories: dict[str, list[Path]] = {}

    for file in base.iterdir():
        if not file.is_file():
            continue
        ext = file.suffix.lower() or ".no_extension"
        if ext not in categories:
            categories[ext] = []
        categories[ext].append(file)

    return categories

# ==================== 实战：批量重命名文件 ====================

def batch_rename(directory: str, pattern: str = "file_", prefix: str = "doc_") -> int:
    """
    批量重命名文件
    将 file_001.txt, file_002.txt 重命名为 doc_001.txt, doc_002.txt
    返回重命名的文件数
    """
    base = Path(directory)
    count = 0

    for file in sorted(base.glob(f"{pattern}*")):
        if not file.is_file():
            continue
        new_name = file.name.replace(pattern, prefix)
        new_path = file.with_name(new_name)
        file.rename(new_path)
        count += 1
        print(f"  {file.name} -> {new_name}")

    return count


# ==================== 实战：清理临时文件 ====================

def cleanup_temp_files(directory: str, days: int = 7) -> int:
    """清理指定天数前的临时文件"""
    import time

    base = Path(directory)
    threshold = time.time() - days * 86400
    count = 0

    for file in base.rglob("*.tmp"):
        if file.is_file() and file.stat().st_mtime < threshold:
            file.unlink()  # 删除文件
            count += 1

    return count
```

```python
def convert_timezone(
    dt: datetime,
    from_tz: str,
    to_tz: str,
) -> datetime:
    """
    时区转换
    需要安装 tzdata：pip install tzdata
    """
    from zoneinfo import ZoneInfo  # Python 3.9+

    # 设置源时区
    dt_with_tz = dt.replace(tzinfo=ZoneInfo(from_tz))
    # 转换为目标时区
    return dt_with_tz.astimezone(ZoneInfo(to_tz))

class Timer:
    """简单的计时器"""

    def __init__(self) -> None:
        self._start: float = 0
        self._end: float = 0

    def __enter__(self) -> "Timer":
        self._start = time.perf_counter()
        return self

    def __exit__(self, *args) -> None:
        self._end = time.perf_counter()

    @property
    def elapsed(self) -> float:
        if self._end == 0:
            return time.perf_counter() - self._start
        return self._end - self._start

    def __repr__(self) -> str:
        return f"Timer(elapsed={self.elapsed:.4f}s)"


# 使用
with Timer() as t:
    sum(range(10_000_000))

print(f"耗时: {t.elapsed:.4f} 秒")
```

```python
    users = [

            print(f"  {row['name']}: {row['email']}")


# ==================== 实战：JSON 转换工具 ====================

def flatten_json(data: dict, parent_key: str = "", sep: str = ".") -> dict:
    """
    将嵌套 JSON 扁平化
    {"user": {"name": "Alice"}} -> {"user.name": "Alice"}
    """
    items: list[tuple[str, object]] = []
    for key, value in data.items():
        new_key = f"{parent_key}{sep}{key}" if parent_key else key
        if isinstance(value, dict):
            items.extend(flatten_json(value, new_key, sep).items())
        else:
            items.append((new_key, value))
    return dict(items)


nested = {"user": {"name": "Alice", "address": {"city": "Beijing"}}}
flat = flatten_json(nested)
print(flat)
# {'user.name': 'Alice', 'user.address.city': 'Beijing'}
```

## re：正则表达式

```python
import re


def regex_examples() -> None:
    """常用正则表达式示例"""

    # ==================== 基础匹配 ====================

    text = "Contact us at alice@example.com or bob@company.org for more info."

    # 查找所有邮箱
    emails = re.findall(r"[\w.+-]+@[\w-]+\.[\w.-]+", text)
    print(emails)  # ['alice@example.com', 'bob@company.org']

    # 提取手机号
    phone_text = "Call me at 13812345678 or 010-87654321"
    phones = re.findall(r"1[3-9]\d{9}|0\d{2,3}-?\d{7,8}", phone_text)
    print(phones)  # ['13812345678', '010-87654321']

    # ==================== 替换 ====================

    # 脱敏手机号
    raw = "Phone: 13812345678, Backup: 13987654321"
    masked = re.sub(r"(1\d{2})\d{4}(\d{4})", r"\1****\2", raw)
    print(masked)  # Phone: 138****5678, Backup: 139****4321

    # 去除多余空格
    messy = "hello   world  this  is   messy"
    clean = re.sub(r"\s+", " ", messy)
    print(clean)  # hello world this is messy

    # ==================== 分割 ====================

    log = "2026-03-29|INFO|User login|alice"
    parts = re.split(r"\|", log)
    print(parts)  # ['2026-03-29', 'INFO', 'User login', 'alice']

    # ==================== 编译正则（提高复用性能） ====================

    EMAIL_PATTERN = re.compile(r"^[\w.+-]+@[\w-]+\.[\w.-]+$")

    def is_valid_email(email: str) -> bool:
        return bool(EMAIL_PATTERN.match(email))

    print(is_valid_email("alice@example.com"))  # True
    print(is_valid_email("invalid-email"))       # False

    # ==================== 命名捕获组 ====================

    log_line = '[2026-03-29 10:30:00] ERROR user_service.py:42 - User not found: id=123'
    pattern = re.compile(
        r"\[(?P<timestamp>.*?)\] "
        r"(?P<level>\w+) "
        r"(?P<source>.*?):(?P<line>\d+) - "
        r"(?P<message>.*)"
    )
    match = pattern.match(log_line)
    if match:
        print(match.group("level"))      # ERROR
        print(match.group("timestamp"))  # 2026-03-29 10:30:00
        print(match.group("message"))    # User not found: id=123
```

## itertools：迭代工具

`itertools` 提供了高效的迭代器操作函数，处理大数据集时尤其有用：

```python
from itertools import (
    chain, groupby, product, combinations,
    permutations, islice, count, cycle,
    accumulate, takewhile, dropwhile,
)
from collections import Counter


def itertools_examples() -> None:
    """itertools 常用示例"""

    # ==================== chain：连接多个可迭代对象 ====================

    users_ids = [[1, 2, 3], [4, 5], [6, 7, 8, 9]]
    all_ids = list(chain.from_iterable(users_ids))
    print(all_ids)  # [1, 2, 3, 4, 5, 6, 7, 8, 9]
    # 比 [id for group in users_ids for id in group] 更高效

    # ==================== groupby：分组（需要先排序） ====================

    transactions = [
        {"date": "2026-01-15", "amount": 100},
        {"date": "2026-01-20", "amount": 200},
        {"date": "2026-02-05", "amount": 150},
        {"date": "2026-02-10", "amount": 300},
        {"date": "2026-03-01", "amount": 50},
    ]

    # 按月份分组
    from operator import itemgetter

    transactions.sort(key=lambda t: t["date"][:7])  # 按年月排序
    monthly: dict[str, list[int]] = {}

    for month, group in groupby(transactions, key=lambda t: t["date"][:7]):
        amounts = [t["amount"] for t in group]
        monthly[month] = amounts
        print(f"  {month}: {sum(amounts)}")

    # ==================== product：笛卡尔积 ====================

    colors = ["red", "blue"]
    sizes = ["S", "M", "L"]
    variants = list(product(colors, sizes))
    print(variants)
    # [('red', 'S'), ('red', 'M'), ('red', 'L'), ('blue', 'S'), ...]

    # ==================== combinations / permutations ====================

    team = ["Alice", "Bob", "Charlie"]
    # 组合（不考虑顺序）
    pairs = list(combinations(team, 2))
    print(pairs)  # [('Alice', 'Bob'), ('Alice', 'Charlie'), ('Bob', 'Charlie')]

    # 排列（考虑顺序）
    orders = list(permutations(team, 2))
    print(orders)  # [('Alice', 'Bob'), ('Bob', 'Alice'), ...]

    # ==================== islice：切片迭代器 ====================

    # 读取文件的前 10 行（不加载整个文件）
    # first_10 = list(islice(open("large_file.txt"), 10))

    # ==================== accumulate：累积运算 ====================

    numbers = [1, 2, 3, 4, 5]
    running_sum = list(accumulate(numbers))
    print(running_sum)  # [1, 3, 6, 10, 15]

    running_product = list(accumulate(numbers, lambda a, b: a * b))
    print(running_product)  # [1, 2, 6, 24, 120]

    # ==================== takewhile / dropwhile ====================

    data = [1, 2, 3, 10, 20, 30, 1, 2]

    # 取到第一个不满足条件的元素为止
    taken = list(takewhile(lambda x: x < 10, data))
    print(taken)  # [1, 2, 3]

    # 跳过直到满足条件
    dropped = list(dropwhile(lambda x: x < 10, data))
    print(dropped)  # [10, 20, 30, 1, 2]
```

```python
from itertools import chain, islice, filterfalse


def data_pipeline() -> None:
    """模拟数据处理管道"""

    # 模拟三个数据源
    source_a = [{"id": 1, "score": 85}, {"id": 2, "score": 92}]
    source_b = [{"id": 3, "score": 45}, {"id": 4, "score": 78}]
```

## functools：函数工具

```python
from functools import cache, lru_cache, reduce, wraps, partial, singledispatch


# ==================== cache / lru_cache：缓存 ====================

@cache
def fibonacci(n: int) -> int:
    """使用缓存优化递归（自动缓存所有结果）"""
    if n < 2:
        return n
    return fibonacci(n - 1) + fibonacci(n - 2)

print(fibonacci(50))  # 瞬间出结果（没有缓存会非常慢）

# lru_cache：限制缓存大小（适合参数空间大的场景）
@lru_cache(maxsize=128)
def get_user_from_db(user_id: int) -> dict:
    """模拟数据库查询（结果会被缓存）"""
    print(f"  Querying DB for user {user_id}")
    return {"id": user_id, "name": f"User{user_id}"}

get_user_from_db(1)  # Querying DB for user 1
get_user_from_db(1)  # 直接从缓存返回（无打印）
get_user_from_db(2)  # Querying DB for user 2

# 查看缓存信息
print(get_user_from_db.cache_info())
# CacheInfo(hits=1, misses=2, maxsize=128, currsize=2)

# ==================== wraps：保留被装饰函数的元信息 ====================

def log_execution(func):
    """日志装饰器（使用 wraps 保留函数元信息）"""
    @wraps(func)
    def wrapper(*args, **kwargs):
        print(f"Calling {func.__name__}")
        result = func(*args, **kwargs)
        print(f"{func.__name__} returned {result}")
        return result
    return wrapper

@log_execution
def calculate(a: int, b: int) -> int:
    """Add two numbers"""
    return a + b

print(calculate.__name__)  # calculate（不是 wrapper）
print(calculate.__doc__)   # Add two numbers（保留了文档字符串）

# 固定部分参数，生成新函数

@singledispatch
```

## subprocess：进程管理

```python
import subprocess


def subprocess_examples() -> None:
    """subprocess 常用示例"""

    # ==================== 运行命令并获取输出 ====================

    # 推荐：使用 run（Python 3.5+）
    result = subprocess.run(
        ["git", "status", "--short"],
        capture_output=True,
        text=True,       # 输出为字符串（不是 bytes）
        timeout=10,      # 超时时间
        check=False,     # 不自动检查返回码
    )

    if result.returncode == 0:
        print(f"Output: {result.stdout.strip()}")
    else:
        print(f"Error: {result.stderr.strip()}")

    # ==================== 运行命令并实时输出 ====================

    result = subprocess.run(
        ["python", "-c", "print('Hello from subprocess')"],
        capture_output=True,
        text=True,
    )
    print(result.stdout)  # Hello from subprocess

    # ==================== 管道：命令链 ====================

    # 模拟 cat file | grep pattern
    p1 = subprocess.Popen(["cat", "config.toml"], stdout=subprocess.PIPE)
    p2 = subprocess.Popen(["grep", "port"], stdin=p1.stdout, stdout=subprocess.PIPE, text=True)
    p1.stdout.close()  # 允许 p1 接收 SIGPIPE
    output = p2.communicate()[0]
    print(f"Found: {output.strip()}")

    # ==================== 检查命令是否存在 ====================

    def command_exists(cmd: str) -> bool:
        try:
            subprocess.run(
                ["which", cmd],
                capture_output=True,
                check=True,
            )
            return True
        except (subprocess.CalledProcessError, FileNotFoundError):
            return False

    print(f"git exists: {command_exists('git')}")
    print(f"docker exists: {command_exists('docker')}")
```

## dataclasses：数据类

```python
from dataclasses import dataclass, field, asdict, astuple


@dataclass
class Point:
    """简单的数据类"""
    x: float
    y: float

    def distance_to(self, other: "Point") -> float:
        return ((self.x - other.x) ** 2 + (self.y - other.y) ** 2) ** 0.5


@dataclass
class User:
    """带默认值和验证的数据类"""
    id: int
    name: str
    email: str
    tags: list[str] = field(default_factory=list)  # 可变默认值必须用 default_factory
    is_active: bool = True

    def __post_init__(self) -> None:
        """初始化后验证"""
        if "@" not in self.email:
            raise ValueError(f"Invalid email: {self.email}")


@dataclass(frozen=True, slots=True)
class Config:
    """
    frozen=True: 不可变（适合作为字典 key 或放入 set）
    slots=True:  减少内存使用（Python 3.10+）
    """
    host: str
    port: int = 8000
    debug: bool = False


# 使用
user = User(id=1, name="Alice", email="alice@example.com", tags=["admin", "dev"])
print(asdict(user))   # 转为字典
print(astuple(user))  # 转为元组

# frozen dataclass 可以 hash
config = Config(host="localhost")
configs = {config}  # 可以放入 set

# 继承
@dataclass
class AdminUser(User):
    permissions: list[str] = field(default_factory=lambda: ["read", "write", "delete"])

admin = AdminUser(id=1, name="Root", email="root@example.com")
print(admin.permissions)  # ['read', 'write', 'delete']
```

```python
import asyncio


def httpx_examples() -> None:
    """httpx 基本用法"""

    # ==================== 同步请求 ====================

    # GET 请求
    with httpx.Client(base_url="https://jsonplaceholder.typicode.com") as client:
        # 获取用户列表
        response = client.get("/users")
        users = response.json()
        print(f"Users: {len(users)}")

        # 带查询参数
        response = client.get("/posts", params={"userId": 1})
        posts = response.json()
        print(f"Posts by user 1: {len(posts)}")

        # POST 请求
        response = client.post(
            "/posts",
            json={"title": "Hello", "body": "World", "userId": 1},
        )
        print(f"Created post: {response.json()['id']}")

        # 超时和重试
        response = client.get("/users/1", timeout=5.0)

    # ==================== 异步请求 ====================

    async def fetch_users() -> list[dict]:
        async with httpx.AsyncClient(base_url="https://jsonplaceholder.typicode.com") as client:
            response = await client.get("/users")
            return response.json()

    async def fetch_multiple() -> None:
        """并发请求多个资源"""
        async with httpx.AsyncClient(base_url="https://jsonplaceholder.typicode.com") as client:
            # 并发请求
            tasks = [
                client.get("/users"),
                client.get("/posts"),
                client.get("/comments"),
            ]
            import asyncio
            responses = await asyncio.gather(*tasks)
            for resp in responses:
                print(f"  {resp.url}: {len(resp.json())} items")

    asyncio.run(fetch_multiple())
```

## 最佳实践

1. **优先使用 pathlib**：`Path("a/b/c")` 比 `os.path.join("a", "b", "c")` 更简洁
2. **datetime 始终带时区**：使用 `datetime.now(timezone.utc)` 而不是 `datetime.now()`
3. **JSON 使用 indent 和 ensure_ascii=False**：输出可读的中文 JSON
4. **正则表达式预编译**：重复使用时用 `re.compile()` 提高性能
5. **itertools 处理大数据**：生成器不会一次性加载到内存
6. **functools.cache 注意内存**：`@cache` 会无限缓存，大量数据用 `@lru_cache(maxsize=N)`
7. **subprocess 优先用 `run()`**：比 `Popen` 更安全，`check=True` 自动检查错误
8. **dataclass + slots**：Python 3.10+ 使用 `slots=True` 减少内存占用

## 常见问题

**Q：pathlib 和 os.path 应该用哪个？**

A：新代码统一使用 `pathlib`。它的 API 更面向对象，方法更丰富。仅在需要兼容非常老的 Python 版本（< 3.4）时才用 `os.path`。

**Q：httpx 和 requests 有什么区别？**

A：httpx 同时支持同步和异步 API，支持 HTTP/2，维护更活跃。requests 只支持同步。如果你在用 FastAPI 或做异步编程，直接选 httpx。

**Q：@cache 和 @lru_cache 如何选择？**

A：如果缓存大小可控（参数空间不大），用 `@cache`（更快，无大小限制）。如果参数空间很大，用 `@lru_cache(maxsize=N)` 避免内存溢出。两者都只适用于纯函数（相同输入返回相同输出）。
