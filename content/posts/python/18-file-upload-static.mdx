# 文件上传与静态资源

## 前言

文件上传是博客系统的核心功能之一：用户头像、文章封面、图片附件等都需要可靠的上传和管理。FastAPI 通过 `UploadFile` 提供了简洁的文件上传接口，配合 Pillow 可以实现图片压缩、缩略图生成等处理。

**本文将带你：**

1. 使用 FastAPI 的 UploadFile 实现文件上传
2. 实现文件类型与大小验证
3. 本地文件存储策略
4. 使用 Pillow 处理图片
5. 配置静态文件服务
6. 了解大文件流式上传和云存储集成

---

## 核心概念

### FastAPI 文件上传原理

FastAPI 的 `UploadFile` 本质上是对 Python 标准库 `SpooledTemporaryFile` 的封装：

```
客户端上传文件
    │
    ▼
FastAPI 接收请求
    │
    ▼
UploadFile 对象（内存缓存 + 临时文件）
    ├── filename: 原始文件名
    ├── content_type: MIME 类型
    ├── file: 文件对象
    ├── read(): 读取内容
    ├── write(): 写入内容
    └── save(): 保存到磁盘
```

### 示例 1：头像上传（类型与大小验证）

```python
"""
博客系统 - 文件上传模块
"""
import os
import uuid
import shutil
from pathlib import Path
from datetime import datetime

from fastapi import FastAPI, UploadFile, File, HTTPException, Form
from fastapi.responses import FileResponse
from PIL import Image
import io

app = FastAPI(title="博客系统 API")

# ========== 配置 ==========

# 上传目录
UPLOAD_DIR = Path("uploads")
AVATAR_DIR = UPLOAD_DIR / "avatars"
POST_IMAGE_DIR = UPLOAD_DIR / "posts"

# 确保目录存在
AVATAR_DIR.mkdir(parents=True, exist_ok=True)
POST_IMAGE_DIR.mkdir(parents=True, exist_ok=True)

# 允许的图片类型
ALLOWED_IMAGE_TYPES = {
    "image/jpeg": ".jpg",
    "image/png": ".png",
    "image/gif": ".gif",
    "image/webp": ".webp",
}

# 文件大小限制（单位：字节）
MAX_AVATAR_SIZE = 2 * 1024 * 1024    # 2MB
MAX_POST_IMAGE_SIZE = 5 * 1024 * 1024  # 5MB


# ========== 工具函数 ==========

def validate_image(
    file: UploadFile,
    max_size: int,
    allowed_types: dict[str, str] = ALLOWED_IMAGE_TYPES,
) -> str:
    """
    验证上传的图片文件
    返回文件扩展名
    """
    # 检查文件类型
    if file.content_type not in allowed_types:
        raise HTTPException(
            status_code=400,
            detail=f"不支持的文件类型: {file.content_type}，"
                   f"允许的类型: {', '.join(allowed_types.keys())}",
        )

    # 检查文件大小
    file.file.seek(0, 2)  # 移动到文件末尾
    file_size = file.file.tell()
    file.file.seek(0)     # 重置到文件开头

    if file_size > max_size:
        max_mb = max_size / (1024 * 1024)
        actual_mb = file_size / (1024 * 1024)
        raise HTTPException(
            status_code=400,
            detail=f"文件大小超出限制: {actual_mb:.1f}MB > {max_mb}MB",
        )

    return allowed_types[file.content_type]


def generate_filename(original_ext: str) -> str:
    """生成唯一的文件名"""
    timestamp = datetime.now().strftime("%Y%m%d")
    unique_id = uuid.uuid4().hex[:8]
    return f"{timestamp}_{unique_id}{original_ext}"


# ========== 头像上传 ==========

@app.post("/api/upload/avatar")
async def upload_avatar(
    file: UploadFile = File(..., description="头像图片"),
    user_id: str = Form(..., description="用户 ID"),
):
    """
    上传用户头像
    - 支持 JPEG、PNG、GIF、WebP
    - 最大 2MB
    - 自动生成缩略图
    """
    # 验证文件
    ext = validate_image(file, MAX_AVATAR_SIZE)

    # 读取图片内容
    content = await file.read()

    # 使用 Pillow 处理图片
    try:
        image = Image.open(io.BytesIO(content))

        # 确保是 RGB 模式（处理 RGBA 的 PNG）
        if image.mode in ("RGBA", "P"):
            image = image.convert("RGB")

        # 裁剪为正方形（居中裁剪）
        width, height = image.size
        min_side = min(width, height)
        left = (width - min_side) // 2
        top = (height - min_side) // 2
        image = image.crop((left, top, min_side + left, min_side + top))

        # 调整大小
        avatar_sizes = {
            "large": (256, 256),
            "medium": (128, 128),
            "small": (48, 48),
        }

        saved_files = {}
        for size_name, (w, h) in avatar_sizes.items():
            resized = image.resize((w, h), Image.Resampling.LANCZOS)

            # 生成文件名
            filename = generate_filename(".jpg")
            filepath = AVATAR_DIR / f"{size_name}" / filename

            # 确保子目录存在
            filepath.parent.mkdir(parents=True, exist_ok=True)

            # 保存（JPEG 格式，质量 85）
            resized.save(filepath, "JPEG", quality=85, optimize=True)
            saved_files[size_name] = f"/static/avatars/{size_name}/{filename}"

        return {
            "message": "头像上传成功",
            "user_id": user_id,
            "sizes": saved_files,
        }

    except Exception as e:
        raise HTTPException(status_code=500, detail=f"图片处理失败: {str(e)}")
```

### 示例 2：批量图片上传

```python
from fastapi import UploadFile, File, HTTPException
from typing import List

@app.post("/api/upload/images")
async def upload_post_images(
    files: List[UploadFile] = File(..., description="文章图片（最多 9 张）"),
    post_id: str = Form(..., description="文章 ID"),
):
    """
    批量上传文章图片
    - 最多 9 张
    - 每张最大 5MB
    - 自动压缩并生成缩略图
    """
    # 数量限制
    if len(files) > 9:
        raise HTTPException(status_code=400, detail="最多上传 9 张图片")

    results = []
    for index, file in enumerate(files):
        try:
            # 验证文件
            ext = validate_image(file, MAX_POST_IMAGE_SIZE)
            content = await file.read()

            # 处理图片
            image = Image.open(io.BytesIO(content))
            if image.mode in ("RGBA", "P"):
                image = image.convert("RGB")

            # 压缩图片（最大宽度 1200px，质量 80）
            max_width = 1200
            if image.width > max_width:
                ratio = max_width / image.width
                new_height = int(image.height * ratio)
                image = image.resize((max_width, new_height), Image.Resampling.LANCZOS)

            # 保存原图
            filename = generate_filename(".jpg")
            original_path = POST_IMAGE_DIR / post_id / "original" / filename
            original_path.parent.mkdir(parents=True, exist_ok=True)
            image.save(original_path, "JPEG", quality=80, optimize=True)

            # 生成缩略图（宽度 400px）
            thumb_height = int(image.height * (400 / image.width))
            thumbnail = image.resize((400, thumb_height), Image.Resampling.LANCZOS)
            thumb_path = POST_IMAGE_DIR / post_id / "thumbnail" / filename
            thumb_path.parent.mkdir(parents=True, exist_ok=True)
            thumbnail.save(thumb_path, "JPEG", quality=75, optimize=True)

            results.append({
                "index": index,
                "original_name": file.filename,
                "saved_name": filename,
                "original_url": f"/static/posts/{post_id}/original/{filename}",
                "thumbnail_url": f"/static/posts/{post_id}/thumbnail/{filename}",
                "size": f"{len(content) / 1024:.1f}KB",
            })

        except HTTPException:
            raise
        except Exception as e:
            results.append({
                "index": index,
                "original_name": file.filename,
                "error": str(e),
            })

    success_count = len([r for r in results if "error" not in r])
    return {
        "message": f"上传完成: {success_count}/{len(files)} 张成功",
        "post_id": post_id,
        "images": results,
    }
```

### 示例 3：图片处理工具（压缩、水印）

```python
"""
图片处理工具模块
"""
from PIL import Image, ImageDraw, ImageFont
import io
from pathlib import Path

class ImageProcessor:
    """图片处理器"""

    @staticmethod
    def compress(
        image_data: bytes,
        max_width: int = 1200,
        quality: int = 80,
    ) -> bytes:
        """压缩图片并返回字节数据"""
        image = Image.open(io.BytesIO(image_data))

        # 调整尺寸
        if image.width > max_width:
            ratio = max_width / image.width
            new_height = int(image.height * ratio)
            image = image.resize((max_width, new_height), Image.Resampling.LANCZOS)

        # 转换颜色模式
        if image.mode in ("RGBA", "P"):
            image = image.convert("RGB")

        # 压缩输出
        buffer = io.BytesIO()
        image.save(buffer, format="JPEG", quality=quality, optimize=True)
        return buffer.getvalue()

    @staticmethod
    def create_thumbnail(
        image_data: bytes,
        width: int = 400,
        height: int = 300,
    ) -> bytes:
        """创建缩略图（居中裁剪填充）"""
        image = Image.open(io.BytesIO(image_data))

        # 居中裁剪
        img_ratio = image.width / image.height
        target_ratio = width / height

        if img_ratio > target_ratio:
            # 图片更宽，裁剪左右
            new_width = int(image.height * target_ratio)
            left = (image.width - new_width) // 2
            image = image.crop((left, 0, left + new_width, image.height))
        else:
            # 图片更高，裁剪上下
            new_height = int(image.width / target_ratio)
            top = (image.height - new_height) // 2
            image = image.crop((0, top, image.width, top + new_height))

        # 缩放到目标尺寸
        image = image.resize((width, height), Image.Resampling.LANCZOS)

        buffer = io.BytesIO()
        image.save(buffer, format="JPEG", quality=80)
        return buffer.getvalue()

    @staticmethod
    def add_watermark(
        image_data: bytes,
        text: str = "MyBlog",
        opacity: int = 128,
    ) -> bytes:
        """添加文字水印"""
        image = Image.open(io.BytesIO(image_data))

        # 确保是 RGBA 模式以支持透明度
        if image.mode != "RGBA":
            image = image.convert("RGBA")

        # 创建水印图层
        watermark_layer = Image.new("RGBA", image.size, (255, 255, 255, 0))
        draw = ImageDraw.Draw(watermark_layer)

        # 计算水印位置和大小
        font_size = max(image.width // 20, 16)
        try:
            font = ImageFont.truetype("Arial.ttf", font_size)
        except OSError:
            font = ImageFont.load_default()

        # 计算文字位置（右下角）
        bbox = draw.textbbox((0, 0), text, font=font)
        text_width = bbox[2] - bbox[0]
        text_height = bbox[3] - bbox[1]
        x = image.width - text_width - 20
        y = image.height - text_height - 20

        # 绘制水印
        draw.text((x, y), text, font=font, fill=(255, 255, 255, opacity))

        # 合并图层
        watermarked = Image.alpha_composite(image, watermark_layer)

        # 转回 RGB
        watermarked = watermarked.convert("RGB")

        buffer = io.BytesIO()
        watermarked.save(buffer, format="JPEG", quality=90)
        return buffer.getvalue()

# 使用示例
@app.post("/api/upload/with-watermark")
async def upload_with_watermark(
    file: UploadFile = File(...),
):
    """上传图片并添加水印"""
    ext = validate_image(file, MAX_POST_IMAGE_SIZE)
    content = await file.read()

    processor = ImageProcessor()

    # 压缩
    compressed = processor.compress(content, max_width=1200, quality=80)
    # 添加水印
    watermarked = processor.add_watermark(compressed, text="MyBlog")
    # 生成缩略图
    thumbnail = processor.create_thumbnail(watermarked, width=400, height=300)

    filename = generate_filename(".jpg")
    # 保存文件...（略）
    return {"filename": filename}
```

### 示例 5：大文件流式上传

```python
import shutil
from fastapi import UploadFile, File, HTTPException
from pathlib import Path

# 大文件配置
CHUNK_SIZE = 1024 * 1024  # 1MB 每块
MAX_FILE_SIZE = 100 * 1024 * 1024  # 100MB
UPLOAD_CHUNK_DIR = Path("uploads/chunks")

@app.post("/api/upload/large")
async def upload_large_file(
    file: UploadFile = File(...),
):
    """
    大文件流式上传
    - 使用分块读取，避免内存溢出
    - 支持断点续传（需要前端配合）
    """
    # 验证文件类型
    if file.content_type not in (
        "application/pdf",
        "application/zip",
        "video/mp4",
    ):
        raise HTTPException(status_code=400, detail="不支持的文件类型")

    # 生成目标路径
    filename = generate_filename(Path(file.filename).suffix if file.filename else ".bin")
    filepath = UPLOAD_CHUNK_DIR / filename
    filepath.parent.mkdir(parents=True, exist_ok=True)

    # 流式写入文件（分块读取，避免一次性加载到内存）
    total_size = 0
    try:
        with open(filepath, "wb") as f:
            while True:
                chunk = await file.read(CHUNK_SIZE)
                if not chunk:
                    break
                total_size += len(chunk)

                # 检查文件大小
                if total_size > MAX_FILE_SIZE:
                    filepath.unlink()  # 删除已写入的文件
                    raise HTTPException(
                        status_code=400,
                        detail=f"文件大小超出限制: {total_size / (1024*1024):.1f}MB",
                    )

                f.write(chunk)

        return {
            "message": "文件上传成功",
            "filename": filename,
            "size": f"{total_size / (1024*1024):.2f}MB",
            "url": f"/static/chunks/{filename}",
        }

    except HTTPException:
        raise
    except Exception as e:
        # 清理不完整的文件
        if filepath.exists():
            filepath.unlink()
        raise HTTPException(status_code=500, detail=f"上传失败: {str(e)}")
```

```python
"""
云存储抽象层（S3 / 阿里云 OSS）
使用 boto3 作为示例（AWS S3）
"""
import boto3
from botocore.exceptions import ClientError
from typing import Optional
from pathlib import Path

class CloudStorage:
    """云存储抽象层"""

    def __init__(
        self,
        endpoint_url: Optional[str] = None,
        aws_access_key_id: str = "",
        aws_secret_access_key: str = "",
        bucket_name: str = "blog-uploads",
        region: str = "us-east-1",
    ):
        self.bucket_name = bucket_name
        self.client = boto3.client(
            "s3",
            endpoint_url=endpoint_url,  # 本地 MinIO 或阿里云 OSS endpoint
            aws_access_key_id=aws_access_key_id,
            aws_secret_access_key=aws_secret_access_key,
            region_name=region,
        )

    def upload(
        self,
        file_data: bytes,
        object_key: str,
        content_type: str = "image/jpeg",
    ) -> str:
        """
        上传文件到云存储
        返回公开访问 URL
        """
        try:
            self.client.put_object(
                Bucket=self.bucket_name,
                Key=object_key,
                Body=file_data,
                ContentType=content_type,
            )
            return f"https://{self.bucket_name}.s3.amazonaws.com/{object_key}"
        except ClientError as e:
            raise RuntimeError(f"上传失败: {e}")

    def delete(self, object_key: str) -> bool:
        """删除云存储文件"""
        try:
            self.client.delete_object(Bucket=self.bucket_name, Key=object_key)
            return True
        except ClientError:
            return False
```

## 常见问题

**Q: UploadFile 的 read() 和 file.read() 有什么区别？**

A: `UploadFile` 的 `read()` 方法是异步的（`async def read`），内部会正确处理临时文件。而 `file.file` 是底层的同步文件对象，直接调用 `file.file.read()` 是同步操作。推荐使用 `await file.read()`。

**Q: 上传的文件存在内存中还是磁盘上？**

A: FastAPI 使用 `SpooledTemporaryFile`，小文件（默认 < 2MB）存在内存中，大文件会自动写入临时磁盘。超过 `max_memory_file_size` 后会写入 `tempfile`。

**Q: 如何处理上传进度？**

A: FastAPI 原生不支持上传进度回调。可以通过 WebSocket 实时推送进度，或者使用分片上传方案（前端将文件分成多个 chunk 逐个上传）。

**Q: 生产环境应该用本地存储还是云存储？**

A: 生产环境推荐使用云存储（S3、OSS、COS），原因：CDN 加速、自动备份、高可用、弹性扩展。本地存储适合开发和小型项目。
