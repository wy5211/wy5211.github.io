## 核心概念

### 认证 vs 授权

| 概念     | 英文           | 问题         | 常见方案               |
| -------- | -------------- | ------------ | ---------------------- |
| **认证** | Authentication | 你是谁？     | 用户名密码、JWT、OAuth |
| **授权** | Authorization  | 你能做什么？ | RBAC、ABAC、ACL        |

### JWT 原理

JWT 由三部分组成，用 `.` 分隔：

```
Header.Payload.Signature
```

```json
// Header（头部）-- 指定算法
{
  "alg": "HS256",
  "typ": "JWT"
}

// Payload（载荷）-- 存放用户信息
{
  "sub": "1",          // 用户 ID
  "username": "alice",  // 用户名
  "role": "user",       // 角色
  "exp": 1700000000     # 过期时间
}

// Signature（签名）-- 防止篡改
HMACSHA256(
  base64UrlEncode(header) + "." + base64UrlEncode(payload),
  your-256-bit-secret
)
```

> **类比理解**：JWT 就像一张带有防伪印章的身份证。Header 和 Payload 是明文（Base64 编码，不是加密），任何人都能看到内容；Signature 确保内容没有被篡改。

# 创建密码上下文 -- 使用 bcrypt 算法
pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")


def hash_password(password: str) -> str:
    """对密码进行哈希"""
    return pwd_context.hash(password)


def verify_password(plain_password: str, hashed_password: str) -> bool:
    """验证密码是否正确"""
    return pwd_context.verify(plain_password, hashed_password)

### 示例 3：JWT Token 管理

```python
# app/core/jwt.py
from datetime import datetime, timedelta, timezone
from typing import Any
from jose import jwt, JWTError
from app.config import settings


def create_access_token(data: dict[str, Any]) -> str:
    """
    创建 Access Token

    Access Token 短期有效，用于 API 请求认证
    """
    to_encode = data.copy()
    expire = datetime.now(timezone.utc) + timedelta(
        minutes=settings.access_token_expire_minutes
    )
    to_encode.update({"exp": expire, "type": "access"})
    return jwt.encode(to_encode, settings.secret_key, algorithm=settings.algorithm)


def create_refresh_token(data: dict[str, Any]) -> str:
    """
    创建 Refresh Token

    Refresh Token 长期有效，用于获取新的 Access Token
    """
    to_encode = data.copy()
    expire = datetime.now(timezone.utc) + timedelta(
        days=settings.refresh_token_expire_days
    )
    to_encode.update({"exp": expire, "type": "refresh"})
    return jwt.encode(to_encode, settings.secret_key, algorithm=settings.algorithm)


def decode_token(token: str) -> dict[str, Any] | None:
    """
    解码并验证 JWT Token

    返回 payload 字典，验证失败返回 None
    """
    try:
        payload = jwt.decode(
            token,
            settings.secret_key,
            algorithms=[settings.algorithm],
        )
        return payload
    except JWTError:
        return None
```

### 示例 4：认证依赖

```python
# app/core/deps.py
from fastapi import Depends, HTTPException, status
from fastapi.security import OAuth2PasswordBearer
from sqlalchemy.ext.asyncio import AsyncSession
from app.database import get_db
from app.core.jwt import decode_token
from app.services.user_service import UserService
from app.models.user import User

# OAuth2 密码流 -- 自动从 Authorization header 中提取 Bearer token
oauth2_scheme = OAuth2PasswordBearer(tokenUrl="/auth/login")


async def get_current_user(
    token: str = Depends(oauth2_scheme),
    db: AsyncSession = Depends(get_db),
) -> User:
    """
    获取当前认证用户的依赖

    在路由中使用 Depends(get_current_user) 即可要求认证
    """
    credentials_exception = HTTPException(
        status_code=status.HTTP_401_UNAUTHORIZED,
        detail="Could not validate credentials",
        headers={"WWW-Authenticate": "Bearer"},
    )

    # 解码 Token
    payload = decode_token(token)
    if payload is None:
        raise credentials_exception

    # 验证 Token 类型
    if payload.get("type") != "access":
        raise credentials_exception

    # 获取用户 ID
    user_id: int | None = payload.get("sub")
    if user_id is None:
        raise credentials_exception

    # 查询用户
    user = await UserService.get_by_id(db, int(user_id))
    if user is None or not user.is_active:
        raise credentials_exception

    return user


class RoleChecker:
    """
    基于角色的权限检查依赖

    用法：Depends(RoleChecker("admin", "editor"))
    """

    def __init__(self, *allowed_roles: str):
        self.allowed_roles = allowed_roles

    def __call__(self, current_user: User = Depends(get_current_user)) -> User:
        if current_user.is_superuser:
            return current_user  # 超级用户拥有所有权限

        if current_user.role not in self.allowed_roles:
            raise HTTPException(
                status_code=status.HTTP_403_FORBIDDEN,
                detail="Not enough permissions",
            )
        return current_user
```

```python
# app/routers/auth.py
from fastapi import APIRouter, Depends, HTTPException, status
from fastapi.security import OAuth2PasswordRequestForm
from sqlalchemy.ext.asyncio import AsyncSession
from app.database import get_db
from app.core.security import hash_password, verify_password
from app.core.jwt import create_access_token, create_refresh_token, decode_token
from app.core.deps import get_current_user, RoleChecker
from app.services.user_service import UserService
from app.schemas.auth import (
    RegisterRequest,
    TokenResponse,
    RefreshRequest,
    UserMeResponse,
)
from app.models.user import User

router = APIRouter(prefix="/auth", tags=["认证"])


@router.post("/register", response_model=UserMeResponse, status_code=201)
async def register(
    user_data: RegisterRequest,
    db: AsyncSession = Depends(get_db),
):
    """用户注册"""
    # 检查用户名是否已存在
    existing = await UserService.get_by_username(db, user_data.username)
    if existing:
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail="Username already exists",
        )

    # 检查邮箱是否已存在
    existing_email = await UserService.get_by_email(db, user_data.email)
    if existing_email:
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail="Email already registered",
        )

    # 创建用户
    user = await UserService.create_with_hashed_password(
        db=db,
        username=user_data.username,
        email=user_data.email,
        hashed_password=hash_password(user_data.password),
    )
    return user


@router.post("/login", response_model=TokenResponse)
async def login(
    form_data: OAuth2PasswordRequestForm = Depends(),  # 兼容 OAuth2 规范
    db: AsyncSession = Depends(get_db),
):
    """
    用户登录

    使用 OAuth2PasswordRequestForm 支持表单提交，
    同时兼容 Swagger UI 的 Authorize 按钮
    """
    # 查找用户
    user = await UserService.get_by_username(db, form_data.username)
    if not user or not verify_password(form_data.password, user.hashed_password):
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Incorrect username or password",
            headers={"WWW-Authenticate": "Bearer"},
        )

    if not user.is_active:
        raise HTTPException(
            status_code=status.HTTP_403_FORBIDDEN,
            detail="User account is disabled",
        )

    # 生成 Token
    token_data = {"sub": str(user.id), "username": user.username, "role": user.role}
    access_token = create_access_token(token_data)
    refresh_token = create_refresh_token(token_data)

    return TokenResponse(
        access_token=access_token,
        refresh_token=refresh_token,
    )


@router.post("/refresh", response_model=TokenResponse)
async def refresh_token(
    request: RefreshRequest,
    db: AsyncSession = Depends(get_db),
):
    """刷新 Access Token"""
    # 验证 Refresh Token
    payload = decode_token(request.refresh_token)
    if payload is None or payload.get("type") != "refresh":
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Invalid refresh token",
        )

    # 获取用户
    user_id = payload.get("sub")
    user = await UserService.get_by_id(db, int(user_id))
    if not user or not user.is_active:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="User not found or inactive",
        )

    # 生成新的 Token
    token_data = {"sub": str(user.id), "username": user.username, "role": user.role}
    new_access_token = create_access_token(token_data)
    new_refresh_token = create_refresh_token(token_data)

    return TokenResponse(
        access_token=new_access_token,
        refresh_token=new_refresh_token,
    )


@router.get("/me", response_model=UserMeResponse)
async def get_me(current_user: User = Depends(get_current_user)):
    """获取当前用户信息"""
    return current_user


@router.get("/admin-only")
async def admin_only(
    current_user: User = Depends(RoleChecker("admin")),
):
    """仅管理员可访问"""
    return {"message": f"Welcome, admin {current_user.username}!"}


@router.get("/editor-or-admin")
async def editor_or_admin(
    current_user: User = Depends(RoleChecker("admin", "editor")),
):
    """编辑和管理员可访问"""
    return {"message": f"Welcome, {current_user.username}!"}
```

### 示例 7：在文章路由中使用认证

```python
# app/routers/posts.py -- 添加认证保护
from fastapi import APIRouter, Depends, HTTPException, Query
from sqlalchemy.ext.asyncio import AsyncSession
from app.database import get_db
from app.core.deps import get_current_user
from app.models.user import User
from app.schemas.post import PostCreate, PostUpdate, PostResponse
from app.schemas.common import ResponseBase, PaginatedResponse
from app.services.post_service import PostService

router = APIRouter(prefix="/posts", tags=["文章管理"])


@router.post("/", response_model=ResponseBase[PostResponse], status_code=201)
async def create_post(
    post_data: PostCreate,
    current_user: User = Depends(get_current_user),  # 需要登录
    db: AsyncSession = Depends(get_db),
):
    """创建文章 -- 需要认证"""
    post = await PostService.create(db, post_data, author_id=current_user.id)
    return ResponseBase(data=post)


@router.get("/", response_model=ResponseBase[PaginatedResponse[PostResponse]])
async def list_posts(
    page: int = Query(1, ge=1),
    page_size: int = Query(10, ge=1, le=100),
    db: AsyncSession = Depends(get_db),
):
    """获取文章列表 -- 公开接口"""
    posts, total = await PostService.get_list(db, page, page_size)
    total_pages = (total + page_size - 1) // page_size
    return ResponseBase(data=PaginatedResponse(
        items=posts, total=total, page=page,
        page_size=page_size, total_pages=total_pages,
    ))


@router.put("/{post_id}", response_model=ResponseBase[PostResponse])
async def update_post(
    post_id: int,
    post_data: PostUpdate,
    current_user: User = Depends(get_current_user),  # 需要登录
    db: AsyncSession = Depends(get_db),
):
    """更新文章 -- 仅作者可更新"""
    post = await PostService.update(db, post_id, post_data, user_id=current_user.id)
    if not post:
        raise HTTPException(status_code=404, detail="Post not found")
    return ResponseBase(data=post)


@router.delete("/{post_id}", status_code=204)
async def delete_post(
    post_id: int,
    current_user: User = Depends(get_current_user),  # 需要登录
    db: AsyncSession = Depends(get_db),
):
    """删除文章 -- 仅作者或管理员可删除"""
    post = await PostService.get_by_id(db, post_id)
    if not post:
        raise HTTPException(status_code=404, detail="Post not found")

    # 作者或管理员可以删除
    if post.author_id != current_user.id and not current_user.is_superuser:
        raise HTTPException(status_code=403, detail="Not enough permissions")

    await PostService.delete(db, post_id, current_user.id)
```

### 示例 8：完整的认证流程测试

```python
# test_auth_flow.py
# 使用 httpx 测试完整认证流程
import pytest
from httpx import AsyncClient, ASGITransport
from app.main import app


@pytest.mark.asyncio
async def test_auth_flow():
    transport = ASGITransport(app=app)
    async with AsyncClient(transport=transport, base_url="http://test") as client:

        # 1. 注册
        resp = await client.post("/auth/register", json={
            "username": "alice",
            "email": "alice@example.com",
            "password": "password123",
        })
        assert resp.status_code == 201
        user = resp.json()
        assert user["username"] == "alice"

        # 2. 登录
        resp = await client.post("/auth/login", data={
            "username": "alice",
            "password": "password123",
        })
        assert resp.status_code == 200
        tokens = resp.json()
        access_token = tokens["access_token"]
        refresh_token = tokens["refresh_token"]

        # 3. 使用 Access Token 访问受保护接口
        resp = await client.get(
            "/auth/me",
            headers={"Authorization": f"Bearer {access_token}"},
        )
        assert resp.status_code == 200
        assert resp.json()["username"] == "alice"

        # 4. 刷新 Token
        resp = await client.post("/auth/refresh", json={
            "refresh_token": refresh_token,
        })
        assert resp.status_code == 200
        new_tokens = resp.json()
        assert new_tokens["access_token"] != access_token

        # 5. 使用无效 Token
        resp = await client.get(
            "/auth/me",
            headers={"Authorization": "Bearer invalid_token"},
        )
        assert resp.status_code == 401
```

## 常见问题

**Q: JWT Token 存在哪里？**

A: Access Token 存在前端内存中（不存 localStorage 防止 XSS），Refresh Token 可以存在 HttpOnly Cookie 中（防止 CSRF）。每次请求在 `Authorization: Bearer <token>` header 中发送。

**Q: Token 泄露了怎么办？**

A: Access Token 有效期短（30 分钟），泄露后影响有限。Refresh Token 泄露风险更大，建议实现 Token 黑名单或使用 Redis 存储已撤销的 Token。生产环境应考虑使用 HTTPS。

**Q: 为什么需要 Refresh Token？**

A: Access Token 有效期短是为了安全，但频繁登录用户体验差。Refresh Token 有效期长，用于在 Access Token 过期后自动获取新的 Token，实现无感续期。

**Q: FastAPI 的 `OAuth2PasswordBearer` 和 `OAuth2PasswordRequestForm` 有什么区别？**

A: `OAuth2PasswordBearer` 是从请求头中提取 Token 的依赖。`OAuth2PasswordRequestForm` 是解析登录表单数据的依赖（支持 `username` 和 `password` 字段）。两者配合使用实现了标准的 OAuth2 密码流。
