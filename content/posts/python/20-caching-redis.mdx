# 缓存策略

## 前言

在高并发场景下，直接查询数据库往往成为性能瓶颈。Redis 作为高性能的内存数据库，是解决这一问题的利器。合理使用缓存可以将 API 响应时间从百毫秒级降低到毫秒级，同时大幅减少数据库压力。

**本文将带你：**

1. 理解 Redis 核心数据结构及其适用场景
2. 在 FastAPI 中集成 Redis（同步 + 异步）
3. 实现通用的缓存装饰器
4. 解决缓存穿透、击穿、雪崩三大问题
5. 实现分布式锁

---

## 核心概念

### Redis 数据结构速览

| 数据结构   | 命令                    | 适用场景             |
| ---------- | ----------------------- | -------------------- |
| String     | GET/SET/SETEX           | 简单值缓存、计数器   |
| Hash       | HGET/HSET/HGETALL       | 对象缓存（用户信息） |
| List       | LPUSH/RPOP/LRANGE       | 消息队列、最新列表   |
| Set        | SADD/SMEMBERS/SISMEMBER | 去重、标签、关注关系 |
| Sorted Set | ZADD/ZRANGE/ZRANK       | 排行榜、按时间排序   |

### 缓存读写模式

```
Cache-Aside 模式（最常用）：

读取：
  1. 查缓存 → 命中 → 返回
  2. 未命中 → 查数据库 → 写入缓存 → 返回

写入：
  1. 更新数据库
  2. 删除缓存（不是更新缓存）
```

```python
"""
博客系统 - Redis 集成
"""
import json
from datetime import datetime
from typing import Any, Optional

import redis.asyncio as redis  # 异步 Redis 客户端
from fastapi import FastAPI, Depends, HTTPException
from pydantic import BaseModel

app = FastAPI(title="博客系统 API")

# ========== Redis 连接配置 ==========

# 创建 Redis 连接池
redis_pool = redis.ConnectionPool.from_url(
    "redis://localhost:6379/0",
    max_connections=20,       # 最大连接数
    decode_responses=True,    # 自动解码为字符串
)

# 创建 Redis 客户端
redis_client = redis.Redis(connection_pool=redis_pool)


# ========== 生命周期管理 ==========

from contextlib import asynccontextmanager

@asynccontextmanager
async def lifespan(app: FastAPI):
    """应用生命周期：启动时连接 Redis，关闭时断开"""
    # 测试连接
    await redis_client.ping()
    print("[Redis] 连接成功")
    yield
    # 关闭连接池
    await redis_client.close()
    print("[Redis] 连接已关闭")

app = FastAPI(lifespan=lifespan)


# ========== 基础操作封装 ==========

class RedisCache:
    """Redis 缓存操作封装"""

    def __init__(self, client: redis.Redis):
        self.client = client

    # ----- String 操作 -----

    async def get(self, key: str) -> Optional[Any]:
        """获取缓存值"""
        value = await self.client.get(key)
        if value is not None:
            try:
                return json.loads(value)
            except (json.JSONDecodeError, TypeError):
                return value
        return None

    async def set(self, key: str, value: Any, ttl: Optional[int] = None) -> bool:
        """
        设置缓存值
        ttl: 过期时间（秒），None 表示永不过期
        """
        serialized = json.dumps(value, ensure_ascii=False, default=str)
        if ttl:
            return await self.client.setex(key, ttl, serialized)
        return await self.client.set(key, serialized)

    async def delete(self, key: str) -> int:
        """删除缓存"""
        return await self.client.delete(key)

    async def exists(self, key: str) -> bool:
        """检查 key 是否存在"""
        return bool(await self.client.exists(key))

    # ----- Hash 操作 -----

    async def hset(self, name: str, key: str, value: Any) -> int:
        """设置 Hash 字段"""
        return await self.client.hset(name, key, json.dumps(value, default=str))

    async def hget(self, name: str, key: str) -> Optional[Any]:
        """获取 Hash 字段"""
        value = await self.client.hget(name, key)
        if value:
            return json.loads(value)
        return None

    async def hgetall(self, name: str) -> dict:
        """获取 Hash 所有字段"""
        data = await self.client.hgetall(name)
        return {k: json.loads(v) for k, v in data.items()}

    async def hdel(self, name: str, *keys: str) -> int:
        """删除 Hash 字段"""
        return await self.client.hdel(name, *keys)

    # ----- 计数器 -----

    async def incr(self, key: str, amount: int = 1) -> int:
        """递增"""
        return await self.client.incrby(key, amount)

    # ----- 排行榜（Sorted Set）-----

    async def zadd(self, name: str, mapping: dict[str, float]) -> int:
        """添加到有序集合"""
        return await self.client.zadd(name, mapping)

    async def zrevrange(self, name: str, start: int = 0, end: int = -1) -> list:
        """按分数降序获取"""
        return await self.client.zrevrange(name, start, end, withscores=True)

# 创建缓存实例
cache = RedisCache(redis_client)
```

### 示例 2：缓存装饰器

```python
"""
通用缓存装饰器
"""
import functools
import hashlib
import json
from typing import Optional, Callable

def cached(
    prefix: str,
    ttl: int = 300,
    key_builder: Optional[Callable] = None,
):
    """
    异步缓存装饰器

    Args:
        prefix: 缓存 key 前缀
        ttl: 过期时间（秒），默认 5 分钟
        key_builder: 自定义 key 生成函数，接收函数参数，返回 key
    """
    def decorator(func):
        @functools.wraps(func)
        async def wrapper(*args, **kwargs):
            # 生成缓存 key
            if key_builder:
                cache_key = key_builder(*args, **kwargs)
            else:
                # 默认使用函数名 + 参数哈希
                key_data = {
                    "func": func.__name__,
                    "args": str(args),
                    "kwargs": str(sorted(kwargs.items())),
                }
                key_hash = hashlib.md5(
                    json.dumps(key_data, sort_keys=True).encode()
                ).hexdigest()[:12]
                cache_key = f"{prefix}:{key_hash}"

            # 尝试从缓存获取
            cached_value = await cache.get(cache_key)
            if cached_value is not None:
                return cached_value

            # 缓存未命中，执行函数
            result = await func(*args, **kwargs)

            # 写入缓存（排除 None 值）
            if result is not None:
                await cache.set(cache_key, result, ttl=ttl)

            return result
        return wrapper
    return decorator


# ========== 使用缓存装饰器 ==========

async def get_user_from_db(user_id: int) -> dict:
    """模拟数据库查询"""
    return {"id": user_id, "username": f"User_{user_id}", "email": f"user{user_id}@blog.com"}

@cached(prefix="user", ttl=600)  # 缓存 10 分钟
async def get_user(user_id: int) -> dict:
    """获取用户信息（带缓存）"""
    return await get_user_from_db(user_id)

@cached(prefix="post_detail", ttl=300)  # 缓存 5 分钟
async def get_post_detail(post_id: int) -> dict:
    """获取文章详情（带缓存）"""
    # 模拟数据库查询
    return {
        "id": post_id,
        "title": f"文章 {post_id}",
        "content": "文章内容...",
        "author": "Admin",
        "views": 100,
    }


# ========== API 响应缓存 ==========

async def get_posts_list(page: int, page_size: int, tag: Optional[str] = None) -> dict:
    """获取文章列表（带缓存）"""
    # 自定义缓存 key
    cache_key = f"posts:page={page}:size={page_size}:tag={tag or 'all'}"
    cached = await cache.get(cache_key)
    if cached:
        return cached

    # 模拟数据库查询
    posts = [
        {"id": i, "title": f"文章 {i}"}
        for i in range((page - 1) * page_size + 1, page * page_size + 1)
    ]

    result = {"posts": posts, "total": 100, "page": page, "page_size": page_size}
    await cache.set(cache_key, result, ttl=60)  # 列表缓存 1 分钟
    return result
```

### 示例 3：解决缓存三大问题

```python
"""
缓存穿透、击穿、雪崩的解决方案
"""
import asyncio
import random

# ========== 1. 缓存穿透（查询不存在的数据）==========

async def get_post_with_null_cache(post_id: int) -> Optional[dict]:
    """
    解决方案：缓存空值 + 布隆过滤器

    当查询结果为空时，也缓存一个空值（短 TTL），
    避免恶意请求反复穿透到数据库。
    """
    cache_key = f"post:{post_id}"

    # 查缓存
    cached = await cache.get(cache_key)
    if cached is not None:
        if cached == "NULL":
            return None  # 缓存了空值
        return cached

    # 查数据库
    post = await query_post_from_db(post_id)  # 假设这个函数查询数据库

    if post is None:
        # 缓存空值，TTL 较短（防止下次穿透）
        await cache.set(cache_key, "NULL", ttl=60)
        return None

    # 缓存结果
    await cache.set(cache_key, post, ttl=300)
    return post


# ========== 2. 缓存击穿（热点 key 过期瞬间大量请求）==========

# 使用互斥锁防止缓存击穿
_locks: dict[str, asyncio.Lock] = {}

async def get_hot_post_with_lock(post_id: int) -> dict:
    """
    解决方案：互斥锁

    当缓存过期时，只允许一个请求去查数据库并更新缓存，
    其他请求等待后直接读缓存。
    """
    cache_key = f"post:{post_id}"
    lock_key = f"lock:post:{post_id}"

    # 查缓存
    cached = await cache.get(cache_key)
    if cached is not None and cached != "NULL":
        return cached

    # 获取互斥锁（使用 Redis SETNX）
    lock_acquired = await redis_client.set(lock_key, "1", ex=10, nx=True)
    if lock_acquired:
        try:
            # 再次检查缓存（double check）
            cached = await cache.get(cache_key)
            if cached is not None and cached != "NULL":
                return cached

            # 查数据库
            post = await query_post_from_db(post_id)
            if post:
                await cache.set(cache_key, post, ttl=300)
                return post
            else:
                await cache.set(cache_key, "NULL", ttl=60)
                return None
        finally:
            # 释放锁
            await redis_client.delete(lock_key)
    else:
        # 未获取到锁，等待后重试
        await asyncio.sleep(0.1)
        return await get_hot_post_with_lock(post_id)


# ========== 3. 缓存雪崩（大量 key 同时过期）==========

def get_cache_ttl(base_ttl: int = 300) -> int:
    """
    解决方案：随机 TTL

    在基础 TTL 上添加随机偏移量，
    避免大量 key 在同一时间过期。
    """
    # 在 base_ttl 的 80%~120% 之间随机
    jitter = int(base_ttl * random.uniform(-0.2, 0.2))
    return max(60, base_ttl + jitter)  # 最少 60 秒

# 使用示例
ttl = get_cache_ttl(300)  # 返回 240~360 之间的值
await cache.set("post:1", post_data, ttl=ttl)
```

### 示例 4：分布式锁

```python
"""
基于 Redis 的分布式锁
"""
import uuid
import asyncio

class DistributedLock:
    """
    Redis 分布式锁

    基本的 RedLock 思路：
    1. 使用 SETNX（SET ... NX）原子性地获取锁
    2. 设置过期时间防止死锁
    3. 使用唯一标识确保只有持有者才能释放锁
    """

    def __init__(self, redis_client: redis.Redis, lock_timeout: int = 10):
        self.client = redis_client
        self.lock_timeout = lock_timeout

    async def acquire(self, lock_name: str, wait_timeout: float = 5.0) -> str | None:
        """
        获取分布式锁

        Args:
            lock_name: 锁名称
            wait_timeout: 最大等待时间（秒）

        Returns:
            锁标识符（用于释放锁），获取失败返回 None
        """
        identifier = str(uuid.uuid4())
        lock_key = f"lock:{lock_name}"
        end_time = asyncio.get_event_loop().time() + wait_timeout

        while asyncio.get_event_loop().time() < end_time:
            # 尝试获取锁（原子操作）
            acquired = await self.client.set(
                lock_key,
                identifier,
                ex=self.lock_timeout,  # 自动过期
                nx=True,              # 只在 key 不存在时设置
            )

            if acquired:
                return identifier

            # 等待后重试（指数退避）
            await asyncio.sleep(0.05)

        return None  # 获取锁超时

    async def release(self, lock_name: str, identifier: str) -> bool:
        """
        释放分布式锁

        使用 Lua 脚本确保原子性：只有持有锁的线程才能释放
        """
        lua_script = """
        if redis.call("get", KEYS[1]) == ARGV[1] then
            return redis.call("del", KEYS[1])
        else
            return 0
        end
        """
        lock_key = f"lock:{lock_name}"
        result = await self.client.eval(lua_script, 1, lock_key, identifier)
        return bool(result)


# ========== 使用示例 ==========

dist_lock = DistributedLock(redis_client, lock_timeout=10)

async def update_post_views_with_lock(post_id: int):
    """使用分布式锁更新文章浏览量"""
    lock_id = await dist_lock.acquire(f"post_views:{post_id}")
    if lock_id is None:
        raise HTTPException(status_code=429, detail="操作太频繁，请稍后再试")

    try:
        # 在锁的保护下执行操作
        # 1. 从数据库获取当前浏览量
        views = await get_views_from_db(post_id)
        # 2. 更新浏览量
        await update_views_to_db(post_id, views + 1)
        # 3. 更新缓存
        await cache.set(f"post:{post_id}:views", views + 1, ttl=300)

        return {"views": views + 1}
    finally:
        # 释放锁
        await dist_lock.release(f"post_views:{post_id}", lock_id)


async def export_user_data_with_lock(user_id: int):
    """使用分布式锁防止重复导出"""
    lock_id = await dist_lock.acquire(f"export:{user_id}", wait_timeout=0)
    if lock_id is None:
        raise HTTPException(status_code=409, detail="已有导出任务在进行中")

    try:
        # 执行导出...
        await asyncio.sleep(5)  # 模拟耗时操作
        return {"message": "导出完成"}
    finally:
        await dist_lock.release(f"export:{user_id}", lock_id)
```

```python
"""
博客系统 - 综合缓存方案

# ========== 1. 用户信息缓存（Hash 结构）==========

async def cache_user_info(user_id: int, user_data: dict):
    """缓存用户信息到 Hash"""
    key = f"user:{user_id}"
    await cache.client.hset(key, mapping={

        "username": user_data["username"],
        "avatar": user_data.get("avatar", ""),
        "role": user_data.get("role", "user"),

async def get_cached_user(user_id: int) -> Optional[dict]:

    key = f"user:{user_id}"
    data = await cache.client.hgetall(key)
    if data:
        return data
    return None

async def invalidate_user_cache(user_id: int):
    """用户信息变更时清除缓存"""
    await cache.delete(f"user:{user_id}")


# ========== 2. 文章浏览量计数（String + INCR）==========

async def increment_post_views(post_id: int) -> int:
    """增加文章浏览量（先写缓存，定期同步到数据库）"""
    key = f"post:{post_id}:views"
    views = await cache.incr(key)

    # 首次访问时设置过期时间
    if views == 1:
        await redis_client.expire(key, 86400)  # 24 小时

    return views


# ========== 3. 热门文章排行榜（Sorted Set）==========

async def update_hot_posts():
    """更新热门文章排行（定时任务调用）"""
    # 按浏览量排序
    hot_key = "posts:hot"
    await redis_client.delete(hot_key)

    # 从数据库获取浏览量 Top 100 的文章
    hot_posts = await get_top_posts_from_db(limit=100)

    # 写入 Sorted Set（分数 = 浏览量）
    if hot_posts:
        mapping = {str(p["id"]): p["views"] for p in hot_posts}
        await cache.zadd(hot_key, mapping)
        await redis_client.expire(hot_key, 3600)  # 1 小时

async def get_hot_posts(limit: int = 10) -> list:
    """获取热门文章排行"""
    hot_key = "posts:hot"
    results = await cache.zrevrange(hot_key, 0, limit - 1)
    return [{"post_id": post_id, "views": int(views)} for post_id, views in results]

async def warmup_cache():

    await cache.delete(f"post:{post_id}")

    # 清除标签相关缓存
    async for key in redis_client.scan_iter("tags:*"):
        await cache.delete(key)
```

## 最佳实践

### 1. 缓存 Key 命名规范

```
格式: 业务模块:数据类型:标识符

用户模块:   user:1, user:1:profile
文章模块:   post:100, posts:page=1:size=10
标签模块:   tag:python, tags:popular
排行榜:     posts:hot, users:active
锁:         lock:post:100
限流:       ratelimit:user:1:20260329
```

### 2. TTL 设置建议

| 数据类型 | 建议 TTL | 说明       |
| -------- | -------- | ---------- |
| 用户信息 | 30 分钟  | 变更不频繁 |
| 文章详情 | 5 分钟   | 中等频率   |
| 文章列表 | 1 分钟   | 变更较频繁 |
| 排行榜   | 1 小时   | 定时更新   |
| 热点计数 | 24 小时  | 定期同步   |

### 3. 缓存与数据库一致性

```python
# 推荐方案：先更新数据库，再删除缓存（而不是更新缓存）
async def update_post(post_id: int, data: dict):
    # 1. 更新数据库
    await db.execute(update(Post).where(Post.id == post_id).values(**data))
    await db.commit()

    # 2. 删除缓存（让下次查询时重新加载）
    await cache.delete(f"post:{post_id}")

    # 注意：这里存在短暂的不一致窗口
    # 对一致性要求极高的场景可以使用延迟双删策略
```

## 常见问题

**Q: Redis 挂了怎么办？**

A: 使用缓存降级策略：捕获 Redis 异常，直接查询数据库。配置 Redis 持久化（RDB + AOF），并使用 Redis Sentinel 或 Cluster 保证高可用。

**Q: 缓存和数据库数据不一致怎么办？**

A: 采用"先更新数据库，后删除缓存"策略。极端情况下仍有短暂不一致，可以使用"延迟双删"：更新数据库后先删一次缓存，延迟几百毫秒再删一次。对强一致性要求极高的场景，考虑使用 `Cache-Aside + 消息队列` 方案。

**Q: 什么时候用同步 Redis，什么时候用异步？**

A: 在 FastAPI 的异步路由中（`async def`），使用 `redis.asyncio`。在同步代码或非异步上下文中（如 Celery Worker），使用同步的 `redis` 库。

**Q: 如何监控缓存命中率？**

A: 使用 Redis 的 `INFO STATS` 命令获取 `keyspace_hits` 和 `keyspace_misses`，计算命中率：`hits / (hits + misses)`。也可以在应用层面统计缓存命中和未命中次数。
