## 前言

经过前面两篇的学习，我们已经分别掌握了 FastAPI 和 SQLAlchemy 2.0 的核心用法。现在是将它们整合起来的时候了。

本文将构建一个**完整的博客系统 CRUD API**，采用分层架构设计。如果你有 NestJS 或 Spring Boot 经验，这个架构你会非常熟悉：

- **NestJS 类比**：Models 对应 Entity，Schemas 对应 DTO，Services 对应 Service 层，Routers 对应 Controller
- **Spring Boot 类比**：分层结构几乎一致，Controller -> Service -> Repository -> Entity

**本文将带你：**

1. 设计规范的项目分层结构
2. 实现 Pydantic Schema 设计（Create/Update/Response）
3. 构建用户、文章、标签、评论的完整 CRUD
4. 实现分页、排序、过滤
5. 设计统一响应格式和软删除

```python
from app.config import settings

# 创建异步引擎
engine = create_async_engine(
    settings.database_url,
    echo=settings.debug,
    pool_size=10,
    max_overflow=20,
)

# 创建异步会话工厂
AsyncSessionLocal = async_sessionmaker(
    engine,
    class_=AsyncSession,
    expire_on_commit=False,  # 提交后对象不过期，避免 lazy load 问题
)


async def get_db() -> AsyncSession:
    """获取数据库会话的依赖注入函数"""
    async with AsyncSessionLocal() as session:
        try:
            yield session
            await session.commit()
        except Exception:
            await session.rollback()
            raise
```

```python
# app/schemas/common.py
from typing import Generic, TypeVar, List
from pydantic import BaseModel, Field


T = TypeVar("T")


class ResponseBase(BaseModel, Generic[T]):
    """统一响应格式"""
    code: int = Field(default=200, description="状态码")
    message: str = Field(default="success", description="消息")
    data: T | None = Field(default=None, description="数据")


class PaginatedResponse(BaseModel, Generic[T]):
    """分页响应格式"""
    items: List[T] = Field(description="数据列表")
    total: int = Field(description="总数")
    page: int = Field(description="当前页")
    page_size: int = Field(description="每页数量")
    total_pages: int = Field(description="总页数")


class PaginationParams(BaseModel):
    """分页参数"""
    page: int = Field(default=1, ge=1, description="页码")
    page_size: int = Field(default=10, ge=1, le=100, description="每页数量")

    @property
    def offset(self) -> int:
        return (self.page - 1) * self.page_size

    @property
    def total_pages(self, total: int) -> int:
        return (total + self.page_size - 1) // self.page_size
```

```python
# app/services/user_service.py
from sqlalchemy import select, func
from sqlalchemy.ext.asyncio import AsyncSession
from app.models.user import User
from app.schemas.user import UserCreate, UserUpdate


class UserService:
    """用户业务逻辑层"""

    @staticmethod
    async def create(db: AsyncSession, user_data: UserCreate) -> User:
        """创建用户"""
        from passlib.context import CryptContext

        pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")
        user = User(
            username=user_data.username,
            email=user_data.email,
            hashed_password=pwd_context.hash(user_data.password),
        )
        db.add(user)
        await db.flush()  # 获取 ID，但不提交
        await db.refresh(user)
        return user

    @staticmethod
    async def get_by_id(db: AsyncSession, user_id: int) -> User | None:
        """根据 ID 获取用户"""
        result = await db.execute(select(User).where(User.id == user_id))
        return result.scalar_one_or_none()

    @staticmethod
    async def get_by_username(db: AsyncSession, username: str) -> User | None:
        """根据用户名获取用户"""
        result = await db.execute(select(User).where(User.username == username))
        return result.scalar_one_or_none()

    @staticmethod
    async def get_list(
        db: AsyncSession,
        page: int = 1,
        page_size: int = 10,
    ) -> tuple[list[User], int]:
        """获取用户列表（分页）"""
        offset = (page - 1) * page_size

        # 查询总数
        total_result = await db.execute(select(func.count(User.id)))
        total = total_result.scalar()

        # 查询列表
        result = await db.execute(
            select(User)
            .order_by(User.created_at.desc())
            .offset(offset)
            .limit(page_size)
        )
        users = result.scalars().all()
        return list(users), total

    @staticmethod
    async def update(db: AsyncSession, user_id: int, user_data: UserUpdate) -> User | None:
        """更新用户"""
        user = await UserService.get_by_id(db, user_id)
        if not user:
            return None

        update_data = user_data.model_dump(exclude_unset=True)
        for field, value in update_data.items():
            setattr(user, field, value)

        await db.flush()
        await db.refresh(user)
        return user

    @staticmethod
    async def delete(db: AsyncSession, user_id: int) -> bool:
        """软删除用户"""
        user = await UserService.get_by_id(db, user_id)
        if not user:
            return False
        user.is_active = False
        await db.flush()
        return True
```

```python
# app/services/post_service.py
from sqlalchemy import select, func, desc
from sqlalchemy.ext.asyncio import AsyncSession
from app.models.post import Post
from app.models.tag import Tag
from app.schemas.post import PostCreate, PostUpdate


class PostService:
    """文章业务逻辑层"""

    @staticmethod
    async def create(db: AsyncSession, post_data: PostCreate, author_id: int) -> Post:
        """创建文章"""
        post = Post(
            title=post_data.title,
            content=post_data.content,
            is_published=post_data.is_published,
            author_id=author_id,
        )

        # 处理标签关联
        if post_data.tag_ids:
            tags = await db.execute(
                select(Tag).where(Tag.id.in_(post_data.tag_ids))
            )
            post.tags = list(tags.scalars().all())

        db.add(post)
        await db.flush()
        await db.refresh(post)
        return post

    @staticmethod
    async def get_list(
        db: AsyncSession,
        page: int = 1,
        page_size: int = 10,
        author_id: int | None = None,
        tag_id: int | None = None,
        is_published: bool | None = None,
        search: str | None = None,
    ) -> tuple[list[Post], int]:
        """获取文章列表（支持分页、过滤、搜索）"""
        query = select(Post)

        # 构建过滤条件
        if author_id is not None:
            query = query.where(Post.author_id == author_id)
        if is_published is not None:
            query = query.where(Post.is_published == is_published)
        if tag_id is not None:
            query = query.join(Post.tags).where(Tag.id == tag_id)
        if search:
            query = query.where(Post.title.ilike(f"%{search}%"))

        # 总数查询
        count_query = select(func.count(Post.id))
        if author_id is not None:
            count_query = count_query.where(Post.author_id == author_id)
        if is_published is not None:
            count_query = count_query.where(Post.is_published == is_published)

        total = (await db.execute(count_query)).scalar()

        # 分页与排序
        offset = (page - 1) * page_size
        result = await db.execute(
            query.order_by(desc(Post.created_at))
            .offset(offset)
            .limit(page_size)
        )
        posts = result.scalars().all()
        return list(posts), total

    @staticmethod
    async def get_by_id(db: AsyncSession, post_id: int) -> Post | None:
        """获取文章详情"""
        result = await db.execute(select(Post).where(Post.id == post_id))
        return result.scalar_one_or_none()

    @staticmethod
    async def update(
        db: AsyncSession, post_id: int, post_data: PostUpdate, user_id: int
    ) -> Post | None:
        """更新文章（仅作者可更新）"""
        post = await PostService.get_by_id(db, post_id)
        if not post or post.author_id != user_id:
            return None

        update_data = post_data.model_dump(exclude_unset=True)

        # 单独处理标签关联
        if "tag_ids" in update_data:
            tag_ids = update_data.pop("tag_ids")
            if tag_ids is not None:
                tags = await db.execute(
                    select(Tag).where(Tag.id.in_(tag_ids))
                )
                post.tags = list(tags.scalars().all())

        for field, value in update_data.items():
            setattr(post, field, value)

        await db.flush()
        await db.refresh(post)
        return post

    @staticmethod
    async def delete(db: AsyncSession, post_id: int, user_id: int) -> bool:
        """删除文章（仅作者可删除）"""
        post = await PostService.get_by_id(db, post_id)
        if not post or post.author_id != user_id:
            return False
        await db.delete(post)
        await db.flush()
        return True
```

```python
from sqlalchemy import select, func, desc
from sqlalchemy.ext.asyncio import AsyncSession
from app.models.comment import Comment
from app.schemas.comment import CommentCreate


class CommentService:
    """评论业务逻辑层"""

    @staticmethod
    async def create(db: AsyncSession, comment_data: CommentCreate, user_id: int, post_id: int) -> Comment:
        """创建评论"""
        comment = Comment(
            content=comment_data.content,
            user_id=user_id,
            post_id=post_id,
        )
        db.add(comment)
        await db.flush()
        await db.refresh(comment)
        return comment

    @staticmethod
    async def get_by_post(
        db: AsyncSession, post_id: int, page: int = 1, page_size: int = 20
    ) -> tuple[list[Comment], int]:
        """获取文章的评论列表"""
        offset = (page - 1) * page_size

        total = (
            await db.execute(
                select(func.count(Comment.id)).where(Comment.post_id == post_id)
            )
        ).scalar()

        result = await db.execute(
            select(Comment)
            .where(Comment.post_id == post_id)
            .order_by(desc(Comment.created_at))
            .offset(offset)
            .limit(page_size)
        )
        comments = result.scalars().all()
        return list(comments), total

    @staticmethod
    async def delete(db: AsyncSession, comment_id: int, user_id: int) -> bool:
        """删除评论（仅作者可删除）"""
        comment = await db.execute(
            select(Comment).where(Comment.id == comment_id)
        ).scalar_one_or_none()

        if not comment or comment.user_id != user_id:
            return False

        await db.delete(comment)
        await db.flush()
```

```python
# app/main.py
from contextlib import asynccontextmanager
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
from app.routers import users, posts, comments, tags


@asynccontextmanager
async def lifespan(app: FastAPI):
    """应用生命周期管理"""
    print("Blog API starting...")
    yield
    print("Blog API shutting down...")


app = FastAPI(
    title="博客系统 API",
    description="FastAPI + SQLAlchemy 博客系统",
    version="1.0.0",
    lifespan=lifespan,
)

# CORS 配置
app.add_middleware(
    CORSMiddleware,
    allow_origins=["http://localhost:3000"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# 注册路由
```

## 最佳实践

### 1. Schema 与 Model 分离

始终将 Pydantic Schema 和 SQLAlchemy Model 分开定义。Schema 负责数据验证（API 层），Model 负责数据存储（数据库层）。

### 2. Service 层不依赖 FastAPI

Service 层只接收 `AsyncSession` 和 Pydantic 模型作为参数，不导入任何 FastAPI 对象。这使得 Service 层可以独立测试和复用。

### 3. 使用 `exclude_unset` 实现部分更新

```python
# 好的做法 -- 只更新用户传入的字段
update_data = user_data.model_dump(exclude_unset=True)
for field, value in update_data.items():
    setattr(user, field, value)

# 不好的做法 -- 会将未传入的字段设为默认值
for field, value in user_data.model_dump().items():
    setattr(user, field, value)
```

## 常见问题

**Q: 为什么要把逻辑放在 Service 层而不是直接写在 Router 里？**

A: 分层架构的核心原则是职责分离。Router 只负责处理 HTTP 请求和响应，Service 负责业务逻辑。这样做的好处：Service 可以被多个 Router 复用、更容易编写单元测试、代码组织更清晰。

**Q: 什么时候用 `flush()` 什么时候用 `commit()`？**

A: `flush()` 将更改发送到数据库但不提交事务，适合在事务中获取自动生成的 ID。`commit()` 提交事务。在我们的架构中，`commit()` 由 `get_db()` 依赖统一管理。

**Q: 分页查询中，总数和列表需要两次查询吗？**

A: 是的。`COUNT` 查询和 `SELECT` 查询是两个独立的 SQL。虽然可以一次查询完成，但在大数据量下分开查询更高效，也更容易添加过滤条件。

**Q: 软删除和硬删除怎么选？**

A: 业务数据（用户、文章、评论）建议软删除，保留数据用于审计和恢复。日志、缓存等非关键数据可以硬删除。
