## 前言

Mongoose 的查询构建器（Query Builder）提供了强大的链式 API，让我们能够构建复杂的查询条件。掌握这些查询技巧对于构建高效的应用至关重要。

本文将深入探讨 Mongoose 的各种查询运算符、查询组合、文本搜索以及高级筛选技巧。

## 核心概念

### 查询构建器链式调用

```
Post.find({ status: 'published' })
  .where('views').gte(100)
  .or([{ author: userId }, { collaborators: userId }])
  .select('title views')
  .sort({ createdAt: -1 })
  .limit(20)
  .lean()
  .exec();
```

### 比较运算符

创建 `src/examples/query-operators.ts`：

```typescript
import Post from "../models/Post";
import connectDB from "../config/database";

await connectDB();

// 创建测试数据
await Post.create([
  {
    title: "Mongoose 入门教程",
    content: "...",
    status: "published",
    views: 150,
    likes: 25,
    tags: ["mongodb", "nodejs", "tutorial"],
  },
  {
    title: "Node.js 实战",
    content: "...",
    status: "published",
    views: 300,
    likes: 50,
    tags: ["nodejs", "javascript"],
  },
  {
    title: "MongoDB 高级技巧",
    content: "...",
    status: "published",
    views: 450,
    likes: 75,
    tags: ["mongodb", "database"],
  },
  {
    title: "TypeScript 指南",
    content: "...",
    status: "draft",
    views: 50,
    likes: 10,
    tags: ["typescript", "javascript"],
  },
]);

console.log("\n=== 比较运算符 ===");

// $gt (大于)
const gtPosts = await Post.find({
  views: { $gt: 200 },
}).select("title views");
console.log(`\n$gt (浏览量 > 200): ${gtPosts.length} 篇`);
gtPosts.forEach((post) => console.log(`  - ${post.title}: ${post.views}`));

// $gte (大于等于)
const gtePosts = await Post.find({
  views: { $gte: 300 },
}).select("title views");
console.log(`\n$gte (浏览量 >= 300): ${gtePosts.length} 篇`);
gtePosts.forEach((post) => console.log(`  - ${post.title}: ${post.views}`));

// $lt (小于)
const ltPosts = await Post.find({
  views: { $lt: 100 },
}).select("title views");
console.log(`\n$lt (浏览量 < 100): ${ltPosts.length} 篇`);
ltPosts.forEach((post) => console.log(`  - ${post.title}: ${post.views}`));

// $lte (小于等于)
const ltePosts = await Post.find({
  views: { $lte: 300 },
}).select("title views");
console.log(`\n$lte (浏览量 <= 300): ${ltePosts.length} 篇`);
ltePosts.forEach((post) => console.log(`  - ${post.title}: ${post.views}`));

// $ne (不等于)
const nePosts = await Post.find({
  status: { $ne: "draft" },
}).select("title status");
console.log(`\n$ne (状态 != draft): ${nePosts.length} 篇`);
nePosts.forEach((post) => console.log(`  - ${post.title}: ${post.status}`));

// $eq (等于，可省略)
const eqPosts = await Post.find({
  status: { $eq: "published" },
}).select("title status");
console.log(`\n$eq (状态 == published): ${eqPosts.length} 篇`);

// 组合比较（范围查询）
const rangePosts = await Post.find({
  views: { $gte: 100, $lte: 400 },
}).select("title views");
console.log(`\n范围 (100 <= 浏览量 <= 400): ${rangePosts.length} 篇`);
rangePosts.forEach((post) => console.log(`  - ${post.title}: ${post.views}`));

// 多字段比较
const multiComparePosts = await Post.find({
  views: { $gte: 100 },
  likes: { $gte: 20 },
}).select("title views likes");
console.log(
  `\n多字段 (浏览量 >= 100 AND 点赞 >= 20): ${multiComparePosts.length} 篇`
);
multiComparePosts.forEach((post) => {
  console.log(`  - ${post.title}: ${post.views} 浏览, ${post.likes} 点赞`);
});

process.exit(0);
```

### 逻辑运算符

创建 `src/examples/query-logical.ts`：

```typescript
import Post from "../models/Post";
import User from "../models/User";
import connectDB from "../config/database";

await connectDB();

const userId = await User.findOne()?.select("_id");

// ==================== $or (逻辑或) ====================
console.log("\n=== $or (逻辑或) ===");

const orPosts = await Post.find({
  $or: [{ status: "draft" }, { views: { $lt: 100 } }],
}).select("title status views");

console.log("$or: (草稿 OR 浏览量<100)");
orPosts.forEach((post) => {
  console.log(`  - ${post.title}: ${post.status}, ${post.views} 浏览`);
});

// 多条件 OR
const multiOrPosts = await Post.find({
  $or: [{ title: /Mongoose/i }, { title: /Node.js/i }, { tags: "mongodb" }],
}).select("title tags");

console.log(
  "\n$or: (标题包含 Mongoose OR 标题包含 Node.js OR 包含 mongodb 标签)"
);
multiOrPosts.forEach((post) => {
  console.log(`  - ${post.title}: ${post.tags.join(", ")}`);
});

// ==================== $and (逻辑与) ====================
console.log("\n=== $and (逻辑与) ===");

const andPosts = await Post.find({
  $and: [
    { status: "published" },
    { views: { $gte: 100 } },
    { tags: "mongodb" },
  ],
}).select("title status views tags");

console.log("$and: (已发布 AND 浏览量>=100 AND 包含 mongodb 标签)");
andPosts.forEach((post) => {
  console.log(
    `  - ${post.title}: ${post.status}, ${post.views} 浏览, [${post.tags.join(", ")}]`
  );
});

// ==================== $not (逻辑非) ====================
console.log("\n=== $not (逻辑非) ===");

const notPosts = await Post.find({
  status: {
    $not: { $eq: "draft" },
  },
}).select("title status");

console.log("$not: (状态 != draft)");
notPosts.forEach((post) => {
  console.log(`  - ${post.title}: ${post.status}`);
});

// 配合正则使用
const notRegexPosts = await Post.find({
  title: {
    $not: /Mongoose/i,
  },
}).select("title");

console.log("\n$not: (标题不包含 Mongoose)");
notRegexPosts.forEach((post) => {
  console.log(`  - ${post.title}`);
});

// ==================== $nor (逻辑或非) ====================
console.log("\n=== $nor (逻辑或非) ===");

const norPosts = await Post.find({
  $nor: [{ status: "draft" }, { views: { $lt: 100 } }],
}).select("title status views");

console.log("$nor: (不是草稿 AND 不是低浏览量)");
norPosts.forEach((post) => {
  console.log(`  - ${post.title}: ${post.status}, ${post.views} 浏览`);
});

// ==================== 嵌套逻辑运算符 ====================
console.log("\n=== 嵌套逻辑运算符 ===");

const nestedPosts = await Post.find({
  $or: [
    {
      $and: [{ status: "published" }, { views: { $gte: 200 } }],
    },
    {
      $and: [{ status: "draft" }, { tags: "mongodb" }],
    },
  ],
}).select("title status views");

console.log("$or + $and: ((已发布 AND 高浏览) OR (草稿 AND 包含 mongodb))");
nestedPosts.forEach((post) => {
  console.log(`  - ${post.title}: ${post.status}, ${post.views} 浏览`);
});

process.exit(0);
```

### 元素运算符

创建 `src/examples/query-element.ts`：

```typescript
import Post from "../models/Post";
import connectDB from "../config/database";

await connectDB();

// ==================== $exists (字段存在) ====================
console.log("\n=== $exists (字段存在) ===");

// 字段存在
const hasMetadataPosts = await Post.find({
  "metadata.seoTitle": { $exists: true },
}).select("title metadata.seoTitle");

console.log("$exists: 有 SEO 标题的文章");
hasMetadataPosts.forEach((post) => {
  console.log(`  - ${post.title}: "${post.metadata.seoTitle}"`);
});

// 字段不存在
const noDescriptionPosts = await Post.find({
  "metadata.seoDescription": { $exists: false },
}).select("title");

console.log("\n$exists: 没有 SEO 描述的文章");
noDescriptionPosts.forEach((post) => {
  console.log(`  - ${post.title}`);
});

// 字段存在且不为 null
const existsNotNullPosts = await Post.find({
  comments: { $exists: true, $not: { $size: 0 } },
}).select("title");

console.log("\n$exists: 有评论的文章");
existsNotNullPosts.forEach((post) => {
  console.log(`  - ${post.title}`);
});

// ==================== $type (字段类型) ====================
console.log("\n=== $type (字段类型) ===");

// _id 是 ObjectId
const objectIdPosts = await Post.find({
  _id: { $type: "objectId" },
});
console.log(`$type: _id 是 ObjectId 的文章: ${objectIdPosts.length} 篇`);

// title 是 string
const stringTitlePosts = await Post.find({
  title: { $type: "string", $exists: true },
});
console.log(`$type: title 是 string 的文章: ${stringTitlePosts.length} 篇`);

// ==================== $mod (取模运算) ====================
console.log("\n=== $mod (取模运算) ===");

// views 能被 100 整除
const modPosts = await Post.find({
  views: { $mod: [100, 0] },
}).select("title views");

console.log("$mod: 浏览量是 100 的倍数");
modPosts.forEach((post) => {
  console.log(`  - ${post.title}: ${post.views} 浏览`);
});

// 复杂条件：浏览量在 100-200 之间，且是 50 的倍数
const complexModPosts = await Post.find({
  views: {
    $gte: 100,
    $lte: 200,
    $mod: [50, 0],
  },
}).select("title views");

console.log("\n复杂条件: 100<=浏览量<=200 且是50的倍数");
complexModPosts.forEach((post) => {
  console.log(`  - ${post.title}: ${post.views} 浏览`);
});

process.exit(0);
```

```typescript
import Post from "../models/Post";

console.log("\n范围: 有 2-3 个标签的文章");

console.log("\n复杂元素匹配: 包含特定评论的文章");
complexElemMatchPosts.forEach((post) => {

// ==================== 数组位置操作符 ====================
console.log("\n=== 数组位置操作符 ===");

// 使用 $ 查询数组

  "tags.0": "mongodb", // 第一个标签是 mongodb
}).select("title tags");

console.log("第一个标签是 mongodb 的文章");

// ==================== 数组查询组合 ====================
```

### 正则表达式查询

创建 `src/examples/query-regex.ts`：

```typescript
import Post from "../models/Post";
import connectDB from "../config/database";

await connectDB();

// ==================== $regex (正则表达式) ====================
console.log("\n=== $regex (正则表达式) ===");

// 基本正则
const regexPosts1 = await Post.find({
  title: { $regex: /Mongoose/i }, // i 忽略大小写
}).select("title");

console.log('$regex: 标题包含 "Mongoose"');
regexPosts1.forEach((post) => {
  console.log(`  - ${post.title}`);
});

// 锚点匹配
const regexPosts2 = await Post.find({
  title: { $regex: /^Node.js/ }, // 以 Node.js 开头
}).select("title");

console.log('\n$regex: 标题以 "Node.js" 开头');
regexPosts2.forEach((post) => {
  console.log(`  - ${post.title}`);
});

// 模式匹配
const regexPosts3 = await Post.find({
  title: { $regex: /.*教程.*$/ }, // 以"教程"结尾
}).select("title");

console.log('\n$regex: 标题以"教程"结尾');
regexPosts3.forEach((post) => {
  console.log(`  - ${post.title}`);
});

// ==================== $options (正则选项) ====================
console.log("\n=== $options (正则选项) ===");

// i: 忽略大小写
// m: 多行模式
// x: 忽略空格和注释
// s: 让点号匹配换行符

const optionsPosts = await Post.find({
  title: {
    $regex: "mongoose",
    $options: "i", // 忽略大小写
  },
}).select("title");

console.log('$options + $regex: 标题包含 "mongoose" (忽略大小写)');
optionsPosts.forEach((post) => {
  console.log(`  - ${post.title}`);
});

// ==================== 字符串方法（简化正则） ====================
console.log("\n=== 字符串方法（简化正则） ===");

// 使用字符串进行正则匹配（自动转换为正则）
const stringPosts = await Post.find({
  title: { $regex: "教程" }, // 等价于 /教程/
}).select("title");

console.log('字符串正则: 标题包含"教程"');
stringPosts.forEach((post) => {
  console.log(`  - ${post.title}`);
});

// ==================== 复杂正则示例 ====================
console.log("\n=== 复杂正则示例 ===");

// 匹配中文 + 英文 + 数字
const complexRegexPosts = await Post.find({
  title: {
    $regex: /^[\u4e00-\u9fa5a-zA-Z0-9\s]+$/,
  },
}).select("title");

console.log("复杂正则: 标题只包含中英文、数字和空格");
complexRegexPosts.forEach((post) => {
  console.log(`  - ${post.title}`);
});

// 匹配邮箱格式的字符串
const emailRegexPosts = await Post.find({
  "metadata.contactEmail": {
    $regex: /^[^\s@]+@[^\s@]+\.[^\s@]+$/,
    $options: "i",
  },
});

console.log("\n邮箱格式验证");

// ==================== 正则表达式性能 ====================
console.log("\n=== 正则表达式性能 ===");

// ✅ 前缀匹配（性能好）
const prefixPosts = await Post.find({
  title: { $regex: /^MongoDB/ },
});

// ❌ 后缀匹配（性能差）
const suffixPosts = await Post.find({
  title: { $regex: /指南$/ },
});

// ✅ 使用索引的字符串开头匹配
const indexedPosts = await Post.find({
  title: /^MongoDB/, // 可以使用索引
});

process.exit(0);
```

### 文本搜索

创建 `src/examples/query-text-search.ts`：

```typescript
import Post from "../models/Post";
import connectDB from "../config/database";

await connectDB();

// ==================== 创建文本索引 ====================
console.log("\n=== 创建文本索引 ===");

try {
  await Post.collection.createIndex({
    title: "text",
    content: "text",
    tags: "text",
  });
  console.log("✅ 文本索引创建成功");
} catch (error: any) {
  if (error.code === 85) {
    console.log("⚠️  文本索引已存在");
  } else {
    console.log("❌ 创建索引失败:", error.message);
  }
}

// ==================== $text (文本搜索) ====================
console.log("\n=== $text (文本搜索) ===");

// 基础文本搜索
const textPosts1 = await Post.find({
  $text: { $search: "Mongoose 教程" },
}).select("title");

console.log('$text: 搜索 "Mongoose 教程"');
textPosts1.forEach((post) => {
  console.log(`  - ${post.title}`);
});

// 精确短语搜索
const textPosts2 = await Post.find({
  $text: { $search: '"Node.js 实战"' }, // 双引号表示精确匹配
}).select("title");

console.log('\n$text: 精确搜索 "Node.js 实战"');
textPosts2.forEach((post) => {
  console.log(`  - ${post.title}`);
});

// 排除词
const textPosts3 = await Post.find({
  $text: { $search: "MongoDB -数据库" }, // - 表示排除
}).select("title");

console.log('\n$text: 搜索 "MongoDB" 但排除"数据库"');
textPosts3.forEach((post) => {
  console.log(`  - ${post.title}`);
});

// ==================== 文本搜索排序 ====================
console.log("\n=== 文本搜索排序 ===");

const sortedTextPosts = await Post.find(
  {
    $text: { $search: "Mongoose" },
  },
  {
    score: { $meta: "textScore" }, // 添加相关性分数
  }
)
  .sort({ score: { $meta: "textScore" } }) // 按相关性排序
  .select("title score")
  .limit(5);

console.log("$text: 按相关性排序");
sortedTextPosts.forEach((post: any) => {
  console.log(`  - ${post.title} (相关度: ${post.score?.toFixed(2)})`);
});

// ==================== 文本搜索选项 ====================
console.log("\n=== 文本搜索选项 ===");

// $language: 指定语言
const langPosts = await Post.find({
  $text: {
    $search: "Mongoose",
    $language: "english", // 或 'none', 'chinese' 等
  },

// $caseSensitive: 大小写敏感
const casePosts = await Post.find({

  },

// $diacriticSensitive: 变音符号敏感
const diacriticPosts = await Post.find({

  },

// ==================== 文本搜索限制 ====================
console.log("\n=== 文本搜索限制 ===");

// 1. 最小字符长度
const shortText = await Post.find({

// 2. 停用词
const stopWordPosts = await Post.find({
  $text: { $search: "the a an" }, // 停用词会被忽略

// 3. 索引字段限制
// 只能在已索引的字段上搜索

// ==================== 高级文本搜索 ====================
console.log("\n=== 高级文本搜索 ===");

// 多字段搜索
const multiFieldPosts = await Post.find({

    $language: "none",
  },
});

// 组合搜索
const combinedTextPosts = await Post.find({
  $and: [{ $text: { $search: "Mongoose" } }, { status: "published" }],

console.log("\n组合搜索: 文本搜索 + 状态筛选");
combinedTextPosts.forEach((post) => {
  console.log(`  - ${post.title}: ${post.status}`);
});

// ==================== 文本搜索权重 ====================
console.log("\n=== 文本搜索权重 ===");

// 为不同字段设置权重
try {
  await Post.collection.dropIndex("title_text");
  await Post.collection.createIndex(
    {
      title: "text",
      content: "text",
      tags: "text",
    },
    {
      weights: {
        title: 10, // 标题权重 10
        content: 5, // 内容权重 5
        tags: 3, // 标签权重 3

    }

} catch (error) {
  console.log("⚠️  索引已存在或创建失败");
}

const weightedPosts = await Post.find(
  {
    $text: { $search: "Mongoose" },
  },
  {
    score: { $meta: "textScore" },
  }
)
  .sort({ score: { $meta: "textScore" } })
  .select("title score")
  .limit(3);

console.log("\n加权文本搜索:");
weightedPosts.forEach((post: any) => {
  console.log(`  - ${post.title} (相关度: ${post.score?.toFixed(2)})`);
```

```typescript
// ✅ 使用索引字段
const posts = await Post.find({ author: userId, status: "published" });

const posts = await Post.find().select("title views createdAt").lean();

const posts = await Post.find().limit(20);
```

```typescript
// ✅ 使用查询构建器
const posts = await Post.find({ status: "published" })
```

## 常见问题

### Q1: $regex 和 $text 的区别？

**A**:

- **$regex**: 标准正则表达式，适合精确模式匹配
- **$text**: 全文搜索，支持相关性排序和语言分析

```typescript
// $regex: 精确匹配
await Post.find({ title: { $regex: /Mongoose/i } });

// $text: 全文搜索（更灵活）
await Post.find({ $text: { $search: "Mongoose 教程" } });
```

### Q2: 如何优化数组查询性能？

**A**:

```typescript
// ✅ 使用索引
postSchema.index({ tags: 1 });

// ✅ 使用 $size（如果可能）
const posts = await Post.find({ tags: { $size: 3 } });

// ✅ 使用 $elemMatch（复杂条件）
const posts = await Post.find({
  comments: {
    $elemMatch: {
      author: userId,
      createdAt: { $gte: date },
    },
  },
});
```

### Q3: 逻辑运算符如何短路？

**A**:

```typescript
// $and 和 $or 都会短路
// MongoDB 会先执行容易的查询，再执行复杂的查询

const posts = await Post.find({
  $and: [
    { status: "published" }, // 先执行（快）
    { views: { $gte: 100 } }, // 后执行（慢）
  ],
});
```
