### 3.2 Context 创建

```typescript
// apps/worker/src/context.ts
import { drizzle } from "drizzle-orm/d1";
import { eq } from "drizzle-orm";
import { users } from "@global-blog/database";
import { verifyJwt } from "./lib/jwt";

export interface Context {
  db: ReturnType<typeof drizzle>;
  cache: KVNamespace;
  storage: R2Bucket;
  userId?: number;
  user?: User;
}

export async function createContext(
  env: Env,
  request: Request
): Promise<Context> {
  const db = drizzle(env.DB, { schema });

  // 获取用户信息
  let userId: number | undefined;
  let user: User | undefined;

  try {
    const authHeader = request.headers.get("Authorization");
    if (authHeader?.startsWith("Bearer ")) {
      const token = authHeader.slice(7);
      const payload = verifyJwt(token, env.JWT_SECRET);
      userId = payload.userId;

      if (userId) {
        const [userRecord] = await db
          .select()
          .from(users)
          .where(eq(users.id, userId))
          .limit(1);

        user = userRecord;
      }
    }
  } catch {
    // 忽略认证错误
  }

  return {
    db,
    cache: env.CACHE,
    storage: env.STORAGE,
    userId,
    user,
  };
}
```

### 3.3 文章 Router

```typescript
// apps/worker/src/routers/posts.ts
import { z } from "zod";
import { eq, desc, sql } from "drizzle-orm";
import { posts, users, tags, postsToTags } from "@global-blog/database";
import { t } from "../trpc";
import { invalidateCache } from "../lib/cache";

export const postsRouter = t.router({
  // 获取文章列表（分页）
  list: t.procedure
    .input(
      z.object({
        page: z.number().min(1).default(1),
        limit: z.number().min(1).max(100).default(10),
        published: z.boolean().optional(),
        tag: z.string().optional(),
      })
    )
    .query(async ({ input, ctx }) => {
      const { page, limit, published, tag } = input;

      // 尝试从缓存读取
      const cacheKey = `posts:${page}:${limit}:${published}:${tag || "all"}`;
      const cached = await ctx.cache.get(cacheKey, "json");
      if (cached) {
        return cached;
      }

      // 构建查询
      const where = [];

      if (published !== undefined) {
        where.push(sql`${posts.published} = ${published ? 1 : 0}`);
      }

      if (tag) {
        where.push(
          sql`${posts.id} IN (SELECT post_id FROM posts_to_tags WHERE tag_id = (SELECT id FROM tags WHERE slug = ${tag}))`
        );
      }

      // 查询文章
      const items = await ctx.db
        .select({
          id: posts.id,
          title: posts.title,
          slug: posts.slug,
          excerpt: posts.excerpt,
          coverImage: posts.coverImage,
          published: posts.published,
          viewCount: posts.viewCount,
          createdAt: posts.createdAt,
          author: {
            id: users.id,
            name: users.name,
            username: users.username,
            avatar: users.avatar,
          },
        })
        .from(posts)
        .innerJoin(users, eq(posts.authorId, users.id))
        .where(
          where.length > 0 ? sql`${sql.join(where, sql` AND `)}` : undefined
        )
        .orderBy(desc(posts.createdAt))
        .limit(limit)
        .offset((page - 1) * limit);

      // 查询总数
      const [totalResult] = await ctx.db
        .select({ count: sql<number>`count(*)` })
        .from(posts)
        .where(
          where.length > 0 ? sql`${sql.join(where, sql` AND `)}` : undefined
        );

      const result = {
        items,
        total: totalResult?.count || 0,
        page,
        limit,
        totalPages: Math.ceil((totalResult?.count || 0) / limit),
      };

      // 写入缓存（5 分钟）
      await ctx.cache.put(cacheKey, JSON.stringify(result), {
        expirationTtl: 300,
      });

      return result;
    }),

  // 根据 slug 获取文章
  bySlug: t.procedure
    .input(z.object({ slug: z.string() }))
    .query(async ({ input, ctx }) => {
      const { slug } = input;

      // 尝试从缓存读取
      const cached = await ctx.cache.get(`post:${slug}`, "json");
      if (cached) {
        // 增加浏览量（异步）
        ctx.db
          .update(posts)
          .set({ viewCount: sql`${posts.viewCount} + 1` })
          .where(eq(posts.slug, slug))
          .run();

        return cached;
      }

      // 查询文章
      const [post] = await ctx.db
        .select({
          id: posts.id,
          title: posts.title,
          slug: posts.slug,
          content: posts.content,
          excerpt: posts.excerpt,
          coverImage: posts.coverImage,
          published: posts.published,
          viewCount: posts.viewCount,
          createdAt: posts.createdAt,
          updatedAt: posts.updatedAt,
          publishedAt: posts.publishedAt,
          author: {
            id: users.id,
            name: users.name,
            username: users.username,
            avatar: users.avatar,
            bio: users.bio,
          },
        })
        .from(posts)
        .innerJoin(users, eq(posts.authorId, users.id))
        .where(eq(posts.slug, slug))
        .limit(1);

      if (!post) {
        throw new Error("Post not found");
      }

      // 查询标签
      const postTags = await ctx.db
        .select({
          id: tags.id,
          name: tags.name,
          slug: tags.slug,
        })
        .from(tags)
        .innerJoin(postsToTags, eq(tags.id, postsToTags.tagId))
        .where(eq(postsToTags.postId, post.id));

      const result = { ...post, tags: postTags };

      // 写入缓存（10 分钟）
      await ctx.cache.put(`post:${slug}`, JSON.stringify(result), {
        expirationTtl: 600,
      });

      // 增加浏览量
      await ctx.db
        .update(posts)
        .set({ viewCount: sql`${posts.viewCount} + 1` })
        .where(eq(posts.slug, slug))
        .run();

      return result;
    }),

  // 创建文章
  create: t.procedure
    .input(
      z.object({
        title: z.string().min(1),
        content: z.string().min(1),
        excerpt: z.string().optional(),
        coverImage: z.string().optional(),
        tagIds: z.array(z.number()).optional(),
      })
    )
    .mutation(async ({ input, ctx }) => {
      if (!ctx.userId) {
        throw new Error("Unauthorized");
      }

      const slug = input.title
        .toLowerCase()
        .replace(/\s+/g, "-")
        .replace(/[^\w-]+/g, "");

      // 创建文章
      const [post] = await ctx.db
        .insert(posts)
        .values({
          ...input,
          slug,
          authorId: ctx.userId,
          published: false,
          createdAt: new Date(),
          updatedAt: new Date(),
        })
        .returning();

      // 关联标签
      if (input.tagIds && input.tagIds.length > 0) {
        await ctx.db.insert(postsToTags).values(
          input.tagIds.map((tagId) => ({
            postId: post.id,
            tagId,
          }))
        );
      }

      // 删除缓存
      await invalidateCache(ctx.cache, "posts");

      return post;
    }),

  // 更新文章
  update: t.procedure
    .input(
      z.object({
        id: z.number(),
        title: z.string().min(1).optional(),
        content: z.string().min(1).optional(),
        excerpt: z.string().optional(),
        coverImage: z.string().optional(),
        published: z.boolean().optional(),
      })
    )
    .mutation(async ({ input, ctx }) => {
      const { id, ...data } = input;

      // 检查权限
      const [post] = await ctx.db
        .select()
        .from(posts)
        .where(eq(posts.id, id))
        .limit(1);

      if (!post) {
        throw new Error("Post not found");
      }

      if (post.authorId !== ctx.userId && ctx.user?.role !== "admin") {
        throw new Error("Forbidden");
      }

      // 更新文章
      const [updated] = await ctx.db
        .update(posts)
        .set({
          ...data,
          updatedAt: new Date(),
          publishedAt: data.published ? new Date() : post.publishedAt,
        })
        .where(eq(posts.id, id))
        .returning();

      // 删除缓存
      await invalidateCache(ctx.cache, "posts");
      await ctx.cache.delete(`post:${post.slug}`);

      return updated;
    }),

  // 删除文章
  delete: t.procedure
    .input(z.object({ id: z.number() }))
    .mutation(async ({ input, ctx }) => {
      const { id } = input;

      // 检查权限
      const [post] = await ctx.db
        .select()
        .from(posts)
        .where(eq(posts.id, id))
        .limit(1);

      if (!post) {
        throw new Error("Post not found");
      }

      if (post.authorId !== ctx.userId && ctx.user?.role !== "admin") {
        throw new Error("Forbidden");
      }

      // 删除文章
      await ctx.db.delete(posts).where(eq(posts.id, id));

      // 删除缓存
      await invalidateCache(ctx.cache, "posts");
      await ctx.cache.delete(`post:${post.slug}`);

      return { success: true };
    }),
});
```

import { notFound } from 'next/navigation';
import { db } from '@/lib/db';

            {tag.name}

    </article>

// apps/worker/src/middleware/analytics.ts
export async function analytics(c: Context, next: Next) {
  const start = Date.now();

  await next();

  const duration = Date.now() - start;

  // 发送到分析服务
  await c.env.CACHE.put(
    `analytics:${Date.now()}`,
    JSON.stringify({
      path: c.req.path,
      method: c.req.method,
      duration,
      status: c.res.status,
    }),
    {
      expirationTtl: 60 * 60 * 24 * 7, // 7 天
    }
  );
}
