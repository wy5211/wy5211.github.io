## 一、为什么选择 Drizzle + tRPC？

### 1.1 类型安全全景

```
┌─────────────────────────────────────────────────────────────┐
│                    类型安全全链路                            │
├─────────────────────────────────────────────────────────────┤
│                                                               │
│  数据库 schema (Drizzle)                                     │
│       ↓ 类型推断                                              │
│  数据库查询 (Drizzle ORM)                                    │
│       ↓ 返回类型                                              │
│  tRPC Router                                                │
│       ↓ 自动生成 API 类型                                     │
│  前端调用 (tRPC Client)                                      │
│       ↓ 类型检查                                              │
│  组件 props                                                 │
│                                                               │
│  修改任何一环，类型错误会立即在整个链路中显示！                │
└─────────────────────────────────────────────────────────────┘
```

### 1.2 核心优势

| 特性          | 传统方案     | Drizzle + tRPC |
| ------------- | ------------ | -------------- |
| API 类型      | 手动定义     | 自动生成       |
| 类型同步      | 需要手动更新 | 自动同步       |
| 请求/响应验证 | 手动编写     | Zod 自动验证   |
| 代码生成      | 需要         | 不需要         |
| Bundle 大小   | 较大         | 极小           |
| 开发体验      | 一般         | 极佳           |

### 3.2 使用共享 Schema

**apps/api/src/routers/users.ts**：

```typescript
import { z } from "zod";
import { eq } from "drizzle-orm";
import { users, type User, type NewUser } from "@my-monorepo/database";
import { t } from "../trpc";

export const usersRouter = t.router({
  // 查询所有用户
  list: t.procedure.query(async ({ ctx }) => {
    return await ctx.db.select().from(users);
  }),

  // 根据 ID 查询用户
  byId: t.procedure
    .input(z.object({ id: z.number() }))
    .query(async ({ input, ctx }) => {
      const [user] = await ctx.db
        .select()
        .from(users)
        .where(eq(users.id, input.id));

      return user; // 类型：User | undefined
    }),

  // 创建用户
  create: t.procedure
    .input(
      z.object({
        email: z.string().email(),
        name: z.string().min(1),
      })
    )
    .mutation(async ({ input, ctx }) => {
      const [user] = await ctx.db
        .insert(users)
        .values(input) // 类型：NewUser
        .returning();

      return user; // 类型：User
    }),

  // 更新用户
  update: t.procedure
    .input(
      z.object({
        id: z.number(),
        email: z.string().email().optional(),
        name: z.string().optional(),
      })
    )
    .mutation(async ({ input, ctx }) => {
      const { id, ...data } = input;

      const [user] = await ctx.db
        .update(users)
        .set(data)
        .where(eq(users.id, id))
        .returning();

      return user;
    }),

  // 删除用户
  delete: t.procedure
    .input(z.object({ id: z.number() }))
    .mutation(async ({ input, ctx }) => {
      await ctx.db.delete(users).where(eq(users.id, input.id));

      return { success: true };
    }),
});
```

## 四、tRPC 后端设置

### 4.1 创建 tRPC 实例

**apps/api/src/trpc.ts**：

```typescript
import { initTRPC, TRPCError } from "@trpc/server";
import { ZodError } from "zod";
import { db } from "@my-monorepo/database";

// 创建 Context 类型
export interface Context {
  db: typeof db;
  userId?: number;
}

// 创建 tRPC 实例
const t = initTRPC.context<Context>().create({
  errorFormatter({ shape, error }) {
    return {
      ...shape,
      data: {
        ...shape.data,
        zodError:
          error.cause instanceof ZodError ? error.cause.flatten() : null,
      },
    };
  },
});

export const t = t;

// 创建中间件
export const middleware = t.middleware;

// 创建受保护的路由
export const protectedProcedure = t.procedure.use(async (opts) => {
  const { ctx } = opts;

  if (!ctx.userId) {
    throw new TRPCError({
      code: "UNAUTHORIZED",
      message: "You must be logged in",
    });
  }

  return opts.next({
    ctx: {
      ...ctx,
      userId: ctx.userId,
    },
  });
});
```

'use client';

import { trpc } from '@/lib/trpc/client';
import { useState } from 'react';

export function CreateUser() {
  const [email, setEmail] = useState('');
  const [name, setName] = useState('');

  const utils = trpc.useContext();

  // input 类型自动推断
  const createUser = trpc.users.create.useMutation({
    onSuccess: () => {
      // 刷新列表
      utils.users.list.invalidate();
      setEmail('');
      setName('');
    },
  });

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    createUser.mutate({ email, name });
  };

  return (
    <form onSubmit={handleSubmit}>
      <input
        type="email"
        value={email}
        onChange={(e) => setEmail(e.target.value)}
        placeholder="Email"
        required
      />
      <input
        type="text"
        value={name}
        onChange={(e) => setName(e.target.value)}
        placeholder="Name"
        required
      />
      <button type="submit" disabled={createUser.isLoading}>

## 七、高级用法

### 7.1 关系查询

```typescript
// apps/api/src/routers/posts.ts
import { eq } from "drizzle-orm";
import { posts, users, type Post } from "@my-monorepo/database";
import { t } from "../trpc";

export const postsRouter = t.router({
  // 查询文章和作者
  list: t.procedure.query(async ({ ctx }) => {
    const result = await ctx.db
      .select({
        id: posts.id,
        title: posts.title,
        content: posts.content,
        author: {
          id: users.id,
          name: users.name,
          email: users.email,
        },
      })
      .from(posts)
      .innerJoin(users, eq(posts.authorId, users.id));

    return result;
    // 类型：{
    //   id: number;
    //   title: string;
    //   content: string;
    //   author: {
    //     id: number;
    //     name: string | null;
    //     email: string;
    //   };
    // }[]
  }),

  // 根据 slug 查询文章（包含作者和标签）
  bySlug: t.procedure
    .input(z.object({ slug: z.string() }))
    .query(async ({ input, ctx }) => {
      const [post] = await ctx.db
        .select()
        .from(posts)
        .where(eq(posts.slug, input.slug))
        .limit(1);

      if (!post) return null;

      // 获取作者
      const [author] = await ctx.db
        .select()
        .from(users)
        .where(eq(users.id, post.authorId))
        .limit(1);

      // 获取标签
      const tags = await ctx.db
        .select()
        .from(tags)
        .innerJoin(postsToTags, eq(tags.id, postsToTags.tagId))
        .where(eq(postsToTags.postId, post.id));

      return {
        ...post,
        author,
        tags: tags.map((t) => t.tags),
      };
    }),
});
```

### 7.2 分页查询

```typescript
// apps/api/src/routers/posts.ts
export const postsRouter = t.router({
  paginated: t.procedure
    .input(
      z.object({
        page: z.number().min(1).default(1),
        limit: z.number().min(1).max(100).default(10),
        published: z.boolean().optional(),
      })
    )
    .query(async ({ input, ctx }) => {
      const { page, limit, published } = input;
      const offset = (page - 1) * limit;

      const where =
        published !== undefined ? eq(posts.published, published) : undefined;

      const items = await ctx.db
        .select()
        .from(posts)
        .where(where)
        .orderBy(desc(posts.createdAt))
        .limit(limit)
        .offset(offset);

      // 获取总数
      const [totalResult] = await ctx.db
        .select({ count: sql<number>`count(*)` })
        .from(posts)
        .where(where);

      return {
        items,
        total: totalResult?.count || 0,
        page,
        limit,
        totalPages: Math.ceil((totalResult?.count || 0) / limit),
      };
    }),
});
```
