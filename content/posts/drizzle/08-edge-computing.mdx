### 2.3 创建 Worker

**src/index.ts**：

```typescript
import { drizzle } from "drizzle-orm/d1";
import { eq } from "drizzle-orm";
import * as schema from "./db/schema";

export interface Env {
  DB: D1Database;
  CACHE: KVNamespace;
  STORAGE: R2Bucket;
}

export default {
  async fetch(request: Request, env: Env): Promise<Response> {
    const url = new URL(request.url);
    const db = drizzle(env.DB, { schema });

    // 路由处理
    if (url.pathname === "/api/users") {
      if (request.method === "GET") {
        return await getUsers(db);
      }
      if (request.method === "POST") {
        return await createUser(request, db);
      }
    }

    if (url.pathname.startsWith("/api/users/")) {
      const id = parseInt(url.pathname.split("/")[3]);
      return await getUser(db, id);
    }

    return new Response("Not found", { status: 404 });
  },
};

async function getUsers(db: DrizzleDB) {
  const users = await db.select().from(schema.users);
  return Response.json(users);
}

async function getUser(db: DrizzleDB, id: number) {
  const [user] = await db
    .select()
    .from(schema.users)
    .where(eq(schema.users.id, id))
    .limit(1);

  if (!user) {
    return new Response("User not found", { status: 404 });
  }

  return Response.json(user);
}

async function createUser(request: Request, db: DrizzleDB) {
  const body = await request.json();

  const [user] = await db.insert(schema.users).values(body).returning();

  return Response.json(user, { status: 201 });
}
```

### 3.5 完整的 CRUD 示例

```typescript
import { drizzle } from "drizzle-orm/d1";
import { eq, desc } from "drizzle-orm";
import * as schema from "./db/schema";

interface Env {
  DB: D1Database;
}

export default {
  async fetch(request: Request, env: Env): Promise<Response> {
    const url = new URL(request.url);
    const db = drizzle(env.DB, { schema });

    if (url.pathname === "/api/posts" && request.method === "GET") {
      return getPosts(db);
    }

    if (url.pathname === "/api/posts" && request.method === "POST") {
      return createPost(request, db);
    }

    if (url.pathname.match(/^\/api\/posts\/\d+$/) && request.method === "GET") {
      const id = parseInt(url.pathname.split("/")[3]);
      return getPost(db, id);
    }

    if (
      url.pathname.match(/^\/api\/posts\/\d+$/) &&
      request.method === "PATCH"
    ) {
      const id = parseInt(url.pathname.split("/")[3]);
      return updatePost(request, db, id);
    }

    if (
      url.pathname.match(/^\/api\/posts\/\d+$/) &&
      request.method === "DELETE"
    ) {
      const id = parseInt(url.pathname.split("/")[3]);
      return deletePost(db, id);
    }

    return new Response("Not found", { status: 404 });
  },
};

async function getPosts(db: DrizzleDB) {
  const posts = await db
    .select()
    .from(schema.posts)
    .orderBy(desc(schema.posts.createdAt));

  return Response.json(posts);
}

async function getPost(db: DrizzleDB, id: number) {
  const [post] = await db
    .select()
    .from(schema.posts)
    .where(eq(schema.posts.id, id))
    .limit(1);

  if (!post) {
    return new Response("Post not found", { status: 404 });
  }

  return Response.json(post);
}

async function createPost(request: Request, db: DrizzleDB) {
  const body = await request.json();

  const [post] = await db
    .insert(schema.posts)
    .values({
      ...body,
      createdAt: new Date(),
    })
    .returning();

  return Response.json(post, { status: 201 });
}

async function updatePost(request: Request, db: DrizzleDB, id: number) {
  const body = await request.json();

  const [post] = await db
    .update(schema.posts)
    .set({
      ...body,
      updatedAt: new Date(),
    })
    .where(eq(schema.posts.id, id))
    .returning();

  if (!post) {
    return new Response("Post not found", { status: 404 });
  }

  return Response.json(post);
}

async function deletePost(db: DrizzleDB, id: number) {
  await db.delete(schema.posts).where(eq(schema.posts.id, id));

  return new Response(null, { status: 204 });
}
```

## 四、Cloudflare KV 缓存

### 4.1 KV 简介

KV 是 Cloudflare 的**全球分布式键值存储**：

- 🌍 **全球复制**：数据自动复制到所有边缘节点
- ⚡ **极低延迟**：< 1ms 读取延迟
- 💾 **最终一致性**：数据在 60 秒内全球同步

### 4.2 创建 KV 命名空间

```bash
# 创建 KV 命名空间
wrangler kv:namespace create CACHE

# 更新 wrangler.toml 中的 id
```

### 4.3 使用 KV 缓存

```typescript
export default {
  async fetch(request: Request, env: Env): Promise<Response> {
    const url = new URL(request.url);
    const db = drizzle(env.DB, { schema });

    if (url.pathname === "/api/users") {
      return getUsersWithCache(db, env.CACHE);
    }

    return new Response("Not found", { status: 404 });
  },
};

async function getUsersWithCache(db: DrizzleDB, cache: KVNamespace) {
  // 1. 尝试从缓存读取
  const cached = await cache.get("users", "json");
  if (cached) {
    return Response.json(cached, {
      headers: {
        "X-Cache": "HIT",
      },
    });
  }

  // 2. 从数据库查询
  const users = await db.select().from(schema.users);

  // 3. 写入缓存（60 秒过期）
  await cache.put("users", JSON.stringify(users), {
    expirationTtl: 60,
  });

  return Response.json(users, {
    headers: {
      "X-Cache": "MISS",
    },
  });
}
```

### 4.4 缓存失效

### 4.5 高级缓存策略

```typescript
// 按用户 ID 缓存
async function getUserWithCache(
  db: DrizzleDB,
  cache: KVNamespace,
  userId: number
) {
  const cacheKey = `user:${userId}`;

  // 尝试从缓存读取
  const cached = await cache.get(cacheKey, "json");
  if (cached) {
    return Response.json(cached, { headers: { "X-Cache": "HIT" } });
  }

  // 从数据库查询
  const [user] = await db
    .select()
    .from(schema.users)
    .where(eq(schema.users.id, userId))
    .limit(1);

  if (!user) {
    return new Response("User not found", { status: 404 });
  }

  // 写入缓存（5 分钟）
  await cache.put(cacheKey, JSON.stringify(user), {
    expirationTtl: 300,
  });

  return Response.json(user, { headers: { "X-Cache": "MISS" } });
}

// 缓存标签（手动实现）
async function invalidateUserCache(cache: KVNamespace, userId: number) {
  await cache.delete(`user:${userId}`);
  await cache.delete("users");
  await cache.delete("users:stats");
}
```

## 八、实战示例

### 8.1 边缘 API 服务

```typescript
import { drizzle } from "drizzle-orm/d1";
import { eq, desc, sql } from "drizzle-orm";
import * as schema from "./db/schema";

interface Env {
  DB: D1Database;
  CACHE: KVNamespace;
}

export default {
  async fetch(request: Request, env: Env): Promise<Response> {
    const url = new URL(request.url);
    const db = drizzle(env.DB, { schema });

    // CORS
    if (request.method === "OPTIONS") {
      return new Response(null, {
        headers: {
          "Access-Control-Allow-Origin": "*",
          "Access-Control-Allow-Methods": "GET, POST, PUT, DELETE, OPTIONS",
          "Access-Control-Allow-Headers": "Content-Type, Authorization",
        },
      });
    }

    // API 路由
    if (url.pathname === "/api/posts" && request.method === "GET") {
      return getPosts(db, env.CACHE);
    }

    if (url.pathname === "/api/posts" && request.method === "POST") {
      return createPost(request, db, env.CACHE);
    }

    if (url.pathname === "/api/posts/stats" && request.method === "GET") {
      return getPostsStats(db, env.CACHE);
    }

    return new Response("Not found", { status: 404 });
  },
};

async function getPosts(db: DrizzleDB, cache: KVNamespace) {
  const cached = await cache.get("posts", "json");
  if (cached) {
    return Response.json(cached);
  }

  const posts = await db
    .select()
    .from(schema.posts)
    .orderBy(desc(schema.posts.createdAt))
    .limit(20);

  await cache.put("posts", JSON.stringify(posts), {
    expirationTtl: 60,
  });

  return Response.json(posts);
}

async function createPost(request: Request, db: DrizzleDB, cache: KVNamespace) {
  const body = await request.json();

  const [post] = await db
    .insert(schema.posts)
    .values({
      ...body,
      createdAt: new Date(),
    })
    .returning();

  // 删除相关缓存
  await cache.delete("posts");
  await cache.delete("posts:stats");

  return Response.json(post, { status: 201 });
}

async function getPostsStats(db: DrizzleDB, cache: KVNamespace) {
  const cached = await cache.get("posts:stats", "json");
  if (cached) {
    return Response.json(cached);
  }

  const [stats] = await db
    .select({
      total: sql<number>`count(*)`,
      published: sql<number>`count(*) filter (where ${schema.posts.published} = true)`,
    })
    .from(schema.posts);

  await cache.put("posts:stats", JSON.stringify(stats), {
    expirationTtl: 300,
  });

  return Response.json(stats);
}
```

### 8.2 边缘图片处理

```typescript
export default {
  async fetch(request: Request, env: Env): Promise<Response> {
    const url = new URL(request.url);

    if (url.pathname.startsWith("/images/") && request.method === "GET") {
      return getImage(request, env);
    }

    if (url.pathname === "/upload" && request.method === "POST") {
      return uploadImage(request, env);
    }

    return new Response("Not found", { status: 404 });
  },
};

async function getImage(request: Request, env: Env) {
  const url = new URL(request.url);
  const key = url.pathname.slice(1);

  // 从 R2 获取图片
  const object = await env.STORAGE.get(key);

  if (!object) {
    return new Response("Not found", { status: 404 });
  }

  // 检查缓存
  const cache = caches.default;
  let response = await cache.match(request);

  if (!response) {
    // 设置缓存头
    const headers = new Headers();
    object.writeHttpMetadata(headers);
    headers.set("Cache-Control", "public, max-age=31536000");
    headers.append("Vary", "Accept-Encoding");

    response = new Response(object.body, { headers });
    await cache.put(request, response.clone());
  }

  return response;
}

async function uploadImage(request: Request, env: Env) {
  const formData = await request.formData();
  const file = formData.get("file") as File;

  const key = `images/${Date.now()}-${file.name}`;

  await env.STORAGE.put(key, file);

  return Response.json({
    url: `https://your-domain.com/${key}`,
  });
}
```

### 9.3 错误处理

```typescript
export default {
  async fetch(request: Request, env: Env): Promise<Response> {
    try {
      return await handleRequest(request, env);
    } catch (error) {
      console.error("Error:", error);

      return new Response(
        JSON.stringify({
          error: "Internal server error",
          message: error instanceof Error ? error.message : "Unknown error",
        }),
        {
          status: 500,
          headers: { "Content-Type": "application/json" },
        }
      );
    }
  },
};
```
