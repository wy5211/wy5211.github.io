# 数据库设计与 TypeScript 类型安全

在全栈开发中，数据库是应用的核心。一个良好设计的数据库可以支撑应用的长期发展，而类型安全的数据库操作则能在编译时和运行时保证数据的正确性。

本文将深入探讨关系型数据库设计、Prisma ORM 的使用，以及如何构建类型安全的数据库层。

## 关系型数据库设计原则

### 数据库设计基础

关系型数据库基于关系模型，使用表格（Table）、行（Row）、列（Column）来存储数据。

**核心概念**：

```sql
-- 用户表
CREATE TABLE users (
  id INT PRIMARY KEY AUTO_INCREMENT,
  name VARCHAR(255) NOT NULL,
  email VARCHAR(255) UNIQUE NOT NULL,
  password_hash VARCHAR(255) NOT NULL,
  role ENUM('admin', 'user') DEFAULT 'user',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- 文章表
CREATE TABLE posts (
  id INT PRIMARY KEY AUTO_INCREMENT,
  title VARCHAR(255) NOT NULL,
  content TEXT NOT NULL,
  author_id INT NOT NULL,
  status ENUM('draft', 'published', 'archived') DEFAULT 'draft',
  published_at TIMESTAMP NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (author_id) REFERENCES users(id) ON DELETE CASCADE
);

-- 评论表
CREATE TABLE comments (
  id INT PRIMARY KEY AUTO_INCREMENT,
  content TEXT NOT NULL,
  author_id INT NOT NULL,
  post_id INT NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (author_id) REFERENCES users(id) ON DELETE CASCADE,
  FOREIGN KEY (post_id) REFERENCES posts(id) ON DELETE CASCADE
);
```

### 规范化（Normalization）

规范化是减少数据冗余、提高数据一致性的设计原则。

**第一范式（1NF）**：每个字段都是原子的，不可再分

```sql
-- ❌ 违反 1NF
CREATE TABLE users (
  id INT PRIMARY KEY,
  name VARCHAR(255),
  tags VARCHAR(255) -- "tech, design, programming"
);

-- ✅ 符合 1NF
CREATE TABLE users (
  id INT PRIMARY KEY,
  name VARCHAR(255)
);

CREATE TABLE user_tags (
  user_id INT,
  tag VARCHAR(50),
  FOREIGN KEY (user_id) REFERENCES users(id)
);
```

**第二范式（2NF）**：非主键字段必须完全依赖于主键

```sql
-- ❌ 违反 2NF
CREATE TABLE order_items (
  id INT PRIMARY KEY,
  order_id INT,
  product_id INT,
  product_name VARCHAR(255), -- 依赖于 product_id，而非 (order_id, product_id)
  quantity INT
);

-- ✅ 符合 2NF
CREATE TABLE order_items (
  id INT PRIMARY KEY,
  order_id INT,
  product_id INT,
  quantity INT
);

CREATE TABLE products (
  id INT PRIMARY KEY,
  name VARCHAR(255)
);
```

**第三范式（3NF）**：非主键字段不能依赖于其他非主键字段

```sql
-- ❌ 违反 3NF
CREATE TABLE users (
  id INT PRIMARY KEY,
  name VARCHAR(255),
  city VARCHAR(255),
  state VARCHAR(255), -- 依赖于 city
  zip_code VARCHAR(10)
);

-- ✅ 符合 3NF
CREATE TABLE users (
  id INT PRIMARY KEY,
  name VARCHAR(255),
  city_id INT,
  FOREIGN KEY (city_id) REFERENCES cities(id)
);

CREATE TABLE cities (
  id INT PRIMARY KEY,
  name VARCHAR(255),
  state VARCHAR(255),
  zip_code VARCHAR(10)
);
```

### 索引设计

索引是提高查询性能的关键，但需要平衡查询性能和写入性能。

**何时创建索引**：

```sql
-- 1. WHERE 子句中的字段
CREATE INDEX idx_user_email ON users(email);

-- 2. JOIN 条件中的字段
CREATE INDEX idx_post_author ON posts(author_id);

-- 3. ORDER BY 中的字段
CREATE INDEX idx_post_published ON posts(published_at DESC);

-- 4. 组合索引（多字段查询）
CREATE INDEX idx_post_status_published ON posts(status, published_at DESC);
```

**索引最佳实践**：

```sql
-- ✅ 为高频查询字段创建索引
CREATE INDEX idx_user_email ON users(email);
CREATE INDEX idx_post_author_status ON posts(author_id, status);

-- ❌ 避免过度索引
CREATE INDEX idx_user_name ON users(name); -- name 很少用于查询
CREATE INDEX idx_user_created ON users(created_at); -- created_at 很少单独使用

-- ✅ 复合索引注意顺序
-- 经常按 status 和 created_at 查询
CREATE INDEX idx_post_status_date ON posts(status, published_at DESC);

-- ✅ 使用覆盖索引
CREATE INDEX idx_post_list ON posts(author_id, status, published_at, title);
-- 查询 SELECT title FROM posts WHERE author_id = ? AND status = ?
-- 只需访问索引，无需回表
```

### 外键和关系

```sql
-- 一对一关系（User -> Profile）
CREATE TABLE profiles (
  user_id INT PRIMARY KEY,
  bio TEXT,
  avatar_url VARCHAR(255),
  FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

-- 一对多关系（User -> Posts）
CREATE TABLE posts (
  id INT PRIMARY KEY AUTO_INCREMENT,
  author_id INT NOT NULL,
  title VARCHAR(255),
  FOREIGN KEY (author_id) REFERENCES users(id) ON DELETE SET NULL
);

-- 多对多关系（User -> Roles）
CREATE TABLE user_roles (
  user_id INT,
  role_id INT,
  assigned_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
  FOREIGN KEY (role_id) REFERENCES roles(id) ON DELETE CASCADE,
  PRIMARY KEY (user_id, role_id)
);
```

### Prisma Schema 设计

```prisma
// prisma/schema.prisma

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql" // 或 "mysql", "sqlite", "mongodb"
  url      = env("DATABASE_URL")
}

// 模型定义
model User {
  id            Int       @id @default(autoincrement())
  name          String
  email         String    @unique
  passwordHash  String    @map("password_hash")
  role          UserRole  @default(USER)
  createdAt     DateTime  @default(now()) @map("created_at")
  updatedAt     DateTime  @updatedAt @map("updated_at")

  // 关系
  posts         Post[]
  comments      Comment[]
  profile       Profile?

  @@map("users") // 指定表名
}

model Profile {
  id        Int      @id @default(autoincrement())
  userId    Int      @unique @map("user_id")
  bio       String?
  avatarUrl String?  @map("avatar_url")
  createdAt DateTime @default(now()) @map("created_at")

  // 关系
  user      User     @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@map("profiles")
}

model Post {
  id          Int          @id @default(autoincrement())
  title       String
  content     String       @db.Text
  status      PostStatus   @default(DRAFT)
  publishedAt DateTime?    @map("published_at")
  createdAt   DateTime     @default(now()) @map("created_at")
  updatedAt   DateTime     @updatedAt @map("updated_at")

  // 关系
  authorId    Int          @map("author_id")
  author      User         @relation(fields: [authorId], references: [id], onDelete: Cascade)
  comments    Comment[]

  @@index([authorId, status])
  @@index([publishedAt(sort: Desc)])
  @@map("posts")
}

model Comment {
  id        Int      @id @default(autoincrement())
  content   String
  createdAt DateTime @default(now()) @map("created_at")

  // 关系
  authorId  Int      @map("author_id")
  author    User     @relation(fields: [authorId], references: [id], onDelete: Cascade)
  postId    Int      @map("post_id")
  post      Post     @relation(fields: [postId], references: [id], onDelete: Cascade)

  @@index([postId, createdAt(sort: Desc)])
  @@map("comments")
}

// 枚举
enum UserRole {
  USER
  ADMIN
}

enum PostStatus {
  DRAFT
  PUBLISHED
  ARCHIVED
}
```

### DTO 和 Entity 转换

```typescript
// 创建 DTO（不包含所有字段）
interface CreateUserDto {
  name: string;
  email: string;
  password: string;
  role?: "USER" | "ADMIN";
}

interface UpdateUserDto {
  name?: string;
  email?: string;
  role?: "USER" | "ADMIN";
}

// 类型安全的 DTO 到 Entity 转换
async function createUser(dto: CreateUserDto): Promise<PublicUser> {
  const passwordHash = await hashPassword(dto.password);

  const user = await prisma.user.create({
    data: {
      name: dto.name,
      email: dto.email,
      passwordHash,
      role: dto.role || "USER",
    },
  });

  return toPublicUser(user);
}

async function updateUser(id: number, dto: UpdateUserDto): Promise<PublicUser> {
  const user = await prisma.user.update({
    where: { id },
    data: dto,
  });

  return toPublicUser(user);
}

// Entity 到 DTO 转换
function toPublicUser(user: User): PublicUser {
  return {
    id: user.id,
    name: user.name,
    email: user.email,
    role: user.role,
    createdAt: user.createdAt,
    updatedAt: user.updatedAt,
  };
}
```

### 分页和排序类型

```typescript
interface PaginationParams {
  page: number;
  limit: number;
  sortBy?: string;
  sortOrder?: "asc" | "desc";
}

interface PaginatedResult<T> {
  data: T[];
  total: number;
  page: number;
  limit: number;
  totalPages: number;
}

async function findPaginatedUsers(
  params: PaginationParams,
): Promise<PaginatedResult<User>> {
  const { page, limit, sortBy = "createdAt", sortOrder = "desc" } = params;

  const [data, total] = await Promise.all([
    prisma.user.findMany({
      skip: (page - 1) * limit,
      take: limit,
      orderBy: {
        [sortBy]: sortOrder,
      },
    }),
    prisma.user.count(),
  ]);

  return {
    data,
    total,
    page,
    limit,
    totalPages: Math.ceil(total / limit),
  };
}
```

### 索引优化策略

```prisma
// prisma/schema.prisma

model User {
  id        String   @id @default(cuid())
  email     String   @unique
  name      String?

  @@index([email]) // 单列索引
  @@index([name])  // 用于 LIKE 查询
}

model Post {
  id        String   @id @default(cuid())
  title     String
  status    PostStatus
  publishedAt DateTime?

  authorId  String
  author    User     @relation(fields: [authorId], references: [id])

  // 复合索引
  @@index([authorId, status]) // 查询作者的特定状态文章
  @@index([status, publishedAt(sort: Desc)]) // 按状态和发布时间查询
  @@index([publishedAt(sort: Desc)]) // 最新文章列表
}
```

```typescript
  });

}

  include: {
    author: true,

const posts = await prisma.post.findMany({
  include: {
    author: {
      include: {
        posts: true,
      },
    },
  },
});

// ✅ 只查询需要的字段

  include: {
    author: {
      select: {
        id: true,
        name: true,
      },
    },
  },
```

```typescript
import { prisma } from "@/lib/db";
import type { Post, Prisma } from "@prisma/client";

export type CreatePostInput = Omit<Prisma.PostCreateInput, "authorId">;

  }

  async findAll(params: {
    page?: number;
    limit?: number;
    authorId?: number;
    status?: Prisma.PostStatusWhereInput;
  }): Promise<{ posts: Post[]; total: number }> {
    const { page = 1, limit = 10, authorId, status } = params;
    const where: Prisma.PostWhereInput = {};

    if (authorId) where.authorId = authorId;
    if (status) where.status = status;

    const [posts, total] = await Promise.all([
      prisma.post.findMany({
        where,
        skip: (page - 1) * limit,
        take: limit,
        orderBy: { publishedAt: "desc" },
      }),
      prisma.post.count({ where }),
    ]);

    return { posts, total };
  }

  async create(authorId: number, data: CreatePostInput): Promise<Post> {
    return prisma.post.create({
      data: {
        ...data,
        authorId,
      },
    });
  }

  async update(id: number, data: Prisma.PostUpdateInput): Promise<Post> {
    return prisma.post.update({ where: { id }, data });
  }

  async delete(id: number): Promise<Post> {
    return prisma.post.delete({ where: { id } });
  }

  async publish(id: number): Promise<Post> {
    return prisma.post.update({
      where: { id },
      data: {
        status: "PUBLISHED",
        publishedAt: new Date(),
      },
    });
```

### Service 层

```typescript
// lib/services/user.service.ts
import { userRepository } from "@/repositories/user.repository";
import type {
  CreateUserInput,
  UpdateUserInput,
} from "@/repositories/user.repository";
import * as bcrypt from "bcrypt";

export class UserService {
  async getPublicUser(id: number) {
    const user = await userRepository.findById(id);

    if (!user) {
      throw new Error("User not found");
    }

    const { passwordHash, ...publicUser } = user;
    return publicUser;
  }

  async createUser(data: CreateUserInput & { password: string }) {
    // 检查邮箱是否已存在
    const existing = await userRepository.findByEmail(data.email);
    if (existing) {
      throw new Error("Email already exists");
    }

    // 哈希密码
    const passwordHash = await bcrypt.hash(data.password, 10);

    const { password, ...userData } = data;

    return userRepository.create({
      ...userData,
      passwordHash,
    });
  }

  async updateUser(id: number, data: UpdateUserInput) {
    return userRepository.update(id, data);
  }

  async deleteUser(id: number) {
    return userRepository.delete(id);
  }

  async verifyPassword(email: string, password: string): Promise<boolean> {
    const user = await userRepository.findByEmail(email);

    if (!user) {
      return false;
    }

    return bcrypt.compare(password, user.passwordHash);
  }
}

export const userService = new UserService();
```

### 核心要点

1. **数据库设计原则**：规范化、合理索引、正确的关系设计
2. **Prisma Schema**：定义数据模型和关系
3. **类型安全**：利用 Prisma 生成的类型，避免手动类型注解
4. **Repository 模式**：封装数据库操作，提高可维护性
5. **性能优化**：合理使用索引、批量操作、缓存

### 最佳实践

- 使用 Prisma 迁移管理数据库 schema

- 实现 Repository 和 Service 层，提高代码可维护性

- 使用缓存减少数据库负载

良好的数据库设计能够支撑应用的长期发展，而类型安全则能保证数据的正确性和一致性。
