## 前言

CRUD（Create、Read、Update、Delete）是数据库操作的四大核心功能，也是日常开发中使用最频繁的语句。掌握 CRUD 是成为数据库高手的必经之路。

本文将通过丰富的实战示例，带你系统掌握增删改查的各种技巧和最佳实践。

## CREATE - 插入数据

### 基本语法

```sql
-- 插入完整数据（所有字段）
INSERT INTO 表名 VALUES (值1, 值2, ...);

-- 插入指定字段
INSERT INTO 表名 (字段1, 字段2) VALUES (值1, 值2);

-- 插入多条数据
INSERT INTO 表名 (字段1, 字段2) VALUES
  (值1, 值2),
  (值3, 值4),
  (值5, 值6);
```

### 单行插入

```sql
-- 准备测试表
CREATE TABLE users (
  id BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  username VARCHAR(50) NOT NULL,
  email VARCHAR(100) NOT NULL,
  age TINYINT UNSIGNED DEFAULT 18,
  status TINYINT UNSIGNED DEFAULT 1,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 方式 1：指定字段（推荐）
INSERT INTO users (username, email, age) VALUES ('张三', 'zhangsan@example.com', 25);

-- 方式 2：所有字段（需按表结构顺序，不推荐）
INSERT INTO users VALUES (0, '李四', 'lisi@example.com', 30, 1, NOW());
-- 0 或 NULL 表示使用 AUTO_INCREMENT 的值

-- 方式 3：使用 SET 语法
INSERT INTO users SET username='王五', email='wangwu@example.com', age=28;

-- 查看结果
SELECT * FROM users;
```

### 批量插入（推荐）

```sql
-- 批量插入效率高
INSERT INTO users (username, email, age) VALUES
  ('user1', 'user1@example.com', 20),
  ('user2', 'user2@example.com', 22),
  ('user3', 'user3@example.com', 24),
  ('user4', 'user4@example.com', 26),
  ('user5', 'user5@example.com', 28);

-- 批量插入大量数据时性能更好
-- 1 次插入 1000 条 vs 1000 次插入 1 条
-- 前者速度是后者的几十倍
```

### INSERT IGNORE - 忽略冲突

```sql
-- 创建有唯一约束的表
CREATE TABLE users (
  id INT PRIMARY KEY,
  username VARCHAR(50) UNIQUE,
  email VARCHAR(100) UNIQUE
);

-- 正常插入
INSERT INTO users VALUES (1, 'user1', 'user1@example.com');

-- 再次插入会报错（Duplicate key error）
INSERT INTO users VALUES (1, 'user2', 'user2@example.com');
-- ERROR 1062: Duplicate entry '1' for key 'PRIMARY'

-- 使用 IGNORE 忽略错误
INSERT IGNORE INTO users VALUES (1, 'user2', 'user2@example.com');
-- 不报错，但不插入数据

-- 实际应用：去重插入
INSERT IGNORE INTO users (id, username, email)
VALUES (2, 'user2', 'user2@example.com');
```

### ON DUPLICATE KEY UPDATE - 冲突时更新

```sql
-- 如果主键或唯一键冲突，则更新
INSERT INTO users (id, username, email) VALUES
  (1, 'user1_updated', 'user1_new@example.com')
ON DUPLICATE KEY UPDATE
  username = VALUES(username),
  email = VALUES(email);

-- 逻辑：
-- 1. 尝试插入
-- 2. 如果 id=1 已存在，则更新 username 和 email
-- 3. VALUES(username) 指的是要插入的新值

-- 实际应用：统计数据
CREATE TABLE page_views (
  date DATE UNIQUE,
  views INT UNSIGNED DEFAULT 0
);

-- 每天统计访问量
INSERT INTO page_views (date, views) VALUES (CURDATE(), 1)
ON DUPLICATE KEY UPDATE views = views + 1;

-- 第一次执行：插入新记录，views=1
-- 第二次执行：更新记录，views=2
```

### INSERT ... SELECT - 从查询插入

```sql
-- 从查询结果插入
CREATE TABLE users_backup (
  id INT PRIMARY KEY,
  username VARCHAR(50),
  email VARCHAR(100)
);

-- 从 users 表复制数据
INSERT INTO users_backup (id, username, email)
SELECT id, username, email FROM users WHERE age > 25;

-- 从其他表聚合数据
CREATE TABLE user_stats (
  user_id INT PRIMARY KEY,
  total_orders INT,
  total_spent DECIMAL(10, 2)
);

INSERT INTO user_stats (user_id, total_orders, total_spent)
SELECT user_id, COUNT(*), SUM(total_amount)
FROM orders
GROUP BY user_id;
```

### REPLACE - 替换数据

```sql
-- 如果存在则删除旧数据再插入（DELETE + INSERT）
REPLACE INTO users (id, username, email) VALUES (1, 'new_user1', 'new1@example.com');

-- 与 ON DUPLICATE KEY UPDATE 的区别：
-- REPLACE: 先删除再插入，会重新生成 AUTO_INCREMENT
-- ON DUPLICATE KEY UPDATE: 直接更新，保持 AUTO_INCREMENT

-- ⚠️ 注意：REPLACE 会影响外键关联
-- 如果有子表引用，使用 ON DUPLICATE KEY UPDATE 更安全
```

## READ - 查询数据

### 基本语法

```sql
-- 基本结构
SELECT 字段列表 FROM 表名 WHERE 条件;

-- 查询所有字段
SELECT * FROM users;

-- 查询指定字段
SELECT id, username, email FROM users;

-- 使用别名
SELECT
  id AS user_id,
  username AS name,
  email AS '邮箱地址'
FROM users;
```

### WHERE 条件过滤

```sql
-- 比较运算符
SELECT * FROM users WHERE age = 25;
SELECT * FROM users WHERE age > 25;
SELECT * FROM users WHERE age >= 25;
SELECT * FROM users WHERE age != 25;
SELECT * FROM users WHERE age <> 25;  -- 不等于（另一种写法）

-- 逻辑运算符
SELECT * FROM users WHERE age = 25 AND status = 1;
SELECT * FROM users WHERE age = 25 OR age = 30;
SELECT * FROM users WHERE NOT (status = 0);

-- 范围查询
SELECT * FROM users WHERE age BETWEEN 20 AND 30;
SELECT * FROM users WHERE age >= 20 AND age <= 30;  -- 等价写法

-- 集合查询
SELECT * FROM users WHERE age IN (20, 25, 30);
SELECT * FROM users WHERE age NOT IN (20, 25, 30);

-- 空值判断（注意不能用 =）
SELECT * FROM users WHERE email IS NULL;
SELECT * FROM users WHERE email IS NOT NULL;

-- 模糊查询
SELECT * FROM users WHERE username LIKE 'user%';      -- user 开头
SELECT * FROM users WHERE username LIKE '%user%';     -- 包含 user
SELECT * FROM users WHERE username LIKE 'user_';      -- user 后一个字符
SELECT * FROM users WHERE username LIKE 'user\%';     -- 转义，查找 user%
```

### ORDER BY 排序

```sql
-- 单字段排序
SELECT * FROM users ORDER BY age ASC;           -- 升序（默认）
SELECT * FROM users ORDER BY age DESC;          -- 降序

-- 多字段排序
SELECT * FROM users ORDER BY status DESC, age ASC;
-- 先按 status 降序，相同则按 age 升序

-- 使用字段位置排序（不推荐）
SELECT id, username, age FROM users ORDER BY 3 DESC;
-- 3 表示第 3 个字段（age）

-- NULL 值排序
SELECT * FROM users ORDER BY email ASC;         -- NULL 排在最前
SELECT * FROM users ORDER BY email IS NULL, email;  -- NULL 排在最后
```

### LIMIT 分页

```sql
-- 限制返回行数
SELECT * FROM users LIMIT 5;                    -- 前 5 条
SELECT * FROM users LIMIT 10;                   -- 前 10 条

-- 分页查询（偏移量 + 数量）
SELECT * FROM users LIMIT 0, 5;                 -- 第 1 页（跳过 0 条，取 5 条）
SELECT * FROM users LIMIT 5, 5;                 -- 第 2 页（跳过 5 条，取 5 条）
SELECT * FROM users LIMIT 10, 5;                -- 第 3 页（跳过 10 条，取 5 条）

-- 标准语法（推荐）
SELECT * FROM users LIMIT 5 OFFSET 0;           -- 第 1 页
SELECT * FROM users LIMIT 5 OFFSET 5;           -- 第 2 页

-- 分页公式
-- page: 页码（从 1 开始）
-- pageSize: 每页数量
-- LIMIT pageSize OFFSET (page - 1) * pageSize
```

### DISTINCT 去重

```sql
-- 去除重复行
SELECT DISTINCT status FROM users;              -- 所有不同的状态值
SELECT DISTINCT age FROM users WHERE age > 20;  -- 所有不同的年龄

-- 多字段去重（组合唯一）
SELECT DISTINCT age, status FROM users;

-- 聚合去重
SELECT COUNT(DISTINCT age) FROM users;          -- 有多少个不同的年龄
SELECT COUNT(DISTINCT email) FROM users;        -- 有多少个不同的邮箱
```

## UPDATE - 更新数据

### 基本语法

```sql
-- 更新单个字段
UPDATE users SET age = 26 WHERE id = 1;

-- 更新多个字段
UPDATE users
SET age = 26, status = 0
WHERE id = 1;

-- ⚠️ 注意：一定要有 WHERE 条件！
-- 否则会更新所有行
UPDATE users SET status = 0;  -- 危险！更新所有记录
```

### 条件更新

```sql
-- 基于条件更新
UPDATE users SET status = 0 WHERE age < 18;
UPDATE users SET age = age + 1 WHERE id = 1;    -- 表达式

-- 基于查询更新
UPDATE users
SET status = 0
WHERE id IN (SELECT user_id FROM orders WHERE total_amount > 10000);

-- 使用 CASE 条件更新
UPDATE users
SET status = CASE
  WHEN age < 18 THEN 0
  WHEN age BETWEEN 18 AND 60 THEN 1
  ELSE 2
END;
```

### UPDATE + JOIN

```sql
-- 基于 JOIN 更新
UPDATE users u
JOIN orders o ON u.id = o.user_id
SET u.status = 1
WHERE o.total_amount > 1000;

-- 关联更新
UPDATE users u
LEFT JOIN user_profiles p ON u.id = p.user_id
SET u.nickname = p.nickname
WHERE u.nickname IS NULL AND p.nickname IS NOT NULL;
```

### 批量更新技巧

```sql
-- 方式 1：CASE WHEN（适用于少量数据）
UPDATE users
SET age = CASE id
  WHEN 1 THEN 25
  WHEN 2 THEN 30
  WHEN 3 THEN 35
  ELSE age
END
WHERE id IN (1, 2, 3);

-- 方式 2：临时表 JOIN（适用于大量数据）
CREATE TEMPORARY TABLE temp_updates (id INT, new_age INT);
INSERT INTO temp_updates VALUES (1, 25), (2, 30), (3, 35);

UPDATE users u
JOIN temp_updates t ON u.id = t.id
SET u.age = t.new_age;

DROP TEMPORARY TABLE temp_updates;
```

## DELETE - 删除数据

### 基本语法

```sql
-- 删除指定行
DELETE FROM users WHERE id = 1;

-- 基于条件删除
DELETE FROM users WHERE age < 18;
DELETE FROM users WHERE status = 0 AND created_at < '2023-01-01';

-- ⚠️ 注意：一定要有 WHERE 条件！
-- 否则会删除所有数据
DELETE FROM users;  -- 危险！删除所有记录
```

### TRUNCATE - 清空表

```sql
-- DELETE：逐行删除，可以回滚，不重置自增ID
DELETE FROM users;

-- TRUNCATE：快速清空，不可回滚，重置自增ID
TRUNCATE TABLE users;

-- 对比
DELETE FROM users;      -- 慢，可回滚，AUTO_INCREMENT 不重置
TRUNCATE TABLE users;   -- 快，不可回滚，AUTO_INCREMENT 重置为 1

-- 使用建议：
-- - 删除部分数据 → DELETE
-- - 清空整个表 → TRUNCATE
-- - 需要回滚 → DELETE（在事务中）
```

### 条件删除

```sql
-- 基于查询删除
DELETE FROM users
WHERE id IN (SELECT user_id FROM banned_users);

-- 基于时间删除
DELETE FROM logs WHERE created_at < DATE_SUB(NOW(), INTERVAL 30 DAY);

-- 删除重复数据（保留一条）
DELETE t1 FROM users t1
INNER JOIN users t2
WHERE t1.id > t2.id AND t1.email = t2.email;
```

### DELETE + JOIN

```sql
-- 基于 JOIN 删除
DELETE users
FROM users
INNER JOIN user_profiles p ON users.id = p.user_id
WHERE p.status = 0;

-- 删除关联数据
DELETE users, user_profiles
FROM users
INNER JOIN user_profiles ON users.id = user_profiles.user_id
WHERE users.status = 0;
```

## 综合实战

### 用户管理系统

```sql
-- 创建用户表
CREATE TABLE users (
  id BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  username VARCHAR(50) NOT NULL UNIQUE,
  email VARCHAR(100) NOT NULL UNIQUE,
  password_hash VARCHAR(255) NOT NULL,
  nickname VARCHAR(50),
  avatar VARCHAR(500),
  gender ENUM('male', 'female', 'other') DEFAULT 'other',
  phone VARCHAR(20),
  status TINYINT UNSIGNED DEFAULT 1 COMMENT '1:正常 0:禁用',
  last_login_at TIMESTAMP NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户表';

-- 1. 用户注册（INSERT）
INSERT INTO users (username, email, password_hash, nickname) VALUES
  ('zhangsan', 'zhangsan@example.com', 'hash123...', '张三');

-- 2. 查询用户信息（SELECT）
SELECT
  id,
  username,
  nickname,
  email,
  gender,
  status,
  created_at
FROM users
WHERE username = 'zhangsan';

-- 3. 用户登录（SELECT + 更新登录时间）
UPDATE users
SET last_login_at = CURRENT_TIMESTAMP
WHERE username = 'zhangsan' AND password_hash = 'hash123...';

-- 4. 更新用户资料（UPDATE）
UPDATE users
SET nickname = '张三同学',
    gender = 'male',
    phone = '13800138000'
WHERE id = 1;

-- 5. 禁用用户（UPDATE）
UPDATE users SET status = 0 WHERE id = 1;

-- 6. 删除用户（DELETE）
DELETE FROM users WHERE id = 1;

-- 7. 批量导入用户（INSERT 批量）
INSERT INTO users (username, email, password_hash, nickname) VALUES
  ('user1', 'user1@example.com', 'hash1', '用户1'),
  ('user2', 'user2@example.com', 'hash2', '用户2'),
  ('user3', 'user3@example.com', 'hash3', '用户3');

-- 8. 分页查询用户（SELECT + LIMIT）
SELECT
  id, username, nickname, email, status, created_at
FROM users
WHERE status = 1
ORDER BY created_at DESC
LIMIT 10 OFFSET 0;  -- 第 1 页

-- 9. 搜索用户（SELECT + LIKE）
SELECT
  id, username, nickname, email
FROM users
WHERE username LIKE '%zhang%' OR nickname LIKE '%zhang%';

-- 10. 统计用户（SELECT + 聚合）
SELECT
  COUNT(*) AS total_users,
  COUNT(CASE WHEN status = 1 THEN 1 END) AS active_users,
  COUNT(CASE WHEN status = 0 THEN 1 END) AS banned_users
FROM users;
```

### 商品管理系统

```sql
-- 创建商品表
CREATE TABLE products (
  id BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  category_id BIGINT UNSIGNED NOT NULL,
  name VARCHAR(200) NOT NULL,
  description TEXT,
  price DECIMAL(10, 2) NOT NULL,
  stock INT UNSIGNED DEFAULT 0,
  sales INT UNSIGNED DEFAULT 0,
  status TINYINT UNSIGNED DEFAULT 1 COMMENT '1:上架 0:下架',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='商品表';

-- 1. 添加商品（INSERT）
INSERT INTO products (category_id, name, description, price, stock) VALUES
  (1, 'iPhone 15 Pro', '苹果最新旗舰手机', 7999.00, 100),
  (1, 'MacBook Pro', '苹果笔记本电脑', 15999.00, 50);

-- 2. 查询商品列表（SELECT + WHERE + ORDER BY）
SELECT
  id, name, price, stock, sales, status
FROM products
WHERE status = 1 AND stock > 0
ORDER BY sales DESC
LIMIT 20;

-- 3. 商品详情（SELECT）
SELECT * FROM products WHERE id = 1;

-- 4. 更新商品信息（UPDATE）
UPDATE products
SET description = '最新款 iPhone，A17 芯片',
    price = 7799.00
WHERE id = 1;

-- 5. 下架商品（UPDATE）
UPDATE products SET status = 0 WHERE id = 1;

-- 6. 扣减库存（UPDATE + 表达式）
UPDATE products
SET stock = stock - 1,
    sales = sales + 1
WHERE id = 1 AND stock >= 1;

-- 7. 批量更新价格（UPDATE + CASE）
UPDATE products
SET price = CASE id
  WHEN 1 THEN 7799.00
  WHEN 2 THEN 15499.00
  ELSE price
END
WHERE id IN (1, 2);

-- 8. 删除商品（DELETE）
DELETE FROM products WHERE id = 1;

-- 9. 搜索商品（SELECT + LIKE）
SELECT
  id, name, price, stock
FROM products
WHERE name LIKE '%iPhone%' OR description LIKE '%iPhone%';

-- 10. 价格区间查询（SELECT + BETWEEN）
SELECT
  id, name, price
FROM products
WHERE price BETWEEN 5000 AND 10000
ORDER BY price ASC;
```

## 最佳实践

### 1. 使用事务保护数据

```sql
-- ✅ 推荐：关键操作使用事务
START TRANSACTION;

UPDATE accounts SET balance = balance - 1000 WHERE id = 1;
UPDATE accounts SET balance = balance + 1000 WHERE id = 2;

COMMIT;
-- 或 ROLLBACK; 回滚
```

### 2. 批量操作优化性能

```sql
-- ✅ 推荐：批量插入
INSERT INTO users (username, email) VALUES
  ('user1', 'user1@example.com'),
  ('user2', 'user2@example.com'),
  ('user3', 'user3@example.com');

-- ❌ 不推荐：循环单条插入
INSERT INTO users (username, email) VALUES ('user1', 'user1@example.com');
INSERT INTO users (username, email) VALUES ('user2', 'user2@example.com');
INSERT INTO users (username, email) VALUES ('user3', 'user3@example.com');
```

### 3. 避免 SELECT \*

```sql
-- ✅ 推荐：只查询需要的字段
SELECT id, username, email FROM users;

-- ❌ 不推荐：查询所有字段
SELECT * FROM users;
-- 缺点：
-- 1. 可能读取不需要的列，浪费 I/O
-- 2. 增加网络传输
-- 3. 增加内存占用
-- 4. 无法使用覆盖索引优化
```

### 4. UPDATE 和 DELETE 时必须加 WHERE

```sql
-- ✅ 推荐：先测试 WHERE 条件
SELECT * FROM users WHERE age < 18;
-- 确认无误后再更新
UPDATE users SET status = 0 WHERE age < 18;

-- ❌ 危险：没有 WHERE 条件
UPDATE users SET status = 0;  -- 更新所有记录！
```

### 5. 使用 LIMIT 限制影响行数

```sql
-- ✅ 推荐：不确定条件时加上 LIMIT
UPDATE users SET status = 0 WHERE age < 18 LIMIT 100;
DELETE FROM logs WHERE created_at < '2023-01-01' LIMIT 1000;

-- 分批执行，避免锁表太久
```

### 6. 使用 EXPLAIN 分析查询

```sql
-- 执行前分析查询计划
EXPLAIN SELECT * FROM users WHERE email = 'user@example.com';

-- 查看是否使用了索引
-- type: ALL（全表扫描）、index（索引扫描）、ref（索引查找）
-- key: 实际使用的索引
-- rows: 预计扫描的行数
```

## 常见问题

### Q1: INSERT 数据时出现重复键错误怎么办？

**A:** 使用 `INSERT IGNORE` 或 `ON DUPLICATE KEY UPDATE`。

```sql
-- 方式 1：忽略重复
INSERT IGNORE INTO users (id, username) VALUES (1, 'user1');

-- 方式 2：重复时更新
INSERT INTO users (id, username) VALUES (1, 'user1')
ON DUPLICATE KEY UPDATE username = VALUES(username);
```

### Q2: 如何高效更新大量数据？

**A:** 分批更新，避免锁表太久。

```sql
-- ✅ 推荐：分批更新
UPDATE users SET status = 0 WHERE age < 18 LIMIT 1000;
-- 重复执行，直到影响行数为 0

-- ✅ 或使用临时表
CREATE TEMPORARY TABLE temp_users AS SELECT id FROM users WHERE age < 18;
UPDATE users u JOIN temp_users t ON u.id = t.id SET u.status = 0;
```

### Q3: DELETE 后自增ID不连续怎么办？

**A:** 这是正常现象，不需要处理。

```sql
-- ID 不连续不影响使用
-- 如果必须重置，可以使用：
TRUNCATE TABLE users;  -- 清空并重置

-- 或修改自增值
ALTER TABLE users AUTO_INCREMENT = 1;
```

### Q4: 如何查询第 N 条到第 M 条数据？

**A:** 使用 LIMIT OFFSET。

```sql
-- 查询第 11-20 条（第 2 页）
SELECT * FROM users LIMIT 10 OFFSET 10;

-- 分页公式
-- pageSize = 10, page = 2
SELECT * FROM users LIMIT 10 OFFSET (2 - 1) * 10;
```

### Q5: UPDATE 时如何处理字段为 NULL？

**A:** 需要使用 IS NULL 或 IFNULL。

```sql
-- ✅ 正确方式
UPDATE users SET nickname = NULL WHERE id = 1;
UPDATE users SET nickname = '默认' WHERE nickname IS NULL;

-- ❌ 错误方式
UPDATE users SET nickname = '默认' WHERE nickname = NULL;  -- 错误！
```
