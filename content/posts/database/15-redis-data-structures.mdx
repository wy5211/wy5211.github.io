## Redis 数据类型

### 1. String（字符串）

```typescript
// 缓存用户信息
await redis.set("user:123", JSON.stringify({ id: 123, name: "张三" }));
await redis.expire("user:123", 3600); // 1小时过期

const user = JSON.parse(await redis.get("user:123"));

// 计数器

```

## 缓存策略

### 1. Cache Aside（旁路缓存）

```typescript
// 读取
async function getUser(userId: string) {
  // 1. 先查缓存
  let user = await redis.get(`user:${userId}`);
  if (user) return JSON.parse(user);

  // 2. 缓存未命中，查数据库
  user = await db.findUser(userId);

  // 3. 写入缓存
  await redis.set(`user:${userId}`, JSON.stringify(user), { EX: 3600 });

  return user;
}

// 更新
async function updateUser(userId: string, data: any) {
  // 1. 更新数据库
  await db.updateUser(userId, data);

  // 2. 删除缓存
  await redis.del(`user:${userId}`);
}
```

---

### 2. 缓存穿透防护

```typescript
// 布隆过滤器
async function getProduct(productId: string) {
  const key = `product:${productId}`;

  // 检查布隆过滤器
  const exists = await redis.exists(`filter:${key}`);
  if (!exists) return null;

  // 查询缓存
  let product = await redis.get(key);
  if (product) return JSON.parse(product);

  // 查询数据库
  product = await db.findProduct(productId);
  if (product) {
    await redis.set(key, JSON.stringify(product), { EX: 3600 });
    await redis.set(`filter:${key}`, "1", { EX: 86400 });
  }

  return product;
}
```

---

### 3. 缓存击穿防护

```typescript
// 互斥锁
async function getProduct(productId: string) {
  const key = `product:${productId}`;
  let product = await redis.get(key);

  if (product) return JSON.parse(product);

  const lockKey = `lock:${key}`;
  const lock = await redis.set(lockKey, "1", { NX: true, EX: 10 });

  if (lock) {
    try {
      const data = await db.findProduct(productId);
      await redis.set(key, JSON.stringify(data), { EX: 3600 });
      return data;
    } finally {
      await redis.del(lockKey);
    }
  } else {
    await new Promise((resolve) => setTimeout(resolve, 100));
    return await getProduct(productId);
  }
}
```

## 分布式锁

```typescript
// 获取锁
const lockKey = "lock:resource:123";
const lockValue = Date.now().toString();
const lock = await redis.set(lockKey, lockValue, { NX: true, EX: 30 });

if (lock) {
  try {
    // 执行业务逻辑
  } finally {
    // Lua 脚本释放锁
    const script = `
      if redis.call("get", KEYS[1]) == ARGV[1] then
        return redis.call("del", KEYS[1])
      else
        return 0
      end
    `;
    await redis.eval(script, { keys: [lockKey], arguments: [lockValue] });
  }
}
```
