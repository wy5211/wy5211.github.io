## 问题引入

在前几篇文章中，我们构建了一个内存中的任务队列系统。但问题来了：

**当程序重启时，所有任务都会丢失**。这在生产环境中是不可接受的。

我们需要将任务持久化到数据库中，这样即使服务重启，任务也不会丢失。同时，我们还需要记录任务的执行历史、失败次数等信息。

Go 提供了两种主要的数据库操作方式：

1. **database/sql**：标准库，提供底层 SQL 接口
2. **GORM**：第三方 ORM，提供更高级的抽象

让我们在任务队列项目中实现数据持久化。

---

## 使用 database/sql

### 为什么选择 database/sql

database/sql 是 Go 的标准库，提供了：

- 统一的数据库接口
- 连接池管理
- 预处理语句防止 SQL 注入
- 事务支持

### 连接数据库

首先，让我们为任务队列创建数据库连接和表结构：

```go
package main

import (
	"database/sql"
	"fmt"
	"log"
	"time"

	_ "github.com/go-sql-driver/mysql" // 导入 MySQL 驱动
)

// Task 表示一个任务
type Task struct {
	ID          int
	Name        string
	Status      string // pending, processing, completed, failed
	Priority    int
	MaxRetries  int
	RetryCount  int
	CreatedAt   time.Time
	UpdatedAt   time.Time
}

// DBConfig 数据库配置
type DBConfig struct {
	Host     string
	Port     int
	User     string
	Password string
	Database string
}

// NewDB 创建数据库连接池
func NewDB(cfg DBConfig) (*sql.DB, error) {
	// 构建数据源名称（DSN）
	// 格式：user:password@tcp(host:port)/database?params
	dsn := fmt.Sprintf("%s:%s@tcp(%s:%d)/%s?charset=utf8mb4&parseTime=True&loc=Local",
		cfg.User,
		cfg.Password,
		cfg.Host,
		cfg.Port,
		cfg.Database,
	)

	// 打开数据库连接
	// 第一个参数是驱动名称，第二个参数是 DSN
	db, err := sql.Open("mysql", dsn)
	if err != nil {
		return nil, fmt.Errorf("打开数据库连接失败: %w", err)
	}

	// 测试连接是否可用
	if err := db.Ping(); err != nil {
		return nil, fmt.Errorf("数据库连接测试失败: %w", err)
	}

	// 配置连接池
	// SetMaxOpenConns 设置数据库的最大打开连接数
	// 过高会导致数据库压力过大，过低会导致并发性能不足
	db.SetMaxOpenConns(25)

	// SetMaxIdleConns 设置空闲连接池中的最大连接数
	// 这些连接可以在没有请求时保持打开，减少建立新连接的开销
	db.SetMaxIdleConns(5)

	// SetConnMaxLifetime 设置连接可复用的最长时间
	// 超过这个时间的连接会被关闭，避免长时间连接导致的问题
	db.SetConnMaxLifetime(5 * time.Minute)

	log.Println("数据库连接成功")
	return db, nil
}

// InitSchema 初始化数据库表
func InitSchema(db *sql.DB) error {
	// 创建任务表
	// 使用 IF NOT EXISTS 避免重复创建
	schema := `
	CREATE TABLE IF NOT EXISTS tasks (
		id INT AUTO_INCREMENT PRIMARY KEY,
		name VARCHAR(255) NOT NULL,
		status VARCHAR(50) NOT NULL DEFAULT 'pending',
		priority INT NOT NULL DEFAULT 0,
		max_retries INT NOT NULL DEFAULT 3,
		retry_count INT NOT NULL DEFAULT 0,
		created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
		updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
		INDEX idx_status (status),
		INDEX idx_priority (priority)
	) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
	`

	// 执行 SQL 语句
	if _, err := db.Exec(schema); err != nil {
		return fmt.Errorf("创建表失败: %w", err)
	}

	log.Println("数据库表初始化成功")
	return nil
}
```

### 基础 CRUD 操作

实现任务的增删改查操作：

```go
// TaskRepository 任务数据访问层
type TaskRepository struct {
	db *sql.DB
}

// NewTaskRepository 创建任务仓库
func NewTaskRepository(db *sql.DB) *TaskRepository {
	return &TaskRepository{db: db}
}

// Create 创建新任务
func (r *TaskRepository) Create(task *Task) error {
	// 使用 ? 作为占位符，防止 SQL 注入
	// MySQL 驱动会自动替换参数
	query := `
		INSERT INTO tasks (name, status, priority, max_retries)
		VALUES (?, ?, ?, ?)
	`

	// Exec 执行 SQL 语句并返回结果
	result, err := r.db.Exec(
		query,
		task.Name,
		task.Status,
		task.Priority,
		task.MaxRetries,
	)
	if err != nil {
		return fmt.Errorf("创建任务失败: %w", err)
	}

	// 获取插入的行 ID
	id, err := result.LastInsertId()
	if err != nil {
		return fmt.Errorf("获取插入 ID 失败: %w", err)
	}

	task.ID = int(id)
	return nil
}

// GetByID 根据 ID 获取任务
func (r *TaskRepository) GetByID(id int) (*Task, error) {
	// QueryRow 查询单行数据
	// Scan 将结果扫描到结构体字段中
	task := &Task{}
	err := r.db.QueryRow(`
		SELECT id, name, status, priority, max_retries, retry_count, created_at, updated_at
		FROM tasks WHERE id = ?
	`, id).Scan(
		&task.ID,
		&task.Name,
		&task.Status,
		&task.Priority,
		&task.MaxRetries,
		&task.RetryCount,
		&task.CreatedAt,
		&task.UpdatedAt,
	)

	if err == sql.ErrNoRows {
		// 没有找到记录
		return nil, fmt.Errorf("任务不存在")
	}
	if err != nil {
		return nil, fmt.Errorf("查询任务失败: %w", err)
	}

	return task, nil
}

// GetTasksByStatus 根据状态获取任务列表
func (r *TaskRepository) GetTasksByStatus(status string, limit int) ([]*Task, error) {
	// Query 查询多行数据
	rows, err := r.db.Query(`
		SELECT id, name, status, priority, max_retries, retry_count, created_at, updated_at
		FROM tasks
		WHERE status = ?
		ORDER BY priority DESC, created_at ASC
		LIMIT ?
	`, status, limit)
	if err != nil {
		return nil, fmt.Errorf("查询任务列表失败: %w", err)
	}
	defer rows.Close() // 确保关闭结果集

	var tasks []*Task
	// Next 遍历每一行数据
	for rows.Next() {
		task := &Task{}
		err := rows.Scan(
			&task.ID,
			&task.Name,
			&task.Status,
			&task.Priority,
			&task.MaxRetries,
			&task.RetryCount,
			&task.CreatedAt,
			&task.UpdatedAt,
		)
		if err != nil {
			return nil, fmt.Errorf("扫描数据失败: %w", err)
		}
		tasks = append(tasks, task)
	}

	// 检查是否有遍历错误
	if err = rows.Err(); err != nil {
		return nil, fmt.Errorf("遍历数据失败: %w", err)
	}

	return tasks, nil
}

// UpdateStatus 更新任务状态
func (r *TaskRepository) UpdateStatus(id int, status string) error {
	// Exec 执行更新语句
	result, err := r.db.Exec(`
		UPDATE tasks
		SET status = ?, updated_at = CURRENT_TIMESTAMP
		WHERE id = ?
	`, status, id)
	if err != nil {
		return fmt.Errorf("更新任务状态失败: %w", err)
	}

	// RowsAffected 返回受影响的行数
	rows, err := result.RowsAffected()
	if err != nil {
		return fmt.Errorf("获取影响行数失败: %w", err)
	}

	if rows == 0 {
		return fmt.Errorf("任务不存在")
	}

	return nil
}

// IncrementRetryCount 增加重试次数
func (r *TaskRepository) IncrementRetryCount(id int) error {
	result, err := r.db.Exec(`
		UPDATE tasks
		SET retry_count = retry_count + 1, updated_at = CURRENT_TIMESTAMP
		WHERE id = ?
	`, id)
	if err != nil {
		return fmt.Errorf("增加重试次数失败: %w", err)
	}

	rows, err := result.RowsAffected()
	if err != nil {
		return fmt.Errorf("获取影响行数失败: %w", err)
	}

	if rows == 0 {
		return fmt.Errorf("任务不存在")
	}

	return nil
}

// Delete 删除任务
func (r *TaskRepository) Delete(id int) error {
	result, err := r.db.Exec(`
		DELETE FROM tasks WHERE id = ?
	`, id)
	if err != nil {
		return fmt.Errorf("删除任务失败: %w", err)
	}

	rows, err := result.RowsAffected()
	if err != nil {
		return fmt.Errorf("获取影响行数失败: %w", err)
	}

	if rows == 0 {
		return fmt.Errorf("任务不存在")
	}

	return nil
}
```

### 事务处理

在任务队列中，我们经常需要执行多个操作，要么全部成功，要么全部失败。例如，从队列中取出一个任务时，需要：

1. 查找最高优先级的待处理任务
2. 更新任务状态为"处理中"
3. 增加重试次数

这三个步骤必须作为一个原子操作执行：

```go
// ClaimTask 领取任务（使用事务）
func (r *TaskRepository) ClaimTask(workerID string) (*Task, error) {
	// Begin 开始一个新事务
	tx, err := r.db.Begin()
	if err != nil {
		return nil, fmt.Errorf("开始事务失败: %w", err)
	}

	// 使用 defer 确保在出现 panic 时回滚事务
	defer func() {
		if p := recover(); p != nil {
			tx.Rollback()
			panic(p)
		}
	}()

	// 查询最高优先级的待处理任务
	// FOR UPDATE 锁定该行，防止其他 worker 同时领取
	task := &Task{}
	err = tx.QueryRow(`
		SELECT id, name, status, priority, max_retries, retry_count, created_at, updated_at
		FROM tasks
		WHERE status = 'pending' AND retry_count < max_retries
		ORDER BY priority DESC, created_at ASC
		LIMIT 1
		FOR UPDATE
	`).Scan(
		&task.ID,
		&task.Name,
		&task.Status,
		&task.Priority,
		&task.MaxRetries,
		&task.RetryCount,
		&task.CreatedAt,
		&task.UpdatedAt,
	)

	if err == sql.ErrNoRows {
		tx.Rollback() // 没有任务，回滚事务
		return nil, nil // 返回 nil 表示没有可用任务
	}
	if err != nil {
		tx.Rollback()
		return nil, fmt.Errorf("查询任务失败: %w", err)
	}

	// 更新任务状态
	_, err = tx.Exec(`
		UPDATE tasks
		SET status = 'processing', updated_at = CURRENT_TIMESTAMP
		WHERE id = ?
	`, task.ID)
	if err != nil {
		tx.Rollback()
		return nil, fmt.Errorf("更新任务状态失败: %w", err)
	}

	// 增加重试次数
	_, err = tx.Exec(`
		UPDATE tasks
		SET retry_count = retry_count + 1, updated_at = CURRENT_TIMESTAMP
		WHERE id = ?
	`, task.ID)
	if err != nil {
		tx.Rollback()
		return nil, fmt.Errorf("增加重试次数失败: %w", err)
	}

	// Commit 提交事务
	if err := tx.Commit(); err != nil {
		return nil, fmt.Errorf("提交事务失败: %w", err)
	}

	return task, nil
}
```

## 使用 GORM

### 为什么选择 GORM

虽然 database/sql 很强大，但需要编写大量 SQL 代码。GORM 提供了：

- 自动迁移表结构
- 更简洁的 API
- 关联关系处理
- 钩子函数
- 更好的可读性

### 安装 GORM

```bash
go get -u gorm.io/gorm
go get -u gorm.io/driver/mysql
```

### 定义模型

```go
package models

import (
	"time"
	"gorm.io/gorm"
)

// Task 任务模型
type Task struct {
	ID          uint           `gorm:"primaryKey" json:"id"`
	Name        string         `gorm:"size:255;not null" json:"name"`
	Status      string         `gorm:"size:50;not null;default:'pending';index" json:"status"`
	Priority    int            `gorm:"not null;default:0;index" json:"priority"`
	MaxRetries  int            `gorm:"not null;default:3" json:"max_retries"`
	RetryCount  int            `gorm:"not null;default:0" json:"retry_count"`
	Result      string         `gorm:"type:text" json:"result"`
	Error       string         `gorm:"type:text" json:"error"`
	CreatedAt   time.Time      `json:"created_at"`
	UpdatedAt   time.Time      `json:"updated_at"`
	DeletedAt   gorm.DeletedAt `gorm:"index" json:"-"`
}

// TableName 指定表名
func (Task) TableName() string {
	return "tasks"
}
```

### GORM 操作

```go
package repository

import (
	"fmt"
	"task-queue/models"
	"gorm.io/gorm"
)

// GormTaskRepository GORM 任务仓库
type GormTaskRepository struct {
	db *gorm.DB
}

// NewGormTaskRepository 创建 GORM 任务仓库
func NewGormTaskRepository(db *gorm.DB) *GormTaskRepository {
	return &GormTaskRepository{db: db}
}

// Create 创建任务
func (r *GormTaskRepository) Create(task *models.Task) error {
	// Create 自动生成 INSERT 语句
	// GORM 会根据结构体字段自动映射到数据库列
	if err := r.db.Create(task).Error; err != nil {
		return fmt.Errorf("创建任务失败: %w", err)
	}
	return nil
}

// GetByID 根据 ID 获取任务
func (r *GormTaskRepository) GetByID(id uint) (*models.Task, error) {
	var task models.Task
	// First 查询第一条记录
	// 如果使用主键查询，GORM 会自动使用 WHERE id = ?
	err := r.db.First(&task, id).Error
	if err == gorm.ErrRecordNotFound {
		return nil, fmt.Errorf("任务不存在")
	}
	if err != nil {
		return nil, fmt.Errorf("查询任务失败: %w", err)
	}
	return &task, nil
}

// GetTasksByStatus 根据状态获取任务列表
func (r *GormTaskRepository) GetTasksByStatus(status string, limit int) ([]*models.Task, error) {
	var tasks []*models.Task
	// Where 添加查询条件
	// Order 指定排序
	// Limit 限制返回数量
	err := r.db.Where("status = ?", status).
		Order("priority DESC, created_at ASC").
		Limit(limit).
		Find(&tasks).Error
	if err != nil {
		return nil, fmt.Errorf("查询任务列表失败: %w", err)
	}
	return tasks, nil
}

// UpdateStatus 更新任务状态
func (r *GormTaskRepository) UpdateStatus(id uint, status string) error {
	// Updates 更新指定字段
	// GORM 会自动更新 UpdatedAt 字段
	result := r.db.Model(&models.Task{}).
		Where("id = ?", id).
		Updates(map[string]interface{}{
			"status": status,
		})

	if result.Error != nil {
		return fmt.Errorf("更新任务状态失败: %w", result.Error)
	}

	if result.RowsAffected == 0 {
		return fmt.Errorf("任务不存在")
	}

	return nil
}

// ClaimTask 领取任务（GORM 事务）
func (r *GormTaskRepository) ClaimTask() (*models.Task, error) {
	var task models.Task

	// Transaction 自动管理事务
	// 如果回调函数返回 error，会自动回滚
	err := r.db.Transaction(func(tx *gorm.DB) error {
		// LockRows 锁定查询的行（相当于 SELECT ... FOR UPDATE）
		err := tx.Where("status = ? AND retry_count < max_retries", "pending").
			Order("priority DESC, created_at ASC").
			First(&task).
			Error

		if err == gorm.ErrRecordNotFound {
			// 没有可用任务
			return nil
		}
		if err != nil {
			return fmt.Errorf("查询任务失败: %w", err)
		}

		// 更新任务状态
		if err := tx.Model(&task).Updates(map[string]interface{}{
			"status":     "processing",
			"retry_count": task.RetryCount + 1,
		}).Error; err != nil {
			return fmt.Errorf("更新任务失败: %w", err)
		}

		return nil
	})

	if err != nil {
		return nil, err
	}

	if task.ID == 0 {
		return nil, nil // 没有可用任务
	}

	return &task, nil
}

// CompleteTask 完成任务
func (r *GormTaskRepository) CompleteTask(id uint, result string) error {
	return r.db.Model(&models.Task{}).
		Where("id = ?", id).
		Updates(map[string]interface{}{
			"status":  "completed",
			"result":  result,
		}).Error
}

// FailTask 标记任务失败
func (r *GormTaskRepository) FailTask(id uint, errMsg string) error {
	// 先获取任务
	task, err := r.GetByID(id)
	if err != nil {
		return err
	}

	// 检查是否超过最大重试次数
	updates := map[string]interface{}{
		"error": errMsg,
	}

	if task.RetryCount >= task.MaxRetries {
		updates["status"] = "failed"
	} else {
		updates["status"] = "pending"
	}

	return r.db.Model(&models.Task{}).
		Where("id = ?", id).
		Updates(updates).Error
}

// GetStats 获取任务统计信息
func (r *GormTaskRepository) GetStats() (map[string]int64, error) {
	var stats []struct {
		Status string
		Count  int64
	}

	// 查询各状态的任务数量
	err := r.db.Model(&models.Task{}).
		Select("status, COUNT(*) as count").
		Group("status").
		Scan(&stats).Error
	if err != nil {
		return nil, fmt.Errorf("查询统计信息失败: %w", err)
	}

	result := make(map[string]int64)
	for _, stat := range stats {
		result[stat.Status] = stat.Count
	}

	return result, nil
}
```

```go
	}
	return nil
```

## 完整示例：集成到任务队列

将数据库操作集成到我们的任务队列中：

```go
package main

import (
	"context"
	"fmt"
	"log"
	"os"
	"os/signal"
	"sync"
	"syscall"
	"time"
)

// TaskQueueWithDB 带数据库的任务队列
type TaskQueueWithDB struct {
	repo   *GormTaskRepository
	workers []*Worker
	ctx     context.Context
	cancel  context.CancelFunc
	wg      sync.WaitGroup
}

// NewTaskQueueWithDB 创建任务队列
func NewTaskQueueWithDB(repo *GormTaskRepository, workerCount int) *TaskQueueWithDB {
	ctx, cancel := context.WithCancel(context.Background())

	q := &TaskQueueWithDB{
		repo:   repo,
		ctx:    ctx,
		cancel: cancel,
	}

	// 创建 workers
	for i := 0; i < workerCount; i++ {
		worker := NewWorker(i, q)
		q.workers = append(q.workers, worker)
	}

	return q
}

// Start 启动任务队列
func (q *TaskQueueWithDB) Start() {
	log.Printf("启动任务队列，worker 数量: %d", len(q.workers))

	for _, worker := range q.workers {
		q.wg.Add(1)
		go worker.Run(&q.wg)
	}
}

// Stop 停止任务队列
func (q *TaskQueueWithDB) Stop() {
	log.Println("正在停止任务队列...")
	q.cancel()
	q.wg.Wait()
	log.Println("任务队列已停止")
}

// Worker worker
type Worker struct {
	id   int
	queue *TaskQueueWithDB
}

// NewWorker 创建 worker
func NewWorker(id int, queue *TaskQueueWithDB) *Worker {
	return &Worker{
		id:    id,
		queue: queue,
	}
}

// Run 运行 worker
func (w *Worker) Run(wg *sync.WaitGroup) {
	defer wg.Done()

	log.Printf("Worker %d 启动", w.id)

	for {
		select {
		case <-w.queue.ctx.Done():
			log.Printf("Worker %d 退出", w.id)
			return
		default:
			// 从数据库领取任务
			task, err := w.queue.repo.ClaimTask()
			if err != nil {
				log.Printf("Worker %d 领取任务失败: %v", w.id, err)
				time.Sleep(1 * time.Second)
				continue
			}

			if task == nil {
				// 没有可用任务
				time.Sleep(1 * time.Second)
				continue
			}

			log.Printf("Worker %d 开始处理任务 %d: %s", w.id, task.ID, task.Name)

			// 处理任务
			result, err := w.processTask(task)
			if err != nil {
				log.Printf("Worker %d 处理任务 %d 失败: %v", w.id, task.ID, err)
				// 标记任务失败
				if failErr := w.queue.repo.FailTask(task.ID, err.Error()); failErr != nil {
					log.Printf("标记任务失败出错: %v", failErr)
				}
			} else {
				log.Printf("Worker %d 完成任务 %d", w.id, task.ID)
				// 标记任务完成
				if completeErr := w.queue.repo.CompleteTask(task.ID, result); completeErr != nil {
					log.Printf("标记任务完成出错: %v", completeErr)
				}
			}
		}
	}
}

// processTask 处理任务
func (w *Worker) processTask(task *models.Task) (string, error) {
	// 模拟任务处理
	time.Sleep(time.Duration(task.Priority) * 100 * time.Millisecond)

	return fmt.Sprintf("任务 %s 处理完成", task.Name), nil
}
```

## Trade-off 分析

### database/sql vs GORM

**database/sql 优势：**

- 标准库，无需额外依赖
- 性能更好，没有 ORM 开销
- SQL 完全可控，优化更灵活
- 适合复杂查询和性能敏感场景

**database/sql 劣势：**

- 需要手写大量 SQL
- 代码冗长，容易出错
- 需要手动管理结构体映射
- 事务处理代码复杂

**GORM 优势：**

- 代码简洁，可读性强
- 自动迁移表结构
- 关联关系处理方便
- 钩子函数和回调
- 开发效率高

**GORM 劣势：**

- 有性能开销
- 复杂查询可能不如原生 SQL 高效
- 学习曲线
- 魔法行为可能导致意外结果

### 何时使用哪个？

**使用 database/sql：**

- 性能要求极高的场景
- 复杂的 SQL 查询
- 需要完全控制 SQL 语句
- 简单的 CRUD 操作

**使用 GORM：**

- 快速开发
- 团队熟悉 ORM
- 需要自动迁移
- 有复杂关联关系

### 数据库设计考虑

**索引策略：**

- `status` 字段建立索引，加速状态查询
- `priority` 字段建立索引，加速优先级排序
- 复合索引 `(status, priority)` 可能更高效

**事务粒度：**

- 事务应尽可能短小
- 避免在事务中进行外部调用
- 及时提交或回滚

**连接池配置：**

- 根据并发量调整连接池大小
- 设置合理的连接生命周期
- 监控连接使用情况

## 总结

我们学习了：

1. **database/sql**：Go 标准数据库接口
   - 连接池管理
   - CRUD 操作
   - 事务处理

2. **GORM**：强大的 ORM 框架
   - 模型定义
   - 自动迁移
   - 钩子函数
   - 关联关系

3. **实践应用**：将数据持久化集成到任务队列
   - 任务存储
   - 状态管理
   - 重试机制

4. **设计决策**：database/sql vs GORM 的选择

在实际项目中，可以根据团队情况选择合适的方案。对于大多数应用，GORM 能提供更好的开发体验；对于性能敏感的模块，可以使用 database/sql 优化。
