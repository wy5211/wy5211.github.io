## 为什么需要 HTTP 编程？

任务队列 API 需要对外提供接口：客户端提交任务、查询状态、获取结果。最自然的方式就是暴露 HTTP API。同时，任务队列本身也可能作为客户端调用其他服务。

Go 的 `net/http` 包提供了完整的 HTTP 客户端和服务器实现，无需第三方框架就能构建高性能的 Web 服务。本文将介绍如何用 Go 构建 RESTful API。

---

## HTTP 服务器基础

### 最简单的服务器

```go
package main

import (
    "fmt"
    "net/http"
)

// handler 函数处理 HTTP 请求
func handler(w http.ResponseWriter, r *http.Request) {
    // w 用于写入响应
    // r 包含请求信息

    fmt.Fprintf(w, "Hello, %s!", r.URL.Path[1:])
}

func main() {
    // 注册路由
    http.HandleFunc("/", handler)

    // 启动服务器
    fmt.Println("服务器启动在 :8080")
    if err := http.ListenAndServe(":8080", nil); err != nil {
        fmt.Printf("服务器启动失败: %v\n", err)
    }
}
```

**核心概念**：

- `http.ResponseWriter`：用于写入 HTTP 响应
- `http.Request`：包含请求的所有信息（URL、Header、Body 等）
- `http.HandleFunc`：注册路由和处理函数
- `ListenAndServe`：启动 HTTP 服务器

### 使用 ServeMux 路由

对于更复杂的路由，可以使用 `ServeMux`：

```go
package main

import (
    "fmt"
    "net/http"
)

func homeHandler(w http.ResponseWriter, r *http.Request) {
    if r.URL.Path != "/" {
        http.NotFound(w, r)
        return
    }
    fmt.Fprintln(w, "欢迎来到任务队列 API")
}

func tasksHandler(w http.ResponseWriter, r *http.Request) {
    fmt.Fprintln(w, "任务列表")
}

func taskByIDHandler(w http.ResponseWriter, r *http.Request) {
    // 从 URL 获取 ID
    id := r.URL.Path[len("/tasks/"):]
    fmt.Fprintf(w, "任务详情: %s", id)
}

func main() {
    // 创建 ServeMux
    mux := http.NewServeMux()

    // 注册路由
    mux.HandleFunc("/", homeHandler)
    mux.HandleFunc("/tasks", tasksHandler)
    mux.HandleFunc("/tasks/", taskByIDHandler)

    // 启动服务器
    fmt.Println("服务器启动在 :8080")
    if err := http.ListenAndServe(":8080", mux); err != nil {
        fmt.Printf("服务器启动失败: %v\n", err)
    }
}
```

`ServeMux` 的优势：

- 更清晰的路由管理
- 可以传递给其他中间件
- 支持路由分组

## 构建任务队列 API

### 数据模型

```go
package main

import "time"

type Task struct {
    ID          int       `json:"id"`
    Title       string    `json:"title"`
    Description string    `json:"description"`
    Status      string    `json:"status"`
    CreatedAt   time.Time `json:"created_at"`
    UpdatedAt   time.Time `json:"updated_at"`
}

type CreateTaskRequest struct {
    Title       string `json:"title"`
    Description string `json:"description"`
}

type UpdateTaskRequest struct {
    Title       *string `json:"title,omitempty"`
    Description *string `json:"description,omitempty"`
    Status      *string `json:"status,omitempty"`
}

type TaskResponse struct {
    ID          int       `json:"id"`
    Title       string    `json:"title"`
    Description string    `json:"description"`
    Status      string    `json:"status"`
    CreatedAt   time.Time `json:"created_at"`
    UpdatedAt   time.Time `json:"updated_at"`
}
```

### 内存存储

```go
package main

import (
    "sync"
    "time"
)

type TaskStore struct {
    tasks map[int]*Task
    nextID int
    mu     sync.RWMutex
}

func NewTaskStore() *TaskStore {
    return &TaskStore{
        tasks: make(map[int]*Task),
        nextID: 1,
    }
}

func (s *TaskStore) Create(title, description string) *Task {
    s.mu.Lock()
    defer s.mu.Unlock()

    task := &Task{
        ID:          s.nextID,
        Title:       title,
        Description: description,
        Status:      "pending",
        CreatedAt:   time.Now(),
        UpdatedAt:   time.Now(),
    }

    s.tasks[s.nextID] = task
    s.nextID++

    return task
}

func (s *TaskStore) Get(id int) (*Task, bool) {
    s.mu.RLock()
    defer s.mu.RUnlock()

    task, ok := s.tasks[id]
    return task, ok
}

func (s *TaskStore) List() []*Task {
    s.mu.RLock()
    defer s.mu.RUnlock()

    tasks := make([]*Task, 0, len(s.tasks))
    for _, task := range s.tasks {
        tasks = append(tasks, task)
    }

    return tasks
}

func (s *TaskStore) Update(id int, req UpdateTaskRequest) (*Task, bool) {
    s.mu.Lock()
    defer s.mu.Unlock()

    task, ok := s.tasks[id]
    if !ok {
        return nil, false
    }

    if req.Title != nil {
        task.Title = *req.Title
    }
    if req.Description != nil {
        task.Description = *req.Description
    }
    if req.Status != nil {
        task.Status = *req.Status
    }
    task.UpdatedAt = time.Now()

    return task, true
}

func (s *TaskStore) Delete(id int) bool {
    s.mu.Lock()
    defer s.mu.Unlock()

    if _, ok := s.tasks[id]; !ok {
        return false
    }

    delete(s.tasks, id)
    return true
}
```

### API 处理器

```go
package main

import (
    "encoding/json"
    "fmt"
    "net/http"
)

func respondJSON(w http.ResponseWriter, status int, data interface{}) {
    w.Header().Set("Content-Type", "application/json")
    w.WriteHeader(status)
    json.NewEncoder(w).Encode(data)
}

func respondError(w http.ResponseWriter, status int, message string) {
    respondJSON(w, status, map[string]string{"error": message})
}

func (s *TaskStore) CreateTaskHandler(w http.ResponseWriter, r *http.Request) {
    if r.Method != http.MethodPost {
        respondError(w, http.StatusMethodNotAllowed, "只支持 POST 方法")
        return
    }

    var req CreateTaskRequest
    if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
        respondError(w, http.StatusBadRequest, "无效的请求格式")
        return
    }

    if req.Title == "" {
        respondError(w, http.StatusBadRequest, "标题不能为空")
        return
    }

    task := s.Create(req.Title, req.Description)
    respondJSON(w, http.StatusCreated, task)
}

func (s *TaskStore) GetTaskHandler(w http.ResponseWriter, r *http.Request) {
    if r.Method != http.MethodGet {
        respondError(w, http.StatusMethodNotAllowed, "只支持 GET 方法")
        return
    }

    idStr := r.URL.Path[len("/tasks/"):]
    var id int
    if _, err := fmt.Sscanf(idStr, "%d", &id); err != nil {
        respondError(w, http.StatusBadRequest, "无效的任务 ID")
        return
    }

    task, ok := s.Get(id)
    if !ok {
        respondError(w, http.StatusNotFound, "任务不存在")
        return
    }

    respondJSON(w, http.StatusOK, task)
}

func (s *TaskStore) ListTasksHandler(w http.ResponseWriter, r *http.Request) {
    if r.Method != http.MethodGet {
        respondError(w, http.StatusMethodNotAllowed, "只支持 GET 方法")
        return
    }

    tasks := s.List()
    respondJSON(w, http.StatusOK, tasks)
}

func (s *TaskStore) UpdateTaskHandler(w http.ResponseWriter, r *http.Request) {
    if r.Method != http.MethodPut && r.Method != http.MethodPatch {
        respondError(w, http.StatusMethodNotAllowed, "只支持 PUT 和 PATCH 方法")
        return
    }

    idStr := r.URL.Path[len("/tasks/"):]
    var id int
    if _, err := fmt.Sscanf(idStr, "%d", &id); err != nil {
        respondError(w, http.StatusBadRequest, "无效的任务 ID")
        return
    }

    var req UpdateTaskRequest
    if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
        respondError(w, http.StatusBadRequest, "无效的请求格式")
        return
    }

    task, ok := s.Update(id, req)
    if !ok {
        respondError(w, http.StatusNotFound, "任务不存在")
        return
    }

    respondJSON(w, http.StatusOK, task)
}

func (s *TaskStore) DeleteTaskHandler(w http.ResponseWriter, r *http.Request) {
    if r.Method != http.MethodDelete {
        respondError(w, http.StatusMethodNotAllowed, "只支持 DELETE 方法")
        return
    }

    idStr := r.URL.Path[len("/tasks/"):]
    var id int
    if _, err := fmt.Sscanf(idStr, "%d", &id); err != nil {
        respondError(w, http.StatusBadRequest, "无效的任务 ID")
        return
    }

    if !s.Delete(id) {
        respondError(w, http.StatusNotFound, "任务不存在")
        return
    }

    w.WriteHeader(http.StatusNoContent)
}
```

### 服务器主函数

```go
package main

import (
    "fmt"
    "net/http"
)

func main() {
    store := NewTaskStore()

    mux := http.NewServeMux()

    // 路由注册
    mux.HandleFunc("/tasks", func(w http.ResponseWriter, r *http.Request) {
        switch r.Method {
        case http.MethodGet:
            store.ListTasksHandler(w, r)
        case http.MethodPost:
            store.CreateTaskHandler(w, r)
        default:
            respondError(w, http.StatusMethodNotAllowed, "方法不允许")
        }
    })

    mux.HandleFunc("/tasks/", func(w http.ResponseWriter, r *http.Request) {
        // 检查是否是 /tasks 后面直接跟 ID
        if len(r.URL.Path) <= len("/tasks/") {
            respondError(w, http.StatusBadRequest, "需要任务 ID")
            return
        }

        switch r.Method {
        case http.MethodGet:
            store.GetTaskHandler(w, r)
        case http.MethodPut, http.MethodPatch:
            store.UpdateTaskHandler(w, r)
        case http.MethodDelete:
            store.DeleteTaskHandler(w, r)
        default:
            respondError(w, http.StatusMethodNotAllowed, "方法不允许")
        }
    })

    // 启动服务器
    fmt.Println("任务队列 API 启动在 :8080")
    if err := http.ListenAndServe(":8080", mux); err != nil {
        fmt.Printf("服务器启动失败: %v\n", err)
    }
}
```

## HTTP 客户端

### GET 请求

```go
package main

import (
    "encoding/json"
    "fmt"
    "io"
    "net/http"
)

func getTasks(baseURL string) ([]Task, error) {
    resp, err := http.Get(baseURL + "/tasks")
    if err != nil {
        return nil, err
    }
    defer resp.Body.Close()

    if resp.StatusCode != http.StatusOK {
        return nil, fmt.Errorf("HTTP 错误: %d", resp.StatusCode)
    }

    body, err := io.ReadAll(resp.Body)
    if err != nil {
        return nil, err
    }

    var tasks []Task
    if err := json.Unmarshal(body, &tasks); err != nil {
        return nil, err
    }

    return tasks, nil
}

func main() {
    tasks, err := getTasks("http://localhost:8080")
    if err != nil {
        fmt.Printf("获取任务失败: %v\n", err)
        return
    }

    fmt.Printf("获取到 %d 个任务\n", len(tasks))
    for _, task := range tasks {
        fmt.Printf("- 任务 %d: %s\n", task.ID, task.Title)
    }
}
```

### POST 请求

```go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "net/http"
)

func createTask(baseURL string, req CreateTaskRequest) (*Task, error) {
    jsonData, err := json.Marshal(req)
    if err != nil {
        return nil, err
    }

    resp, err := http.Post(
        baseURL+"/tasks",
        "application/json",
        bytes.NewBuffer(jsonData),
    )
    if err != nil {
        return nil, err
    }
    defer resp.Body.Close()

    if resp.StatusCode != http.StatusCreated {
        return nil, fmt.Errorf("HTTP 错误: %d", resp.StatusCode)
    }

    var task Task
    if err := json.NewDecoder(resp.Body).Decode(&task); err != nil {
        return nil, err
    }

    return &task, nil
}

func main() {
    task, err := createTask("http://localhost:8080", CreateTaskRequest{
        Title:       "新任务",
        Description: "任务描述",
    })
    if err != nil {
        fmt.Printf("创建任务失败: %v\n", err)
        return
    }

    fmt.Printf("创建任务成功: ID=%d, Title=%s\n", task.ID, task.Title)
}
```

### 使用自定义 Client

对于更复杂的需求（超时、重试、连接池），使用自定义 `http.Client`：

```go
package main

import (
    "fmt"
    "net/http"
    "time"
)

func main() {
    // 创建自定义 Client
    client := &http.Client{
        Timeout: 10 * time.Second,
        Transport: &http.Transport{
            MaxIdleConns:        100,
            MaxIdleConnsPerHost: 10,
            IdleConnTimeout:     90 * time.Second,
        },
    }

    // 使用自定义 Client 发送请求
    resp, err := client.Get("http://localhost:8080/tasks")
    if err != nil {
        fmt.Printf("请求失败: %v\n", err)
        return
    }
    defer resp.Body.Close()

    fmt.Printf("状态码: %d\n", resp.StatusCode)
}
```

**自定义 Client 的优势**：

- 设置超时时间
- 配置连接池
- 添加重试逻辑
- 自定义 Transport

## 中间件

中间件可以拦截请求，在处理前后添加逻辑（如认证、日志、CORS）。

### 日志中间件

```go
package main

import (
    "log"
    "net/http"
    "time"
)

func loggingMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        start := time.Now()

        // 调用下一个处理器
        next.ServeHTTP(w, r)

        // 记录请求日志
        log.Printf(
            "%s %s %s",
            r.Method,
            r.RequestURI,
            time.Since(start),
        )
    })
}

func main() {
    mux := http.NewServeMux()
    mux.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) {
        w.Write([]byte("Hello, World!"))
    })

    // 使用中间件
    handler := loggingMiddleware(mux)

    http.ListenAndServe(":8080", handler)
}
```

### CORS 中间件

```go
package main

import (
    "net/http"
)

func corsMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        // 设置 CORS 头
        w.Header().Set("Access-Control-Allow-Origin", "*")
        w.Header().Set("Access-Control-Allow-Methods", "GET, POST, PUT, DELETE, OPTIONS")
        w.Header().Set("Access-Control-Allow-Headers", "Content-Type, Authorization")

        // 处理预检请求
        if r.Method == http.MethodOptions {
            w.WriteHeader(http.StatusOK)
            return
        }

        next.ServeHTTP(w, r)
    })
}
```

### 认证中间件

```go
package main

import (
    "net/http"
    "strings"
)

func authMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        // 获取 Authorization 头
        auth := r.Header.Get("Authorization")
        if auth == "" {
            http.Error(w, "未授权", http.StatusUnauthorized)
            return
        }

        // 检查 Bearer token
        if !strings.HasPrefix(auth, "Bearer ") {
            http.Error(w, "无效的认证格式", http.StatusUnauthorized)
            return
        }

        token := strings.TrimPrefix(auth, "Bearer ")
        if !isValidToken(token) {
            http.Error(w, "无效的 token", http.StatusUnauthorized)
            return
        }

        next.ServeHTTP(w, r)
    })
}

func isValidToken(token string) bool {
    // 实际应用中应该验证 JWT 或查询数据库
    return token == "valid-token"
}
```

### 组合中间件

```go
package main

import (
    "net/http"
)

func chainMiddleware(handler http.Handler, middlewares ...func(http.Handler) http.Handler) http.Handler {
    for _, mw := range middlewares {
        handler = mw(handler)
    }
    return handler
}

func main() {
    mux := http.NewServeMux()
    mux.HandleFunc("/tasks", func(w http.ResponseWriter, r *http.Request) {
        w.Write([]byte("任务列表"))
    })

    // 组合多个中间件
    handler := chainMiddleware(mux,
        loggingMiddleware,
        corsMiddleware,
        authMiddleware,
    )

    http.ListenAndServe(":8080", handler)
}
```

## 使用 Gin 框架

虽然 `net/http` 足够强大，但实际项目常用框架简化开发。Gin 是最流行的 Go Web 框架之一。

### 安装 Gin

```bash
go get -u github.com/gin-gonic/gin
```

### 使用 Gin 构建任务队列 API

```go
package main

import (
    "net/http"
    "strconv"

    "github.com/gin-gonic/gin"
)

type Task struct {
    ID          int    `json:"id"`
    Title       string `json:"title"`
    Description string `json:"description"`
    Status      string `json:"status"`
}

var (
    tasks   = make(map[int]*Task)
    nextID  = 1
)

func main() {
    r := gin.Default()

    // 路由组
    api := r.Group("/api")
    {
        api.GET("/tasks", listTasks)
        api.POST("/tasks", createTask)
        api.GET("/tasks/:id", getTask)
        api.PUT("/tasks/:id", updateTask)
        api.DELETE("/tasks/:id", deleteTask)
    }

    r.Run(":8080")
}

func listTasks(c *gin.Context) {
    taskList := make([]*Task, 0, len(tasks))
    for _, task := range tasks {
        taskList = append(taskList, task)
    }
    c.JSON(http.StatusOK, taskList)
}

func createTask(c *gin.Context) {
    var req CreateTaskRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
        return
    }

    task := &Task{
        ID:          nextID,
        Title:       req.Title,
        Description: req.Description,
        Status:      "pending",
    }
    tasks[nextID] = task
    nextID++

    c.JSON(http.StatusCreated, task)
}

func getTask(c *gin.Context) {
    id, err := strconv.Atoi(c.Param("id"))
    if err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": "无效的任务 ID"})
        return
    }

    task, ok := tasks[id]
    if !ok {
        c.JSON(http.StatusNotFound, gin.H{"error": "任务不存在"})
        return
    }

    c.JSON(http.StatusOK, task)
}

func updateTask(c *gin.Context) {
    id, err := strconv.Atoi(c.Param("id"))
    if err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": "无效的任务 ID"})
        return
    }

    task, ok := tasks[id]
    if !ok {
        c.JSON(http.StatusNotFound, gin.H{"error": "任务不存在"})
        return
    }

    var req UpdateTaskRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
        return
    }

    if req.Title != nil {
        task.Title = *req.Title
    }
    if req.Description != nil {
        task.Description = *req.Description
    }
    if req.Status != nil {
        task.Status = *req.Status
    }

    c.JSON(http.StatusOK, task)
}

func deleteTask(c *gin.Context) {
    id, err := strconv.Atoi(c.Param("id"))
    if err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": "无效的任务 ID"})
        return
    }

    if _, ok := tasks[id]; !ok {
        c.JSON(http.StatusNotFound, gin.H{"error": "任务不存在"})
        return
    }

    delete(tasks, id)
    c.Status(http.StatusNoContent)
}
```

**Gin 的优势**：

- 更简洁的路由语法
- 内置 JSON 绑定和验证
- 更好的错误处理
- 丰富的中间件生态
- 高性能

## Trade-off 分析

### net/http vs 框架（Gin/Echo）

**net/http**：

- 优势：标准库，无依赖，学习 Go 的好方式
- 劣势：路由匹配弱，需要手写很多代码

**Gin/Echo**：

- 优势：功能丰富，开发效率高，社区支持好
- 劣势：额外依赖，学习框架成本

**建议**：

- 学习和简单项目：用 `net/http`
- 生产项目：用 Gin 或 Echo

### JSON vs 其他格式

**JSON**：

- 优势：人类可读，语言无关，广泛支持
- 劣势：解析慢，体积大

**Protobuf/MessagePack**：

- 优势：性能高，体积小
- 劣势：不可读，需要 schema

**建议**：内部服务用 Protobuf，对外 API 用 JSON

### 同步处理 vs 异步处理

**同步处理**（本文示例）：

- 优势：实现简单，适合快速响应的操作
- 劣势：阻塞请求，吞吐量低

**异步处理**（任务队列应该用这个）：

- 优势：高吞吐量，解耦处理和响应
- 劣势：实现复杂，需要状态管理

Go 的 HTTP 编程简洁高效，`net/http` 包提供了构建 Web 服务所需的一切。通过本文，你应该能够：

1. 使用 `net/http` 构建 RESTful API
2. 处理 JSON 请求和响应
3. 编写中间件
4. 使用 Gin 等框架简化开发

在任务队列 API 中，HTTP 编程用于：

- 接收任务提交请求
- 提供任务状态查询接口
- 对外暴露管理接口

记住几个要点：

1. 使用正确的 HTTP 方法和状态码
2. 并发访问需要加锁保护
3. 使用中间件处理横切关注点
4. 生产环境考虑使用框架和认证
