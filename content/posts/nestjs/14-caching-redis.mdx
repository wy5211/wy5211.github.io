## 二、配置缓存

### 2.1 模块配置

```typescript
import { CacheModule } from "@nestjs/cache-manager";
import { redisStore } from "cache-manager-ioredis";

@Module({
  imports: [
    CacheModule.register({
      store: redisStore,
      host: "localhost",
      port: 6379,
      ttl: 300, // 5 分钟
      max: 100, // 最大缓存数
    }),
  ],
})
export class AppModule {}
```

### 2.2 全局缓存

```typescript
async function bootstrap() {
  const app = await NestFactory.create(AppModule);

  // 启用全局缓存
  app.enableShutdownHooks();

  await app.listen(3000);
}
```

## 三、使用缓存装饰器

### 3.1 基础缓存

```typescript
import { CacheTTL } from "@nestjs/cache-manager";

@Controller("posts")
export class PostsController {
  @Get()
  @CacheTTL(60) // 缓存 60 秒
  findAll() {
    return this.postsService.findAll();
  }
}
```

### 3.2 自动缓存失效

```typescript
@Injectable()
export class PostsService {
  @CacheTTL(300)
  async findAll() {
    return this.prisma.post.findMany();
  }

  // 更新时清除缓存
  async update(id: string, data: any) {
    const CACHE_KEY = "posts:all";
    await this.cacheManager.del(CACHE_KEY);

    return this.prisma.post.update({ where: { id }, data });
  }
}
```

## 四、手动缓存控制

### 4.1 设置缓存

```typescript
@Injectable()
export class PostsService {
  constructor(@Inject(CACHE_MANAGER) private cacheManager: Cache) {}

  async findOne(id: string) {
    // 检查缓存
    const cached = await this.cacheManager.get(`post:${id}`);
    if (cached) {
      return cached;
    }

    // 查询数据库
    const post = await this.prisma.post.findUnique({ where: { id } });

    // 设置缓存
    await this.cacheManager.set(`post:${id}`, post, 300);

    return post;
  }
}
```

### 4.2 清除缓存

```typescript
async update(id: string, data: any) {
  // 清除缓存
  await this.cacheManager.del(`post:${id}`);

  return this.prisma.post.update({ where: { id }, data });
}

async delete(id: string) {
  // 清除缓存
  await this.cacheManager.del(`post:${id}`);

  return this.prisma.post.delete({ where: { id } });
}
```

## 五、缓存策略

### 5.1 Cache-Aside

```typescript
// 应用程序负责缓存管理
async getUser(id: string) {
  // 1. 先查缓存
  let user = await this.cacheManager.get(`user:${id}`);

  // 2. 缓存未命中，查数据库
  if (!user) {
    user = await this.prisma.user.findUnique({ where: { id } });

    // 3. 写入缓存
    if (user) {
      await this.cacheManager.set(`user:${id}`, user, 300);
    }
  }

  return user;
}
```

### 5.2 Write-Through

```typescript
// 写入时同时更新缓存和数据库
async createUser(data: CreateUserDto) {
  const user = await this.prisma.user.create({ data });

  // 同时更新缓存
  await this.cacheManager.set(`user:${user.id}`, user, 300);

  return user;
}
```
