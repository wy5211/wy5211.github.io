## 一、什么是管道（Pipe）？

### 1.1 管道的概念

**管道（Pipe）** 是 NestJS 中用于处理输入数据的类，它有两个典型应用场景：

1. **数据转换**：将输入数据转换为所需格式（如字符串转数字）
2. **数据验证**：验证输入数据是否符合要求

**管道的工作流程**：

```
请求 → 管道 → 控制器
           ↓
      验证/转换数据
           ↓
      如果失败 → 抛出异常
      如果成功 → 传递给控制器
```

### 1.2 内置管道

NestJS 提供了多个内置管道：

| 管道               | 用途         | 示例                                      |
| ------------------ | ------------ | ----------------------------------------- |
| `ValidationPipe`   | 数据验证     | 验证 DTO                                  |
| `ParseIntPipe`     | 转换为整数   | `@Param('id', ParseIntPipe)`              |
| `ParseBoolPipe`    | 转换为布尔值 | `@Query('published', ParseBoolPipe)`      |
| `ParseFloatPipe`   | 转换为浮点数 | `@Query('price', ParseFloatPipe)`         |
| `ParseUUIDPipe`    | 验证 UUID    | `@Param('id', ParseUUIDPipe)`             |
| `ParseEnumPipe`    | 验证枚举     | `@Query('status', ParseEnumPipe)`         |
| `DefaultValuePipe` | 设置默认值   | `@Query('page', new DefaultValuePipe(1))` |

## 二、ValidationPipe 详解

### 2.1 全局配置 ValidationPipe

在 `main.ts` 中配置：

```typescript
import { ValidationPipe } from "@nestjs/common";

async function bootstrap() {
  const app = await NestFactory.create(AppModule);

  app.useGlobalPipes(
    new ValidationPipe({
      whitelist: true, // 移除未定义的属性
      forbidNonWhitelisted: true, // 如果有未定义的属性则报错
      transform: true, // 自动转换类型
      transformOptions: {
        enableImplicitConversion: true, // 隐式类型转换
      },
      disableErrorMessages: false, // 显示错误信息
    })
  );

  await app.listen(3000);
}
bootstrap();
```

**配置说明**：

```typescript
{
  // 移除未在 DTO 中定义的属性
  whitelist: true,

  // 如果有未定义的属性则抛出错误（比 whitelist 更严格）
  forbidNonWhitelisted: true,

  // 自动转换类型（如字符串 "123" 转为数字 123）
  transform: true,

  // 禁用错误信息（生产环境可考虑关闭）
  disableErrorMessages: false,

  // 验证所有嵌套对象
  always: true,

  // 只验证特定组
  groups: [],

  // 跳过缺失的属性
  skipMissingProperties: false,

  // 跳过 null 值
  skipNullProperties: false,

  // 跳过 undefined 值
  skipUndefinedProperties: false,
}
```

### 2.2 局部使用 ValidationPipe

在控制器或方法上使用：

```typescript
import { UsePipes, ValidationPipe } from "@nestjs/common";

@Controller("posts")
export class PostsController {
  // 整个控制器使用
  @UsePipes(new ValidationPipe({ whitelist: true }))
  @Post()
  create() {}

  // 单个方法使用
  @Post()
  @UsePipes(new ValidationPipe({ forbidNonWhitelisted: true }))
  create() {}
}
```

## 三、class-validator 装饰器

### 3.1 常用验证装饰器

```typescript
import {
  IsString,
  IsNumber,
  IsInt,
  IsBoolean,
  IsArray,
  IsObject,
  IsDate,
  IsEmail,
  IsUrl,
  IsUUID,
  IsEnum,
  IsNotEmpty,
  IsOptional,
  Min,
  Max,
  MinLength,
  MaxLength,
  Matches,
  In,
  NotIn,
  ArrayNotEmpty,
  ArrayMinSize,
  ArrayMaxSize,
} from "class-validator";

export class CreateUserDto {
  // 字符串验证
  @IsString()
  @IsNotEmpty()
  name: string;

  @IsString()
  @IsEmail()
  email: string;

  // 数字验证
  @IsNumber()
  @IsInt()
  @Min(18)
  @Max(100)
  age: number;

  // 布尔验证
  @IsBoolean()
  isActive: boolean;

  // 数组验证
  @IsArray()
  @ArrayNotEmpty()
  @ArrayMinSize(1)
  @ArrayMaxSize(5)
  tags: string[];

  // 枚举验证
  @IsEnum(Role)
  role: Role;

  // 长度验证
  @IsString()
  @MinLength(8)
  @MaxLength(20)
  password: string;

  // 正则验证
  @IsString()
  @Matches(/^[A-Za-z0-9]+$/, {
    message: "密码只能包含字母和数字",
  })
  username: string;

  // UUID 验证
  @IsUUID("4")
  userId: string;

  // URL 验证
  @IsUrl()
  avatar: string;

  // 可选字段
  @IsOptional()
  @IsString()
  bio?: string;

  // 包含验证
  @IsString()
  @In(["male", "female", "other"])
  gender: string;
}
```

## 四、自定义验证装饰器

### 4.1 创建自定义验证器

```typescript
// src/common/validators/is-password-strong.validator.ts
import {
  ValidatorConstraint,
  ValidatorConstraintInterface,
  registerDecorator,
  ValidationOptions,
} from "class-validator";

@ValidatorConstraint({ name: "isPasswordStrong", async: false })
export class IsPasswordStrongConstraint implements ValidatorConstraintInterface {
  validate(password: string) {
    // 至少 8 个字符
    if (password.length < 8) return false;

    // 至少一个大写字母
    if (!/[A-Z]/.test(password)) return false;

    // 至少一个小写字母
    if (!/[a-z]/.test(password)) return false;

    // 至少一个数字
    if (!/\d/.test(password)) return false;

    // 至少一个特殊字符
    if (!/[!@#$%^&*(),.?":{}|<>]/.test(password)) return false;

    return true;
  }

  defaultMessage() {
    return "密码必须包含至少 8 个字符，包括大小写字母、数字和特殊字符";
  }
}

// 创建装饰器
export function IsPasswordStrong(validationOptions?: ValidationOptions) {
  return function (object: Object, propertyName: string) {
    registerDecorator({
      target: object.constructor,
      propertyName: propertyName,
      options: validationOptions,
      constraints: [],
      validator: IsPasswordStrongConstraint,
    });
  };
}
```

使用自定义验证器：

```typescript
export class RegisterDto {
  @IsString()
  @IsPasswordStrong({
    message: "密码强度不够",
  })
  password: string;
}
```

### 4.2 异步验证器

```typescript
@ValidatorConstraint({ name: "isUserAlreadyExist", async: true })
export class IsUserAlreadyExistConstraint implements ValidatorConstraintInterface {
  constructor(private prisma: PrismaService) {}

  async validate(email: string) {
    const user = await this.prisma.user.findUnique({
      where: { email },
    });
    return !user; // 如果用户不存在则验证通过
  }

  defaultMessage() {
    return "该邮箱已被注册";
  }
}

export function IsUserAlreadyExist(validationOptions?: ValidationOptions) {
  return function (object: Object, propertyName: string) {
    registerDecorator({
      target: object.constructor,
      propertyName: propertyName,
      options: validationOptions,
      constraints: [],
      validator: IsUserAlreadyExistConstraint,
    });
  };
}
```

## 五、自定义管道

### 5.1 创建验证管道

```typescript
// src/common/pipes/validation.pipe.ts
import {
  PipeTransform,
  Injectable,
  ArgumentMetadata,
  BadRequestException,
} from "@nestjs/common";
import { validate } from "class-validator";
import { plainToInstance } from "class-transformer";

@Injectable()
export class ValidationPipe implements PipeTransform<any> {
  async transform(value: any, { metatype }: ArgumentMetadata) {
    if (!metatype || !this.toValidate(metatype)) {
      return value;
    }

    const object = plainToInstance(metatype, value);
    const errors = await validate(object);

    if (errors.length > 0) {
      const messages = errors.map((error) => {
        return {
          field: error.property,
          constraints: error.constraints,
        };
      });

      throw new BadRequestException({
        message: "验证失败",
        errors: messages,
      });
    }

    return object;
  }

  private toValidate(metatype: Function): boolean {
    const types: Function[] = [String, Boolean, Number, Array, Object];
    return !types.includes(metatype);
  }
}
```

### 5.2 创建转换管道

```typescript
// src/common/parse-int.pipe.ts
import {
  PipeTransform,
  Injectable,
  ArgumentMetadata,
  BadRequestException,
} from '@nestjs/common';

@Injectable()
export class ParseIntPipe implements PipeTransform<string, number> {
  transform(value: string, metadata: ArgumentMetadata): number {
    const val = parseInt(value, 10);

    if (isNaN(val)) {
      throw new BadRequestException(
        `参数 "${metadata.data}" 必须是数字`,
      );
    }

    return val;
  }
}

// 使用
@Get(':id')
findOne(@Param('id', ParseIntPipe) id: number) {
  // id 现在是 number 类型
}
```

### 5.3 创建默认值管道

```typescript
// src/common/pipes/default-value.pipe.ts
import {
  PipeTransform,
  Injectable,
  ArgumentMetadata,
} from '@nestjs/common';

@Injectable()
export class DefaultValuePipe<T = any>
  implements PipeTransform<T, T>
{
  constructor(private readonly defaultValue: T) {}

  transform(value: T, metadata: ArgumentMetadata): T {
    return value ?? this.defaultValue;
  }
}

// 使用
@Get()
findAll(
  @Query('page', new DefaultValuePipe(1)) page: number,
  @Query('limit', new DefaultValuePipe(10)) limit: number,
) {
  // 如果没有提供 page，默认为 1
  // 如果没有提供 limit，默认为 10
}
```

## 六、验证组

### 6.1 定义验证组

```typescript
export class UserDto {
  @IsString()
  @IsNotEmpty()
  name: string;

  @IsEmail()
  @IsNotEmpty({ groups: ["create"] })
  email: string;

  @IsString()
  @MinLength(8)
  @IsNotEmpty({ groups: ["create"] })
  password: string;
}
```

### 6.2 在控制器中使用组

```typescript
@Controller("users")
export class UsersController {
  @Post()
  @UsePipes(new ValidationPipe({ groups: ["create"] }))
  create(@Body() dto: UserDto) {
    // 只有 create 组的验证会执行
  }

  @Patch(":id")
  @UsePipes(new ValidationPipe({ groups: ["update"] }))
  update(@Body() dto: UserDto) {
    // 只有 update 组的验证会执行
  }
}
```

## 八、错误处理

### 8.1 自定义错误响应

```typescript
// src/common/filters/validation-exception.filter.ts
import {
  ExceptionFilter,
  Catch,
  ArgumentsHost,
  HttpException,
  BadRequestException,
} from "@nestjs/common";

@Catch(BadRequestException)
export class ValidationExceptionFilter implements ExceptionFilter {
  catch(exception: BadRequestException, host: ArgumentsHost) {
    const ctx = host.switchToHttp();
    const response = ctx.getResponse();
    const request = ctx.getRequest();
    const status = exception.getStatus();
    const exceptionResponse = exception.getResponse();

    const errors =
      typeof exceptionResponse === "object" && "message" in exceptionResponse
        ? (exceptionResponse as any).message
        : "验证失败";

    response.status(status).json({
      success: false,
      statusCode: status,
      timestamp: new Date().toISOString(),
      path: request.url,
      errors: Array.isArray(errors) ? errors : [errors],
    });
  }
}
```

## 九、常见问题

### Q1: 如何验证可选字段？

**A**: 使用 `@IsOptional()` 装饰器：

```typescript
export class UpdateUserDto {
  @IsOptional()
  @IsString()
  @MinLength(2)
  name?: string;

  @IsOptional()
  @IsEmail()
  email?: string;
}
```

### Q2: 如何验证嵌套数组？

**A**: 使用 `@ValidateNested({ each: true })`：

```typescript
export class CreateUserDto {
  @ValidateNested({ each: true })
  @Type(() => AddressDto)
  addresses: AddressDto[];
}
```

### Q3: 如何禁用某个字段的验证？

**A**: 使用 `@Allow()` 装饰器（从 class-validator 导入）：

```typescript
import { Allow } from "class-validator";

export class CreateUserDto {
  @IsString()
  name: string;

  @Allow() // 不验证此字段
  anyData: any;
}
```
