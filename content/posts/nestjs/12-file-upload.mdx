## 前言

文件上传是 Web 应用的常见功能。在 NestJS 中，我们可以使用 Multer 来轻松实现文件上传。

**本文将带你：**

1. 配置文件上传

## 二、基础文件上传

### 2.1 单文件上传

```typescript
import {
  Controller,
  Post,
  UseInterceptors,
  UploadedFile,
} from "@nestjs/common";
import { FileInterceptor } from "@nestjs/platform-express";
import { diskStorage } from "multer";

@Controller("upload")
export class UploadController {
  @Post("avatar")
  @UseInterceptors(
    FileInterceptor("avatar", {
      storage: diskStorage({
        destination: "./uploads/avatars",
        filename: (req, file, cb) => {
          const filename = `${Date.now()}-${file.originalname}`;
          cb(null, filename);
        },
      }),
      fileFilter: (req, file, cb) => {
        if (!file.originalname.match(/\.(jpg|jpeg|png|gif)$/)) {
          return cb(new Error("只允许上传图片"), false);
        }
        cb(null, true);
      },
      limits: {
        fileSize: 5 * 1024 * 1024, // 5MB
      },
    })
  )
  uploadAvatar(@UploadedFile() file: Express.Multer.File) {
    return {
      filename: file.filename,
      originalname: file.originalname,
      size: file.size,
    };
  }
}
```

### 2.2 多文件上传

```typescript
import { FilesInterceptor } from '@nestjs/platform-express';

@Post('gallery')
@UseInterceptors(
  FilesInterceptor('files', 10, {
    storage: diskStorage({
      destination: './uploads/gallery',
      filename: (req, file, cb) => {
        const filename = `${Date.now()}-${file.originalname}`;
        cb(null, filename);
      },
    }),
    limits: {
      fileSize: 10 * 1024 * 1024, // 10MB
    },
  }),
)
uploadGallery(@UploadedFiles() files: Express.Multer.File[]) {
  return files.map(file => ({
    filename: file.filename,
    originalname: file.originalname,
    size: file.size,
  }));
}
```

### 2.3 字段文件上传

```typescript
import { FileFieldsInterceptor } from '@nestjs/platform-express';

@Post('documents')
@UseInterceptors(
  FileFieldsInterceptor([
    { name: 'idCard', maxCount: 1 },
    { name: 'passport', maxCount: 1 },
  ]),
)
uploadDocuments(@UploadedFiles() files: any) {
  return {
    idCard: files.idCard?.[0],
    passport: files.passport?.[0],
  };
}
```

## 三、图片处理

### 3.1 安装 Sharp

```bash
pnpm add sharp
```

### 3.2 图片压缩和缩略图

```typescript
import {
  Controller,
  Post,
  UseInterceptors,
  UploadedFile,
} from "@nestjs/common";
import { FileInterceptor } from "@nestjs/platform-express";
import * as sharp from "sharp";
import { extname } from "path";

@Controller("images")
export class ImagesController {
  @Post("upload")
  @UseInterceptors(
    FileInterceptor("image", {
      storage: diskStorage({
        destination: "./uploads/images",
        filename: async (req, file, cb) => {
          const filename = `${Date.now()}${extname(file.originalname)}`;
          cb(null, filename);
        },
      }),
    })
  )
  async uploadImage(@UploadedFile() file: Express.Multer.File) {
    const originalPath = file.path;
    const thumbnailPath = `./uploads/images/thumbnails/${file.filename}`;

    // 生成缩略图
    await sharp(originalPath)
      .resize(200, 200, { fit: "cover" })
      .toFile(thumbnailPath);

    return {
      original: file.filename,
      thumbnail: `thumbnails/${file.filename}`,
    };
  }
}
```

## 四、阿里云 OSS

### 4.1 安装 SDK

```bash
pnpm add ali-oss @nestjs/config
```

### 4.2 OSS 服务

```typescript
import { Injectable } from "@nestjs/common";
import { ConfigService } from "@nestjs/config";
import * as OSS from "ali-oss";

@Injectable()
export class OssService {
  private client: OSS;

  constructor(private configService: ConfigService) {
    this.client = new OSS({
      region: this.configService.get("OSS_REGION"),
      accessKeyId: this.configService.get("OSS_ACCESS_KEY_ID"),
      accessKeySecret: this.configService.get("OSS_ACCESS_KEY_SECRET"),
      bucket: this.configService.get("OSS_BUCKET"),
    });
  }

  async uploadFile(filename: string, file: Buffer) {
    const result = await this.client.put(filename, file);
    return result.url;
  }

  async deleteFile(filename: string) {
    await this.client.delete(filename);
  }
}
```
