## 一、文件上传原理

### 1.1 前端发送的文件长什么样？

当我们使用 FormData 上传文件时，浏览器会将数据编码为 `multipart/form-data` 格式。

**前端代码示例**：

```typescript
// 前端发送请求
const formData = new FormData();
formData.append("avatar", file); // file 是 File 对象
formData.append("userId", "123"); // 可以同时发送其他字段

fetch("http://localhost:3000/upload/avatar", {
  method: "POST",
  body: formData,
  // 注意：不需要设置 Content-Type，浏览器会自动设置
  // 并生成 boundary 分隔符
});
```

**实际的 HTTP 请求**：

```http
POST /upload/avatar HTTP/1.1
Host: localhost:3000
Content-Type: multipart/form-data; boundary=----WebKitFormBoundary7MA4YWxkTrZu0gW

------WebKitFormBoundary7MA4YWxkTrZu0gW
Content-Disposition: form-data; name="avatar"; filename="photo.jpg"
Content-Type: image/jpeg

[这里是文件的二进制数据...]
------WebKitFormBoundary7MA4YWxkTrZu0gW
Content-Disposition: form-data; name="userId"

123
------WebKitFormBoundary7MA4YWxkTrZu0gW--
```

**关键点**：

1. **Content-Type**: `multipart/form-data; boundary=----WebKitFormBoundary...`
2. **boundary**: 浏览器自动生成的分隔符，用于区分不同的字段
3. **每个字段**都包含：
   - `Content-Disposition: form-data; name="字段名"; filename="文件名"`
   - `Content-Type`: 文件的 MIME 类型
   - 文件的二进制数据

### 1.2 后端如何解析？

NestJS 使用 Express 框架，Express 使用 **Multer** 中间件来处理 `multipart/form-data`。

**Multer 的作用**：

```
浏览器发送的原始数据
    ↓
Multer 解析 multipart/form-data
    ↓
将文件保存到磁盘/内存
    ↓
在 req 对象中添加 file/files 属性
    ↓
控制器通过 @UploadedFile() 装饰器获取文件
```

## 三、后端接收到的文件对象

### 3.1 Express.Multer.File 对象结构

当文件上传成功后，`@UploadedFile()` 装饰器会返回一个 `Express.Multer.File` 对象：

```typescript
{
  // 文件在服务器上的存储路径（磁盘存储）
  path: string;

  // 文件名（可能是重命名后的文件名）
  filename: string;

  // 原始文件名（用户上传时的文件名）
  originalname: string;

  // 文件的 MIME 类型
  mimetype: string;

  // 文件大小（字节）
  size: number;

  // 文件存储在内存中的 Buffer（内存存储）
  buffer: Buffer;

  // 字段名（FormData 中使用的字段名）
  fieldname: string;

  // 文件编码
  encoding: string;
}
```

**实际示例**：

```typescript
// 用户上传 photo.jpg
{
  fieldname: 'avatar',
  originalname: 'photo.jpg',
  encoding: '7bit',
  mimetype: 'image/jpeg',
  size: 245678,
  filename: '1710876543210-photo.jpg',
  path: '/uploads/avatars/1710876543210-photo.jpg'
}
```

### 3.2 文件存储方式

Multer 支持两种存储方式：

**1. 内存存储（Memory Storage）**

```typescript
{
  storage: multer.memoryStorage();
}
```

- 文件存储在内存中（Buffer）
- 适合小文件或需要直接处理的文件
- 处理完后需要手动保存

**2. 磁盘存储（Disk Storage）**

```typescript
{
  storage: multer.diskStorage({
    destination: "./uploads",
    filename: (req, file, cb) => {
      cb(null, file.originalname);
    },
  });
}
```

- 文件保存到磁盘
- 适合大文件
- 可以自定义文件名

## 四、单文件上传

### 4.1 基础实现

**创建上传模块**：

```bash
nest g module upload
nest g controller upload
```

**实现单文件上传**：

```typescript
// src/upload/upload.controller.ts
import {
  Controller,
  Post,
  UseInterceptors,
  UploadedFile,
} from "@nestjs/common";
import { FileInterceptor } from "@nestjs/platform-express";
import { diskStorage } from "multer";
import { extname } from "path";

@Controller("upload")
export class UploadController {
  @Post("avatar")
  @UseInterceptors(
    FileInterceptor("avatar", {
      storage: diskStorage({
        destination: "./uploads/avatars",
        filename: (req, file, cb) => {
          // 生成唯一文件名：时间戳 + 随机数 + 原始扩展名
          const uniqueSuffix =
            Date.now() + "-" + Math.round(Math.random() * 1e9);
          const ext = extname(file.originalname);
          const filename = `${uniqueSuffix}${ext}`;
          cb(null, filename);
        },
      }),
      // 文件过滤器
      fileFilter: (req, file, cb) => {
        // 只允许图片
        if (!file.originalname.match(/\.(jpg|jpeg|png|gif)$/i)) {
          return cb(new Error("只允许上传图片！"), false);
        }
        cb(null, true);
      },
      // 限制文件大小：5MB
      limits: {
        fileSize: 5 * 1024 * 1024,
      },
    })
  )
  uploadAvatar(@UploadedFile() file: Express.Multer.File) {
    // 返回文件信息
    return {
      success: true,
      message: "文件上传成功",
      data: {
        filename: file.filename,
        originalname: file.originalname,
        mimetype: file.mimetype,
        size: file.size,
        url: `/uploads/avatars/${file.filename}`,
      },
    };
  }
}
```

**测试上传**：

```bash

  -F "avatar=@/path/to/photo.jpg"
```

```typescript
import { FilesInterceptor } from "@nestjs/platform-express";

@Controller("upload")

        filename: file.filename,
        originalname: file.originalname,
        size: file.size,

  }
```

## 六、图片处理

### 6.1 使用 Sharp 处理图片

Sharp 是一个高性能的图片处理库，支持压缩、裁剪、缩放等操作。

```typescript
// src/images/images.controller.ts
import {
  Controller,
  Post,
  UseInterceptors,
  UploadedFile,
  BadRequestException,
} from "@nestjs/common";
import { FileInterceptor } from "@nestjs/platform-express";
import { diskStorage } from "multer";
import { extname } from "path";
import * as sharp from "sharp";

@Controller("images")
export class ImagesController {
  @Post("upload")
  @UseInterceptors(
    FileInterceptor("image", {
      storage: diskStorage({
        destination: "./uploads/images",
        filename: (req, file, cb) => {
          const uniqueSuffix = Date.now();
          const ext = extname(file.originalname);
          cb(null, `${uniqueSuffix}${ext}`);
        },
      }),
      fileFilter: (req, file, cb) => {
        if (!file.mimetype.startsWith("image/")) {
          return cb(new BadRequestException("只允许上传图片"), false);
        }
        cb(null, true);
      },
      limits: {
        fileSize: 10 * 1024 * 1024, // 10MB
      },
    })
  )
  async uploadImage(@UploadedFile() file: Express.Multer.File) {
    const originalPath = file.path;
    const thumbnailDir = "./uploads/images/thumbnails";
    const thumbnailPath = `${thumbnailDir}/${file.filename}`;

    try {
      // 生成缩略图：200x200，裁剪模式
      await sharp(originalPath)
        .resize(200, 200, {
          fit: "cover",
          position: "center",
        })
        .jpeg({ quality: 80 })
        .toFile(thumbnailPath);

      // 生成中等尺寸：800x800，保持比例
      const mediumPath = `./uploads/images/medium-${file.filename}`;
      await sharp(originalPath)
        .resize(800, 800, {
          fit: "inside",
          withoutEnlargement: true,
        })
        .jpeg({ quality: 85 })
        .toFile(mediumPath);

      return {
        success: true,
        data: {
          original: `/uploads/images/${file.filename}`,
          thumbnail: `/uploads/images/thumbnails/${file.filename}`,
          medium: `/uploads/images/medium-${file.filename}`,
        },
      };
    } catch (error) {
      throw new BadRequestException("图片处理失败");
    }
  }
}
```

### 6.2 图片裁剪

```typescript
@Post('crop')
@UseInterceptors(FileInterceptor('image'))
async cropImage(
  @UploadedFile() file: Express.Multer.File,
  @Body() cropDto: { x: number; y: number; width: number; height: number },
) {
  const outputPath = `./uploads/cropped/${Date.now()}-${file.originalname}`;

  await sharp(file.buffer)
    .extract({
      left: cropDto.x,
      top: cropDto.y,
      width: cropDto.width,
      height: cropDto.height,
    })
    .toFile(outputPath);

  return { success: true, url: outputPath };
}
```

### 6.3 图片压缩

```typescript
@Post('compress')
@UseInterceptors(FileInterceptor('image'))
async compressImage(@UploadedFile() file: Express.Multer.File) {
  const outputPath = `./uploads/compressed/${Date.now()}-${file.originalname}`;

  // 获取原始图片信息
  const metadata = await sharp(file.buffer).metadata();

  // 根据原始大小决定压缩比例
  let quality = 80;
  if (metadata.size && metadata.size > 5 * 1024 * 1024) {
    quality = 60; // 大于 5MB，压缩更多
  }

  await sharp(file.buffer)
    .jpeg({ quality })
    .toFile(outputPath);

  return { success: true, url: outputPath };
}
```

### 7.2 创建 OSS 服务

```typescript
// src/oss/oss.service.ts
import { Injectable } from "@nestjs/common";
import { ConfigService } from "@nestjs/config";
import * as OSS from "ali-oss";

@Injectable()
export class OssService {
  private client: OSS;

  constructor(private configService: ConfigService) {
    this.client = new OSS({
      region: this.configService.get("OSS_REGION"),
      accessKeyId: this.configService.get("OSS_ACCESS_KEY_ID"),
      accessKeySecret: this.configService.get("OSS_ACCESS_KEY_SECRET"),
      bucket: this.configService.get("OSS_BUCKET"),
    });
  }

  /**
   * 上传文件到 OSS
   * @param filename 文件名
   * @param file 文件 Buffer
   * @param folder 文件夹路径
   */
  async uploadFile(filename: string, file: Buffer, folder: string = "uploads") {
    const key = `${folder}/${Date.now()}-${filename}`;

    try {
      const result = await this.client.put(key, file);
      return {
        url: result.url,
        name: result.name,
      };
    } catch (error) {
      throw new Error(`OSS 上传失败: ${error.message}`);
    }
  }

  /**
   * 删除文件
   */
  async deleteFile(filename: string) {
    try {
      await this.client.delete(filename);
      return { success: true };
    } catch (error) {
      throw new Error(`OSS 删除失败: ${error.message}`);
    }
  }

  /**
   * 获取文件签名 URL（临时访问）
   */
  async getSignedUrl(filename: string, expires: number = 3600) {
    try {
      const url = this.client.signatureUrl(filename, { expires });
      return url;
    } catch (error) {
      throw new Error(`生成签名 URL 失败: ${error.message}`);
    }
  }
}
```

## 八、文件验证

### 8.1 文件类型验证

```typescript
// src/common/utils/file.utils.ts
export const ALLOWED_IMAGE_TYPES = [
  "image/jpeg",
  "image/jpg",
  "image/png",
  "image/gif",
  "image/webp",
];

export const ALLOWED_DOCUMENT_TYPES = [
  "application/pdf",
  "application/msword",
  "application/vnd.openxmlformats-officedocument.wordprocessingml.document",
  "text/plain",
];

export const FILE_SIZE_LIMITS = {
  avatar: 5 * 1024 * 1024, // 5MB
  image: 10 * 1024 * 1024, // 10MB
  document: 20 * 1024 * 1024, // 20MB
};

export function validateFileType(
  file: Express.Multer.File,
  allowedTypes: string[]
): boolean {
  return allowedTypes.includes(file.mimetype);
}

export function validateFileSize(
  file: Express.Multer.File,
  maxSize: number
): boolean {
  return file.size <= maxSize;
}
```

### 8.2 自定义文件过滤器

```typescript
// src/common/interceptors/file-upload.interceptor.ts
import { BadRequestException } from "@nestjs/common";
import { FileInterceptor } from "@nestjs/platform-express";
import { diskStorage } from "multer";
import { extname } from "path";
import {
  ALLOWED_IMAGE_TYPES,
  FILE_SIZE_LIMITS,
  validateFileType,
  validateFileSize,
} from "../utils/file.utils";

export function ImageFileInterceptor(fieldName: string) {
  return FileInterceptor(fieldName, {
    storage: diskStorage({
      destination: "./uploads/images",
      filename: (req, file, cb) => {
        const uniqueSuffix = Date.now() + "-" + Math.round(Math.random() * 1e9);
        const ext = extname(file.originalname);
        cb(null, `${uniqueSuffix}${ext}`);
      },
    }),
    fileFilter: (req, file, cb) => {
      // 验证文件类型
      if (!validateFileType(file, ALLOWED_IMAGE_TYPES)) {
        return cb(new BadRequestException("不支持的文件类型"), false);
      }

      // 验证文件大小
      if (!validateFileSize(file, FILE_SIZE_LIMITS.image)) {
        return cb(new BadRequestException("文件大小超过限制（10MB）"), false);
      }

      cb(null, true);
    },
    limits: {
      fileSize: FILE_SIZE_LIMITS.image,
    },
  });
}
```

```typescript
  ExceptionFilter,
  Catch,
  ArgumentsHost,
  HttpException,
} from "@nestjs/common";
import { Response } from "express";

@Catch(Error)
export class FileUploadFilter implements ExceptionFilter {
  catch(exception: Error, host: ArgumentsHost) {
    const ctx = host.switchToHttp();
    const response = ctx.getResponse<Response>();

    if (exception.name === "MulterError") {
      // Multer 错误
      let message = "文件上传失败";
      if (exception.message === "File too large") {
        message = "文件大小超过限制";
      } else if (exception.message === "Unexpected field") {
        message = "字段名错误";
      }

      return response.status(400).json({
        success: false,
        message,
        code: "MULTER_ERROR",
      });
    }

    // 其他错误
    return response.status(500).json({
      success: false,
      message: "服务器错误",
      code: "INTERNAL_SERVER_ERROR",
    });
  }
}
```

```typescript
// src/users/users.controller.ts
import {
  Controller,
  Post,
  UseInterceptors,
  UploadedFile,
  ParseFilePipe,
  MaxFileSizeValidator,
  FileTypeValidator,
  BadRequestException,
} from "@nestjs/common";
import { FileInterceptor } from "@nestjs/platform-express";
import { diskStorage } from "multer";
import { extname } from "path";
import { UsersService } from "./users.service";

@Controller("users")
export class UsersController {
  constructor(private usersService: UsersService) {}

  @Post(":id/avatar")
  @UseInterceptors(
    FileInterceptor("avatar", {
      storage: diskStorage({
        destination: "./uploads/avatars",
        filename: (req, file, cb) => {
          // 使用用户 ID 作为文件名前缀
          const userId = req.params.id;
          const ext = extname(file.originalname);
          cb(null, `user-${userId}${ext}`);
        },
      }),
      fileFilter: (req, file, cb) => {
        if (!file.mimetype.startsWith("image/")) {
          return cb(new BadRequestException("只允许上传图片"), false);
        }
        cb(null, true);
      },
      limits: {
        fileSize: 5 * 1024 * 1024, // 5MB
      },
    })
  )
  async uploadAvatar(
    @Param("id") id: string,
    @UploadedFile(
      new ParseFilePipe({
        validators: [
          new MaxFileSizeValidator({ maxSize: 5 * 1024 * 1024 }),
          new FileTypeValidator({ fileType: /(jpg|jpeg|png|gif)$/ }),
        ],
        exceptionFactory: (error) => {
          throw new BadRequestException("文件验证失败");
        },
      })
    )
    file: Express.Multer.File
  ) {
    // 更新数据库中的头像路径
    const avatarUrl = await this.usersService.updateAvatar(id, file.filename);

    return {
      success: true,
      message: "头像上传成功",
      data: {
        avatarUrl,
      },
    };
  }
}
```

```typescript
// 前端组件
import { useState } from 'react';

export function AvatarUpload() {
  const [uploading, setUploading] = useState(false);
  const [preview, setPreview] = useState<string | null>(null);

  const handleFileChange = async (e: React.ChangeEvent<HTMLInputElement>) => {
    const file = e.target.files?.[0];
    if (!file) return;

    // 客户端验证
    if (!file.type.startsWith('image/')) {
      alert('请选择图片文件');
      return;
    }

    if (file.size > 5 * 1024 * 1024) {
      alert('文件大小不能超过 5MB');
      return;
    }

    // 预览
    setPreview(URL.createObjectURL(file));

    // 上传
    setUploading(true);
    try {
      const formData = new FormData();
      formData.append('avatar', file);

      const response = await fetch(`/api/users/${userId}/avatar`, {
        method: 'POST',
        body: formData,
      });

      const result = await response.json();

      if (result.success) {
        alert('头像上传成功！');
        // 刷新用户信息
      } else {
        alert(result.message);
      }
    } catch (error) {
      alert('上传失败，请重试');
    } finally {
      setUploading(false);
    }
  };

  return (
    <div>
      <input
        type="file"
        accept="image/*"
        onChange={handleFileChange}
        disabled={uploading}
      />
      {preview && (
        <img src={preview} alt="预览" style={{ width: 200, height: 200 }} />
      )}
      {uploading && <p>上传中...</p>}
    </div>
  );
}
```

```typescript
// 定期清理临时文件
import * as fs from 'fs';
import * as path from 'path';

@Cron('0 0 * * *') // 每天凌晨执行
async cleanTempFiles() {
  const tempDir = './uploads/temp';
  const files = fs.readdirSync(tempDir);
  const now = Date.now();
  const maxAge = 24 * 60 * 60 * 1000; // 24 小时

  for (const file of files) {
    const filePath = path.join(tempDir, file);
    const stats = fs.statSync(filePath);

    if (now - stats.mtimeMs > maxAge) {
      fs.unlinkSync(filePath);
      this.logger.log(`删除临时文件: ${file}`);
    }
  }
}
```
