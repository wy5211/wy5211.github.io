## 二、CI/CD 集成

### 2.1 GitHub Actions

```yaml
# .github/workflows/claude-code-review.yml
name: Claude Code Review

on:
  pull_request:
    types: [opened, synchronize]

jobs:
  review:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout code
        uses: actions/checkout@v3
        with:
          fetch-depth: 0

      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: "20"

      - name: Install Claude Code CLI
        run: npm install -g @anthropic-ai/claude-code

      - name: Configure Claude Code
        env:
          ANTHROPIC_API_KEY: ${{ secrets.ANTHROPIC_API_KEY }}
        run: |
          claude config set api_key "$ANTHROPIC_API_KEY"

      - name: Run code review
        run: |
          claude review --pr ${{ github.event.number }} \
            --output review.md \
            --format markdown

      - name: Comment PR
        uses: actions/github-script@v6
        with:
          script: |
            const fs = require('fs');
            const review = fs.readFileSync('review.md', 'utf8');
            github.rest.issues.createComment({
              issue_number: context.issue.number,
              owner: context.repo.owner,
              repo: context.repo.repo,
              body: review
            });

      - name: Check for critical issues
        run: |
          if grep -q "Critical" review.md; then
            echo "❌ Critical issues found!"
            exit 1
          fi
```

### 3.2 Playwright 集成

```typescript
// e2e/claude-helper.ts
import { Page, test } from "@playwright/test";

export class ClaudeE2EHelper {
  constructor(private page: Page) {}

  // 智能等待元素
  async smartWait(selector: string) {
    // 使用 Claude Code 理解元素加载逻辑
    const strategy = await this.analyzeElementLoad(selector);

    switch (strategy.type) {
      case "immediate":
        return this.page.waitForSelector(selector, { state: "attached" });
      case "lazy":
        return this.page.waitForSelector(selector, { state: "visible" });
      case "async":
        // 等待网络请求完成
        return this.page.waitForLoadState("networkidle");
    }
  }

  // 智能表单填写
  async smartFill(formSelector: string, data: Record<string, any>) {
    // 分析表单结构
    const formStructure = await this.analyzeForm(formSelector);

    // 根据字段类型填写
    for (const [field, value] of Object.entries(data)) {
      const fieldInfo = formStructure.fields[field];

      switch (fieldInfo.type) {
        case "select":
          await this.page.selectOption(
            `${formSelector} [name="${field}"]`,
            value
          );
          break;
        case "checkbox":
          if (value) {
            await this.page.check(`${formSelector} [name="${field}"]`);
          }
          break;
        default:
          await this.page.fill(`${formSelector} [name="${field}"]`, value);
      }
    }
  }

  // 自动化截图对比
  async compareScreenshot(name: string) {
    const screenshot = await this.page.screenshot();

    // 使用 Claude Code 分析视觉差异
    const comparison = await this.analyzeScreenshot(screenshot, {
      baseline: `screenshots/baseline/${name}.png`,
      threshold: 0.05,
    });

    if (comparison.difference > 0.05) {
      // 失败并高亮差异区域
      await this.highlightDifferences(comparison.diffs);
      throw new Error(`Screenshot mismatch: ${comparison.difference * 100}%`);
    }
  }
}

// 使用示例
test("user registration flow", async ({ page }) => {
  const helper = new ClaudeE2EHelper(page);

  await page.goto("/register");

  // 智能填写表单
  await helper.smartFill("#registration-form", {
    email: "test@example.com",
    password: "SecurePass123",
    acceptTerms: true,
  });

  // 智能提交
  await helper.smartClick("#register-button");

  // 验证结果
  await helper.smartWait("#success-message");
  await helper.compareScreenshot("registration-success");
});
```

### 5.1 Jira 集成

```typescript
// claude-jira-integration.ts
import ClaudeCode from "@anthropic-ai/claude-code";

const claude = new ClaudeCode({
  apiKey: process.env.ANTHROPIC_API_KEY,
});

// 自动创建 Jira 任务
async function createJiraTaskFromClaude(prNumber: number) {
  // 分析 PR
  const analysis = await claude.analyzePR(prNumber);

  // 提取任务信息
  const task = {
    project: "BACKEND",
    summary: analysis.title,
    description: analysis.description,
    issueType: "Story",
    priority: analysis.priority,
    labels: analysis.labels,
  };

  // 创建 Jira 任务
  const jiraResponse = await fetch(
    "https://api.atlassian.com/ex/jira/rest/api/2/issue",
    {
      method: "POST",
      headers: {
        Authorization: `Basic ${process.env.JIRA_TOKEN}`,
        "Content-Type": "application/json",
      },
      body: JSON.stringify(task),
    }
  );

  return jiraResponse.json();
}

// 自动更新任务状态
async function updateTaskStatus(issueId: string, status: string) {
  await fetch(
    `https://api.atlassian.com/ex/jira/rest/api/2/issue/${issueId}/transitions`,
    {
      method: "POST",
      headers: {
        Authorization: `Basic ${process.env.JIRA_TOKEN}`,
        "Content-Type": "application/json",
      },
      body: JSON.stringify({
        transition: { name: status },
      }),
    }
  );
}
```
