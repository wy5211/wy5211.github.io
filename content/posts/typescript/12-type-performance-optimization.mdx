## 前言

随着项目规模的增长，TypeScript 类型检查的速度可能会成为开发瓶颈。复杂的类型定义、过度的类型操作、不当的泛型使用都可能导致编译器性能问题。

本文将介绍如何编写高性能的类型定义，避免常见的性能陷阱，确保你的 TypeScript 项目保持快速的类型检查。

## 核心概念

### TypeScript 编译器的性能瓶颈

1. **类型实例化**：泛型的实例化成本
2. **类型分配**：创建新的类型对象
3. **类型推断**：复杂的类型推断算法
4. **类型联合**：大型联合类型的展开
5. **递归深度**：深度递归的类型操作

### 性能分析工具

## 高级技巧

### 技巧 1：避免过度的条件类型嵌套

```typescript
// ❌ 性能差：深层嵌套的条件类型
type DeepValue<T> = T extends object
  ? T extends { value: infer V }
    ? V extends object
      ? V extends { value: infer W }
        ? W
        : V
      : V
    : T
  : T;

// ✅ 性能好：使用辅助类型简化
type Value<T> = T extends { value: infer V } ? V : T;

type DeepValueOptimized<T> = T extends object ? Value<Value<T>> : T;
```

---

### 技巧 2：缓存类型实例化

```typescript
// ❌ 性能差：多次实例化同一个复杂类型
type Bad<T> = {
  a: ComplexType<T>;
  b: ComplexType<T>;
  c: ComplexType<T>;
};

// ✅ 性能好：使用类型别名缓存
type CachedComplexType<T> = ComplexType<T>;

type Good<T> = {
  a: CachedComplexType<T>;
  b: CachedComplexType<T>;
  c: CachedComplexType<T>;
};
```

---

### 技巧 3：限制递归深度

```typescript
// ❌ 性能差：无限递归
type DeepKeys<T> =
  T extends object
    ? {
        [K in keyof T]: K | DeepKeys<T[K]>
      }[keyof T]
    : never

// ✅ 性能好：限制递归深度
type DeepKeysLimited<T, Depth extends number = 5> =
  Depth extends 0
    ? never
    : T extends object
      ? {
          [K in keyof T]: K | DeepKeysLimited<T[K], Depth extends 10 ? 9 : Depth - 1>
        }[keyof T]
      : never
```

---

### 技巧 4：避免大型联合类型的展开

```typescript
// ❌ 性能差：大型联合类型
type AllTypes =
  | string
  | number
  | boolean
  | null
  | undefined
  | Function
  | Date
  | RegExp
  | Error
  | Array<any>
  | Object
  | Promise<any>;
// ... 更多类型

// ✅ 性能好：使用基础类型和条件类型
type Value<T> = T extends AllTypes ? T : never;

// 或者使用类型别名
type BasicType = string | number | boolean | null | undefined;

type ComplexType =
  | Function
  | Date
  | RegExp
  | Error
  | Array<any>
  | Object
  | Promise<any>;

type AllTypesOptimized = BasicType | ComplexType;
```

---

### 技巧 5：使用分布式条件类型

```typescript
// ❌ 性能差：非分布式条件类型
type NonDistributed<T> = [T] extends [string] ? "string" : never;

type Result = NonDistributed<"a" | "b" | "c">;
// 'string' (不分发)

// ✅ 性能好：分布式条件类型
type Distributed<T> = T extends string ? "string" : never;

type Result2 = Distributed<"a" | "b" | "c">;
// 'string' | 'string' | 'string' = 'string' (分发后简化)
```

---

### 技巧 6：优化映射类型

```typescript
// ❌ 性能差：复杂的映射类型
type BadMapped<T> = {
  [K in keyof T as T[K] extends Function
    ? never
    : K extends string
      ? `get${Capitalize<K>}`
      : never
  ]: T[K]
}

// ✅ 性能好：分步骤处理
type FunctionKeys<T> = {
  [K in keyof T]: T[K] extends Function ? K : never
}[keyof T]

type NonFunctionKeys<T> = Exclude<keyof T, FunctionKeys<T>>

type GetterKeys<T> = {
  [K in NonFunctionKeys<T> as K extends string
    ? `get${Capitalize<K>}`
    : never
]: T[K]
```

## 实战优化

### 优化大型项目的类型定义

```typescript
// ❌ 优化前：一次性处理所有类型
type AllData<T> = {
  [K in keyof T]: T[K] extends Array<infer U>
    ? Array<DeepReadonly<U>>
    : T[K] extends object
      ? DeepReadonly<T[K]>
      : T[K];
};

// ✅ 优化后：分步处理
type ProcessArray<T> = T extends Array<infer U> ? Array<DeepReadonly<U>> : T;

type ProcessObject<T> = T extends object ? DeepReadonly<T> : T;

type AllDataOptimized<T> = {
  [K in keyof T]: ProcessArray<ProcessObject<T[K]>>;
};
```

### 优化条件类型链

```typescript
// ❌ 优化前：长链的条件类型
type LongChain<T> = T extends string
  ? "string"
  : T extends number
    ? "number"
    : T extends boolean
      ? "boolean"
      : T extends null
        ? "null"
        : T extends undefined
          ? "undefined"
          : T extends Function
            ? "function"
            : "object";

// ✅ 优化后：使用对象映射
type TypeMap = {
  string: "string";
  number: "number";
  boolean: "boolean";
  null: "null";
  undefined: "undefined";
  function: "function";
};

type ShortChain<T> = T extends keyof TypeMap ? TypeMap[T] : "object";
```

### 优化模板字面量类型

```typescript
// ❌ 优化前：复杂的模板字面量操作
type ComplexTemplate<T> =
  T extends `${infer A}_${infer B}_${infer C}_${infer D}_${infer E}`
    ? `${A}-${B}-${C}-${D}-${E}`
    : T;

// ✅ 优化后：使用正则表达式或简化逻辑
type SimpleTemplate<T> = T extends `${infer A}_${infer Rest}`
  ? `${A}-${SimpleTemplate<Rest>}`
  : T;

// 或者使用辅助类型
type ReplaceUnderscore<T> = T extends `${infer A}_${infer B}`
  ? ReplaceUnderscore<`${A}-${B}`>
  : T;
```

## 性能监控

### 使用性能分析工具

```typescript
// 创建 performance.ts
import * as ts from "typescript";

function measurePerformance() {
  const program = ts.createProgram(["tsconfig.json"]);

  const start = Date.now();
  const diagnostics = program.getSemanticDiagnostics();
  const end = Date.now();

  console.log(`Type checking took ${end - start}ms`);
  console.log(`Found ${diagnostics.length} diagnostics`);

  // 分析最复杂的类型
  const files = program.getSourceFiles();
  for (const file of files) {
    const lineCount = file.getLineStarts().length;
    console.log(`${file.fileName}: ${lineCount} lines`);
  }
}

measurePerformance();
```

## 避坑指南

### ❌ 错误 1：过度使用条件类型

```typescript
// ❌ 性能差：复杂的条件类型
type ComplexType<T> = T extends string
  ? "string"
  : T extends number
    ? "number"
    : T extends boolean
      ? "boolean"
      : "unknown";

// ✅ 性能好：使用类型映射
type SimpleTypeMap = {
  string: "string";
  number: "number";
  boolean: "boolean";
};

type SimpleType<T extends keyof SimpleTypeMap> = SimpleTypeMap[T];
```

---

### ❌ 错误 2：忽略类型推断的成本

```typescript
// ❌ 性能差：复杂的类型推断
function process<T>(
  value: T
): T extends Array<infer U> ? U[] : T extends Promise<infer V> ? V : T {
  return value as any;
}

// ✅ 性能好：显式类型参数
function processExplicit<T, U = T extends Array<infer V> ? V : T>(value: T): U {
  return value as any;
}
```

---

### ❌ 错误 3：滥用递归类型

```typescript
// ❌ 性能差：深度递归
type DeepReadonly<T> = {
  readonly [P in keyof T]: T[P] extends object
    ? DeepReadonly<T[P]>
    : T[P]
}

// ✅ 性能好：限制深度或使用辅助类型
type DeepReadonlyLimited<T, Depth extends number = 3> =
  Depth extends 0
    ? T
    : {
        readonly [P in keyof T]: T[P] extends object
          ? DeepReadonlyLimited<T[P], Depth extends 10 ? 9 : Depth - 1>
          : T[P]
      }
```

---

### ❌ 错误 4：大型联合类型的滥用

```typescript
// ❌ 性能差：大型联合类型
type AllPossibleValues = "value1" | "value2" | "value3" | "value4" | "value5";
// ... 100+ 个值

// ✅ 性能好：使用枚举或常量对象
enum Values {
  Value1 = "value1",
  Value2 = "value2",
  Value3 = "value3",
}

type AllPossibleValuesOptimized = Values;
```
