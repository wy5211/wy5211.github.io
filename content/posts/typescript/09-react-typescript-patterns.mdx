## 前言

React 和 TypeScript 的结合已经成为现代前端开发的标准配置。但仅仅掌握基础的类型定义是不够的，我们需要学习如何在 React 中应用 TypeScript 的高级特性，构建真正类型安全的组件和应用。

本文将深入探讨泛型组件、高级 Hooks 类型推导、表单验证等实用模式。

## 核心概念

### 组件 Props 类型设计

良好的 Props 类型设计是类型安全的基础：

```typescript
// ✅ 使用 interface 定义组件 Props
interface ButtonProps {
  variant?: 'primary' | 'secondary' | 'danger'
  size?: 'small' | 'medium' | 'large'
  disabled?: boolean
  onClick?: () => void
  children: React.ReactNode
}

function Button({ variant = 'primary', size = 'medium', children, ...props }: ButtonProps) {
  return (
    <button
      className={`btn btn-${variant} btn-${size}`}
      {...props}
    >
      {children}
    </button>
  )
}
```

### 使用 Polymorphic Components

创建可以渲染不同 HTML 元素的组件：

```typescript
type AsProps<T extends React.ElementType> = {
  as?: T
}

type PropsToAs<T extends AsProps<React.ElementType>, P> = Omit<
  P,
  keyof AsProps<T>
> &
  AsProps<T>

type PolymorphicComponentProps<
  T extends React.ElementType,
  P = {}
> = PropsToAs<T, P> & React.ComponentPropsWithoutRef<T>

// 泛型组件
type PolymorphicComponent<T extends React.ElementType, P = {}> =
  React.ForwardRefExoticComponent<
    PolymorphicComponentProps<T, P> & React.RefAttributes<HTMLElement>
  >

// 使用示例
function Box<C extends React.ElementType = 'div'>(
  { as, ...props }: PolymorphicComponentProps<C, { className?: string }>,
  ref: React.Ref<HTMLElement>
) {
  const Component = as || 'div'
  return <Component ref={ref} {...props} />
}

const ForwardedBox = React.forwardRef(Box)

// ✅ 类型安全
ForwardedBox({ as: 'button', className: 'btn' }) // button 元素
ForwardedBox({ as: 'a', href: '/' }) // a 元素，但有 href 的类型错误
```

### 技巧 1：泛型组件实现

创建可以处理不同数据类型的组件：

```typescript
// 泛型 Table 组件
interface TableProps<T> {
  data: T[]
  columns: Column<T>[]
  onRowClick?: (row: T) => void
  keyExtractor: (row: T) => string | number
}

interface Column<T> {
  key: keyof T
  title: string
  render?: (value: T[keyof T], row: T) => React.ReactNode
  sortable?: boolean
}

function Table<T>({ data, columns, onRowClick, keyExtractor }: TableProps<T>) {
  return (
    <table>
      <thead>
        <tr>
          {columns.map((col) => (
            <th key={String(col.key)}>{col.title}</th>
          ))}
        </tr>
      </thead>
      <tbody>
        {data.map((row) => (
          <tr
            key={keyExtractor(row)}
            onClick={() => onRowClick?.(row)}
          >
            {columns.map((col) => (
              <td key={String(col.key)}>
                {col.render
                  ? col.render(row[col.key], row)
                  : String(row[col.key])}
              </td>
            ))}
          </tr>
        ))}
      </tbody>
    </table>
  )
}

// 使用示例
interface User {
  id: number
  name: string
  email: string
  role: 'admin' | 'user'
}

const userColumns: Column<User>[] = [
  { key: 'id', title: 'ID' },
  { key: 'name', title: 'Name' },
  {
    key: 'email',
    title: 'Email',
    render: (email) => <a href={`mailto:${email}`}>{email}</a>
  },
  {
    key: 'role',
    title: 'Role',
    render: (role) => (
      <span className={`badge badge-${role}`}>{role}</span>
    )
  }
]

function UsersPage() {
  const users: User[] = [
    { id: 1, name: 'Alice', email: 'alice@example.com', role: 'admin' },
    { id: 2, name: 'Bob', email: 'bob@example.com', role: 'user' }
  ]

  return (
    <Table
      data={users}
      columns={userColumns}
      keyExtractor={(user) => user.id}
      onRowClick={(user) => console.log('Clicked:', user)}
    />
  )
}
```

### 技巧 2：类型安全的表单组件

构建完整的类型安全表单系统：

```typescript
// 表单字段类型
type FieldValue = string | number | boolean | Date | File | null

interface FieldConfig<T> {
  initialValue: T
  validation?: (value: T) => string | undefined
  transform?: (value: T) => T
}

type FormFields<T extends Record<string, FieldValue>> = {
  [K in keyof T]: FieldConfig<T[K]>
}

// 表单状态
type FormState<T> = {
  values: T
  errors: Partial<Record<keyof T, string>>
  touched: Partial<Record<keyof T, boolean>>
}

// 表单 Hook
function useForm<T extends Record<string, FieldValue>>(
  fields: FormFields<T>
) {
  const [form, setForm] = React.useState<FormState<T>>({
    values: Object.keys(fields).reduce((acc, key) => {
      acc[key as keyof T] = fields[key as keyof T].initialValue
      return acc
    }, {} as T),
    errors: {},
    touched: {}
  })

  const setValue = <K extends keyof T>(field: K, value: T[K]) => {
    const config = fields[field]
    const transformedValue = config.transform
      ? config.transform(value)
      : value

    setForm((prev) => ({
      ...prev,
      values: {
        ...prev.values,
        [field]: transformedValue
      }
    }))

    // 验证字段
    if (config.validation) {
      const error = config.validation(transformedValue)
      setForm((prev) => ({
        ...prev,
        errors: {
          ...prev.errors,
          [field]: error
        }
      }))
    }
  }

  const setError = <K extends keyof T>(field: K, error: string) => {
    setForm((prev) => ({
      ...prev,
      errors: {
        ...prev.errors,
        [field]: error
      }
    }))
  }

  const setTouched = <K extends keyof T>(field: K, touched: boolean) => {
    setForm((prev) => ({
      ...prev,
      touched: {
        ...prev.touched,
        [field]: touched
      }
    }))
  }

  const validate = (): boolean => {
    const errors: Partial<Record<keyof T, string>> = {}

    for (const field in fields) {
      const config = fields[field]
      if (config.validation) {
        const error = config.validation(form.values[field])
        if (error) {
          errors[field] = error
        }
      }
    }

    setForm((prev) => ({
      ...prev,
      errors
    }))

    return Object.keys(errors).length === 0
  }

  const reset = () => {
    setForm({
      values: Object.keys(fields).reduce((acc, key) => {
        acc[key as keyof T] = fields[key as keyof T].initialValue
        return acc
      }, {} as T),
      errors: {},
      touched: {}
    })
  }

  return {
    form,
    setValue,
    setError,
    setTouched,
    validate,
    reset
  }
}

// 表单字段组件
interface FormFieldProps<T, K extends keyof T> {
  name: K
  label: string
  form: ReturnType<typeof useForm<T>>
  type?: 'text' | 'email' | 'password' | 'number'
  placeholder?: string
}

function FormField<T, K extends keyof T>({
  name,
  label,
  form,
  type = 'text',
  placeholder
}: FormFieldProps<T, K>) {
  const error = form.form.errors[name]
  const touched = form.form.touched[name]

  return (
    <div className="form-field">
      <label>{label}</label>
      <input
        type={type}
        value={form.form.values[name] as string}
        onChange={(e) =>
          form.setValue(name, e.target.value as T[K])
        }
        onBlur={() => form.setTouched(name, true)}
        placeholder={placeholder}
      />
      {touched && error && <span className="error">{error}</span>}
    </div>
  )
}

// 使用示例
interface LoginFormData {
  email: string
  password: string
  remember: boolean
}

const loginFormConfig: FormFields<LoginFormData> = {
  email: {
    initialValue: '',
    validation: (value) => {
      if (!value) return 'Email is required'
      if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value)) {
        return 'Invalid email format'
      }
      return undefined
    }
  },
  password: {
    initialValue: '',
    validation: (value) => {
      if (!value) return 'Password is required'
      if (value.length < 8) return 'Password must be at least 8 characters'
      return undefined
    }
  },
  remember: {
    initialValue: false
  }
}

function LoginForm() {
  const form = useForm(loginFormConfig)

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault()
    if (form.validate()) {
      console.log('Form is valid:', form.form.values)
      // 提交逻辑
    }
  }

  return (
    <form onSubmit={handleSubmit}>
      <FormField
        name="email"
        label="Email"
        form={form}
        type="email"
        placeholder="Enter your email"
      />
      <FormField
        name="password"
        label="Password"
        form={form}
        type="password"
        placeholder="Enter your password"
      />
      <button type="submit">Login</button>
    </form>
  )
}
```

### 技巧 3：高级自定义 Hook

创建类型推导能力强大的自定义 Hook：

```typescript
// useLocalStorage Hook
function useLocalStorage<T>(
  key: string,
  initialValue: T
): [T, (value: T | ((prev: T) => T)) => void] {
  const [storedValue, setStoredValue] = React.useState<T>(() => {
    try {
      const item = window.localStorage.getItem(key)
      return item ? JSON.parse(item) : initialValue
    } catch (error) {
      console.error(`Error loading ${key} from localStorage:`, error)
      return initialValue
    }
  })

  const setValue = (value: T | ((prev: T) => T)) => {
    try {
      const valueToStore =
        value instanceof Function ? value(storedValue) : value
      setStoredValue(valueToStore)
      window.localStorage.setItem(key, JSON.stringify(valueToStore))
    } catch (error) {
      console.error(`Error saving ${key} to localStorage:`, error)
    }
  }

  return [storedValue, setValue]
}

// useFetch Hook
type FetchState<T> =
  | { status: 'idle' }
  | { status: 'loading' }
  | { status: 'success'; data: T }
  | { status: 'error'; error: Error }

function useFetch<T>(url: string): FetchState<T> {
  const [state, setState] = React.useState<FetchState<T>>({
    status: 'idle'
  })

  React.useEffect(() => {
    setState({ status: 'loading' })

    fetch(url)
      .then((res) => res.json())
      .then((data) => setState({ status: 'success', data }))
      .catch((error) => setState({ status: 'error', error }))
  }, [url])

  return state
}

// 使用示例
interface User {
  id: number
  name: string
  email: string
}

function UserProfile() {
  const [user, setUser] = useLocalStorage<User>('user', {
    id: 0,
    name: '',
    email: ''
  })

  // user 类型被正确推导为 User
  return (
    <div>
      <input
        value={user.name}
        onChange={(e) =>
          setUser({ ...user, name: e.target.value })
        }
      />
    </div>
  )
}
```

### 技巧 4：类型安全的 Context

创建类型安全的 React Context：

```typescript
// 创建类型安全的 Context
type createContext<T>() => [
  React.Context<T | undefined>,
  () => T
]

function createSafeContext<T>(defaultValue: T) {
  const context = React.createContext<T | undefined>(defaultValue)

  const useContext = (): T => {
    const value = React.useContext(context)
    if (value === undefined) {
      throw new Error('useContext must be used within a Provider')
    }
    return value
  }

  return [context, useContext] as const
}

// 使用示例
interface UserContextValue {
  user: User | null
  login: (email: string, password: string) => Promise<void>
  logout: () => void
}

const [UserContext, useUserContext] = createSafeContext<UserContextValue>({
  user: null,
  login: async () => {},
  logout: () => {}
})

function UserProvider({ children }: { children: React.ReactNode }) {
  const [user, setUser] = React.useState<User | null>(null)

  const login = async (email: string, password: string) => {
    // 登录逻辑
    const userData = await api.login(email, password)
    setUser(userData)
  }

  const logout = () => {
    setUser(null)
  }

  return (
    <UserContext.Provider value={{ user, login, logout }}>
      {children}
    </UserContext.Provider>
  )
}

function UserProfile() {
  const { user, logout } = useUserContext() // 类型安全

  return (
    <div>
      <h1>Welcome, {user?.name}</h1>
      <button onClick={logout}>Logout</button>
    </div>
  )
}
```

// 受控输入组件
type InputProps<T> = {
  value: T
  onChange: (value: T) => void
  type?: 'text' | 'number' | 'email' | 'password'
  placeholder?: string
  disabled?: boolean
}

function TextInput({
  value,
  onChange,
  type = 'text',
  placeholder,
  disabled
}: InputProps<string>) {
  return (
    <input
      type={type}
      value={value}
      onChange={(e) => onChange(e.target.value)}
      placeholder={placeholder}
      disabled={disabled}
    />
  )
}

function NumberInput({
  value,
  onChange,
  placeholder,
  disabled
}: InputProps<number>) {
  return (
    <input
      type="number"
      value={value}
      onChange={(e) => onChange(parseFloat(e.target.value) || 0)}
      placeholder={placeholder}
      disabled={disabled}
    />
  )
}

// 使用示例
function SearchForm() {

  const [minPrice, setMinPrice] = React.useState(0)

  return (
    <form>
      <TextInput
        value={query}
        onChange={setQuery}
        placeholder="Search..."
      />
      <NumberInput
        value={minPrice}
        onChange={setMinPrice}
        placeholder="Min price"
      />
    </form>
  )
}

### 技巧 6：类型受控的选择器组件

```typescript
// 选择器组件
interface SelectOption<T> {
  value: T
  label: string
}

type SelectProps<T> = {
  options: SelectOption<T>[]
  value: T
  onChange: (value: T) => void
  placeholder?: string
}

function Select<T>({ options, value, onChange, placeholder }: SelectProps<T>) {
  return (
    <select
      value={String(value)}
      onChange={(e) => {
        const option = options.find(
          (opt) => String(opt.value) === e.target.value
        )
        if (option) {
          onChange(option.value)
        }
      }}
    >
      {placeholder && <option value="">{placeholder}</option>}
      {options.map((option) => (
        <option key={String(option.value)} value={String(option.value)}>
          {option.label}
        </option>
      ))}
    </select>
  )
}

// 使用示例
type UserRole = 'admin' | 'user' | 'guest'

const roleOptions: SelectOption<UserRole>[] = [
  { value: 'admin', label: 'Administrator' },
  { value: 'user', label: 'User' },
  { value: 'guest', label: 'Guest' }
]

function UserForm() {
  const [role, setRole] = React.useState<UserRole>('user')

  return (
    <Select
      options={roleOptions}
      value={role}
      onChange={setRole}
      placeholder="Select a role"
    />
  )
}
```

### 构建类型安全的 API 客户端

```typescript
// API 客户端类型
type ApiResponse<T> =
  | { success: true; data: T }
  | { success: false; error: string };

type ApiMethod = "GET" | "POST" | "PUT" | "PATCH" | "DELETE";

type ApiEndpoint<TMethod extends ApiMethod, TParams, TResponse> = {
  method: TMethod;
  path: string;
};

// API 客户端
class ApiClient {
  constructor(private baseUrl: string) {}

  async request<TMethod extends ApiMethod, TParams, TResponse>(
    endpoint: ApiEndpoint<TMethod, TParams, TResponse>,
    params?: TParams
  ): Promise<ApiResponse<TResponse>> {
    try {
      const url = new URL(endpoint.path, this.baseUrl);

      let options: RequestInit = {
        method: endpoint.method,
      };

      if (endpoint.method === "GET") {
        if (params) {
          Object.entries(params as any).forEach(([key, value]) => {
            url.searchParams.set(key, String(value));
          });
        }
      } else {
        options.body = JSON.stringify(params);
        options.headers = {
          "Content-Type": "application/json",
        };
      }

      const response = await fetch(url.toString(), options);

      if (!response.ok) {
        return {
          success: false,
          error: `HTTP error! status: ${response.status}`,
        };
      }

      const data = await response.json();
      return { success: true, data };
    } catch (error) {
      return {
        success: false,
        error: error instanceof Error ? error.message : "Unknown error",
      };
    }
  }
}

// API 定义
interface User {
  id: number;
  name: string;
  email: string;
}

interface CreateUserData {
  name: string;
  email: string;
}

interface UpdateUserData {
  name?: string;
  email?: string;
}

const api = new ApiClient("https://api.example.com");

// ✅ 类型安全
const result = await api.request(
  { method: "POST", path: "/users" },
  { name: "Alice", email: "alice@example.com" }
);

if (result.success) {
  // result.data: User
  console.log(result.data.name);
} else {
  console.error(result.error);
}
```

## 避坑指南

### ❌ 错误 1：使用 `any` 作为 Props 类型

```typescript
// ❌ 丢失类型安全
function Component({ data }: { data: any }) {
  return <div>{data.name}</div>
}

// ✅ 使用泛型
function Component<T extends { name: string }>({ data }: { data: T }) {
  return <div>{data.name}</div>
}
```

---

### ❌ 错误 2：忘记处理 undefined 的 Context

```typescript
// ❌ 可能导致运行时错误
const MyContext = React.createContext<User | undefined>(undefined);

function useMyContext() {
  return React.useContext(MyContext); // 可能是 undefined
}

// ✅ 创建类型安全的 Hook
function useMyContext(): User {
  const context = React.useContext(MyContext);
  if (!context) {
    throw new Error("useMyContext must be used within a Provider");
  }
  return context;
}
```

---

### ❌ 错误 3：在事件处理器中类型断言

```typescript
// ❌ 使用类型断言
function handleChange(e: any) {
  setValue(e.target.value);
}

// ✅ 使用正确的事件类型
function handleChange(e: React.ChangeEvent<HTMLInputElement>) {
  setValue(e.target.value);
}
```
