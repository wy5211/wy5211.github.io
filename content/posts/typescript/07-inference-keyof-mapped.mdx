## 前言

TypeScript 4.1 引入了**Key Remapping**（键重映射）功能，让我们在映射类型中使用 `as` 子句来重新映射键。这开启了许多之前无法实现的类型操作可能性。

结合条件类型和类型推断，我们可以创建出非常强大和灵活的类型转换工具。

## 核心概念

### Key Remapping 基础

使用 `as` 子句重新映射类型的键：

```typescript
// 基础重映射
type Getters<T> = {
  [K in keyof T as `get${Capitalize<string & K>}`]: () => T[K];
};

interface User {
  name: string;
  age: number;
}

type UserGetters = Getters<User>;
// {
//   getName: () => string
//   getAge: () => number
// }
```

### 条件过滤

通过返回 `never` 来过滤键：

```typescript
// 只保留函数类型的属性
type OnlyFunctions<T> = {
  [K in keyof T as T[K] extends Function ? K : never]: T[K];
};

interface User {
  id: number;
  name: string;
  greet(): string;
  update(): void;
}

type UserMethods = OnlyFunctions<User>;
// {
//   greet: () => string
//   update: () => void
// }

// 只保留字符串类型的属性
type OnlyStringProperties<T> = {
  [K in keyof T as T[K] extends string ? K : never]: T[K];
};

type StringProps = OnlyStringProperties<User>;
// {
//   name: string
// }
```

// 创建 getter

### 技巧 2：类型安全的路径访问

创建类型安全的对象路径访问：

```typescript
// 获取对象的所有路径
type Paths<T, K extends keyof T = keyof T> = K extends string | number
  ? T[K] extends object
    ? K | `${K}.${Paths<T[K]>}`
    : K
  : never;

// 根据路径获取类型
type PathValue<T, P extends Paths<T>> = P extends `${infer K}.${infer Rest}`
  ? K extends keyof T
    ? PathValue<T[K], Rest>
    : never
  : P extends keyof T
    ? T[P]
    : never;

// 使用示例
interface User {
  id: number;
  name: string;
  profile: {
    email: string;
    address: {
      city: string;
      country: string;
    };
  };
}

type UserPaths = Paths<User>;
// 'id' | 'name' | 'profile' | 'profile.email' | 'profile.address' | 'profile.address.city' | 'profile.address.country'

type CityType = PathValue<User, "profile.address.city">;
// string

// 实现路径访问函数
function getPath<T, P extends Paths<T>>(obj: T, path: P): PathValue<T, P> {
  const keys = (path as string).split(".");
  let result: any = obj;
  for (const key of keys) {
    result = result?.[key];
  }
  return result;
}

const user: User = {
  id: 1,
  name: "Alice",
  profile: {
    email: "alice@example.com",
    address: {
      city: "New York",
      country: "USA",
    },
  },
};

const city = getPath(user, "profile.address.city");
// city 类型: string
```

### 技巧 3：事件处理器类型

创建类型安全的事件系统：

```typescript
// 事件处理器类型
type EventHandlers<T extends Record<string, any>> = {
  [K in keyof T as `on${Capitalize<string & K>}`]: (
    handler: (data: T[K]) => void
  ) => void;
};

// 事件发射器类型
type EventEmitters<T extends Record<string, any>> = {
  [K in keyof T as `emit${Capitalize<string & K>}`]: (data: T[K]) => void;
};

// 完整的事件系统类型
type EventEmitter<T extends Record<string, any>> = EventHandlers<T> &
  EventEmitters<T>;

// 使用示例
interface AppEvents {
  userCreated: { id: number; name: string };
  userUpdated: { id: number; changes: Partial<User> };
  userDeleted: { id: number };
}

type TypedEventEmitter = EventEmitter<AppEvents>;
// {
//   onUserCreated: (handler: (data: { id: number; name: string }) => void) => void
//   onUserUpdated: (handler: (data: { id: number; changes: Partial<User> }) => void) => void
//   onUserDeleted: (handler: (data: { id: number }) => void) => void
//   emitUserCreated: (data: { id: number; name: string }) => void
//   emitUserUpdated: (data: { id: number; changes: Partial<User> }) => void
//   emitUserDeleted: (data: { id: number }) => void
// }

// 实现
class EventManager implements TypedEventEmitter {
  private handlers = new Map<keyof AppEvents, Set<Function>>();

  onUserCreated(handler: (data: AppEvents["userCreated"]) => void): void {
    this.addHandler("userCreated", handler);
  }

  onUserUpdated(handler: (data: AppEvents["userUpdated"]) => void): void {
    this.addHandler("userUpdated", handler);
  }

  onUserDeleted(handler: (data: AppEvents["userDeleted"]) => void): void {
    this.addHandler("userDeleted", handler);
  }

  emitUserCreated(data: AppEvents["userCreated"]): void {
    this.emit("userCreated", data);
  }

  emitUserUpdated(data: AppEvents["userUpdated"]): void {
    this.emit("userUpdated", data);
  }

  emitUserDeleted(data: AppEvents["userDeleted"]): void {
    this.emit("userDeleted", data);
  }

  private addHandler<K extends keyof AppEvents>(
    event: K,
    handler: (data: AppEvents[K]) => void
  ) {
    if (!this.handlers.has(event)) {
      this.handlers.set(event, new Set());
    }
    this.handlers.get(event)!.add(handler);
  }

  private emit<K extends keyof AppEvents>(event: K, data: AppEvents[K]) {
    this.handlers.get(event)?.forEach((handler) => handler(data));
  }
}

// 使用
const emitter = new EventManager();

emitter.onUserCreated((data) => {
  console.log(`User created: ${data.name}`);
});

emitter.emitUserCreated({ id: 1, name: "Alice" });
```

### 技巧 4：类型状态的机转换

创建类型安全的状态机：

```typescript
// 定义状态和事件
type States = {
  idle: { type: "idle" };
  loading: { type: "loading"; progress: number };
  success: { type: "success"; data: any };
  error: { type: "error"; error: Error };
};

type StateTransitions = {
  idle: States["loading"];
  loading: States["success"] | States["error"];
  success: States["idle"];
  error: States["idle"];
};

// 为每个状态创建转换方法
type StateMachine<T extends Record<string, any>> = {
  [K in keyof T as `to${Capitalize<string & K>}`]: (state: T[K]) => void;
};

// 使用示例
type MyStateMachine = StateMachine<StateTransitions>;
// {
//   toIdle: (state: { type: 'idle' }) => void
//   toLoading: (state: { type: 'loading'; progress: number }) => void
//   toSuccess: (state: { type: 'success'; data: any }) => void
//   toError: (state: { type: 'error'; error: Error }) => void
// }

class StateManager {
  private currentState: States[keyof States] = { type: "idle" };

  toIdle(): void {
    this.currentState = { type: "idle" };
  }

  toLoading(state: States["loading"]): void {
    this.currentState = state;
  }

  toSuccess(state: States["success"]): void {
    this.currentState = state;
  }

  toError(state: States["error"]): void {
    this.currentState = state;
  }

  getCurrentState(): States[keyof States] {
    return this.currentState;
  }
}
```

### 技巧 5：类型安全的查询构建器

创建类型安全的数据库查询构建器：

```typescript
// 查询操作符
type QueryOperator = "=" | "!=" | ">" | "<" | ">=" | "<=" | "like" | "in";

// 查询条件
type QueryCondition<T, K extends keyof T> =
  | {
      field: K;
      operator: Extract<QueryOperator, "=" | "!=" | ">" | "<" | ">=" | "<=">;
      value: T[K];
    }
  | { field: K; operator: "like"; value: string }
  | { field: K; operator: "in"; value: T[K][] };

// 查询构建器
type QueryBuilder<T> = {
  where<K extends keyof T>(
    field: K,
    operator: QueryOperator,
    value: T[K]
  ): QueryBuilder<T>;
  orderBy<K extends keyof T>(
    field: K,
    direction?: "ASC" | "DESC"
  ): QueryBuilder<T>;
  limit(count: number): QueryBuilder<T>;
  execute(): Promise<T[]>;
};

// 类型安全的 where 条件
type WhereClauses<T> = {
  [K in keyof T as `whereBy${Capitalize<string & K>}`]: (
    operator: QueryOperator,
    value: T[K]
  ) => QueryBuilder<T>;
};

type OrderByClauses<T> = {
  [K in keyof T as `orderBy${Capitalize<string & K>}`]: (
    direction?: "ASC" | "DESC"
  ) => QueryBuilder<T>;
};

// 组合所有查询方法
type TypedQueryBuilder<T> = WhereClauses<T> &
  OrderByClauses<T> & {
    limit(count: number): QueryBuilder<T>;
    execute(): Promise<T[]>;
  };

// 使用示例
interface User {
  id: number;
  name: string;
  email: string;
  age: number;
  createdAt: Date;
}

class UserQuery implements TypedQueryBuilder<User> {
  private conditions: string[] = [];
  private orders: string[] = [];
  private limitCount?: number;

  whereById(operator: QueryOperator, value: number): QueryBuilder<User> {
    this.conditions.push(`id ${operator} ${value}`);
    return this;
  }

  whereByName(operator: QueryOperator, value: string): QueryBuilder<User> {
    this.conditions.push(`name ${operator} '${value}'`);
    return this;
  }

  whereByEmail(operator: QueryOperator, value: string): QueryBuilder<User> {
    this.conditions.push(`email ${operator} '${value}'`);
    return this;
  }

  whereByAge(operator: QueryOperator, value: number): QueryBuilder<User> {
    this.conditions.push(`age ${operator} ${value}`);
    return this;
  }

  orderById(direction: "ASC" | "DESC" = "ASC"): QueryBuilder<User> {
    this.orders.push(`id ${direction}`);
    return this;
  }

  orderByName(direction: "ASC" | "DESC" = "ASC"): QueryBuilder<User> {
    this.orders.push(`name ${direction}`);
    return this;
  }

  orderByAge(direction: "ASC" | "DESC" = "ASC"): QueryBuilder<User> {
    this.orders.push(`age ${direction}`);
    return this;
  }

  limit(count: number): QueryBuilder<User> {
    this.limitCount = count;
    return this;
  }

  async execute(): Promise<User[]> {
    // 实际执行查询
    return [];
  }
}

// 使用
const query = new UserQuery();
const users = await query
  .whereByAge(">", 18)
  .orderByName("ASC")
  .limit(10)
  .execute();
```

### 技巧 6：类型属性的别名

为类型属性创建别名：

```typescript
// 创建只读别名
type ReadonlyAliases<T> = {
  [K in keyof T as `readonly${Capitalize<string & K>}`]: Readonly<T[K]>;
};

// 创建可写别名
type MutableAliases<T> = {
  [K in keyof T as `mutable${Capitalize<string & K>}`]: T[K];
};

interface Config {
  apiVersion: string;
  maxRetries: number;
  timeout: number;
}

type ConfigWithAliases = Config & ReadonlyAliases<Config>;
// {
//   apiVersion: string
//   maxRetries: number
//   timeout: number
//   readonlyApiVersion: string
//   readonlyMaxRetries: number
//   readonlyTimeout: number
// }

// 高级：创建带前缀的属性
type Prefixed<T, P extends string> = {
  [K in keyof T as `${P}${Capitalize<string & K>}`]: T[K];
};

type Suffixed<T, S extends string> = {
  [K in keyof T as `${K}${Capitalize<S>}`]: T[K];
};

interface User {
  id: number;
  name: string;
  email: string;
}

type UserWithPrefixed = Prefixed<User, "user">;
// {
//   userId: number
//   userName: string
//   userEmail: string
// }

type UserWithSuffixed = Suffixed<User, "Prop">;
// {
//   idProp: number
//   nameProp: string
//   emailProp: string
// }
```

## 实战应用

### 构建类型安全的表单系统

```typescript
// 表单字段类型
type FormField<T> = {
  value: T;
  error?: string;
  touched: boolean;
};

// 创建表单状态类型
type FormState<T> = {
  [K in keyof T as `${K}Value`]: T[K];
} & {
  [K in keyof T as `${K}Error`]: string | undefined;
} & {
  [K in keyof T as `${K}Touched`]: boolean;
};

// 创建表单字段类型
type FormFields<T> = {
  [K in keyof T]: FormField<T[K]>;
};

// 创建表单验证器类型
type FormValidator<T> = {
  [K in keyof T]?: (value: T[K]) => string | undefined;
};

// 使用示例
interface LoginForm {
  email: string;
  password: string;
  remember: boolean;
}

type LoginFormData = FormFields<LoginForm>;

const loginValidators: FormValidator<LoginForm> = {
  email: (value) => {
    if (!value) return "Email is required";
    if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value)) {
      return "Invalid email format";
    }
    return undefined;
  },
  password: (value) => {
    if (!value) return "Password is required";
    if (value.length < 8) return "Password must be at least 8 characters";
    return undefined;
  },
};

class FormManager<T extends Record<string, any>> {
  private fields: FormFields<T>;

  constructor(initialValues: T) {
    this.fields = Object.keys(initialValues).reduce((acc, key) => {
      acc[key as keyof T] = {
        value: initialValues[key as keyof T],
        touched: false,
      };
      return acc;
    }, {} as FormFields<T>);
  }

  setValue<K extends keyof T>(field: K, value: T[K]): void {
    this.fields[field].value = value;
  }

  setTouched<K extends keyof T>(field: K, touched: boolean): void {
    this.fields[field].touched = touched;
  }

  validate<K extends keyof T>(
    field: K,
    validator: (value: T[K]) => string | undefined
  ): boolean {
    const fieldData = this.fields[field];
    const error = validator(fieldData.value);
    fieldData.error = error;
    return !error;
  }

  validateAll(validator: FormValidator<T>): boolean {
    let isValid = true;
    for (const key in validator) {
      const fieldValidator = validator[key];
      if (fieldValidator) {
        const valid = this.validate(key, fieldValidator);
        if (!valid) isValid = false;
      }
    }
    return isValid;
  }

  getValues(): T {
    return Object.keys(this.fields).reduce((acc, key) => {
      acc[key as keyof T] = this.fields[key as keyof T].value;
      return acc;
    }, {} as T);
  }
}

// 使用
const loginForm = new FormManager<LoginForm>({
  email: "",
  password: "",
  remember: false,
});

loginForm.setValue("email", "user@example.com");
loginForm.setValue("password", "password123");

loginForm.setTouched("email", true);
loginForm.validate("email", loginValidators.email!);

if (loginForm.validateAll(loginValidators)) {
  const values = loginForm.getValues();
  console.log("Form is valid:", values);
}
```

## 避坑指南

### ❌ 错误 1：Key Remapping 中的类型断言

```typescript
// ❌ 过度使用类型断言
type Getters<T> = {
  [K in keyof T as `get${Capitalize<string & K>}`]: T[K]
}

// ✅ 更安全的写法
type SafeGetters<T> = {
  [K in keyof T as K extends string
    ? `get${Capitalize<K>}`
    : never
  ]: T[K]
```

---

### ❌ 错误 2：忽略条件中的 never 返回

```typescript
// ❌ 没有正确过滤
type NonNullable<T> = {
  [K in keyof T]: null extends T[K] ? never : K;
};

// ✅ 正确过滤
type NonNullableKeys<T> = {
  [K in keyof T]: null extends T[K] ? never : K;
}[keyof T];
```

---

### ❌ 错误 3：递归映射导致性能问题

```typescript
// ❌ 可能导致性能问题
type DeepKeys<T> = T extends object
  ? {
      [K in keyof T]: K | `${K}.${DeepKeys<T[K]>}`;
    }[keyof T]
  : never;

// ✅ 使用辅助类型优化
type DeepKeysHelper<T, K extends keyof T> =
  | K
  | (T[K] extends object ? `${K}.${DeepKeys<T[K]>}` : K);

type DeepKeys<T> = T extends object ? DeepKeysHelper<T, keyof T> : never;
```

## 总结

### 核心要点

1. **Key Remapping**：使用 `as` 子句重新映射类型的键
2. **条件过滤**：返回 `never` 来过滤不需要的键
3. **模板字面量**：结合 Key Remapping 创建动态键名
4. **类型推断**：在映射中使用 `infer` 推断类型
5. **性能优化**：避免过度复杂的递归映射

### 实用模式

```typescript
// 创建 getter/setter
type Getters<T> = {
  [K in keyof T as `get${Capitalize<string & K>}`]: () => T[K];
};

// 过滤类型
type OnlyFunctions<T> = {
  [K in keyof T as T[K] extends Function ? K : never]: T[K];
};

// 路径类型
type Paths<T, K extends keyof T = keyof T> = K extends string | number
  ? T[K] extends object
    ? K | `${K}.${Paths<T[K]>}`
    : K
  : never;
```
