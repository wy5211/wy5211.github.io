## 前言

TypeScript 的类型系统在编译期提供强大的类型检查，但在运行时，JavaScript 的值是动态的。**类型守卫**（Type Guards）和**断言函数**（Assertion Functions）帮助我们在运行时检查类型，并让 TypeScript 理解这些检查，从而实现精确的类型收窄。

掌握类型守卫和断言技巧，可以让你的代码既类型安全又灵活。

## 核心概念

### 类型收窄（Type Narrowing）

TypeScript 能够根据条件语句自动收窄类型：

```typescript
function process(value: string | number) {
  if (typeof value === "string") {
    // 这里 value 的类型是 string
    console.log(value.toUpperCase());
  } else {
    // 这里 value 的类型是 number
    console.log(value.toFixed(2));
  }
}
```

### 类型谓词（Type Predicates）

使用 `parameterName is Type` 语法自定义类型守卫：

```typescript
function isString(value: unknown): value is string {
  return typeof value === "string";
}

function process(value: unknown) {
  if (isString(value)) {
    // 这里 value 被收窄为 string
    console.log(value.toUpperCase());
  }
}
```

### 断言函数（Assertion Functions）

使用 `asserts` 关键字断言类型：

```typescript
function assertIsString(value: unknown): asserts value is string {
  if (typeof value !== "string") {
    throw new Error("value is not a string");
  }
}

function process(value: unknown) {
  assertIsString(value);
  // 这里 value 被断言为 string
  console.log(value.toUpperCase());
}
```

## 高级技巧

### 技巧 1：自定义类型守卫

编写复杂的自定义类型守卫：

```typescript
// 检查是否为对象
function isObject(value: unknown): value is Record<string, unknown> {
  return typeof value === "object" && value !== null && !Array.isArray(value);
}

// 检查是否为 User 对象
interface User {
  id: number;
  name: string;
  email: string;
}

function isUser(value: unknown): value is User {
  return (
    isObject(value) &&
    typeof value.id === "number" &&
    typeof value.name === "string" &&
    typeof value.email === "string" &&
    "id" in value &&
    "name" in value &&
    "email" in value
  );
}

// 使用
function processData(data: unknown) {
  if (isUser(data)) {
    // data 被收窄为 User
    console.log(`User: ${data.name}`);
  } else {
    console.log("Not a user");
  }
}

processData({ id: 1, name: "Alice", email: "alice@example.com" }); // ✅
processData({ id: 1, name: "Alice" }); // ❌ 缺少 email
```

### 技巧 2：可辨识联合（Discriminated Unions）

使用共同属性区分联合类型成员：

```typescript
// 定义可辨识联合
interface Success<T> {
  type: "success";
  data: T;
}

interface Error {
  type: "error";
  error: Error;
}

type Result<T> = Success<T> | Error;

// 类型守卫
function isSuccess<T>(result: Result<T>): result is Success<T> {
  return result.type === "success";
}

function isError<T>(result: Result<T>): result is Error {
  return result.type === "error";
}

// 使用
async function fetchUser(id: number): Promise<Result<User>> {
  try {
    const user = await api.getUser(id);
    return { type: "success", data: user };
  } catch (error) {
    return { type: "error", error: error as Error };
  }
}

// 处理结果
const result = await fetchUser(1);

if (isSuccess(result)) {
  // result.data: User
  console.log(result.data.name);
} else if (isError(result)) {
  // result.error: Error
  console.error(result.error.message);
}
```

**实际应用**：API 响应处理

```typescript
type APIResponse<T> =
  | { status: 200; data: T }
  | { status: 400; error: string }
  | { status: 401; error: string }
  | { status: 404; error: string }
  | { status: 500; error: string };

function isSuccess<T>(
  response: APIResponse<T>
): response is { status: 200; data: T } {
  return response.status === 200;
}

function isClientError(
  response: APIResponse<any>
): response is { status: 400 | 401 | 404; error: string } {
  return response.status >= 400 && response.status < 500;
}

function isServerError(
  response: APIResponse<any>
): response is { status: 500; error: string } {
  return response.status === 500;
}

// 使用
async function handleResponse<T>(response: APIResponse<T>) {
  if (isSuccess(response)) {
    return response.data;
  } else if (isClientError(response)) {
    throw new Error(`Client error: ${response.error}`);
  } else {
    throw new Error(`Server error: ${response.error}`);
  }
}
```

### 技巧 3：断言函数模式

使用断言函数强制类型收窄：

```typescript
// 基础断言函数
function assertDefined<T>(value: T | null | undefined): asserts value is T {
  if (value === undefined || value === null) {
    throw new Error("value is not defined");
  }
}

// 使用
function process(value: string | undefined) {
  assertDefined(value);
  // value 现在是 string
  console.log(value.toUpperCase());
}

// 断言对象有某个属性
function assertHasKey<K extends string>(
  obj: any,
  key: K
): asserts obj is Record<K, unknown> {
  if (!(key in obj)) {
    throw new Error(`Object does not have key "${key}"`);
  }
}

// 使用
function processObject(obj: unknown) {
  assertHasKey(obj, "name");
  assertHasKey(obj, "age");

  if (typeof obj.name === "string" && typeof obj.age === "number") {
    console.log(`${obj.name} is ${obj.age} years old`);
  }
}
```

**实用断言函数集合**：

```typescript
// 断言不为 null/undefined
function assertDefined<T>(
  value: T | null | undefined,
  message?: string
): asserts value is T {
  if (value === undefined || value === null) {
    throw new Error(message || 'Expected value to be defined')
  }
}

// 断言为字符串
function assertString(
  value: unknown,
  message?: string
): asserts value is string {
  if (typeof value !== 'string') {
    throw new Error(message || `Expected string, got ${typeof value}`)
  }
}

// 断言为数字
function assertNumber(
  value: unknown,
  message?: string
): asserts value is number {
  if (typeof value !== 'number' || isNaN(value)) {
    throw new Error(message || `Expected number, got ${typeof value}`)
  }
}

// 断言为数组
function assertArray<T>(
  value: unknown,
  guard: (item: unknown) => item is T,
  message?: string
): asserts value is T[] {
  if (!Array.isArray(value)) {
    throw new Error(message || `Expected array, got ${typeof value}`)
  }
  if (!value.every(guard)) {
    throw new Error(message || 'Array contains invalid items')
  }
}

// 使用示例
function processData(data: unknown) {
  assertString(data, 'Data must be a string')
  console.log(data.toUpperCase())

  assertNumber(123)
  console.log(123.toFixed(2))
}
```

### 技巧 4：类型守卫工厂

创建可复用的类型守卫工厂：

```typescript
// 创建对象属性守卫
function hasProperty<T extends string>(
  key: T
): (obj: unknown) => obj is { [K in T]: unknown } {
  return (obj: unknown): obj is { [K in T]: unknown } => {
    return typeof obj === "object" && obj !== null && key in obj;
  };
}

// 创建类型守卫
function hasType<T>(
  check: (value: unknown) => boolean
): (value: unknown) => value is T {
  return (value: unknown): value is T => check(value);
}

// 使用
const hasId = hasProperty("id");
const hasName = hasProperty("name");
const isString = hasType<string>((v) => typeof v === "string");
const isNumber = hasType<number>((v) => typeof v === "number" && !isNaN(v));

function processUser(user: unknown) {
  if (
    hasId(user) &&
    hasName(user) &&
    isString(user.id) &&
    isString(user.name)
  ) {
    // user: { id: string; name: string }
    console.log(`User ${user.name} has ID ${user.id}`);
  }
}

// 高级：组合类型守卫
function and<T, U>(
  guard1: (value: unknown) => value is T,
  guard2: (value: unknown) => value is U
): (value: unknown) => value is T & U {
  return (value: unknown): value is T & U => guard1(value) && guard2(value);
}

function or<T, U>(
  guard1: (value: unknown) => value is T,
  guard2: (value: unknown) => value is U
): (value: unknown) => value is T | U {
  return (value: unknown): value is T | U => guard1(value) || guard2(value);
}

// 使用
const hasStringId = and(
  hasProperty("id"),
  hasProperty("id" as "id" & { type: "string" })
);
const isStringOrNumber = or(isString, isNumber);
```

### 技巧 5：模板字面量类型守卫

结合模板字面量类型创建类型守卫：

```typescript
// 定义事件类型
type Event = `${string}:${string}`;

function isEvent(value: string): value is Event {
  return /^[a-z]+:[a-z]+$/.test(value);
}

// 事件处理器
interface EventHandlers {
  "user:login": (userId: string, timestamp: number) => void;
  "user:logout": (userId: string) => void;
  "post:create": (postId: string, authorId: string) => void;
}

function isEventHandlerKey<K extends PropertyKey>(
  key: K,
  handlers: EventHandlers
): key is keyof EventHandlers {
  return key in handlers;
}

// 使用
class EventEmitter {
  private handlers: Partial<EventHandlers> = {};

  on<K extends keyof EventHandlers>(event: K, handler: EventHandlers[K]): void {
    this.handlers[event] = handler;
  }

  emit<K extends string>(event: K, ...args: any[]) {
    if (isEventHandlerKey(event, this.handlers)) {
      const handler = this.handlers[event];
      if (handler) {
        handler(...(args as any));
      }
    }
  }
}

const emitter = new EventEmitter();

emitter.on("user:login", (userId, timestamp) => {
  console.log(`User ${userId} logged in at ${timestamp}`);
});

emitter.emit("user:login", "123", Date.now()); // ✅
emitter.emit("unknown:event", "data"); // ⚠️ 编译期警告
```

### 技巧 6：运行时类型验证库

创建一个完整的类型验证系统：

```typescript
// 验证器类型
type Validator<T> = (value: unknown) => value is T;

// 基础验证器
const isUnknown: Validator<unknown> = () => true;
const isNever: Validator<never> = () => false;

const isString: Validator<string> = (value): value is string =>
  typeof value === "string";

const isNumber: Validator<number> = (value): value is number =>
  typeof value === "number" && !isNaN(value);

const isBoolean: Validator<boolean> = (value): value is boolean =>
  typeof value === "boolean";

const isObject: Validator<Record<string, unknown>> = (
  value
): value is Record<string, unknown> =>
  typeof value === "object" && value !== null && !Array.isArray(value);

// 数组验证器
function isArray<T>(itemValidator: Validator<T>): Validator<T[]> {
  return (value): value is T[] =>
    Array.isArray(value) && value.every(itemValidator);
}

// 对象验证器
function isObjectWith<T extends Record<string, Validator<any>>>(
  shape: T
): Validator<{ [K in keyof T]: T[K] extends Validator<infer V> ? V : never }> {
  return (value): value is any => {
    if (!isObject(value)) return false;

    for (const key in shape) {
      const validator = shape[key];
      if (!validator(value[key])) {
        return false;
      }
    }
    return true;
  };
}

// 联合验证器
function isUnion<T extends Validator<any>[]>(
  ...validators: T
): Validator<T[number] extends Validator<infer V> ? V : never> {
  return (value): value is any =>
    validators.some((validator) => validator(value));
}

// 交集验证器
function isIntersection<T extends Validator<any>[]>(
  ...validators: T
): Validator<T[number] extends Validator<infer V> ? V : never> {
  return (value): value is any =>
    validators.every((validator) => validator(value));
}

// 使用示例
interface User {
  id: number;
  name: string;
  email: string;
  age?: number;
}

const isUser = isObjectWith({
  id: isNumber,
  name: isString,
  email: isString,
  age: (value): value is number | undefined =>
    value === undefined || isNumber(value),
});

// 测试
const data: unknown = {
  id: 1,
  name: "Alice",
  email: "alice@example.com",
  age: 30,
};

if (isUser(data)) {
  // data 类型被推断为 User
  console.log(data.name);
}

// 数组验证
const isUserArray = isArray(isUser);

if (isUserArray(unknownData)) {
  // unknownData 类型被推断为 User[]
  unknownData.forEach((user) => console.log(user.name));
}
```

## 实战应用

### 类型安全的表单验证

```typescript
// 表单字段验证器
interface FieldValidator<T> {
  validate: (value: unknown) => value is T;
  required?: boolean;
  message?: string;
}

// 创建字段验证器
function createFieldValidator<T>(
  check: (value: unknown) => value is T,
  options?: { required?: boolean; message?: string }
): FieldValidator<T> {
  return {
    validate: (value: unknown) => {
      if (value === undefined || value === null) {
        return !options?.required;
      }
      return check(value);
    },
    required: options?.required,
    message: options?.message,
  };
}

// 常用验证器
const requiredString = createFieldValidator(
  (value): value is string => typeof value === "string" && value.length > 0,
  { required: true, message: "This field is required" }
);

const optionalString = createFieldValidator(
  (value): value is string | undefined =>
    value === undefined || typeof value === "string"
);

const requiredEmail = createFieldValidator(
  (value): value is string => {
    if (typeof value !== "string") return false;
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    return emailRegex.test(value);
  },
  { required: true, message: "Invalid email format" }
);

const requiredNumber = createFieldValidator(
  (value): value is number => typeof value === "number" && !isNaN(value),
  { required: true, message: "This field must be a number" }
);

const optionalNumber = createFieldValidator(
  (value): value is number | undefined =>
    value === undefined || (typeof value === "number" && !isNaN(value))
);

// 表单配置
interface LoginForm {
  email: string;
  password: string;
  remember?: boolean;
}

const loginFormValidators = {
  email: requiredEmail,
  password: createFieldValidator(
    (value): value is string => typeof value === "string" && value.length >= 8,
    { required: true, message: "Password must be at least 8 characters" }
  ),
  remember: createFieldValidator(
    (value): value is boolean | undefined =>
      value === undefined || typeof value === "boolean"
  ),
};

// 验证函数
function validateForm<T extends Record<string, unknown>, U>(
  data: T,
  validators: { [K in keyof U]: FieldValidator<any> }
):
  | { success: true; data: U }
  | { success: false; errors: Partial<Record<keyof T, string>> } {
  const errors: Partial<Record<keyof T, string>> = {};

  for (const key in validators) {
    const validator = validators[key];
    if (!validator.validate(data[key])) {
      errors[key] = validator.message || "Invalid value";
    }
  }

  if (Object.keys(errors).length > 0) {
    return { success: false, errors };
  }

  return { success: true, data: data as any };
}

// 使用示例
const formData = {
  email: "user@example.com",
  password: "password123",
  remember: true,
};

const result = validateForm<typeof formData, LoginForm>(
  formData,
  loginFormValidators
);

if (result.success) {
  // result.data 类型是 LoginForm
  console.log("Form is valid:", result.data);
} else {
  // result.errors 包含错误信息
  console.error("Validation errors:", result.errors);
}
```

## 避坑指南

### ❌ 错误 1：类型守卫与类型断言混淆

```typescript
// ❌ 错误：类型守卫应该返回 boolean
function badGuard(value: unknown): string {
  if (typeof value === "string") {
    return value;
  }
  return "";
}

// ✅ 正确：使用类型谓词
function goodGuard(value: unknown): value is string {
  return typeof value === "string";
}
```

---

### ❌ 错误 2：断言函数没有抛出错误

```typescript
// ❌ 错误：断言函数必须在条件不满足时抛出错误
function badAssert(value: unknown): asserts value is string {
  if (typeof value !== "string") {
    return; // 没有抛出错误！
  }
}

// ✅ 正确：必须抛出错误
function goodAssert(value: unknown): asserts value is string {
  if (typeof value !== "string") {
    throw new Error("value is not a string");
  }
}
```

---

### ❌ 错误 3：忽略 null 检查

```typescript
function process(value: string | null) {
  // ❌ 忘记检查 null
  console.log(value.toUpperCase());

  // ✅ 正确检查
  if (value !== null) {
    console.log(value.toUpperCase());
  }

  // ✅ 使用可选链
  console.log(value?.toUpperCase());
}
```

### ❌ 错误 4：复杂类型守卫性能问题

```typescript
// ❌ 每次都创建新的正则表达式
function isEmail(value: unknown): value is string {
  return typeof value === "string" && /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value);
}

// ✅ 缓存正则表达式
const EMAIL_REGEX = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;

function isEmailOptimized(value: unknown): value is string {
  return typeof value === "string" && EMAIL_REGEX.test(value);
}
```

## 总结

### 核心要点

1. **类型收窄**：TypeScript 自动根据条件收窄类型
2. **类型谓词**：`value is Type` 自定义类型守卫
3. **断言函数**：`asserts value is Type` 强制断言类型
4. **可辨识联合**：使用共同属性区分联合类型
5. **守卫组合**：组合多个类型守卫实现复杂验证

### 实用工具

```typescript
// 基础守卫
const isString = (v: unknown): v is string => typeof v === "string";
const isNumber = (v: unknown): v is number =>
  typeof v === "number" && !isNaN(v);
const isObject = (v: unknown): v is Record<string, unknown> =>
  typeof v === "object" && v !== null && !Array.isArray(v);

// 断言函数
function assertDefined<T>(value: T | null | undefined): asserts value is T {
  if (value === undefined || value === null) {
    throw new Error("value is not defined");
  }
}

// 组合守卫
function and<T, U>(
  guard1: (v: unknown) => v is T,
  guard2: (v: unknown) => v is U
): (v: unknown) => v is T & U {
  return (v): v is T & U => guard1(v) && guard2(v);
}
```
