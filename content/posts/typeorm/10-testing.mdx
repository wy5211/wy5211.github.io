## 前言

测试是保障代码质量的关键环节。TypeORM 提供了多种测试方式，从轻量级的单元测试到完整的 E2E 测试，让我们可以全面验证数据访问层的正确性。

**本篇你将学到**：

- 测试环境配置
- 内存数据库（SQLite）测试
- Mock DataSource
- 单元测试（Service 层）
- 集成测试（Repository 层）
- E2E 测试（API 层）
- 测试工厂（Factory Pattern）

## 测试环境配置

### 安装测试依赖

```bash
# Jest（NestJS 默认使用）
npm install --save-dev jest @types/jest ts-jest

# 测试用的 SQLite 驱动
npm install --save-dev better-sqlite3 @types/better-sqlite3

# NestJS 测试工具
npm install --save-dev @nestjs/testing
```

### 测试用的 DataSource

```typescript
// test/test-data-source.ts
import { DataSource } from "typeorm";
import { User, UserRole } from "../src/modules/auth/entities/user.entity";
import { Post, PostStatus } from "../src/modules/posts/entities/post.entity";
import { Category } from "../src/modules/posts/entities/category.entity";
import { Tag } from "../src/modules/posts/entities/tag.entity";

export async function createTestDataSource(): Promise<DataSource> {
  const dataSource = new DataSource({
    type: "better-sqlite3",
    database: ":memory:", // 内存数据库，测试完即销毁
    entities: [User, Post, Category, Tag],
    synchronize: true, // 测试环境自动同步
    logging: false, // 测试时关闭日志
  });

  await dataSource.initialize();
  return dataSource;
}
```

### 测试工具函数

```typescript
// test/test-utils.ts
import { DataSource, Repository } from "typeorm";
import * as bcrypt from "bcrypt";
import { User, UserRole } from "../src/modules/auth/entities/user.entity";

// 清空所有表
export async function cleanAllTables(dataSource: DataSource) {
  const entities = dataSource.entityMetadatas;
  for (const entity of entities) {
    const repository = dataSource.getRepository(entity.name);
    await repository.query(`DELETE FROM ${entity.tableName}`);
  }
}

// 创建测试用户
export async function createTestUser(
  userRepo: Repository<User>,
  overrides: Partial<User> = {}
): Promise<User> {
  const passwordHash = await bcrypt.hash("password123", 10);

  const user = userRepo.create({
    name: "测试用户",
    username: `testuser_${Date.now()}`,
    email: `test_${Date.now()}@example.com`,
    passwordHash,
    role: UserRole.READER,
    isActive: true,
    ...overrides,
  });

  return userRepo.save(user);
}
```

## 单元测试（Service 层）

单元测试使用 Mock 隔离外部依赖，只测试业务逻辑。

### Mock Repository

```typescript
// test/modules/posts/posts.service.spec.ts
import { Test, TestingModule } from "@nestjs/testing";
import { getRepositoryToken } from "@nestjs/typeorm";
import { NotFoundException, ConflictException } from "@nestjs/common";
import { Repository } from "typeorm";
import { PostsService } from "../../../src/modules/posts/posts.service";
import {
  Post,
  PostStatus,
} from "../../../src/modules/posts/entities/post.entity";

describe("PostsService", () => {
  let service: PostsService;
  let postRepo: jest.Mocked<Repository<Post>>;

  // 创建 Mock Repository
  const mockPostRepo = () => ({
    create: jest.fn(),
    save: jest.fn(),
    find: jest.fn(),
    findOne: jest.fn(),
    findOneBy: jest.fn(),
    findAndCount: jest.fn(),
    update: jest.fn(),
    increment: jest.fn(),
    softRemove: jest.fn(),
    createQueryBuilder: jest.fn(() => ({
      leftJoinAndSelect: jest.fn().mockReturnThis(),
      where: jest.fn().mockReturnThis(),
      andWhere: jest.fn().mockReturnThis(),
      orderBy: jest.fn().mockReturnThis(),
      skip: jest.fn().mockReturnThis(),
      take: jest.fn().mockReturnThis(),
      getManyAndCount: jest.fn(),
      getOne: jest.fn(),
    })),
  });

  beforeEach(async () => {
    const module: TestingModule = await Test.createTestingModule({
      providers: [
        PostsService,
        {
          provide: getRepositoryToken(Post),
          useFactory: mockPostRepo,
        },
      ],
    }).compile();

    service = module.get<PostsService>(PostsService);
    postRepo = module.get(getRepositoryToken(Post));
  });

  describe("create", () => {
    it("应该成功创建文章", async () => {
      const createDto = {
        title: "测试文章",
        slug: "test-post",
        content: "这是测试内容",
        authorId: 1,
      };

      // 设置 Mock 返回值
      postRepo.findOneBy.mockResolvedValue(null); // slug 不重复
      postRepo.create.mockReturnValue({ id: 1, ...createDto } as Post);
      postRepo.save.mockResolvedValue({ id: 1, ...createDto } as Post);

      const result = await service.create(createDto);

      expect(result).toBeDefined();
      expect(result.title).toBe("测试文章");
      expect(postRepo.save).toHaveBeenCalled();
    });

    it("slug 重复时应该抛出异常", async () => {
      const createDto = {
        title: "测试文章",
        slug: "existing-slug",
        content: "内容",
        authorId: 1,
      };

      postRepo.findOneBy.mockResolvedValue({
        id: 1,
        slug: "existing-slug",
      } as Post);

      await expect(service.create(createDto)).rejects.toThrow(
        ConflictException
      );
    });
  });

  describe("findOne", () => {
    it("应该返回文章", async () => {
      const mockPost = {
        id: 1,
        title: "测试文章",
        content: "内容",
      } as Post;

      postRepo.findOne.mockResolvedValue(mockPost);

      const result = await service.findOne(1);

      expect(result).toEqual(mockPost);
      expect(postRepo.findOne).toHaveBeenCalledWith({
        where: { id: 1 },
        relations: ["category", "tags"],
      });
    });

    it("文章不存在时应该抛出异常", async () => {
      postRepo.findOne.mockResolvedValue(null);

      await expect(service.findOne(999)).rejects.toThrow(NotFoundException);
    });
  });
});
```

## 集成测试（Repository 层）

集成测试使用真实的数据库（内存 SQLite）验证 SQL 操作。

### 使用内存数据库

```typescript
// test/modules/posts/posts.repository.spec.ts
import { DataSource, Repository } from "typeorm";
import { createTestDataSource } from "../../test-data-source";
import {
  Post,
  PostStatus,
} from "../../../src/modules/posts/entities/post.entity";
import { Category } from "../../../src/modules/posts/entities/category.entity";
import { Tag } from "../../../src/modules/posts/entities/tag.entity";

describe("Post Repository (Integration)", () => {
  let dataSource: DataSource;
  let postRepo: Repository<Post>;
  let categoryRepo: Repository<Category>;
  let tagRepo: Repository<Tag>;

  beforeAll(async () => {
    dataSource = await createTestDataSource();
    postRepo = dataSource.getRepository(Post);
    categoryRepo = dataSource.getRepository(Category);
    tagRepo = dataSource.getRepository(Tag);
  });

  afterAll(async () => {
    await dataSource.destroy();
  });

  afterEach(async () => {
    // 清空数据
    await postRepo.query("DELETE FROM posts");
    await categoryRepo.query("DELETE FROM categories");
    await tagRepo.query("DELETE FROM tags");
  });

  describe("创建文章", () => {
    it("应该成功创建文章", async () => {
      const post = postRepo.create({
        title: "测试文章",
        slug: "test-post",
        content: "这是测试内容",
        authorId: 1,
        status: PostStatus.DRAFT,
      });

      const saved = await postRepo.save(post);

      expect(saved.id).toBeDefined();
      expect(saved.title).toBe("测试文章");
      expect(saved.status).toBe(PostStatus.DRAFT);
      expect(saved.createdAt).toBeDefined();
    });
  });

  describe("查询文章", () => {
    let category: Category;
    let tag1: Tag;
    let tag2: Tag;

    beforeEach(async () => {
      category = await categoryRepo.save({
        name: "技术",
        slug: "tech",
      });

      tag1 = await tagRepo.save({ name: "TypeScript", slug: "typescript" });
      tag2 = await tagRepo.save({ name: "Node.js", slug: "nodejs" });

      await postRepo.save([
        {
          title: "文章一",
          slug: "post-1",
          content: "内容一",
          authorId: 1,
          status: PostStatus.PUBLISHED,
          category,
          tags: [tag1],
          publishedAt: new Date(),
        },
        {
          title: "文章二",
          slug: "post-2",
          content: "内容二",
          authorId: 1,
          status: PostStatus.DRAFT,
          category,
          tags: [tag1, tag2],
        },
      ]);
    });

    it("查询已发布文章", async () => {
      const posts = await postRepo.find({
        where: { status: PostStatus.PUBLISHED },
        relations: ["category", "tags"],
      });

      expect(posts).toHaveLength(1);
      expect(posts[0].title).toBe("文章一");
    });

    it("查询文章及其标签", async () => {
      const post = await postRepo.findOne({
        where: { slug: "post-2" },
        relations: ["tags"],
      });

      expect(post.tags).toHaveLength(2);
      expect(post.tags.map((t) => t.name)).toContain("TypeScript");
    });

    it("分页查询", async () => {
      const [posts, total] = await postRepo.findAndCount({
        skip: 0,
        take: 1,
        order: { createdAt: "DESC" },
      });

      expect(posts).toHaveLength(1);
      expect(total).toBe(2);
    });
  });

  describe("软删除", () => {
    it("应该支持软删除和恢复", async () => {
      const post = await postRepo.save({
        title: "待删除",
        slug: "to-delete",
        content: "内容",
        authorId: 1,
      });

      await postRepo.softDelete(post.id);

      // 正常查询找不到
      const found = await postRepo.findOneBy({ id: post.id });
      expect(found).toBeNull();

      // 包含软删除
      const withDeleted = await postRepo.findOne({
        where: { id: post.id },
        withDeleted: true,
      });
      expect(withDeleted).toBeDefined();
      expect(withDeleted.deletedAt).toBeDefined();

      // 恢复
      await postRepo.restore(post.id);
      const restored = await postRepo.findOneBy({ id: post.id });
      expect(restored).toBeDefined();
    });
  });
});
```

## E2E 测试（API 层）

E2E 测试验证完整的请求-响应流程。

```typescript
// test/modules/posts/posts.e2e-spec.ts
import { Test, TestingModule } from "@nestjs/testing";
import { INestApplication, ValidationPipe } from "@nestjs/common";
import * as request from "supertest";
import { TypeOrmModule } from "@nestjs/typeorm";
import {
  Post,
  PostStatus,
} from "../../../src/modules/posts/entities/post.entity";
import { Category } from "../../../src/modules/posts/entities/category.entity";
import { Tag } from "../../../src/modules/posts/entities/tag.entity";
import { PostsModule } from "../../../src/modules/posts/posts.module";

describe("Posts API (E2E)", () => {
  let app: INestApplication;

  beforeAll(async () => {
    const moduleFixture: TestingModule = await Test.createTestingModule({
      imports: [
        TypeOrmModule.forRoot({
          type: "better-sqlite3",
          database: ":memory:",
          entities: [Post, Category, Tag],
          synchronize: true,
          logging: false,
        }),
        PostsModule,
      ],
    }).compile();

    app = moduleFixture.createNestApplication();

    // 配置与生产一致
    app.useGlobalPipes(
      new ValidationPipe({
        whitelist: true,
        forbidNonWhitelisted: true,
        transform: true,
      })
    );

    await app.init();
  });

  afterAll(async () => {
    await app.close();
  });

  describe("POST /posts", () => {
    it("应该创建文章", () => {
      return request(app.getHttpServer())
        .post("/posts")
        .send({
          title: "测试文章",
          slug: "test-post",
          content: "这是测试内容",
          authorId: 1,
        })
        .expect(201)
        .expect((res) => {
          expect(res.body.success).toBe(true);
          expect(res.body.data.title).toBe("测试文章");
        });
    });

    it("缺少必填字段应该返回 400", () => {
      return request(app.getHttpServer())
        .post("/posts")
        .send({
          title: "测试文章",
          // 缺少 slug, content, authorId
        })
        .expect(400);
    });
  });

  describe("GET /posts", () => {
    beforeEach(async () => {
      // 插入测试数据
      const postRepo = app.get(TypeOrmModule).getRepository(Post);
      await postRepo.save([
        {
          title: "文章一",
          slug: "post-1",
          content: "内容一",
          authorId: 1,
          status: PostStatus.PUBLISHED,
        },
        {
          title: "文章二",
          slug: "post-2",
          content: "内容二",
          authorId: 1,
          status: PostStatus.PUBLISHED,
        },
      ]);
    });

    it("应该返回文章列表", () => {
      return request(app.getHttpServer())
        .get("/posts?page=1&limit=10")
        .expect(200)
        .expect((res) => {
          expect(res.body.success).toBe(true);
          expect(res.body.data.data).toHaveLength(2);
        });
    });
  });

  describe("GET /posts/:id", () => {
    it("应该返回文章详情", async () => {
      const postRepo = app.get(TypeOrmModule).getRepository(Post);
      const post = await postRepo.save({
        title: "测试文章",
        slug: "detail-test",
        content: "详细内容",
        authorId: 1,
      });

      return request(app.getHttpServer())
        .get(`/posts/${post.id}`)
        .expect(200)
        .expect((res) => {
          expect(res.body.data.title).toBe("测试文章");
        });
    });

    it("文章不存在应该返回 404", () => {
      return request(app.getHttpServer()).get("/posts/99999").expect(404);
    });
  });
});
```

## 测试工厂（Factory Pattern）

测试工厂可以方便地创建测试数据。

### 定义工厂

```typescript
// test/factories/user.factory.ts
import { Repository } from "typeorm";
import { User, UserRole } from "../../../src/modules/auth/entities/user.entity";
import * as bcrypt from "bcrypt";

let counter = 0;

export async function createUser(
  userRepo: Repository<User>,
  overrides: Partial<User> = {}
): Promise<User> {
  counter++;
  const passwordHash = await bcrypt.hash("password123", 10);

  const user = userRepo.create({
    name: `测试用户${counter}`,
    username: `user${counter}`,
    email: `user${counter}@test.com`,
    passwordHash,
    role: UserRole.READER,
    isActive: true,
    ...overrides,
  });

  return userRepo.save(user);
}

// test/factories/post.factory.ts
import { Repository } from "typeorm";
import {
  Post,
  PostStatus,
} from "../../../src/modules/posts/entities/post.entity";

let counter = 0;

export async function createPost(
  postRepo: Repository<Post>,
  overrides: Partial<Post> = {}
): Promise<Post> {
  counter++;
  const post = postRepo.create({
    title: `测试文章${counter}`,
    slug: `test-post-${counter}-${Date.now()}`,
    content: `这是第 ${counter} 篇测试文章的内容`,
    authorId: 1,
    status: PostStatus.DRAFT,
    ...overrides,
  });

  return postRepo.save(post);
}
```

### 使用工厂

```typescript
describe("文章查询", () => {
  it("应该按作者查询文章", async () => {
    // 创建测试数据
    const author1 = await createUser(userRepo, { role: UserRole.AUTHOR });
    const author2 = await createUser(userRepo, { role: UserRole.AUTHOR });

    await createPost(postRepo, {
      authorId: author1.id,
      status: PostStatus.PUBLISHED,
    });
    await createPost(postRepo, {
      authorId: author1.id,
      status: PostStatus.PUBLISHED,
    });
    await createPost(postRepo, { authorId: author2.id });

    // 查询
    const posts = await postRepo.find({
      where: { authorId: author1.id, status: PostStatus.PUBLISHED },
    });

    expect(posts).toHaveLength(2);
  });
});
```

## 测试最佳实践

### 1. 测试文件命名

```
posts.service.spec.ts    # 单元测试
posts.repository.spec.ts # 集成测试
posts.e2e-spec.ts        # E2E 测试
```

### 2. 测试结构（AAA 模式）

```typescript
it("应该返回文章", async () => {
  // Arrange（准备）
  const mockPost = { id: 1, title: "测试" };
  postRepo.findOne.mockResolvedValue(mockPost);

  // Act（执行）
  const result = await service.findOne(1);

  // Assert（断言）
  expect(result).toEqual(mockPost);
});
```

### 3. 测试覆盖率

```json
// package.json
{
  "scripts": {
    "test": "jest",
    "test:watch": "jest --watch",
    "test:cov": "jest --coverage",
    "test:e2e": "jest --config ./test/jest-e2e.json"
  }
}
```

### 4. 测试数据隔离

```typescript
// ✅ 每个测试用例前后清理数据
afterEach(async () => {
  await cleanAllTables(dataSource);
});

// ✅ 使用唯一数据避免冲突
const email = `test_${Date.now()}_${Math.random()}@example.com`;

// ✅ 使用事务回滚（可选）
beforeEach(async () => {
  await dataSource.query("BEGIN");
});

afterEach(async () => {
  await dataSource.query("ROLLBACK");
});
```

## 总结

### 核心要点

1. **测试环境**
   - 使用内存 SQLite（快速、隔离）
   - 独立的测试 DataSource

2. **单元测试**
   - Mock Repository 隔离数据库
   - 只测试业务逻辑
   - 覆盖正常和异常路径

3. **集成测试**
   - 使用真实数据库（内存 SQLite）
   - 验证 SQL 操作的正确性
   - 测试关联、分页、软删除等

4. **E2E 测试**
   - 测试完整的 HTTP 请求流程
   - 验证请求参数、响应格式、状态码
   - 使用 supertest 发送请求

5. **测试工厂**
   - 封装测试数据创建逻辑
   - 支持覆盖默认值
   - 保持测试代码简洁
