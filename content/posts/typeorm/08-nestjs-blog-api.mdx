```typescript
// src/app.module.ts
import { Module } from "@nestjs/common";
import { ConfigModule, ConfigService } from "@nestjs/config";
import { TypeOrmModule } from "@nestjs/typeorm";
import { PostsModule } from "./modules/posts/posts.module";

@Module({
  imports: [
    // 配置模块
    ConfigModule.forRoot({
      isGlobal: true,
      envFilePath: `.env.${process.env.NODE_ENV || "development"}`,
    }),

    // TypeORM 配置
    TypeOrmModule.forRootAsync({
      imports: [ConfigModule],
      inject: [ConfigService],
      useFactory: (configService: ConfigService) => ({
        type: "postgres",
        host: configService.get("DB_HOST", "localhost"),
        port: configService.get<number>("DB_PORT", 5432),
        username: configService.get("DB_USERNAME", "postgres"),
        password: configService.get("DB_PASSWORD", "postgres"),
        database: configService.get("DB_DATABASE", "typeorm_demo"),
        entities: [__dirname + "/**/*.entity{.ts,.js}"],
        synchronize: configService.get("NODE_ENV") === "development",
        logging: configService.get("NODE_ENV") === "development",
        autoLoadEntities: true,
      }),
    }),

    PostsModule,
  ],
})
export class AppModule {}
```

```typescript
// src/modules/posts/entities/post.entity.ts
import {
  Entity,
  PrimaryGeneratedColumn,
  Column,
  CreateDateColumn,
  UpdateDateColumn,
  DeleteDateColumn,
  ManyToOne,
  ManyToMany,
  JoinTable,
  Index,
} from "typeorm";
import { Category } from "./category.entity";
import { Tag } from "./tag.entity";

export enum PostStatus {
  DRAFT = "DRAFT",
  PUBLISHED = "PUBLISHED",
  ARCHIVED = "ARCHIVED",
}

@Entity("posts")
@Index(["status", "publishedAt"])
export class Post {
  @PrimaryGeneratedColumn()
  id: number;

  @Column({ length: 255 })
  title: string;

  @Column({ unique: true, length: 255 })
  slug: string;

  @Column({ nullable: true, length: 500 })
  excerpt: string;

  @Column({ type: "text" })
  content: string;

  @Column({ nullable: true, name: "cover_image", length: 500 })
  coverImage: string;

  @Column({ name: "author_id" })
  authorId: number;

  @ManyToOne(() => Category, { nullable: true, onDelete: "SET NULL" })
  category: Category;

  @Column({ nullable: true })
  categoryId: number;

  @ManyToMany(() => Tag, (tag) => tag.posts, { eager: true })
  @JoinTable({
    name: "post_tags",
    joinColumn: { name: "post_id", referencedColumnName: "id" },
    inverseJoinColumn: { name: "tag_id", referencedColumnName: "id" },
  })
  tags: Tag[];

  @Column({
    type: "enum",
    enum: PostStatus,
    default: PostStatus.DRAFT,
  })
  status: PostStatus;

  @Column({ default: 0, name: "view_count" })
  viewCount: number;

  @Column({ default: false, name: "is_featured" })
  isFeatured: boolean;

  @Column({ nullable: true, name: "published_at", type: "timestamptz" })
  publishedAt: Date;

  @CreateDateColumn({ name: "created_at" })
  createdAt: Date;

  @UpdateDateColumn({ name: "updated_at" })
  updatedAt: Date;

  @DeleteDateColumn({ name: "deleted_at" })
  deletedAt: Date;
}
```

```typescript
import { IsOptional, IsInt, Min, Max } from "class-validator";
import { Type } from "class-transformer";
import { ApiPropertyOptional } from "@nestjs/swagger";

export class PaginationDto {
  @ApiPropertyOptional({ default: 1 })
  @IsOptional()
  @Type(() => Number)
  @IsInt()
  @Min(1)
  page: number = 1;

  @ApiPropertyOptional({ default: 10 })
  @IsOptional()
  @Type(() => Number)
  @IsInt()
  @Min(1)
  @Max(100)
  limit: number = 10;

  @ApiPropertyOptional()
  @IsOptional()
  orderBy: string = "createdAt";

  @ApiPropertyOptional({ enum: ["ASC", "DESC"] })
  @IsOptional()
  orderDirection: "ASC" | "DESC" = "DESC";

  get skip(): number {
    return (this.page - 1) * this.limit;
  }
}
```

## Service 层

```typescript
// src/modules/posts/posts.service.ts
import {
  Injectable,
  NotFoundException,
  ConflictException,
} from "@nestjs/common";
import { InjectRepository } from "@nestjs/typeorm";
import { Repository, Like, ILike, In } from "typeorm";
import { Post, PostStatus } from "./entities/post.entity";
import { CreatePostDto } from "./dto/create-post.dto";
import { UpdatePostDto } from "./dto/update-post.dto";
import { PaginationDto } from "../../common/dto/pagination.dto";

@Injectable()
export class PostsService {
  constructor(
    @InjectRepository(Post)
    private postRepo: Repository<Post>
  ) {}

  // 创建文章
  async create(createPostDto: CreatePostDto): Promise<Post> {
    // 检查 slug 是否已存在
    const existing = await this.postRepo.findOneBy({
      slug: createPostDto.slug,
    });
    if (existing) {
      throw new ConflictException(`Slug "${createPostDto.slug}" 已被使用`);
    }

    const post = this.postRepo.create({
      ...createPostDto,
      status: (createPostDto.status as PostStatus) || PostStatus.DRAFT,
      tags: createPostDto.tagIds?.map((id) => ({ id })),
    });

    // 如果发布，设置发布时间
    if (post.status === PostStatus.PUBLISHED) {
      post.publishedAt = new Date();
    }

    return this.postRepo.save(post);
  }

  // 查询文章列表（分页）
  async findAll(pagination: PaginationDto) {
    const { page, limit, skip, orderBy, orderDirection } = pagination;

    const [data, total] = await this.postRepo.findAndCount({
      skip,
      take: limit,
      order: { [orderBy]: orderDirection },
      relations: ["category"],
    });

    return {
      data,
      total,
      page,
      limit,
      totalPages: Math.ceil(total / limit),
    };
  }

  // 查询已发布文章
  async findPublished(pagination: PaginationDto) {
    const { page, limit, skip, orderBy, orderDirection } = pagination;

    const [data, total] = await this.postRepo.findAndCount({
      where: { status: PostStatus.PUBLISHED },
      skip,
      take: limit,
      order: { [orderBy]: orderDirection },
      relations: ["category"],
    });

    return {
      data,
      total,
      page,
      limit,
      totalPages: Math.ceil(total / limit),
    };
  }

  // 搜索文章
  async search(keyword: string, pagination: PaginationDto) {
    const { page, limit, skip } = pagination;

    const qb = this.postRepo
      .createQueryBuilder("post")
      .leftJoinAndSelect("post.category", "category")
      .leftJoinAndSelect("post.tags", "tag")
      .where("post.status = :status", { status: PostStatus.PUBLISHED })
      .andWhere(
        "(post.title ILIKE :kw OR post.content ILIKE :kw OR post.excerpt ILIKE :kw)",
        { kw: `%${keyword}%` }
      )
      .orderBy("post.publishedAt", "DESC")
      .skip(skip)
      .take(limit);

    const [data, total] = await qb.getManyAndCount();

    return {
      data,
      total,
      page,
      limit,
      totalPages: Math.ceil(total / limit),
    };
  }

  // 查询单篇文章
  async findOne(id: number): Promise<Post> {
    const post = await this.postRepo.findOne({
      where: { id },
      relations: ["category", "tags"],
    });

    if (!post) {
      throw new NotFoundException(`文章 #${id} 不存在`);
    }

    return post;
  }

  // 通过 slug 查询
  async findBySlug(slug: string): Promise<Post> {
    const post = await this.postRepo.findOne({
      where: { slug },
      relations: ["category", "tags"],
    });

    if (!post) {
      throw new NotFoundException(`文章 "${slug}" 不存在`);
    }

    return post;
  }

  // 更新文章
  async update(id: number, updatePostDto: UpdatePostDto): Promise<Post> {
    const post = await this.findOne(id);

    // 如果 slug 被修改，检查唯一性
    if (updatePostDto.slug && updatePostDto.slug !== post.slug) {
      const existing = await this.postRepo.findOneBy({
        slug: updatePostDto.slug,
      });
      if (existing) {
        throw new ConflictException(`Slug "${updatePostDto.slug}" 已被使用`);
      }
    }

    // 更新标签
    if (updatePostDto.tagIds) {
      post.tags = updatePostDto.tagIds.map((id) => ({ id })) as any;
    }

    // 如果从草稿变为发布，设置发布时间
    if (
      updatePostDto.status === PostStatus.PUBLISHED &&
      post.status !== PostStatus.PUBLISHED &&
      !post.publishedAt
    ) {
      post.publishedAt = new Date();
    }

    Object.assign(post, {
      ...updatePostDto,
      tags: post.tags,
      publishedAt: post.publishedAt,
    });

    return this.postRepo.save(post);
  }

  // 增加浏览量
  async incrementViewCount(id: number): Promise<void> {
    await this.postRepo.increment({ id }, "viewCount", 1);
  }

  // 软删除文章
  async remove(id: number): Promise<void> {
    const post = await this.findOne(id);
    await this.postRepo.softRemove(post);
  }
}
```

## Controller 层

```typescript
// src/modules/posts/posts.controller.ts
import {
  Controller,
  Get,
  Post,
  Put,
  Delete,
  Body,
  Param,
  Query,
  ParseIntPipe,
  HttpCode,
  HttpStatus,
} from "@nestjs/common";
import { PostsService } from "./posts.service";
import { CreatePostDto } from "./dto/create-post.dto";
import { UpdatePostDto } from "./dto/update-post.dto";
import { PaginationDto } from "../../common/dto/pagination.dto";

@Controller("posts")
export class PostsController {
  constructor(private readonly postsService: PostsService) {}

  // POST /posts - 创建文章
  @Post()
  @HttpCode(HttpStatus.CREATED)
  async create(@Body() createPostDto: CreatePostDto) {
    const post = await this.postsService.create(createPostDto);
    return {
      success: true,
      data: post,
      message: "文章创建成功",
    };
  }

  // GET /posts - 查询所有文章（管理用）
  @Get()
  async findAll(@Query() pagination: PaginationDto) {
    const result = await this.postsService.findAll(pagination);
    return {
      success: true,
      data: result,
    };
  }

  // GET /posts/published - 查询已发布文章
  @Get("published")
  async findPublished(@Query() pagination: PaginationDto) {
    const result = await this.postsService.findPublished(pagination);
    return {
      success: true,
      data: result,
    };
  }

  // GET /posts/search?keyword=xxx - 搜索文章
  @Get("search")
  async search(
    @Query("keyword") keyword: string,
    @Query() pagination: PaginationDto
  ) {
    const result = await this.postsService.search(keyword, pagination);
    return {
      success: true,
      data: result,
    };
  }

  // GET /posts/slug/:slug - 通过 slug 查询
  @Get("slug/:slug")
  async findBySlug(@Param("slug") slug: string) {
    const post = await this.postsService.findBySlug(slug);
    // 增加浏览量（异步执行，不阻塞响应）
    this.postsService.incrementViewCount(post.id);
    return {
      success: true,
      data: post,
    };
  }

  // GET /posts/:id - 查询单篇文章
  @Get(":id")
  async findOne(@Param("id", ParseIntPipe) id: number) {
    const post = await this.postsService.findOne(id);
    return {
      success: true,
      data: post,
    };
  }

  // PUT /posts/:id - 更新文章
  @Put(":id")
  async update(
    @Param("id", ParseIntPipe) id: number,
    @Body() updatePostDto: UpdatePostDto
  ) {
    const post = await this.postsService.update(id, updatePostDto);
    return {
      success: true,
      data: post,
      message: "文章更新成功",
    };
  }

  // DELETE /posts/:id - 删除文章
  @Delete(":id")
  @HttpCode(HttpStatus.NO_CONTENT)
  async remove(@Param("id", ParseIntPipe) id: number) {
    await this.postsService.remove(id);
  }
}
```

## 统一响应格式

### 响应拦截器

```typescript
// src/common/interceptors/transform.interceptor.ts
import {
  Injectable,
  NestInterceptor,
  ExecutionContext,
  CallHandler,
} from "@nestjs/common";
import { Observable, map } from "rxjs";

export interface Response<T> {
  success: boolean;
  data: T;
  message?: string;
}

@Injectable()
export class TransformInterceptor<T> implements NestInterceptor<
  T,
  Response<T>
> {
  intercept(
    context: ExecutionContext,
    next: CallHandler
  ): Observable<Response<T>> {
    return next.handle().pipe(
      map((data) => {
        // 如果已经是标准格式，直接返回
        if (data?.success !== undefined) {
          return data;
        }
        return {
          success: true,
          data,
        };
      })
    );
  }
}
```

### 异常过滤器

```typescript
// src/common/filters/http-exception.filter.ts
import {
  ExceptionFilter,
  Catch,
  ArgumentsHost,
  HttpException,
  HttpStatus,
} from "@nestjs/common";
import { Response } from "express";

@Catch()
export class AllExceptionsFilter implements ExceptionFilter {
  catch(exception: unknown, host: ArgumentsHost) {
    const ctx = host.switchToHttp();
    const response = ctx.getResponse<Response>();

    let status = HttpStatus.INTERNAL_SERVER_ERROR;
    let message = "服务器内部错误";

    if (exception instanceof HttpException) {
      status = exception.getStatus();
      const exceptionResponse = exception.getResponse();
      message =
        typeof exceptionResponse === "string"
          ? exceptionResponse
          : (exceptionResponse as any).message || exception.message;
    }

    response.status(status).json({
      success: false,
      statusCode: status,
      message: Array.isArray(message) ? message : [message],
      timestamp: new Date().toISOString(),
    });
  }
}
```

### main.ts 配置

```typescript
// src/main.ts
import { NestFactory } from "@nestjs/core";
import { ValidationPipe } from "@nestjs/common";
import { AppModule } from "./app.module";
import { TransformInterceptor } from "./common/interceptors/transform.interceptor";
import { AllExceptionsFilter } from "./common/filters/http-exception.filter";

async function bootstrap() {
  const app = await NestFactory.create(AppModule);

  // 全局验证管道
  app.useGlobalPipes(
    new ValidationPipe({
      whitelist: true, // 过滤掉 DTO 中没有的属性
      forbidNonWhitelisted: true, // 禁止传入 DTO 中没有的属性
      transform: true, // 自动类型转换
      transformOptions: {
        enableImplicitConversion: true,
      },
    })
  );

  // 全局响应拦截器
  app.useGlobalInterceptors(new TransformInterceptor());

  // 全局异常过滤器
  app.useGlobalFilters(new AllExceptionsFilter());

  // CORS
  app.enableCors();

  // API 前缀
  app.setGlobalPrefix("api");

  await app.listen(3000);
  console.log("应用已启动：http://localhost:3000/api");
}
bootstrap();
```

## 总结

### 核心要点

1. **TypeOrmModule 配置**
   - `forRoot()`：全局数据库连接
   - `forFeature()`：模块级 Repository 注册
   - 使用 `ConfigService` 管理环境变量

2. **完整模块结构**
   - Entity：数据模型
   - DTO：数据验证
   - Service：业务逻辑
   - Controller：路由处理

3. **DTO 验证**
   - `class-validator`：字段验证
   - `class-transformer`：类型转换
   - `ValidationPipe`：全局验证管道

4. **统一响应**
   - 拦截器：包装成功响应
   - 异常过滤器：统一错误格式
