# Node.js 性能优化：从内存泄漏到并发处理

Node.js 虽然高性能，但用不好也会出现内存泄漏、CPU 飙升、并发瓶颈等问题。本文从实战角度讲解性能优化技巧。

## 一、理解 Node.js 性能特点

### 1.1 单线程 + 事件循环

```
┌───────────────────────────────────┐
│           Event Loop              │
│  ┌─────────┐ ┌─────────┐ ┌──────┐│
│  │ Timers  │→│Poll     │→│Check ││
│  └─────────┘ └─────────┘ └──────┘│
└───────────────────────────────────┘
```

**关键点：**
- 主线程只有一个，阻塞操作会卡住整个进程
- 异步 I/O 由 libuv 线程池处理
- CPU 密集任务需要用 Worker Threads

### 1.2 内存模型

```javascript
// 内存结构
```

## 二、内存优化

### 2.1 常见内存泄漏

**1. 全局变量**
```javascript
// ❌ 错误：全局数组无限增长
const cache = []
app.get('/data', (req, res) => {
  cache.push(req.body)  // 永远不会被回收
  res.json({ ok: true })
})

// ✅ 正确：限制大小
const cache = new Map()
const MAX_CACHE = 1000

app.get('/data', (req, res) => {
  const key = req.body.id
  cache.set(key, req.body)
  if (cache.size > MAX_CACHE) {
    const firstKey = cache.keys().next().value
    cache.delete(firstKey)
  }
  res.json({ ok: true })
})
```

**2. 闭包引用**
```javascript
// ❌ 错误：闭包持有大对象引用
function processLargeData(data) {
  const handler = () => {
    console.log(data.length)  // data 永远不会被释放
  }
  setInterval(handler, 1000)
}

// ✅ 正确：只保留需要的数据
function processLargeData(data) {
  const length = data.length  // 只保留需要的信息
  const handler = () => {
    console.log(length)
  }
  setInterval(handler, 1000)
}
```

**3. 事件监听器**
```javascript
// ❌ 错误：重复添加监听器
function connect() {
  const client = new Client()
  client.on('data', (data) => {
    // 处理数据
  })
  // 每次连接都添加新的监听器
}

// ✅ 正确：移除旧监听器或使用 once
function connect() {
  const client = new Client()
  const handler = (data) => {
    // 处理数据
  }
  client.on('data', handler)
  client.once('end', () => {
    client.removeListener('data', handler)
  })
}
```

### 2.2 内存监控

## 三、事件循环优化

### 3.1 避免阻塞事件循环

```javascript
// ❌ 错误：同步大计算
function heavyComputation(n) {
  let result = 0
  for (let i = 0; i < n; i++) {
    result += Math.sqrt(i)
  }
  return result
}

// ✅ 正确：分片处理
async function heavyComputation(n, chunkSize = 10000) {
  let result = 0
  let i = 0
  
  while (i < n) {
    const end = Math.min(i + chunkSize, n)
    for (; i < end; i++) {
      result += Math.sqrt(i)
    }
    // 让出事件循环
    await new Promise(setImmediate)
  }
  
  return result
}
```

### 3.2 使用 Worker Threads

```javascript
// 主线程
const { Worker } = require('worker_threads')

function runHeavyTask(data) {
  return new Promise((resolve, reject) => {
    const worker = new Worker('./heavy-task.js', {
      workerData: data
    })
    
    worker.on('message', resolve)
    worker.on('error', reject)
    worker.on('exit', (code) => {
      if (code !== 0) reject(new Error(`Worker stopped with code ${code}`))
    })
  })
}

// heavy-task.js
const { workerData, parentPort } = require('worker_threads')

const result = heavyComputation(workerData)
parentPort.postMessage(result)
```

## 四、并发处理

### 4.1 连接池

```javascript
// ❌ 错误：每次请求创建新连接
app.get('/users', async (req, res) => {
  const connection = await mysql.createConnection(config)
  const [rows] = await connection.execute('SELECT * FROM users')
  await connection.end()
  res.json(rows)

// ✅ 正确：使用连接池
const pool = mysql.createPool({
  ...config,
  connectionLimit: 10,
  queueLimit: 0
})

app.get('/users', async (req, res) => {
  const [rows] = await pool.execute('SELECT * FROM users')
  res.json(rows)
})
```

### 4.2 请求限流

```javascript
const rateLimit = require('express-rate-limit')

// 全局限流
const limiter = rateLimit({
  windowMs: 60 * 1000,  // 1 分钟
  max: 100,             // 最多 100 次请求
  message: 'Too many requests'

  windowMs: 60 * 1000,
  max: 30,
  keyGenerator: (req) => req.ip + req.path
})

app.use('/api/', apiLimiter)
```

### 4.3 并发控制

```javascript
// 控制并发数量
async function processWithConcurrency(tasks, concurrency = 5) {
  const results = []
  const executing = new Set()
  
  for (const task of tasks) {
    const promise = task().then(result => {
      executing.delete(promise)
      return result
    })
    
    executing.add(promise)
    results.push(promise)
    
    if (executing.size >= concurrency) {
      await Promise.race(executing)
    }
  }
  
  return Promise.all(results)
}

// 使用
const urls = ['url1', 'url2', 'url3', ...]
const results = await processWithConcurrency(
  urls.map(url => () => fetch(url)),
  5  // 最多 5 个并发
)
```

### 4.4 队列处理


## 五、数据库优化

### 5.1 查询优化

```javascript
// ❌ 错误：N+1 查询
const users = await db.query('SELECT * FROM users')
for (const user of users) {
  user.posts = await db.query('SELECT * FROM posts WHERE user_id = ?', [user.id])
}

// ✅ 正确：批量查询
const users = await db.query('SELECT * FROM users')
const userIds = users.map(u => u.id)
const posts = await db.query(
  'SELECT * FROM posts WHERE user_id IN (?)',
  [userIds]
)
// 在代码中关联
const postsByUser = _.groupBy(posts, 'user_id')
users.forEach(user => {
  user.posts = postsByUser[user.id] || []
})
```

### 5.3 缓存策略

```javascript
const Redis = require('ioredis')
const redis = new Redis()

async function getUserWithCache(id) {
  const cacheKey = `user:${id}`
  
  // 先查缓存
  const cached = await redis.get(cacheKey)
  if (cached) {
    return JSON.parse(cached)
  }
  
  // 查数据库
  const user = await db.query('SELECT * FROM users WHERE id = ?', [id])
  
  // 写入缓存
  await redis.setex(cacheKey, 3600, JSON.stringify(user))  // 1 小时
  
  return user
}
```

### 6.3 健康检查

```javascript
app.get('/health', async (req, res) => {
  const health = {
    uptime: process.uptime(),
    timestamp: Date.now(),
    memory: process.memoryUsage(),
    status: 'ok'
  }
  
  // 检查数据库连接
  try {
    await db.ping()
    health.database = 'ok'
  } catch (e) {
    health.database = 'error'

  }

    await redis.ping()
    health.redis = 'ok'
  } catch (e) {
    health.redis = 'error'

  }
  
  const statusCode = health.status === 'ok' ? 200 : 503
  res.status(statusCode).json(health)
})
```

## 总结

Node.js 性能优化是系统工程。

**核心原则：**
1. **不阻塞事件循环** - 分片或用 Worker
2. **控制内存增长** - 限制缓存，检查泄漏
3. **合理并发** - 连接池、限流、队列
4. **持续监控** - 内存、延迟、慢请求

**优化顺序：**
```
监控定位 → 分析瓶颈 → 针对优化 → 验证效果
```

性能优化没有银弹，需要根据实际情况分析。先监控，再优化。
