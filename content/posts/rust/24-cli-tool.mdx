## 前言

恭喜你来到了 Rust 学习系列的最后一篇！

Rust 非常适合开发命令行工具（CLI）——编译为单一二进制文件、无需运行时、启动速度快、内存占用小。本文将使用 `clap` 库开发一个类似 `grep` 的文件搜索工具，综合运用前面学到的所有知识。

```bash
cd rfind
```

## clap 基础

### 定义命令行参数

```rust
use clap::{Parser, Subcommand};

#[derive(Parser)]
#[command(name = "rfind")]
#[command(author = "Your Name")]
#[command(version = "1.0")]
#[command(about = "一个简单的文件搜索工具", long_about = None)]
struct Cli {
    /// 要搜索的模式
    pattern: String,

    /// 搜索的目录
    #[arg(short, long, default_value = ".")]
    path: String,

    /// 是否区分大小写
    #[arg(short = 'i', long)]
    ignore_case: bool,

    /// 只显示文件名，不显示匹配内容
    #[arg(short, long)]
    files_only: bool,

    /// 显示行号
    #[arg(short = 'n', long)]
    line_number: bool,
}

fn main() {
    let cli = Cli::parse();

    println!("搜索模式: {}", cli.pattern);
    println!("搜索目录: {}", cli.path);
    println!("忽略大小写: {}", cli.ignore_case);
}
```

### 子命令

```rust
#[derive(Parser)]
struct Cli {
    #[command(subcommand)]
    command: Commands,
}

#[derive(Subcommand)]
enum Commands {
    /// 搜索文件内容
    Search {
        pattern: String,
        #[arg(short, long, default_value = ".")]
        path: String,
    },
    /// 统计文件信息
    Stats {
        path: String,
    },
}

fn main() {
    let cli = Cli::parse();

    match cli.command {
        Commands::Search { pattern, path } => {
            println!("搜索 '{}' 在 '{}'", pattern, path);
        }
        Commands::Stats { path } => {
            println!("统计 '{}'", path);
        }
    }
}
```

## 完整实现：文件搜索工具

### 核心逻辑

```rust
use clap::Parser;
use colored::*;
use std::fs;
use std::path::Path;
use walkdir::WalkDir;

#[derive(Parser)]
#[command(name = "rfind")]
#[command(version = "1.0")]
#[command(about = "在文件中搜索文本")]
struct Cli {
    /// 搜索模式
    pattern: String,

    /// 搜索目录
    #[arg(short, long, default_value = ".")]
    path: String,

    /// 忽略大小写
    #[arg(short = 'i', long)]
    ignore_case: bool,

    /// 显示行号
    #[arg(short, long, default_value_t = true)]
    line_number: bool,

    /// 文件扩展名过滤
    #[arg(short, long)]
    extension: Option<String>,

    /// 最大搜索深度
    #[arg(short = 'd', long)]
    max_depth: Option<usize>,
}

struct SearchOptions {
    pattern: String,
    ignore_case: bool,
    show_line_number: bool,
    extension: Option<String>,
    max_depth: Option<usize>,
}

struct MatchResult {
    file_path: String,
    line_number: usize,
    line: String,
}

fn search_file(path: &Path, options: &SearchOptions) -> Result<Vec<MatchResult>, String> {
    let content = fs::read_to_string(path).map_err(|e| e.to_string())?;
    let mut results = Vec::new();

    let pattern = if options.ignore_case {
        options.pattern.to_lowercase()
    } else {
        options.pattern.clone()
    };

    for (i, line) in content.lines().enumerate() {
        let line_to_check = if options.ignore_case {
            line.to_lowercase()
        } else {
            line.to_string()
        };

        if line_to_check.contains(&pattern) {
            results.push(MatchResult {
                file_path: path.display().to_string(),
                line_number: i + 1,
                line: line.to_string(),
            });
        }
    }

    Ok(results)
}

fn should_search(path: &Path, options: &SearchOptions) -> bool {
    if !path.is_file() {
        return false;
    }

    // 跳过隐藏文件
    if let Some(name) = path.file_name() {
        if name.to_string_lossy().starts_with('.') {
            return false;
        }
    }

    // 跳过二进制文件（通过扩展名过滤）
    if let Some(ext) = path.extension() {
        let ext = ext.to_string_lossy().to_lowercase();
        let binary_exts = ["png", "jpg", "jpeg", "gif", "ico", "pdf", "zip", "gz"];
        if binary_exts.contains(&ext.as_str()) {
            return false;
        }

        // 扩展名过滤
        if let Some(ref target_ext) = options.extension {
            if ext != target_ext.to_lowercase() {
                return false;
            }
        }
    }

    true
}

fn search_dir(path: &Path, options: &SearchOptions) -> Vec<MatchResult> {
    let mut all_results = Vec::new();

    let mut walker = WalkDir::new(path);
    if let Some(max_depth) = options.max_depth {
        walker = walker.max_depth(max_depth);
    }

    for entry in walker.into_iter().filter_map(|e| e.ok()) {
        let entry_path = entry.path();

        if !should_search(entry_path, options) {
            continue;
        }

        if let Ok(results) = search_file(entry_path, options) {
            all_results.extend(results);
        }
    }

    all_results
}

fn print_results(results: &[MatchResult], options: &SearchOptions) {
    let mut current_file = String::new();

    for result in results {
        // 文件名变化时打印文件名
        if result.file_path != current_file {
            current_file = result.file_path.clone();
            println!();
            println!("{}", current_file.green().bold());
        }

        if options.show_line_number {
            let line_num = format!("{:>4}: ", result.line_number).cyan();
            let highlighted = highlight_match(&result.line, &options.pattern, options.ignore_case);
            println!("  {}{}", line_num, highlighted);
        } else {
            let highlighted = highlight_match(&result.line, &options.pattern, options.ignore_case);
            println!("  {}", highlighted);
        }
    }
}

fn highlight_match(line: &str, pattern: &str, ignore_case: bool) -> ColoredString {
    let (line_lower, pattern_lower) = if ignore_case {
        (line.to_lowercase(), pattern.to_lowercase())
    } else {
        (line.to_string(), pattern.to_string())
    };

    if let Some(pos) = line_lower.find(&pattern_lower) {
        let before = &line[..pos];
        let match_str = &line[pos..pos + pattern.len()];
        let after = &line[pos + pattern.len()..];

        let mut result = String::new();
        result.push_str(before);
        result.push_str(&format!("{}", match_str.red().bold()));
        result.push_str(after);
        result.normal()
    } else {
        line.normal()
    }
}

fn main() {
    let cli = Cli::parse();

    let options = SearchOptions {
        pattern: cli.pattern,
        ignore_case: cli.ignore_case,
        show_line_number: cli.line_number,
        extension: cli.extension,
        max_depth: cli.max_depth,
    };

    let path = Path::new(&cli.path);

    if !path.exists() {
        eprintln!("错误: 路径 '{}' 不存在", cli.path);
        std::process::exit(1);
    }

    let results = search_dir(path, &options);

    if results.is_empty() {
        println!("没有找到匹配 '{}'", options.pattern);
    } else {
        println!("找到 {} 个匹配", results.len().to_string().yellow().bold());
        print_results(&results, &options);
    }
}
```

### 使用示例

```bash
# 基本搜索
cargo run -- "TODO" src/

# 忽略大小写
cargo run -- "rust" --ignore-case src/

# 只搜索 .rs 文件
cargo run -- "async" --extension rs

# 不显示行号
cargo run -- "error" --no-line-number

# 限制搜索深度
cargo run -- "test" --max-depth 2

# 编译为二进制
cargo build --release
./target/release/rfind "pattern" /path/to/search
```
