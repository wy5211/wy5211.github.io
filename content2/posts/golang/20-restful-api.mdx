## 项目概述

### 功能需求

我们的任务队列 API 需要实现以下功能：

1. **任务管理**
   - 创建任务
   - 查询任务状态
   - 列出任务

### Task 模型

```go
package models

import (
	"time"
	"gorm.io/gorm"
)

// Task 任务模型
type Task struct {
	ID          uint           `gorm:"primaryKey" json:"id"`
	Name        string         `gorm:"size:255;not null" json:"name"`
	Status      string         `gorm:"size:50;not null;default:'pending';index" json:"status"`
	Priority    int            `gorm:"not null;default:0;index" json:"priority"`
	MaxRetries  int            `gorm:"not null;default:3" json:"max_retries"`
	RetryCount  int            `gorm:"not null;default:0" json:"retry_count"`
	Result      string         `gorm:"type:text" json:"result,omitempty"`
	Error       string         `gorm:"type:text" json:"error,omitempty"`
	StartedAt   *time.Time     `json:"started_at,omitempty"`
	CompletedAt *time.Time     `json:"completed_at,omitempty"`
	CreatedAt   time.Time      `json:"created_at"`
	UpdatedAt   time.Time      `json:"updated_at"`
	DeletedAt   gorm.DeletedAt `gorm:"index" json:"-"`
}

// TableName 指定表名
func (Task) TableName() string {
	return "tasks"
}

// BeforeCreate 创建前钩子
func (t *Task) BeforeCreate(tx *gorm.DB) error {
	// 设置默认值
	if t.Status == "" {
		t.Status = "pending"
	}
	if t.MaxRetries == 0 {
		t.MaxRetries = 3
	}
	return nil
}

// CanRetry 检查是否可以重试
func (t *Task) CanRetry() bool {
	return t.RetryCount < t.MaxRetries
}

// MarkProcessing 标记为处理中
func (t *Task) MarkProcessing() {
	t.Status = "processing"
	now := time.Now()
	t.StartedAt = &now
}

// MarkCompleted 标记为完成
func (t *Task) MarkCompleted(result string) {
	t.Status = "completed"
	t.Result = result
	now := time.Now()
	t.CompletedAt = &now
}

// MarkFailed 标记为失败
func (t *Task) MarkFailed(err error) {
	t.Error = err.Error()
	t.RetryCount++

	if !t.CanRetry() {
		t.Status = "failed"
	} else {
		t.Status = "pending"
	}
}
```

## 数据访问层

```go
package repository

import (
	"fmt"
	"task-queue-api/models"
	"gorm.io/gorm"
)

// TaskRepository 任务仓库接口
type TaskRepository interface {
	Create(task *models.Task) error
	GetByID(id uint) (*models.Task, error)
	List(offset, limit int) ([]*models.Task, int64, error)
	Update(task *models.Task) error
	Delete(id uint) error
	GetStats() (*models.TaskStats, error)
	ClaimTask() (*models.Task, error)
}

// gormTaskRepository GORM 实现
type gormTaskRepository struct {
	db *gorm.DB
}

// NewTaskRepository 创建任务仓库
func NewTaskRepository(db *gorm.DB) TaskRepository {
	return &gormTaskRepository{db: db}
}

// Create 创建任务
func (r *gormTaskRepository) Create(task *models.Task) error {
	return r.db.Create(task).Error
}

// GetByID 根据 ID 获取任务
func (r *gormTaskRepository) GetByID(id uint) (*models.Task, error) {
	var task models.Task
	err := r.db.First(&task, id).Error
	if err == gorm.ErrRecordNotFound {
		return nil, fmt.Errorf("任务不存在")
	}
	return &task, err
}

// List 列出任务
func (r *gormTaskRepository) List(offset, limit int) ([]*models.Task, int64, error) {
	var tasks []*models.Task
	var total int64

	// 获取总数
	if err := r.db.Model(&models.Task{}).Count(&total).Error; err != nil {
		return nil, 0, err
	}

	// 获取列表
	err := r.db.Order("priority DESC, created_at ASC").
		Offset(offset).
		Limit(limit).
		Find(&tasks).Error

	return tasks, total, err
}

// Update 更新任务
func (r *gormTaskRepository) Update(task *models.Task) error {
	return r.db.Save(task).Error
}

// Delete 删除任务
func (r *gormTaskRepository) Delete(id uint) error {
	result := r.db.Delete(&models.Task{}, id)
	if result.Error != nil {
		return result.Error
	}
	if result.RowsAffected == 0 {
		return fmt.Errorf("任务不存在")
	}
	return nil
}

// GetStats 获取统计信息
func (r *gormTaskRepository) GetStats() (*models.TaskStats, error) {
	var stats models.TaskStats

	// 统计各状态任务数量
	r.db.Model(&models.Task{}).Count(&stats.Total)
	r.db.Model(&models.Task{}).Where("status = ?", "pending").Count(&stats.Pending)
	r.db.Model(&models.Task{}).Where("status = ?", "processing").Count(&stats.Processing)
	r.db.Model(&models.Task{}).Where("status = ?", "completed").Count(&stats.Completed)
	r.db.Model(&models.Task{}).Where("status = ?", "failed").Count(&stats.Failed)

	return &stats, nil
}

// ClaimTask 领取任务
func (r *gormTaskRepository) ClaimTask() (*models.Task, error) {
	var task models.Task

	// 使用事务确保原子性
	err := r.db.Transaction(func(tx *gorm.DB) error {
		// 查找并锁定任务
		err := tx.Where("status = ? AND retry_count < max_retries", "pending").
			Order("priority DESC, created_at ASC").
			First(&task).
			Error

		if err == gorm.ErrRecordNotFound {
			// 没有可用任务
			return nil
		}
		if err != nil {
			return err
		}

		// 更新状态
		return tx.Model(&task).Updates(map[string]interface{}{
			"status":     "processing",
			"retry_count": task.RetryCount + 1,
		}).Error
	})

	if err != nil {
		return nil, err
	}

	if task.ID == 0 {
		return nil, nil // 没有可用任务
	}

	return &task, nil
}
```

```go
package service

import (
	"context"
	"fmt"
	"task-queue-api/models"
	"task-queue-api/repository"
	"time"
)

// TaskService 任务服务接口
type TaskService interface {
	CreateTask(ctx context.Context, name string, priority int) (*models.Task, error)
	GetTask(ctx context.Context, id uint) (*models.Task, error)
	ListTasks(ctx context.Context, page, pageSize int) ([]*models.Task, int64, error)
	DeleteTask(ctx context.Context, id uint) error
	GetStats(ctx context.Context) (*models.TaskStats, error)
}

// taskService 任务服务实现
type taskService struct {
	repo   repository.TaskRepository
	taskCh chan *models.Task
}

// NewTaskService 创建任务服务
func NewTaskService(repo repository.TaskRepository) TaskService {
	return &taskService{
		repo:   repo,
		taskCh: make(chan *models.Task, 100),
	}
}

// CreateTask 创建任务
func (s *taskService) CreateTask(ctx context.Context, name string, priority int) (*models.Task, error) {
	// 参数验证
	if name == "" {
		return nil, fmt.Errorf("任务名称不能为空")
	}
	if priority < 0 || priority > 10 {
		return nil, fmt.Errorf("优先级必须在 0-10 之间")
	}

	// 创建任务
	task := &models.Task{
		Name:     name,
		Priority: priority,
		Status:   "pending",
	}

	// 保存到数据库
	if err := s.repo.Create(task); err != nil {
		return nil, fmt.Errorf("创建任务失败: %w", err)
	}

	// 发送到任务队列
	select {
	case s.taskCh <- task:
		return task, nil
	case <-ctx.Done():
		return nil, ctx.Err()
	}
}

// GetTask 获取任务
func (s *taskService) GetTask(ctx context.Context, id uint) (*models.Task, error) {
	return s.repo.GetByID(id)
}

// ListTasks 列出任务
func (s *taskService) ListTasks(ctx context.Context, page, pageSize int) ([]*models.Task, int64, error) {
	if page < 1 {
		page = 1
	}
	if pageSize < 1 || pageSize > 100 {
		pageSize = 10
	}

	offset := (page - 1) * pageSize
	return s.repo.List(offset, pageSize)
}

// DeleteTask 删除任务
func (s *taskService) DeleteTask(ctx context.Context, id uint) error {
	return s.repo.Delete(id)
}

// GetStats 获取统计信息
func (s *taskService) GetStats(ctx context.Context) (*models.TaskStats, error) {
	return s.repo.GetStats()
}

// TaskChan 获取任务通道
func (s *taskService) TaskChan() <-chan *models.Task {
	return s.taskCh
}
```

## HTTP 处理器

```go
package handler

import (
	"net/http"
	"task-queue-api/models"
	"task-queue-api/service"
	"github.com/gin-gonic/gin"
)

// TaskHandler 任务处理器
type TaskHandler struct {
	service service.TaskService
}

// NewTaskHandler 创建任务处理器
func NewTaskHandler(service service.TaskService) *TaskHandler {
	return &TaskHandler{service: service}
}

// CreateTaskRequest 创建任务请求
type CreateTaskRequest struct {
	Name     string `json:"name" binding:"required"`
	Priority int    `json:"priority" binding:"min=0,max=10"`
}

// CreateTask 创建任务
func (h *TaskHandler) CreateTask(c *gin.Context) {
	var req CreateTaskRequest
	if err := c.ShouldBindJSON(&req); err != nil {
		c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
		return
	}

	task, err := h.service.CreateTask(c.Request.Context(), req.Name, req.Priority)
	if err != nil {
		c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
		return
	}

	c.JSON(http.StatusCreated, task)
}

// GetTask 获取任务
func (h *TaskHandler) GetTask(c *gin.Context) {
	id := c.Param("id")
	var taskID uint
	if _, err := fmt.Sscanf(id, "%d", &taskID); err != nil {
		c.JSON(http.StatusBadRequest, gin.H{"error": "无效的任务 ID"})
		return
	}

	task, err := h.service.GetTask(c.Request.Context(), taskID)
	if err != nil {
		c.JSON(http.StatusNotFound, gin.H{"error": err.Error()})
		return
	}

	c.JSON(http.StatusOK, task)
}

// ListTasks 列出任务
func (h *TaskHandler) ListTasks(c *gin.Context) {
	page := c.DefaultQuery("page", "1")
	pageSize := c.DefaultQuery("page_size", "10")

	var p, ps int
	fmt.Sscanf(page, "%d", &p)
	fmt.Sscanf(pageSize, "%d", &ps)

	tasks, total, err := h.service.ListTasks(c.Request.Context(), p, ps)
	if err != nil {
		c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
		return
	}

	c.JSON(http.StatusOK, gin.H{
		"tasks": tasks,
		"total": total,
		"page":  p,
		"page_size": ps,
	})
}

// DeleteTask 删除任务
func (h *TaskHandler) DeleteTask(c *gin.Context) {
	id := c.Param("id")
	var taskID uint
	if _, err := fmt.Sscanf(id, "%d", &taskID); err != nil {
		c.JSON(http.StatusBadRequest, gin.H{"error": "无效的任务 ID"})
		return
	}

	if err := h.service.DeleteTask(c.Request.Context(), taskID); err != nil {
		c.JSON(http.StatusNotFound, gin.H{"error": err.Error()})
		return
	}

	c.Status(http.StatusNoContent)
}

// GetStats 获取统计信息
func (h *TaskHandler) GetStats(c *gin.Context) {
	stats, err := h.service.GetStats(c.Request.Context())
	if err != nil {
		c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
		return
	}

	c.JSON(http.StatusOK, stats)
}
```

## Worker 实现

```go
package worker

import (
	"context"
	"fmt"
	"log"
	"time"
	"task-queue-api/models"
	"task-queue-api/repository"
)

// Worker worker
type Worker struct {
	id   int
	repo repository.TaskRepository
}

// NewWorker 创建 worker
func NewWorker(id int, repo repository.TaskRepository) *Worker {
	return &Worker{
		id:   id,
		repo: repo,
	}
}

// Start 启动 worker
func (w *Worker) Start(ctx context.Context) {
	log.Printf("Worker %d 启动", w.id)

	ticker := time.NewTicker(1 * time.Second)
	defer ticker.Stop()

	for {
		select {
		case <-ctx.Done():
			log.Printf("Worker %d 收到停止信号", w.id)
			return

		case <-ticker.C:
			// 领取并处理任务
			if err := w.processTask(ctx); err != nil {
				log.Printf("Worker %d 处理任务出错: %v", w.id, err)
			}
		}
	}
}

// processTask 处理任务
func (w *Worker) processTask(ctx context.Context) error {
	// 从数据库领取任务
	task, err := w.repo.ClaimTask()
	if err != nil {
		return fmt.Errorf("领取任务失败: %w", err)
	}

	if task == nil {
		// 没有可用任务
		return nil
	}

	log.Printf("Worker %d 开始处理任务 %d: %s", w.id, task.ID, task.Name)

	// 更新状态为处理中
	task.MarkProcessing()
	if err := w.repo.Update(task); err != nil {
		return fmt.Errorf("更新任务状态失败: %w", err)
	}

	// 执行任务
	result, err := w.executeTask(ctx, task)

	// 更新任务结果
	if err != nil {
		task.MarkFailed(err)
		log.Printf("Worker %d 任务 %d 失败: %v", w.id, task.ID, err)
	} else {
		task.MarkCompleted(result)
		log.Printf("Worker %d 任务 %d 完成", w.id, task.ID)
	}

	if err := w.repo.Update(task); err != nil {
		return fmt.Errorf("更新任务结果失败: %w", err)
	}

	return nil
}

// executeTask 执行任务逻辑
func (w *Worker) executeTask(ctx context.Context, task *models.Task) (string, error) {
	// 模拟任务执行
	// 实际应用中，这里可能是 HTTP 请求、数据库操作、文件处理等
	executionTime := time.Duration(task.Priority) * 100 * time.Millisecond

	select {
	case <-time.After(executionTime):
		result := fmt.Sprintf("任务 %s 执行完成，耗时 %v", task.Name, executionTime)
		return result, nil

	case <-ctx.Done():
		return "", ctx.Err()
	}
}

// WorkerPool Worker 池
type WorkerPool struct {
	workers []*Worker
	ctx     context.Context
	cancel  context.CancelFunc
}

// NewWorkerPool 创建 Worker 池
func NewWorkerPool(workerCount int, repo repository.TaskRepository) *WorkerPool {
	ctx, cancel := context.WithCancel(context.Background())

	pool := &WorkerPool{
		ctx:     ctx,
		cancel:  cancel,
		workers: make([]*Worker, workerCount),
	}

	for i := 0; i < workerCount; i++ {
		pool.workers[i] = NewWorker(i, repo)
	}

	return pool
}

// Start 启动 Worker 池
func (p *WorkerPool) Start() {
	for _, worker := range p.workers {
		go worker.Start(p.ctx)
	}
	log.Printf("Worker 池已启动，Worker 数量: %d", len(p.workers))
}

// Stop 停止 Worker 池
func (p *WorkerPool) Stop() {
	log.Println("正在停止 Worker 池...")
	p.cancel()
	log.Println("Worker 池已停止")
}
```

```go
package router

import (
	"task-queue-api/handler"
	"task-queue-api/middleware"

	"github.com/gin-gonic/gin"
)

// SetupRouter 设置路由
func SetupRouter(taskHandler *handler.TaskHandler) *gin.Engine {
	r := gin.Default()

	// 添加日志中间件
	r.Use(middleware.Logger())

	// API 路由组
	v1 := r.Group("/api/v1")
	{
		// 任务路由
		tasks := v1.Group("/tasks")
		{
			tasks.POST("", taskHandler.CreateTask)           // 创建任务
			tasks.GET("", taskHandler.ListTasks)             // 列出任务
			tasks.GET("/:id", taskHandler.GetTask)           // 获取任务
			tasks.DELETE("/:id", taskHandler.DeleteTask)     // 删除任务
		}

		// 统计路由
		v1.GET("/stats", taskHandler.GetStats)              // 获取统计
	}

	return r
}
```

```go
package middleware

import (
	"log"
	"time"

	"github.com/gin-gonic/gin"
)

// Logger 日志中间件
func Logger() gin.HandlerFunc {
	return func(c *gin.Context) {
		// 记录开始时间
		start := time.Now()

		// 处理请求
		c.Next()

		// 记录日志
		duration := time.Since(start)
		log.Printf("[%s] %s %s %d %v",
			c.Request.Method,
			c.Request.URL.Path,
			c.ClientIP(),
			c.Writer.Status(),
			duration,
		)
	}
}
```

```go
package config

import (

// Config 配置
type Config struct {

	Database DatabaseConfig
	Worker   WorkerConfig

	Port         string
	ReadTimeout  time.Duration
	WriteTimeout time.Duration

	Host     string
	Port     string
	User     string
	Password string
	Database string
}

// WorkerConfig Worker 配置
type WorkerConfig struct {
	Count int
}

// Load 加载配置
func Load() *Config {
	return &Config{
		Server: ServerConfig{
			Port:         getEnv("SERVER_PORT", "8080"),
			ReadTimeout:  10 * time.Second,
			WriteTimeout: 10 * time.Second,
		},
		Database: DatabaseConfig{
			Host:     getEnv("DB_HOST", "localhost"),
			Port:     getEnv("DB_PORT", "3306"),
			User:     getEnv("DB_USER", "root"),
			Password: getEnv("DB_PASSWORD", ""),
			Database: getEnv("DB_NAME", "task_queue"),
		},
		Worker: WorkerConfig{
			Count: getEnvInt("WORKER_COUNT", 3),
		},
	}
}

func getEnv(key, defaultValue string) string {
	if value := os.Getenv(key); value != "" {
		return value
	}
	return defaultValue
}

func getEnvInt(key string, defaultValue int) int {
	if value := os.Getenv(key); value != "" {
		var i int
		fmt.Sscanf(value, "%d", &i)
		return i
	}
	return defaultValue
}
```

```go
	defer workerPool.Stop()

	// 设置路由
	r := router.SetupRouter(taskHandler)

	// 创建 HTTP 服务器
	srv := &http.Server{
		Addr:         ":" + cfg.Server.Port,
		Handler:      r,
		ReadTimeout:  cfg.Server.ReadTimeout,
		WriteTimeout: cfg.Server.WriteTimeout,
	}

	// 启动服务器
	go func() {
		log.Printf("服务器启动在端口 %s", cfg.Server.Port)
		if err := srv.ListenAndServe(); err != nil && err != http.ErrServerClosed {
			log.Fatal("服务器启动失败:", err)
		}
	}()

	// 优雅关闭
	quit := make(chan os.Signal, 1)
	signal.Notify(quit, syscall.SIGINT, syscall.SIGTERM)
	<-quit

	log.Println("正在关闭服务器...")

	ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
	defer cancel()

	if err := srv.Shutdown(ctx); err != nil {
		log.Fatal("服务器关闭失败:", err)
	}

	log.Println("服务器已关闭")
}

// initDB 初始化数据库
func initDB(cfg *config.Config) (*gorm.DB, error) {
	dsn := fmt.Sprintf("%s:%s@tcp(%s:%s)/%s?charset=utf8mb4&parseTime=True&loc=Local",
		cfg.Database.User,
		cfg.Database.Password,
		cfg.Database.Host,
		cfg.Database.Port,
		cfg.Database.Database,
	)

	db, err := gorm.Open(mysql.Open(dsn), &gorm.Config{})
	if err != nil {
		return nil, err
	}

	// 配置连接池
	sqlDB, err := db.DB()
	if err != nil {
		return nil, err
	}

	sqlDB.SetMaxOpenConns(25)
	sqlDB.SetMaxIdleConns(5)
	sqlDB.SetConnMaxLifetime(5 * time.Minute)

	return db, nil
}
```

## Trade-off 分析

### 架构选择

**分层架构 vs 单体架构**

我们选择了分层架构（Handler → Service → Repository）：

优势：

- 职责清晰，易于维护
- 便于测试和mock
- 可以独立演进各层

劣势：

- 代码量增加
- 可能有过度设计

对于这个规模的项目，分层架构是合理的选择。

### 数据库选择

**MySQL vs SQLite vs PostgreSQL**

选择了 MySQL：

- 适合生产环境
- 事务支持完善
- 工具生态成熟

对于小型项目，SQLite 更简单；对于需要高级特性的项目，PostgreSQL 更强大。

### 并发模型

**Channel vs 数据库轮询**

我们选择了数据库轮询：

- 简单可靠
- 支持多实例部署
- 持久化保证

对于高吞吐场景，基于 Channel 的内存队列更高效。

### 测试策略

**单元测试 vs 集成测试**

我们采用了混合策略：

- 核心逻辑：单元测试
- API 端点：集成测试
- 平衡了覆盖率和开发成本
