## 为什么需要理解数据类型

在构建任务队列 API 时，你需要用数组或切片来存储任务列表，用 map 来快速查找任务，用结构体来定义任务的数据模型。选择正确的数据类型不仅影响代码的清晰度，还直接影响程序的性能。

Go 的类型系统在简单和强大之间找到了平衡。切片是对动态数组的抽象，map 提供了哈希表功能，指针让你能高效地传递大对象。这篇文章会用任务队列的实际场景来讲解这些核心数据类型。

## 数组：固定大小的序列

数组是固定长度的同类型元素序列。在 Go 中，数组的大小是类型的一部分，这意味着 `[3]int` 和 `[5]int` 是两种完全不同的类型。

数组在任务队列中可以用于固定大小的缓冲区或预先知道的元素集合。

```go
package main

import "fmt"

func main() {
    // 声明一个固定大小的数组（零值初始化）
    var taskBuffer [5]string
    fmt.Printf("初始缓冲区: %v\n", taskBuffer)  // [    ]

    // 初始化数组
    taskQueue := [5]string{
        "任务1: 数据备份",
        "任务2: 日志清理",
        "任务3: 报表生成",
    }
    fmt.Printf("任务队列: %v\n", taskQueue)

    // 让编译器计算数组长度
    taskIDs := [...]int{1001, 1002, 1003, 1004}
    fmt.Printf("任务 ID 数组: %v\n", taskIDs)
    fmt.Printf("数组长度: %d\n", len(taskIDs))

    // 访问和修改数组元素
    fmt.Printf("第一个任务: %s\n", taskQueue[0])
    taskQueue[0] = "任务1: 紧急数据备份"
    fmt.Printf("更新后的第一个任务: %s\n", taskQueue[0])

    // 遍历数组
    fmt.Println("\n=== 遍历任务队列 ===")
    for i, task := range taskQueue {
        if task != "" {  // 跳过空元素
            fmt.Printf("位置 %d: %s\n", i, task)
        }
    }

    // 数组是值类型，赋值时会复制整个数组
    queueCopy := taskQueue
    queueCopy[0] = "任务1: 已修改"
    fmt.Printf("\n原始队列[0]: %s\n", taskQueue[0])
    fmt.Printf("复制队列[0]: %s\n", queueCopy[0])
}
```

## 切片：动态数组

切片是对数组的抽象，提供了动态大小的灵活序列。切片是 Go 中最常用的数据结构，几乎所有的集合操作都用切片。

切片由三部分组成：指向底层数组的指针、长度和容量。

```go
package main

import "fmt"

func main() {
    // 创建切片的几种方式
    // 1. 直接声明并初始化
    tasks := []string{
        "数据备份",
        "日志清理",
        "报表生成",
        "邮件发送",
    }
    fmt.Printf("任务列表: %v\n", tasks)
    fmt.Printf("长度: %d, 容量: %d\n", len(tasks), cap(tasks))

    // 2. 使用 make 创建（指定长度和容量）
    taskResults := make([]string, 0, 10)  // 长度0，容量10
    fmt.Printf("\n初始结果列表: %v\n", taskResults)
    fmt.Printf("长度: %d, 容量: %d\n", len(taskResults), cap(taskResults))

    // 3. 从数组切片
    taskArray := [5]string{"A", "B", "C", "D", "E"}
    priorityTasks := taskArray[1:4]  // 包含索引 1,2,3（不包括4）
    fmt.Printf("\n高优先级任务: %v\n", priorityTasks)

    // 添加元素到切片
    taskResults = append(taskResults, "成功")
    taskResults = append(taskResults, "失败", "超时")
    fmt.Printf("\n添加结果后的列表: %v\n", taskResults)

    // 合并切片
    moreTasks := []string{"系统更新", "安全扫描"}
    allTasks := append(tasks, moreTasks...)
    fmt.Printf("\n合并后的任务列表: %v\n", allTasks)

    // 切片的切片
    fmt.Println("\n=== 任务批次处理 ===")
    batchSize := 2
    for i := 0; i < len(allTasks); i += batchSize {
        end := i + batchSize
        if end > len(allTasks) {
            end = len(allTasks)
        }
        batch := allTasks[i:end]
        fmt.Printf("批次 %d: %v\n", i/batchSize+1, batch)
    }

    // 复制切片（避免共享底层数组）
    originalTasks := []string{"任务1", "任务2", "任务3"}
    copiedTasks := make([]string, len(originalTasks))
    copy(copiedTasks, originalTasks)

    copiedTasks[0] = "修改后的任务1"
    fmt.Printf("\n原始任务: %v\n", originalTasks)
    fmt.Printf("复制任务: %v\n", copiedTasks)
}
```

切片在 Go 中如此重要，是因为它在动态性和性能之间取得了平衡。当你不确定数据量时，用切片；当你需要频繁添加删除元素时，切片是最佳选择。

## 映射：键值对集合

map 是 Go 的哈希表实现，提供了快速的键值对存储和查找。在任务队列中，你可以用 map 来存储任务 ID 到任务详情的映射，实现 O(1) 时间复杂度的查找。

```go
package main

import "fmt"

func main() {
    // 创建 map 的几种方式
    // 1. 使用 make（推荐）
    taskMap := make(map[string]int)

    // 2. 直接初始化
    taskPriority := map[string]int{
        "数据备份":    8,
        "日志清理":    3,
        "报表生成":    5,
    }
    fmt.Printf("任务优先级: %v\n", taskPriority)

    // 添加和修改元素
    taskMap["task_1001"] = 1001
    taskMap["task_1002"] = 1002
    fmt.Printf("\n任务映射: %v\n", taskMap)

    // 获取值（并检查键是否存在）
    priority, exists := taskPriority["数据备份"]
    if exists {
        fmt.Printf("\n数据备份的优先级: %d\n", priority)
    } else {
        fmt.Println("\n任务不存在")
    }

    // 删除元素
    delete(taskPriority, "日志清理")
    fmt.Printf("\n删除日志清理后的映射: %v\n", taskPriority)

    // 遍历 map（注意：遍历顺序是随机的）
    fmt.Println("\n=== 所有任务优先级 ===")
    for task, priority := range taskPriority {
        fmt.Printf("%s: %d\n", task, priority)
    }

    // 实际应用：任务状态追踪
    type TaskStatus struct {
        Status    string
        Progress  int
        StartTime int64
    }

    taskStatusMap := make(map[int]*TaskStatus)

    // 添加任务状态
    taskStatusMap[1001] = &TaskStatus{
        Status:    "running",
        Progress:  45,
        StartTime: 1684567890,
    }

    taskStatusMap[1002] = &TaskStatus{
        Status:    "pending",
        Progress:  0,
        StartTime: 0,
    }

    // 查询任务状态
    fmt.Println("\n=== 任务状态查询 ===")
    taskID := 1001
    if status, ok := taskStatusMap[taskID]; ok {
        fmt.Printf("任务 %d 状态: %s, 进度: %d%%\n",
            taskID, status.Status, status.Progress)
    }
}
```

map 的使用场景很广泛：缓存、计数器、索引、配置存储等。但要注意，map 不是线程安全的，如果需要在多个 goroutine 中访问，需要加锁或使用 `sync.Map`。

## 指针：内存地址

指针存储变量的内存地址。Go 的指针比 C 的指针简单，不支持指针运算，这让代码更安全。

指针的主要用途是：避免大对象的复制、在函数间修改原始数据、实现引用语义。

```go
package main

import "fmt"

func main() {
    // 基本指针操作
    taskCount := 100
    ptr := &taskCount  // 获取 taskCount 的地址

    fmt.Printf("任务数量: %d\n", taskCount)
    fmt.Printf("任务数量的地址: %p\n", ptr)
    fmt.Printf("通过指针访问的值: %d\n", *ptr)

    // 通过指针修改原值
    *ptr = 150
    fmt.Printf("修改后的任务数量: %d\n", taskCount)

    // 指针的零值是 nil
    var nilPtr *int
    fmt.Printf("\nnil 指针的值: %v\n", nilPtr)

    // 使用指针作为函数参数，避免复制大对象
    task := Task{
        ID:          1001,
        Name:        "数据备份任务",
        Description: "每日全量数据备份",
        Priority:    8,
    }

    fmt.Printf("\n修改前: %s (优先级: %d)\n", task.Name, task.Priority)
    updateTaskPriority(&task, 9)
    fmt.Printf("修改后: %s (优先级: %d)\n", task.Name, task.Priority)

    // 指针和值的区别
    task1 := Task{ID: 1001, Name: "任务1"}
    task2 := task1           // 值复制：task2 是 task1 的副本
    task3 := &task1          // 引用：task3 指向 task1

    task2.Name = "修改后的任务2"
    task3.Name = "通过指针修改"

    fmt.Printf("\ntask1.Name: %s\n", task1.Name)
    fmt.Printf("task2.Name: %s\n", task2.Name)
    fmt.Printf("task3.Name: %s\n", task3.Name)
}

type Task struct {
    ID          int
    Name        string
    Description string
    Priority    int
}

func updateTaskPriority(task *Task, newPriority int) {
    task.Priority = newPriority
}
```

指针在 Go 中是必要的，但不要滥用。对于小对象，值传递通常更简单且性能足够。对于大对象或需要修改原值的情况，指针是正确的选择。

## 结构体：自定义数据类型

结构体是 Go 中组织数据的主要方式。它把不同类型的数据字段组合成一个逻辑单元，类似于其他语言中的类，但没有继承和方法重载的复杂性。

在任务队列中，你会用结构体来定义任务、Worker、结果等实体。

```go
package main

import "fmt"
import "time"

// Task 定义任务结构
type Task struct {
    ID          int
    Name        string
    Type        string
    Priority    int
    Status      string
    CreatedAt   time.Time
    CompletedAt *time.Time  // 指针类型，表示可选字段（零值为 nil）
}

// 方法：计算任务耗时
func (t Task) Duration() time.Duration {
    if t.CompletedAt == nil {
        return 0
    }
    return t.CompletedAt.Sub(t.CreatedAt)
}

// 方法：标记任务完成
func (t *Task) Complete() {
    now := time.Now()
    t.Status = "completed"
    t.CompletedAt = &now
}

func main() {
    // 创建结构体实例的几种方式
    // 1. 按字段名初始化（推荐）
    task1 := Task{
        ID:        1001,
        Name:      "数据备份",
        Type:      "backup",
        Priority:  8,
        Status:    "pending",
        CreatedAt: time.Now(),
    }

    // 2. 按顺序初始化（不推荐，易错）
    task2 := Task{1002, "日志清理", "cleanup", 5, "running", time.Now(), nil}

    // 3. 使用 new 关键字（返回指针）
    task3 := new(Task)
    task3.ID = 1003
    task3.Name = "报表生成"
    task3.Priority = 6
    task3.Status = "pending"
    task3.CreatedAt = time.Now()

    fmt.Printf("任务1: %+v\n", task1)
    fmt.Printf("任务2: %+v\n", task2)
    fmt.Printf("任务3: %+v\n", *task3)

    // 调用方法
    fmt.Printf("\n任务1 耗时: %v\n", task1.Duration())

    // 完成任务
    task1.Complete()
    fmt.Printf("任务1 完成后: %+v\n", task1)
    fmt.Printf("任务1 耗时: %v\n", task1.Duration())

    // 结构体数组
    taskQueue := []Task{
        {ID: 1001, Name: "数据备份", Priority: 8, Status: "pending", CreatedAt: time.Now()},
        {ID: 1002, Name: "日志清理", Priority: 5, Status: "running", CreatedAt: time.Now()},
        {ID: 1003, Name: "报表生成", Priority: 6, Status: "pending", CreatedAt: time.Now()},
    }

    fmt.Println("\n=== 按优先级排序的任务队列 ===")
    // 简单排序（后续会详细介绍排序）
    for i := 0; i < len(taskQueue); i++ {
        for j := i + 1; j < len(taskQueue); j++ {
            if taskQueue[i].Priority < taskQueue[j].Priority {
                taskQueue[i], taskQueue[j] = taskQueue[j], taskQueue[i]
            }
        }
    }

    for _, task := range taskQueue {
        fmt.Printf("ID: %d, 名称: %s, 优先级: %d, 状态: %s\n",
            task.ID, task.Name, task.Priority, task.Status)
    }
}
```

结构体是 Go 面向对象编程的基础。它通过组合而非继承来实现代码复用，这种方式更灵活且避免了继承链的复杂性。

## 实战例子：简单的任务管理器

把学到的数据类型整合起来，实现一个基本的任务管理器。

```go
package main

import (
    "fmt"
    "time"
)

// Task 任务结构
type Task struct {
    ID          int
    Name        string
    Priority    int
    Status      string
    CreatedAt   time.Time
    CompletedAt *time.Time
}

// TaskManager 任务管理器
type TaskManager struct {
    tasks    map[int]*Task      // 任务存储
    queue    []int              // 任务队列（存储任务ID）
    nextID   int                // 下一个任务ID
}

func NewTaskManager() *TaskManager {
    return &TaskManager{
        tasks: make(map[int]*Task),
        queue: make([]int, 0),
        nextID: 1001,
    }
}

func (tm *TaskManager) AddTask(name string, priority int) int {
    task := &Task{
        ID:        tm.nextID,
        Name:      name,
        Priority:  priority,
        Status:    "pending",
        CreatedAt: time.Now(),
    }

    tm.tasks[task.ID] = task
    tm.queue = append(tm.queue, task.ID)
    tm.nextID++

    return task.ID
}

func (tm *TaskManager) GetTask(id int) (*Task, bool) {
    task, exists := tm.tasks[id]
    return task, exists
}

func (tm *TaskManager) CompleteTask(id int) bool {
    task, exists := tm.tasks[id]
    if !exists {
        return false
    }

    now := time.Now()
    task.Status = "completed"
    task.CompletedAt = &now
    return true
}

func (tm *TaskManager) ListTasks() []*Task {
    result := make([]*Task, 0, len(tm.queue))
    for _, id := range tm.queue {
        if task := tm.tasks[id]; task != nil {
            result = append(result, task)
        }
    }
    return result
}

func main() {
    manager := NewTaskManager()

    // 添加任务
    fmt.Println("=== 添加任务 ===")
    id1 := manager.AddTask("数据备份", 8)
    id2 := manager.AddTask("日志清理", 5)
    id3 := manager.AddTask("报表生成", 6)

    fmt.Printf("添加任务 ID: %d, %d, %d\n", id1, id2, id3)

    // 查询任务
    fmt.Println("\n=== 查询任务 ===")
    if task, ok := manager.GetTask(id1); ok {
        fmt.Printf("任务 %d: %s (优先级: %d, 状态: %s)\n",
            task.ID, task.Name, task.Priority, task.Status)
    }

    // 完成任务
    fmt.Println("\n=== 完成任务 ===")
    if manager.CompleteTask(id1) {
        fmt.Printf("任务 %d 已完成\n", id1)
    }

    // 列出所有任务
    fmt.Println("\n=== 所有任务 ===")
    for _, task := range manager.ListTasks() {
        duration := "未完成"
        if task.CompletedAt != nil {
            duration = task.CompletedAt.Sub(task.CreatedAt).String()
        }
        fmt.Printf("ID: %d, 名称: %s, 优先级: %d, 状态: %s, 耗时: %s\n",
            task.ID, task.Name, task.Priority, task.Status, duration)
    }
}
```

## 权衡与考虑

Go 的数据类型设计体现了实用主义。数组固定大小但零拷贝，切片动态但可能涉及内存重新分配。map 提供了 O(1) 查找但遍历顺序不确定。指针提供了引用语义但增加了复杂性。

选择数据类型时要考虑：数据量是否确定、是否需要频繁增删、查找性能要求、内存使用效率。切片是最通用的选择，map 适合快速查找，数组适合固定大小的场景。

结构体是 Go 组织数据的核心，通过组合而非继承实现了灵活性。指针接收者和值接收者的选择会影响性能和语义，需要根据具体场景决定。
